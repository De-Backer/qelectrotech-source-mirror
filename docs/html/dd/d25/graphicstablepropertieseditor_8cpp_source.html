119 QUndoCommand *GraphicsTablePropertiesEditor::associatedUndo() const
120 {
121  if (m_live_edit)
122  {
123  if (!qFuzzyCompare(ui->m_x_pos->value(), m_table_item->pos().x())) {
124  auto undo = new QPropertyUndoCommand(m_table_item.data(), "x", m_table_item->pos().x(), ui->m_x_pos->value());
125  undo->setAnimated(true, false);
126  undo->setText(tr("Déplacer un tableau"));
127  return undo;
128  }
129 
130  if (!qFuzzyCompare(ui->m_y_pos->value(), m_table_item->pos().y())) {
131  auto undo = new QPropertyUndoCommand(m_table_item.data(), "y", m_table_item->pos().y(), ui->m_y_pos->value());
132  undo->setAnimated(true, false);
133  undo->setText(tr("Déplacer un tableau"));
134  return undo;
135  }
136 
137  if (ui->m_display_n_row_sb->value() != m_table_item->displayNRow()) {
138  auto undo = new QPropertyUndoCommand(m_table_item.data(), "displayNRow", m_table_item->displayNRow(), ui->m_display_n_row_sb->value());
139  undo->setText(tr("Modifier le nombre de ligne affiché par un tableau"));
140  return undo;
141  }
142 
143  QMargins edited_header_margins(ui->m_header_left_margin->value(),
144  ui->m_header_top_margin->value(),
145  ui->m_header_right_margin->value(),
146  ui->m_header_bottom_margin->value());

157  ui->m_table_top_margin->value(),
158  ui->m_table_right_margin->value(),

161  if (edited_table_margins != model_margins)
162  {
163  auto undo = new ModelIndexCommand(m_table_item->model(), m_table_item->model()->index(0,0));
164  undo->setData(QETUtils::marginsToString(edited_table_margins), Qt::UserRole+1);
165  undo->setText(tr("Modifier les marges d'un tableau"));

170  {
171  auto undo = new ModelHeaderDataCommand(m_table_item->model());
172  undo->setData(0, Qt::Horizontal, m_header_button_group->checkedId(), Qt::TextAlignmentRole);
173  undo->setText(tr("Modifier l'alignement d'une en tête de tableau"));
174  return undo;
175  }
176

281  m_other_table_vector.append(item_);
282  ui->m_previous_table_cb->addItem(item_->tableName(), m_other_table_vector.indexOf(item_));

284  }

311 }

313 void GraphicsTablePropertiesEditor::updateInfoLabel()
314 {
315  auto table_ = m_table_item;
316  while (table_->previousTable()) { table_ = table_->previousTable();}
317 
318  int count_ = 0;
319  bool infinite = false;
320  if (table_->displayNRow() <= 0) {
321  infinite = true;
322  } else {
323  count_ = table_->displayNRow();
324  }
325 
326  while (table_->nextTable())
327  {
328  table_ = table_->nextTable();
329  if (table_->displayNRow() <= 0) {
330  infinite = true;
331  } else {
332  count_ += table_->displayNRow();
333  }
334  }
335 
336  auto value = m_table_item->model()->rowCount() - count_;
337  if (value > 0 && !infinite) {
338  ui->m_info_label->setText(tr("<center>ATTENTION :</center>\n il manque %1 lignes afin d'afficher l'intégralité des informations").arg(value));
339  ui->m_info_label->show();
340  } else {
341  ui->m_info_label->hide();
342  }
343 }

425 void GraphicsTablePropertiesEditor::on_m_apply_geometry_to_linked_table_pb_clicked()
426 {
427  if (m_table_item.isNull() || !m_table_item->diagram() || (!m_table_item->nextTable() && !m_table_item->previousTable())) {
428  return;
429  }
430  auto first_table = m_table_item;
431  while (first_table->previousTable()) {
432  first_table = first_table->previousTable();
433  }
434 
435  //Get all linked tables.
436  QVector<QetGraphicsTableItem*> vector_;
437  vector_ << first_table;
438  while (first_table->nextTable())
439  {
440  vector_ << first_table->nextTable();
441  first_table = first_table->nextTable();
442  }
443  vector_.removeAll(m_table_item);
444 
445 
446  auto new_pos = m_table_item->pos();
447  auto new_size = m_table_item->size();
448  auto new_displayN_row = m_table_item->displayNRow();
449  //Apply to all linked table
450  auto parent_undo = new QUndoCommand(tr("Appliquer la géometrie d'un tableau aux tableau liée à celui-ci"));
451  for (auto table : vector_)
452  {
453  new QPropertyUndoCommand(table, "pos", table->pos(), new_pos, parent_undo);
454  new QPropertyUndoCommand(table, "size", table->size(), new_size, parent_undo);
455  new QPropertyUndoCommand(table, "displayNRow", table->displayNRow(), new_displayN_row, parent_undo);
456  }
457  m_table_item->diagram()->undoStack().push(parent_undo);
458 }
