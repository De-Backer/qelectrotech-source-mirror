120 QVector<QetGraphicsTableItem *> ElementProvider::table(
121  QetGraphicsTableItem *table,
122  QAbstractItemModel *model)
123 {
124  QVector<QetGraphicsTableItem *> v_;
125  QVector<QetGraphicsTableItem *> linked_vector;
126 
127  if (table)
128  {
129  //table can be inside a chain, at the head of a chain or alone
130  auto linked_table = table->previousTable()
131  ? table->previousTable()
132  : table->nextTable();
133  while (linked_table) { //Go to the first table
134  if (linked_table->previousTable())
135  linked_table = linked_table->previousTable();
136  else
137  break;
138  }
139  while (linked_table) { //Store each linked table in linked_vector
140  linked_vector.append(linked_table);
141  linked_table = linked_table->nextTable();
142  }
143  }
144 
145  for (auto d : m_diagram_list) {
146  for (auto item_ : d->items())
147  {
148  if(item_->type() == QetGraphicsTableItem::Type)
149  {
150  auto found_table = static_cast<QetGraphicsTableItem *>(item_);
151 
152  if (linked_vector.contains(found_table)) {
153  continue;
154  }
155 
156  if (!model ||
157  (found_table->model() &&
158  model->metaObject()->className() == found_table->model()->metaObject()->className()))
159  {v_.append(found_table);}
160  }
161  }
162  }
163 
164  return v_;
165 }
