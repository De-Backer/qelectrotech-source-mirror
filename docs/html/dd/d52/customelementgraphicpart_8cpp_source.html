60 void CustomElementGraphicPart::drawCross(const QPointF &center,
61  QPainter *painter)
62 {
63  painter -> save();
64  painter -> setRenderHint(QPainter::Antialiasing, false);
65  painter -> setPen((painter -> brush().color()
66  == QColor(Qt::black)
67  && painter -> brush().isOpaque())
68  ? Qt::yellow
69  : Qt::blue);
70  painter -> drawLine(QLineF(center.x() - 2.0,
71  center.y(),
72  center.x() + 2.0,
73  center.y()));
74  painter -> drawLine(QLineF(center.x(),
75  center.y() - 2.0,
76  center.x(),
77  center.y() + 2.0));
78  painter -> restore();
79 }
