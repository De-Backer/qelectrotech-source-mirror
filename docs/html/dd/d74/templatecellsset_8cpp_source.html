98 TitleBlockTemplateVisualCell *TitleBlockTemplateCellsSet::topLeftCell() const {
99  if (empty()) return(nullptr);
100  if (count() == 1) return(first());
101 
102  // look for cells at the top
103  QMultiMap<int, TitleBlockTemplateVisualCell *> top_cells;
104  foreach (TitleBlockTemplateVisualCell *cell_view, *this) {
105  if (TitleBlockCell *cell = cell_view -> cell()) {
106  top_cells.insert(cell -> num_row, cell_view);
107  }
108  }
109  QList<TitleBlockTemplateVisualCell *> candidates = top_cells.values(top_cells.keys().first());
110  if (candidates.count() == 1) return(candidates.first());
111 
112  // look for the cell at the left
113  int lowest_num_col = 100000;
114  TitleBlockTemplateVisualCell *candidate = nullptr;
115  foreach (TitleBlockTemplateVisualCell *cell_view, candidates) {
116  if (TitleBlockCell *cell = cell_view -> cell()) {
117  if (cell -> num_col < lowest_num_col) {
118  lowest_num_col = cell -> num_col;
119  candidate = cell_view;
120  }
121  }
122  }
123  return(candidate);
124 }

129 TitleBlockTemplateVisualCell *TitleBlockTemplateCellsSet::bottomRightCell() const {

136  bottom_cells.insert(cell_view -> geometry().bottom(), cell_view);

139  if (candidates.count() == 1) return(candidates.first());
