61 bool DynamicTextFieldEditor::setPart(CustomElementPart *part) {

78 bool DynamicTextFieldEditor::setParts(QList <CustomElementPart *> parts) {
79  if (parts.isEmpty()) {
80  m_parts.clear();
81  if (m_text_field) {
82  disconnectConnections();
83  }
84  m_text_field = nullptr;
85  return true;
86  }
87 
88  if (PartDynamicTextField *part= static_cast<PartDynamicTextField *>(parts.first())) {
89  if (m_text_field) {
90  disconnectConnections();
91  }
92 
93  m_text_field = part;
94  m_parts.clear();
95  m_parts.append(part);
96  for (int i=1; i < parts.length(); i++)
97  m_parts.append(static_cast<PartDynamicTextField*>(parts[i]));
98 
99  setUpConnections();
100  updateForm();
101  return true;
102  }
103  return(false);
104 }

123 void DynamicTextFieldEditor::updateForm() {
124  if(m_text_field) {
125  ui -> m_x_sb -> setValue(m_text_field.data() -> x());
126  ui -> m_y_sb -> setValue(m_text_field.data() ->y ());
127  ui -> m_rotation_sb -> setValue(QET::correctAngle(m_text_field.data() -> rotation()));
128  ui -> m_frame_cb -> setChecked(m_text_field.data() -> frame());
129  ui -> m_user_text_le -> setText(m_text_field.data() -> text());
130  ui -> m_size_sb -> setValue(m_text_field.data() -> font().pointSize());
131  ui -> m_color_kpb -> setColor(m_text_field.data() -> color());
132  ui -> m_width_sb -> setValue(m_text_field.data() -> textWidth());
133  ui -> m_font_pb -> setText(m_text_field -> font().family());
134 
135  switch (m_text_field.data() -> textFrom()) {
136  case DynamicElementTextItem::UserText: {
137  ui -> m_text_from_cb -> setCurrentIndex(0);
138  break;
139  }
140  case DynamicElementTextItem::ElementInfo: {
141  ui -> m_text_from_cb -> setCurrentIndex(1);
142  ui -> m_elmt_info_cb -> setCurrentIndex(
143  ui -> m_elmt_info_cb -> findData(m_text_field.data() -> infoName()));
144  break;
145  }
146  case DynamicElementTextItem::CompositeText: {
147  ui -> m_text_from_cb -> setCurrentIndex(2);
148  }
149  }
150 
151  on_m_text_from_cb_activated(ui -> m_text_from_cb -> currentIndex()); //For enable the good widget
152  }
153 }

193 void DynamicTextFieldEditor::fillInfoComboBox() {

196  QStringList strl;
197  QString type = elementEditor() -> elementScene() -> elementType();
198 
199  if(type.contains("report")) {
200  strl << "function" << "tension_protocol" << "conductor_color" << "conductor_section";
201  }

209  info_map.insert(QETApp::elementTranslatedInfoKey(str), str);
210 
211  for (const QString& key : info_map.keys())
212  ui -> m_elmt_info_cb -> addItem(key, info_map.value(key));
