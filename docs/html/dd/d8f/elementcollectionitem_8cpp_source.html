116 int ElementCollectionItem::rowForInsertItem(const QString &name)
117 {
118  if (name.isEmpty())
119  return -1;
120 
121  QList <ElementCollectionItem *> child;
122  //The item to insert is an element we search from element child
123  if (name.endsWith(".elmt"))
124  {
125  child = elementsDirectChild();
126  //There isn't element, we insert at last position
127  if (child.isEmpty())
128  return rowCount();
129  }
130  //The item is a directory, we search from directory child
131  else
132  {
133  child = directoriesDirectChild();
134  //There isn't directory, we insert at first position
135  if(child.isEmpty())
136  return 0;
137  }
138 
139  foreach (ElementCollectionItem *eci, child)
140  if (eci->name() > name)
141  return model()->indexFromItem(eci).row();
142 
143  return (model()->indexFromItem(child.last()).row() + 1);
144 }
