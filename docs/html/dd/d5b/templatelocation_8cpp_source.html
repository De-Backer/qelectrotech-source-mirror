94 void TitleBlockTemplateLocation::fromString(const QString &loc_str) {
95  collection_ = QETApp::titleBlockTemplatesCollection(QUrl(loc_str).scheme());
96 
97  QRegExp name_from_url("^[^:]*:\\/\\/(.*)$");
98  if (name_from_url.exactMatch(loc_str)) {
99  name_ = name_from_url.capturedTexts().at(1);
100  } else {
101  name_ = QString();
102  }
103 }
