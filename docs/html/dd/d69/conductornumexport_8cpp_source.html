112 void ConductorNumExport::fillHash()
113 {
114  //We used this rx to avoid insert num composed only withe white space.
115  QRegularExpression rx("^ *$");
116  for (Diagram *d : m_project->diagrams())
117  {
118  DiagramContent dc(d, false);
119  for (Conductor *c : dc.conductors())
120  {
121  QString num = c->textItem()->toPlainText();
122  if (num.isEmpty() || num.contains(rx)) {
123  continue;
124  }
125 
126  //We must to define if the connected terminal is a folio report, if it is the case
127  //we don't add the num to the hash because the terminal doesn't represent a real terminal.
128  if(!(c->terminal1->parentElement()->linkType() & Element::AllReport))
129  {
130  int value = m_hash.value(num, 0);
131  ++value;
132  m_hash.insert(num, value);
133  }
134  if(!(c->terminal2->parentElement()->linkType() & Element::AllReport))
135  {
136  int value = m_hash.value(num, 0);
137  ++value;
138  m_hash.insert(num, value);
139  }
140  }
141  }
142 }
