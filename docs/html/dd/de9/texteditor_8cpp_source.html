103 bool TextEditor::setPart(CustomElementPart *part) {
104  if (!part) {
105  m_text = nullptr;

107  return true;
108  }
109 
110  if (PartText *part_text = static_cast<PartText *>(part)) {
111  if (part_text == m_text) {
112  return true;

116  updateForm();
117  return true;

122 bool TextEditor::setParts(QList <CustomElementPart *> parts) {
123  if (parts.isEmpty()) {
124  m_parts.clear();
125  if (m_text) {
126  disconnectChangeConnection();
127  }
128  m_text = nullptr;
129  return true;
130  }
131 
132  if (PartText *part = static_cast<PartText *>(parts.first())) {
133  if (m_text) {
134  disconnectChangeConnection();
135  }
136  m_text = part;
137  m_parts.clear();
138  m_parts.append(part);
139  for (int i=1; i < parts.length(); i++) {
140  m_parts.append(static_cast<PartText*>(parts[i]));
141  }
142  setUpChangeConnection(m_text);
143  updateForm();
144  return true;
145  }
146  return(false);
147 }

170 void TextEditor::setUpEditConnection() {
171  disconnectEditConnection();
172 
173  m_edit_connection << connect(ui -> m_line_edit, &QLineEdit::editingFinished, [this]() {
174  QString text_ = ui -> m_line_edit -> text();
175  for (int i=0; i < m_parts.length(); i++) {
176  PartText* partText = m_parts[i];
177  if (text_ != partText -> toPlainText()) {
178  QPropertyUndoCommand *undo = new QPropertyUndoCommand(partText, "text", partText -> toPlainText(), text_);
179  undo -> setText(tr("Modifier le contenu d'un champ texte"));
180  undoStack().push(undo);
181  }
182  }
183  });
184  m_edit_connection << connect(ui->m_x_sb, QOverload<int>::of(&QSpinBox::valueChanged), [this]() {
185  QPointF pos(ui -> m_x_sb -> value(), 0);
186  for (int i=0; i < m_parts.length(); i++) {
187  PartText* partText = m_parts[i];
188  pos.setY(partText -> pos().y());
189  if (pos != partText -> pos()) {
190  QPropertyUndoCommand *undo = new QPropertyUndoCommand(partText, "pos", partText -> pos(), pos);
191  undo -> setText(tr("Déplacer un champ texte"));
192  undo -> setAnimated(true, false);
193  undoStack().push(undo);
194  }
195  }
196  });
197  m_edit_connection << connect(ui -> m_y_sb, QOverload<int>::of(&QSpinBox::valueChanged), [this]() {
198  QPointF pos(0, ui -> m_y_sb -> value());
199  for (int i=0; i < m_parts.length(); i++) {
200  PartText* partText = m_parts[i];
201  pos.setX(partText -> pos().x());
202  if (pos != partText -> pos()) {
203  QPropertyUndoCommand *undo = new QPropertyUndoCommand(partText, "pos", partText -> pos(), pos);
204  undo -> setText(tr("Déplacer un champ texte"));
205  undo -> setAnimated(true, false);
206  undoStack().push(undo);
207  }
208  }
209  });
210  m_edit_connection << connect(ui -> m_rotation_sb, QOverload<int>::of(&QSpinBox::valueChanged), [this]() {
211  for (int i=0; i < m_parts.length(); i++) {
212  PartText* partText = m_parts[i];
213  if (ui -> m_rotation_sb -> value() != partText -> rotation()) {
214  QPropertyUndoCommand *undo = new QPropertyUndoCommand(
215  partText, "rotation", partText -> rotation(), ui -> m_rotation_sb -> value());
216  undo -> setText(tr("Pivoter un champ texte"));
217  undo -> setAnimated(true, false);
218  undoStack().push(undo);
219  }
220  }
221  });
222  m_edit_connection << connect(ui->m_size_sb, QOverload<int>::of(&QSpinBox::valueChanged), [this]() {
223  for (int i=0; i < m_parts.length(); i++) {
224  PartText* partText = m_parts[i];
225  if (partText -> font().pointSize() != ui -> m_size_sb -> value()) {
226  QFont font_ = partText -> font();
227  font_.setPointSize(ui -> m_size_sb -> value());
228  QPropertyUndoCommand *undo = new QPropertyUndoCommand(partText, "font", partText -> font(), font_);
229  undo -> setText(tr("Modifier la police d'un texte"));
230  undoStack().push(undo);
231  }
232  }
233  });
234 }
