232 void PartText::handleUserTransformation(const QRectF &initial_selection_rect, const QRectF &new_selection_rect) {
233  // let's try the naive approach
234  QPointF new_pos = mapPoints(initial_selection_rect, new_selection_rect, QList<QPointF>() << saved_point_).first();
235  setPos(new_pos);
236 
237  // adjust the font size following the vertical scale factor
238  qreal sy = new_selection_rect.height() / initial_selection_rect.height();
239  qreal new_font_size = saved_font_size_ * sy;

344  QPropertyUndoCommand *undo = new QPropertyUndoCommand(this, "text", previous_text, new_text);
345  undo -> setText(tr("Modifier un champ texte"));
