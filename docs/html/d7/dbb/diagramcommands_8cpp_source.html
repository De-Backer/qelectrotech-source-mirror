87 void PasteDiagramCommand::redo()
88 {
89  diagram -> showMe();
90  QSettings settings;
91 
92  if (first_redo)
93  {
94  first_redo = false;
95 
96  //this is the first paste, we do some actions for the new element
97  const QList <Element *> elmts_list = content.m_elements;
98  for (Element *e : elmts_list)
99  {
100  //make new uuid, because old uuid are the uuid of the copied element
101  e -> newUuid();
102 
103  if (settings.value("diagramcommands/erase-label-on-copy", true).toBool())
104  {
105  //Reset the information about the label, the comment and location
106  DiagramContext dc = e->elementInformations();
107  dc.addValue("formula", "");
108  dc.addValue("label", "");
109  dc.addValue("comment", "");
110  dc.addValue("location", "");
111  e->setElementInformations(dc);
112 
113  //Reset the text of conductors
114  const QList <Conductor *> conductors_list = content.m_conductors_to_move;
115  for (Conductor *c : conductors_list)
116  {
117  ConductorProperties cp = c -> properties();
118  cp.text = c->diagram() ? c -> diagram() -> defaultConductorProperties.text : "_";
119  c -> setProperties(cp);
120  }
121  }
122  }
123  }
124  else
125  {
126  const QList <QGraphicsItem *> qgis_list = content.items(filter);
127  for (QGraphicsItem *item : qgis_list) {
128  diagram->addItem(item);
129  }
130  }
131 
132  const QList <QGraphicsItem *> qgis_list = content.items();
133  for (QGraphicsItem *qgi : qgis_list)
134  qgi -> setSelected(true);
135 }

248 void MoveElementsCommand::move(const QPointF &actual_movement)
249 {
250  typedef DiagramContent dc;
251 
252  //Move every movable items, except conductor
253  for (QGraphicsItem *qgi : content_to_move.items(dc::Elements
254  | dc::TextFields
255  | dc::Images
256  | dc::Shapes
257  | dc::TextGroup
258  | dc::ElementTextFields
259  | dc::Tables))
260  {
261  //If curent item have parent, and parent item is in content_to_move
262  //we don't apply movement to this item, because this item will be moved by is parent.
263  if (qgi->parentItem())
264  if (content_to_move.items().contains(qgi->parentItem()))
265  continue;
266 
267  if(qgi->toGraphicsObject())
268  setupAnimation(qgi->toGraphicsObject(), "pos",
269  qgi->pos(), qgi->pos() + actual_movement);
270  else if(qgi->type() == QGraphicsItemGroup::Type)
271  {
272  //ElementTextItemGroup is a QObject but not a QGraphicsObject
273  if(ElementTextItemGroup *etig = dynamic_cast<ElementTextItemGroup *>(qgi))
274  setupAnimation(etig, "pos", etig->pos(),
275  etig->pos() + actual_movement);
276  }
277  else qgi -> setPos(qgi->pos() + actual_movement);
278  }
279 
280  // Move some conductors
281  for (Conductor *conductor : content_to_move.m_conductors_to_move)
282  setupAnimation(conductor, "pos", conductor->pos(),
283  conductor->pos() + actual_movement);
284 
285  // Recalcul the path of other conductor
286  for (Conductor *conductor : content_to_move.m_conductors_to_update)
287  setupAnimation(conductor, "animPath", 1, 1);
288 }

381 void MoveConductorsTextsCommand::addTextMovement(ConductorTextItem *text_item,
382  const QPointF &old_pos,
383  const QPointF &new_pos,
384  bool already_moved) {
385  // si le champ de texte est deja connu de l'objet d'annulation, il sera ignore
386  if (texts_to_move_.contains(text_item)) return;
387 
388  // on memorise le champ de texte,
389  //en l'associant au mouvement effectue et a son etat avant le deplacement
390  texts_to_move_.insert(text_item, qMakePair(new_pos - old_pos, already_moved));
391 
392  // met a jour la description de l'objet d'annulation
393  regenerateTextLabel();
394 }
