70 void ElementTextItemGroup::addToGroup(QGraphicsItem *item)
71 {
72  if(item->type() == DynamicElementTextItem::Type)
73  {
74  //Befor add text to this group we must to set the text at the same rotation of this group

101 void ElementTextItemGroup::removeFromGroup(QGraphicsItem *item)
102 {
103  QGraphicsItemGroup::removeFromGroup(item);
104  //the item transformation is not reseted, we must to do it, because for exemple if the group rotation is 45°
105  //When item is removed from group, visually the item is unchanged (so 45°) but if we call item->rotation() the returned value is 0.
106  item->resetTransform();
107  item->setRotation(this->rotation());
108  item->setFlag(QGraphicsItem::ItemIsSelectable, true);
109  updateAlignment();
110 
111  if(DynamicElementTextItem *deti = qgraphicsitem_cast<DynamicElementTextItem *>(item))
112  {
113  disconnect(deti, &DynamicElementTextItem::fontChanged, this, &ElementTextItemGroup::updateAlignment);
114  disconnect(deti, &DynamicElementTextItem::textChanged, this, &ElementTextItemGroup::updateAlignment);
115  disconnect(deti, &DynamicElementTextItem::textFromChanged, this, &ElementTextItemGroup::updateAlignment);

162 void ElementTextItemGroup::updateAlignment()
163 {
164  if(m_block_alignment_update)
165  return;
166 
167  prepareGeometryChange();
168 
169  QList <DynamicElementTextItem *> texts = this->texts();
170 
171  qreal rotation_ = rotation();
172 
173  //Set the rotation of this group to 0° relative to the scene
174  qreal rot = rotation();
175  QGraphicsItem *parent = parentItem();
176  while (parent) {
177  rot += parent->rotation();
178  parent = parent->parentItem();
179  }
180  if(rot != 0)
181  setRotation(rotation() - rot);
182 
183 
184  if(texts.size() == 1)
185  {
186  prepareGeometryChange();
187 
188  QGraphicsItem *first = texts.first();
189  setPos(mapFromScene(first->mapToScene(pos())));
190  first->setPos(0,0);
191  }
192  else if (texts.size() > 1)
193  {
194  qreal width = 0;
195  for(QGraphicsItem *item : texts)
196  if(item->boundingRect().width() > width)
197  width = item->boundingRect().width();
198 
199  prepareGeometryChange();
200  std::sort(texts.begin(), texts.end(), sorting);
201 
202  qreal y_offset = 0;
203 
204  if(m_alignment == Qt::AlignLeft)
205  {
206  QPointF ref = texts.first()->pos();
207 
208  for(QGraphicsItem *item : texts)
209  {
210  item->setPos(0, ref.y()+y_offset);
211  y_offset+=item->boundingRect().height() + m_vertical_adjustment;
212  }
213  }
214  else if(m_alignment == Qt::AlignVCenter)
215  {
216  QPointF ref(width/2,0);
217 
218  for(QGraphicsItem *item : texts)
219  {
220  item->setPos(ref.x() - item->boundingRect().width()/2,
221  ref.y() + y_offset);
222  y_offset+=item->boundingRect().height() + m_vertical_adjustment;
223  }
224  }
225  else if (m_alignment == Qt::AlignRight)
226  {
227  QPointF ref(width,0);
228 
229  for(QGraphicsItem *item : texts)
230  {
231  item->setPos(ref.x() - item->boundingRect().width(),
232  ref.y() + y_offset);
233  y_offset+=item->boundingRect().height() + m_vertical_adjustment;
234  }
235  }
236  }
237 
238  //Restor the rotation
239  setRotation(rotation_);
240 
241  if(m_Xref_item)
242  m_Xref_item->autoPos();
243  if(m_slave_Xref_item)
244  adjustSlaveXrefPos();
245  if(m_hold_to_bottom_of_page)
246  autoPos();
247 }

274 void ElementTextItemGroup::setHoldToBottomPage(bool hold)
275 {
276  if(m_hold_to_bottom_of_page == hold)
277  return;
278 
279  m_hold_to_bottom_of_page = hold;
280  if(m_hold_to_bottom_of_page)
281  {
282  setFlag(QGraphicsItem::ItemIsSelectable, false);
283  setFlag(QGraphicsItem::ItemIsMovable, false);
284  connect(m_parent_element, &Element::yChanged, this, &ElementTextItemGroup::autoPos);
285  connect(m_parent_element, &Element::rotationChanged, this, &ElementTextItemGroup::autoPos);
286  if(m_parent_element->linkType() == Element::Master)
287  {
288  //We use timer to let the time of the parent element xref to be updated, befor update the position of this group
289  //because the position of this group is related to the size of the parent element Xref
290  m_linked_changed_timer = connect(m_parent_element, &Element::linkedElementChanged,
291  [this]() {QTimer::singleShot(200, this, &ElementTextItemGroup::autoPos);});
292  if(m_parent_element->diagram())
293  m_XrefChanged_timer = connect(m_parent_element->diagram()->project(), &QETProject::XRefPropertiesChanged,
294  [this]() {QTimer::singleShot(200, this, &ElementTextItemGroup::autoPos);});
295  }
296  autoPos();
297  }
298  else
299  {
300  setFlag(QGraphicsItem::ItemIsSelectable, true);
301  setFlag(QGraphicsItem::ItemIsMovable, true);
302  disconnect(m_parent_element, &Element::yChanged, this, &ElementTextItemGroup::autoPos);
303  disconnect(m_parent_element, &Element::rotationChanged, this, &ElementTextItemGroup::autoPos);
304  if(m_parent_element->linkType() == Element::Master)
305  {
306  disconnect(m_linked_changed_timer);
307  if(m_XrefChanged_timer)
308  disconnect(m_XrefChanged_timer);
309  }
310  }
311 
312  emit holdToBottomPageChanged(hold);
313 }

405 void ElementTextItemGroup::fromXml(QDomElement &dom_element)
406 {
407  if (dom_element.tagName() != xmlTaggName()) {
408  qDebug() << "ElementTextItemGroup::fromXml : Wrong tagg name";
409  return;
410  }
411 
412  setName(dom_element.attribute("name", "no name"));
413  QMetaEnum me = QMetaEnum::fromType<Qt::Alignment>();
414  setAlignment(Qt::Alignment(me.keyToValue(dom_element.attribute("alignment").toStdString().data())));
415 
416  setPos(dom_element.attribute("x", QString::number(0)).toDouble(),
417  dom_element.attribute("y", QString::number(0)).toDouble());
418 
419  setRotation(dom_element.attribute("rotation", QString::number(0)).toDouble());
420  setVerticalAdjustment(dom_element.attribute("vertical_adjustment").toInt());
421  setFrame(dom_element.attribute("frame", "false") == "true"? true : false);
422 
423  QString hold = dom_element.attribute("hold_to_bottom_page", "false");
424  setHoldToBottomPage(hold == "true" ? true : false);
425 
426  if(parentElement())
427  {
428  m_block_alignment_update = true;
429  for(const QDomElement& text : QET::findInDomElement(dom_element, "texts", "text"))
430  {
431  DynamicElementTextItem *deti = nullptr;
432  QUuid uuid(text.attribute("uuid"));
433 
434  for(DynamicElementTextItem *txt : parentElement()->dynamicTextItems())
435  if(txt->uuid() == uuid)
436  deti = txt;
437 
438  if (deti)
439  parentElement()->addTextToGroup(deti, this);
440  }
441  m_block_alignment_update = false;
442  }

451 void ElementTextItemGroup::paint(QPainter *painter, const QStyleOptionGraphicsItem *option, QWidget *widget)
452 {
453  Q_UNUSED(option);
454  Q_UNUSED(widget);
455  if(isSelected())
456  {
457  painter->save();
458  QPen t;
459  t.setColor(Qt::gray);
460  t.setStyle(Qt::DashDotLine);
461  t.setCosmetic(true);
462  painter->setPen(t);
463  painter->drawRoundedRect(boundingRect().adjusted(1, 1, -1, -1), 10, 10);
464 
465  painter->restore();
466  }
467  if(m_frame)
468  {
469  qreal font_size = 1;
470  QRectF rect;
471  for(DynamicElementTextItem *deti : this->texts())
472  {
473  font_size = std::max(font_size, deti->font().pointSizeF());
474  rect = rect.united(mapFromItem(deti, deti->frameRect()).boundingRect());
475  }
476 
477  //Adjust the thickness according to the font size
478  qreal w=0.3;
479  if (font_size >= 5)
480  {
481  w = font_size*0.1;
482  if(w > 2.5)
483  w = 2.5;
484  }
485 
486  painter->save();
487  QPen pen;
488  pen.setWidthF(w);
489  painter->setPen(pen);
490  painter->setRenderHint(QPainter::Antialiasing);
491 
492  //Adjust the rounding of the rectangle according to the size of the font
493  qreal ro = font_size/3;
494  painter->drawRoundedRect(rect, ro, ro);
495  painter->restore();
496  }
497 }

503 QRectF ElementTextItemGroup::boundingRect() const
504 {
505  //If we refer to the Qt doc, the bounding rect of a QGraphicsItemGroup,
506  //is the bounding of all childrens in the group
507  //When add an item in the group, the bounding rect is good, but
508  //if we move an item already in the group, the bounding rect of the group stay unchanged.
509  //We reimplement this function to avoid this behavior.
510  QRectF rect;
511  for(QGraphicsItem *qgi : texts())
512  {
513  QRectF r(qgi->pos(), QSize(qgi->boundingRect().width(), qgi->boundingRect().height()));
514  rect = rect.united(r);
515  }
516  return rect;
517 }

565 void ElementTextItemGroup::mouseMoveEvent(QGraphicsSceneMouseEvent *event)
566 {
567  if((event->buttons() & Qt::LeftButton) && (flags() & ItemIsMovable))
568  {
569  if(diagram() && m_first_move)
570  diagram()->elementTextsMover().beginMovement(diagram(), this);

574  m_initial_position = pos();

576  parentElement()->setHighlighted(true);

578 
579  QPointF current_parent_pos;
580  QPointF button_down_parent_pos;
581  current_parent_pos = mapToParent(mapFromScene(event->scenePos()));
582  button_down_parent_pos = mapToParent(mapFromScene(event->buttonDownScenePos(Qt::LeftButton)));
583 
584  QPointF new_pos = m_initial_position + current_parent_pos - button_down_parent_pos;
585  event->modifiers() == Qt::ControlModifier ? setPos(new_pos) : setPos(Diagram::snapToGrid(new_pos));
586 
587  if(diagram())
588  diagram()->elementTextsMover().continueMovement(event);
589  } else {
590  event->ignore();
591  }
592

698 void ElementTextItemGroup::updateXref()
699 {
700  if(m_parent_element->diagram())
701  {
702  QETProject *project = m_parent_element->diagram()->project();
703 
704  if(m_parent_element->linkType() == Element::Master &&
705  !m_parent_element->linkedElements().isEmpty())
706  {
707 
708  XRefProperties xrp = project->defaultXRefProperties(m_parent_element->kindInformations()["type"].toString());
709 
710  if(xrp.snapTo() == XRefProperties::Label)
711  {
712  //At least one text owned by this group must be set with
713  //textFrom -> element info and element info name -> label
714  //for display a xref
715  for(DynamicElementTextItem *deti : texts())
716  {
717  if(deti->textFrom() == DynamicElementTextItem::ElementInfo &&
718  deti->infoName() == "label")
719  {
720  if(!m_Xref_item)
721  m_Xref_item = new CrossRefItem(m_parent_element, this);
722  m_Xref_item->autoPos();
723  return;
724  }
725  }
726  }
727  }
728  else if(m_parent_element->linkType() == Element::Slave &&
729  !m_parent_element->linkedElements().isEmpty())
730  {
731  Element *master_elmt = m_parent_element->linkedElements().first();
732  for(DynamicElementTextItem *deti : texts())
733  {
734  if((deti->textFrom() == DynamicElementTextItem::ElementInfo && deti->infoName() == "label") ||
735  (deti->textFrom() == DynamicElementTextItem::CompositeText && deti->compositeText().contains("%{label")))
736  {
737  XRefProperties xrp = project->defaultXRefProperties(master_elmt->kindInformations()["type"].toString());
738  QString xref_label = xrp.slaveLabel();
739  xref_label = autonum::AssignVariables::formulaToLabel(xref_label, master_elmt->rSequenceStruct(), master_elmt->diagram(), master_elmt);
740 
741  if(!m_slave_Xref_item)
742  {
743  m_slave_Xref_item = new QGraphicsTextItem(xref_label, this);
744  m_slave_Xref_item->setFont(QETApp::diagramTextsFont(5));
745 
746  m_update_slave_Xref_connection << connect(master_elmt, &Element::xChanged, this, &ElementTextItemGroup::updateXref);
747  m_update_slave_Xref_connection << connect(master_elmt, &Element::yChanged, this, &ElementTextItemGroup::updateXref);
748  m_update_slave_Xref_connection << connect(master_elmt, &Element::elementInfoChange, this, &ElementTextItemGroup::updateXref);
