321 void ElementsPanel::filter(const QString &m, QET::Filtering filtering) {
322  QList<QTreeWidgetItem *> items = findItems("*", Qt::MatchRecursive | Qt::MatchWildcard);
323  const int expanded_role = 42; // magic number? So you consider Douglas Adams wrote about magic?
324 
325  if (filtering == QET::BeginFilter) {
326  foreach (QTreeWidgetItem *item, items) {
327  item -> setData(0, expanded_role, item -> isExpanded());
328  }
329  }
330 
331  if (filtering != QET::EndFilter) {
332  filter_ = m;
333  applyCurrentFilter(items);
334  } else { // filtering == QET::EndFilter
335  filter_ = QString();
336  QTreeWidgetItem *current_item = currentItem();
337 
338  // restore the tree as it was before the filtering
339  foreach (QTreeWidgetItem *qtwi, items) {
340  qtwi -> setHidden(false);
341  qtwi -> setExpanded(qtwi -> data(0, expanded_role).toBool());
342  }
343 
344  // avoid hiding the currently selected item
345  if (current_item) {
346  ensureHierarchyIsVisible(QList<QTreeWidgetItem *>() << current_item);
347  scrollToItem(current_item);
348  }
349  }
350 }

392 void ElementsPanel::applyCurrentFilter(const QList<QTreeWidgetItem *> &items) {
393  if (filter_.isEmpty()) return;
394  buildFilterList();
395  QList<QTreeWidgetItem *> matching_items;
396 
397  foreach (QTreeWidgetItem *item, items) {
398  bool item_matches = true;
399 
400  foreach( QString filter , filter_list_ ) {
401  filter = filter.trimmed();
402  if ( !filter.isEmpty() ) {
403  item_matches &= matchesFilter(item, filter);
404  }
405  }
406  if (item_matches) matching_items << item;
407  item -> setHidden(!item_matches);
408  }
409  ensureHierarchyIsVisible(matching_items);
410 }

416 void ElementsPanel::ensureHierarchyIsVisible(const QList<QTreeWidgetItem *> &items) {
417  // remonte l'arborescence pour lister les categories contenant les elements filtres
418  QSet<QTreeWidgetItem *> parent_items;
419  foreach(QTreeWidgetItem *item, items) {
420  for (QTreeWidgetItem *parent_qtwi = item -> parent() ; parent_qtwi ; parent_qtwi = parent_qtwi -> parent()) {
421  parent_items << parent_qtwi;
422  }
423  }
424 
425  // etend les parents
426  foreach(QTreeWidgetItem *parent_qtwi, parent_items) {
427  if (!parent_qtwi -> isExpanded()) parent_qtwi -> setExpanded(true);
428  }
429 
430  // affiche les parents
431  foreach(QTreeWidgetItem *parent_qtwi, parent_items) {
432  if (parent_qtwi -> isHidden()) parent_qtwi -> setHidden(false);
433  }
434 }
