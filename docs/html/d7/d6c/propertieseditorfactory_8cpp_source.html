75 PropertiesEditorWidget *PropertiesEditorFactory::propertiesEditor(
76  QList<QGraphicsItem *> items,
77  PropertiesEditorWidget *editor,
78  QWidget *parent)
79 {
80  int count_ = items.size();
81 
82  //The editor widget can only edit one item
83  //or several items of the same type
84  if (count_ != 1)
85  {
86  if (count_ == 0) {
87  return nullptr;
88  }
89 
90  int type_ = items.first()->type();
91  for (QGraphicsItem *qgi : items) {
92  if (qgi->type() != type_) {
93  return nullptr;
94  }
95  }
96  }
97 
98  QGraphicsItem *item = items.first();
99  const int type_ = item->type();
100  QString class_name;
101  if (editor) {
102  class_name = editor->metaObject()->className();
103  }
104 
105  switch (type_)
106  {
107  case Element::Type: //1000
108  {
109  if (count_ > 1) {
110  return nullptr;
111  }
112  auto elmt = static_cast<Element*>(item);
113  //auto created_editor = new ElementPropertiesWidget(elmt, parent);
114 
115  //We already edit an element, just update the editor with a new element
116  if (class_name == ElementPropertiesWidget::staticMetaObject.className())
117  {
118  static_cast<ElementPropertiesWidget*>(editor)->setElement(elmt);
119  return editor;
120  }
121  return new ElementPropertiesWidget(elmt, parent);
122  }
123  case IndependentTextItem::Type: //1005
124  {
125  QList<IndependentTextItem *> text_list;
126  for (QGraphicsItem *qgi : items) {
127  text_list.append(static_cast<IndependentTextItem*>(qgi));
128  }
129 
130  if (class_name == IndiTextPropertiesWidget::staticMetaObject.className())
131  {
132  static_cast<IndiTextPropertiesWidget*>(editor)->setText(text_list);
133  return editor;
134  }
135 
136  return new IndiTextPropertiesWidget(text_list, parent);
137  }
138  case DiagramImageItem::Type: //1007
139  {
140  if (count_ > 1) {
141  return nullptr;
142  }
143  return new ImagePropertiesWidget(static_cast<DiagramImageItem*>(item), parent);
144  }
145  case QetShapeItem::Type: //1008
146  {
147  QList<QetShapeItem *> shapes_list;
148  for (QGraphicsItem *qgi : items) {
149  shapes_list.append(static_cast<QetShapeItem*>(qgi));
150  }
151 
152  if (class_name == ShapeGraphicsItemPropertiesWidget::staticMetaObject.className())
153  {
154  static_cast<ShapeGraphicsItemPropertiesWidget*>(editor)->setItems(shapes_list);
155  return editor;
156  }
157 
158  return new ShapeGraphicsItemPropertiesWidget(shapes_list, parent);
159  }
160  case DynamicElementTextItem::Type: //1010
161  {
162  if (count_ > 1) {
163  return nullptr;
164  }
165 
166  DynamicElementTextItem *deti = static_cast<DynamicElementTextItem *>(item);
167  //For dynamic element text, we open the element editor to edit it
168  //If we already edit an element, just update the editor with a new element
169  if (class_name == ElementPropertiesWidget::staticMetaObject.className())
170  {
171  static_cast<ElementPropertiesWidget*>(editor)->setDynamicText(deti);
172  return editor;
173  }
174  return new ElementPropertiesWidget(deti, parent);
175  }
176  case QGraphicsItemGroup::Type:
177  {
178  if (count_ > 1) {
179  return nullptr;
180  }
181 
182  if(ElementTextItemGroup *group = dynamic_cast<ElementTextItemGroup *>(item))
183  {
184  //For element text item group, we open the element editor to edit it
185  //If we already edit an element, just update the editor with a new element
186  if(class_name == ElementPropertiesWidget::staticMetaObject.className())
187  {
188  static_cast<ElementPropertiesWidget *>(editor)->setTextsGroup(group);
189  return editor;
190  }
191  return new ElementPropertiesWidget(group, parent);
192  }
193  break;
194  }
195  case QetGraphicsTableItem::Type:
196  {
197  if (count_ > 1) {
198  return nullptr;
199  }
200 
201  auto table = static_cast<QetGraphicsTableItem*>(item);
202  if (class_name == GraphicsTablePropertiesEditor::staticMetaObject.className())
203  {
204  static_cast<GraphicsTablePropertiesEditor*>(editor)->setTable(table);
205  return editor;
206  }
207  return new GraphicsTablePropertiesEditor(table, parent);
208  }
209  default:
210  return nullptr;
211  }
212 
213  return nullptr;
214 }
