249  if (ok) setColumnsWidth(cols_width);

251  // backward compatibility:
252  // diagrams saved with 0.1 version have a "height" attribute
253  if (xml_elmt.hasAttribute("rows") && xml_elmt.hasAttribute("rowsize")) {
254  // rows counts
255  int rows_count = xml_elmt.attribute("rows").toInt(&ok);
256  if (ok) setRowsCount(rows_count);

260  if (ok) setRowsHeight(rows_size);

262  // hauteur du schema
263  double height = xml_elmt.attribute("height").toDouble(&ok);
264  if (ok) setDiagramHeight(height);
265  }
266 
267  // rows and columns display
268  displayColumns(xml_elmt.attribute("displaycols") != "false");
269  displayRows(xml_elmt.attribute("displayrows") != "false");
270 
271  updateRectangles();
272 }

504 void BorderTitleBlock::draw(QPainter *painter)
505 {
506  //Set the QPainter
507  painter -> save();
508  QPen pen(Qt::black);
509  pen.setCosmetic(true);
510  painter -> setPen(pen);
511  painter -> setBrush(Qt::NoBrush);
512 
513  QSettings settings;
514 
515  //Draw the borer
516  if (display_border_) painter -> drawRect(diagram_rect_);
517 
518  painter -> setFont(QETApp::diagramTextsFont());
519 
520  //Draw the empty case at the top left of diagram when there is header
521  if (display_border_ && (display_columns_ || display_rows_))
522  {
523  QRectF first_rectangle(
524  diagram_rect_.topLeft().x(),
525  diagram_rect_.topLeft().y(),
526  rows_header_width_,
527  columns_header_height_
528  );
529  painter -> drawRect(first_rectangle);
530  }
531 
532  //Draw the nums of columns
533  if (display_border_ && display_columns_) {
534  for (int i = 1 ; i <= columns_count_ ; ++ i) {
535  QRectF numbered_rectangle = QRectF(
536  diagram_rect_.topLeft().x()
537  + (rows_header_width_
538  + ((i - 1) * columns_width_)),
539  diagram_rect_.topLeft().y(),
540  columns_width_,
541  columns_header_height_
542  );
543  painter -> drawRect(numbered_rectangle);
544  if (settings.value("border-columns_0", true).toBool()){
545  painter -> drawText(numbered_rectangle,
546  Qt::AlignVCenter
547  | Qt::AlignCenter,
548  QString("%1").arg(i - 1));
549  }else{
550  painter -> drawText(numbered_rectangle,
551  Qt::AlignVCenter
552  | Qt::AlignCenter,
553  QString("%1").arg(i));
554  }
555  }
556  }
557 
558  //Draw the nums of rows
559  if (display_border_ && display_rows_) {
560  QString row_string("A");
561  for (int i = 1 ; i <= rows_count_ ; ++ i) {
562  QRectF lettered_rectangle = QRectF(
563  diagram_rect_.topLeft().x(),
564  diagram_rect_.topLeft().y()
565  + (
566  columns_header_height_
567  + ((i - 1)* rows_height_)
568  ),
569  rows_header_width_,
570  rows_height_
571  );
572  painter -> drawRect(lettered_rectangle);
573  painter -> drawText(lettered_rectangle,
574  Qt::AlignVCenter
575  | Qt::AlignCenter,
576  row_string);
577  row_string = incrementLetters(row_string);
578  }
579  }
580 
581  // render the titleblock, using the TitleBlockTemplate object
582  if (display_titleblock_) {
583  QRectF tbt_rect = titleBlockRectForQPainter();
584  if (m_edge == Qt::BottomEdge)
585  {
586  painter -> translate(tbt_rect.topLeft());
587  m_titleblock_template_renderer -> render(
588  painter,
589  tbt_rect.width());
590  painter -> translate(-tbt_rect.topLeft());
591  }
592  else
593  {
594  painter->translate(tbt_rect.topLeft());
595  painter->rotate(-90);
596  m_titleblock_template_renderer -> render(
597  painter,
598  tbt_rect.width());
599  painter->rotate(90);
600  painter -> translate(-tbt_rect.topLeft());
601  }
602  }
603 
604  painter -> restore();
605 }

615 void BorderTitleBlock::drawDxf(
616  int width,
617  int height,
618  bool keep_aspect_ratio,
619  QString &file_path,
620  int color) {
621  Q_UNUSED (width); Q_UNUSED (height); Q_UNUSED (keep_aspect_ratio);
622 
623  // Transform to DXF scale.
624  columns_header_height_ *= Createdxf::yScale;
625  rows_height_ *= Createdxf::yScale;
626  rows_header_width_ *= Createdxf::xScale;
627  columns_width_ *= Createdxf::xScale;
628 
629  // draw the empty box that appears as soon as there is a header
630  // dessine la case vide qui apparait des qu'il y a un entete
631  if (display_border_ &&
632  (display_columns_ ||
633  display_rows_)
634  ) {
635  Createdxf::drawRectangle(
636  file_path,
637  double(diagram_rect_.topLeft().x()) * Createdxf::xScale,
638  Createdxf::sheetHeight
639  - double(diagram_rect_.topLeft().y())
640  * Createdxf::yScale
641  - columns_header_height_,
642  rows_header_width_,
643  columns_header_height_,
644  color
645  );
646  }
647 
648  QSettings settings;
649 
650  // draw the numbering of the columns
651  // dessine la numerotation des colonnes
652  if (display_border_ &&
653  display_columns_) {
654  for (int i = 1 ; i <= columns_count_ ; ++ i) {
655  double xCoord = diagram_rect_.topLeft().x() * Createdxf::xScale +
656  (rows_header_width_ + ((i - 1) *
657  columns_width_));
658  double yCoord = Createdxf::sheetHeight
659  - diagram_rect_.topLeft().y()*Createdxf::yScale
660  - columns_header_height_;
661  double recWidth = columns_width_;
662  double recHeight = columns_header_height_;
663  Createdxf::drawRectangle(file_path, xCoord, yCoord,
664  recWidth, recHeight, color);
665  if (settings.value("border-columns_0", true).toBool()){
666  Createdxf::drawTextAligned(file_path,
667  QString::number(i - 1),
668  xCoord+recWidth/4,
669  yCoord + recHeight*0.2,
670  recHeight*0.7,
671  0,
672  0,
673  1,
674  2,
675  xCoord+recWidth/2,
676  1,
677  color,
678  0);
679  }else{
680  Createdxf::drawTextAligned(file_path,
681  QString::number(i),
682  xCoord+recWidth/4,
683  yCoord + recHeight*0.2,
684  recHeight*0.7,
685  0,
686  0,
687  1,
688  2,
689  xCoord+recWidth/2,
690  1,
691  color,
692  0);
693  }
694  }
695  }
696 
697  // draw line numbering
698  // dessine la numerotation des lignes
699  if (display_border_ && display_rows_) {
700  QString row_string("A");
701  for (int i = 1 ; i <= rows_count_ ; ++ i) {
702  double xCoord = diagram_rect_.topLeft().x()
703  * Createdxf::xScale;
704  double yCoord = Createdxf::sheetHeight
705  - diagram_rect_.topLeft().y()
706  *Createdxf::yScale
707  - (
708  columns_header_height_
709  + ((i - 1)
710  * rows_height_) )
711  - rows_height_;
712  double recWidth = rows_header_width_;
713  double recHeight = rows_height_;
714  Createdxf::drawRectangle(file_path, xCoord, yCoord,
715  recWidth, recHeight, color);
716  Createdxf::drawTextAligned(file_path,
717  row_string,
718  xCoord+recWidth*0.1,
719  yCoord + recHeight*0.4,
720  recWidth*0.7,
721  0,
722  0,
723  1,
724  2,
725  xCoord+recWidth/2,
726  1,
727  color,
728  0);
729  row_string = incrementLetters(row_string);
730  }
731  }
732 
733  // render the titleblock, using the TitleBlockTemplate object
734  if (display_titleblock_) {
735  //qp -> translate(titleblock_rect_.topLeft());
736  QRectF rect = titleBlockRect();
737  m_titleblock_template_renderer -> renderDxf(rect,
738  rect.width(),
739  file_path,
740  color);
741  //qp -> translate(-titleblock_rect_.topLeft());
742  }
743 
744  // Transform back to QET scale
745  columns_header_height_ /= Createdxf::yScale;
746  rows_height_ /= Createdxf::yScale;
747  rows_header_width_ /= Createdxf::xScale;
748  columns_width_ /= Createdxf::xScale;
749 
750 }

980 QString BorderTitleBlock::incrementLetters(const QString &string) {
981  if (string.isEmpty()) {
982  return("A");
983  } else {
984  // separate previous digits from last digit
985  // separe les digits precedents du dernier digit
986  QString first_digits(string.left(string.count() - 1));
987  QChar last_digit(string.at(string.count() - 1));
988  if (last_digit != 'Z') {
989  // increments the last digit
990  // incremente le dernier digit
991  last_digit = last_digit.toLatin1() + 1;
992  return(first_digits + QString(last_digit));
