228 void AddPartCommand::redo() {
229  // le premier appel a redo, lors de la construction de l'objet, ne doit pas se faire
230  if (first_redo) {
231  if (!part -> zValue()) {
232  // the added part has no specific zValue already defined, we put it
233  // above existing items (but still under terminals)
234  QList<QGraphicsItem *> existing_items = m_scene -> zItems(ElementScene::SortByZValue | ElementScene::SelectedOrNot);
235  qreal z = existing_items.count() ? existing_items.last() -> zValue() + 1 : 1;
236  part -> setZValue(z);
237  }
238  m_scene -> clearSelection();
239  first_redo = false;
240  return;
241  }
242  m_scene -> addItem(part);
243 }

284 ChangeZValueCommand::ChangeZValueCommand(
285  ElementScene *elmt,
286  QET::DepthOption o,
287  QUndoCommand *parent
288 ) :
289  ElementEditionCommand(elmt, nullptr, parent),
290  m_option(o)
291 {
292  // retrieve all primitives but terminals
293  QList<QGraphicsItem *> items_list = m_scene -> zItems(ElementScene::SortByZValue | ElementScene::SelectedOrNot);
294 
295  // prend un snapshot des zValues
296  foreach(QGraphicsItem *qgi, items_list) undo_hash.insert(qgi, qgi -> zValue());
297 
298  // choisit le nom en fonction du traitement
299  if (m_option == QET::BringForward) {
300  setText(QObject::tr("amener au premier plan", "undo caption"));
301  applyBringForward(items_list);
302  } else if (m_option == QET::Raise) {
303  setText(QObject::tr("rapprocher", "undo caption"));
304  applyRaise(items_list);
305  } else if (m_option == QET::Lower) {
306  setText(QObject::tr("éloigner", "undo caption"));
307  applyLower(items_list);
308  } else if (m_option == QET::SendBackward) {
309  setText(QObject::tr("envoyer au fond", "undo caption"));
310  applySendBackward(items_list);
311  }
312 }

332 void ChangeZValueCommand::applyBringForward(const QList<QGraphicsItem *> &items_list) {
333  QList<QGraphicsItem *> non_selected_items = items_list;
334  QList<QGraphicsItem *> selected_items;
335  foreach(QGraphicsItem *qgi, non_selected_items) {
336  if (qgi -> isSelected()) {
337  selected_items << qgi;
338  non_selected_items.removeAt(non_selected_items.indexOf(qgi));
339  }
340  }
341  int z = 1;
342  foreach(QGraphicsItem *qgi, non_selected_items) redo_hash.insert(qgi, z ++);
343  foreach(QGraphicsItem *qgi, selected_items) redo_hash.insert(qgi, z ++);
344 }

350 void ChangeZValueCommand::applyRaise(const QList<QGraphicsItem *> &items_list) {
351  QList<QGraphicsItem *> my_items_list = items_list;
352 
353  for (int i = my_items_list.count() - 2 ; i >= 0 ; -- i) {
354  if (my_items_list[i] -> isSelected()) {
355  if (!my_items_list[i +1] -> isSelected()) {
356 #if QT_VERSION < QT_VERSION_CHECK(5, 13, 0) // ### Qt 6: remove
357  my_items_list.swap(i, i + 1);
358 #else
359 #pragma message("@TODO remove code for QT 5.13 or later")
360  my_items_list.swapItemsAt(i, i + 1);
361 #endif
362  }
363  }
364  }
365  int z = 1;
366  foreach(QGraphicsItem *qgi, my_items_list) redo_hash.insert(qgi, z ++);
367 }

521 void ScalePartsCommand::scale(const QRectF &before, const QRectF &after) {
522  if (!scaled_primitives_.count()) return;
523  if (before == after) return;
524  if (!before.width() || !before.height()) return; // cowardly flee division by zero FIXME?
525 
526  foreach (CustomElementPart *part_item, scaled_primitives_) {
527  part_item -> startUserTransformation(before);
528  part_item -> handleUserTransformation(before, after);
529  }
530 }
