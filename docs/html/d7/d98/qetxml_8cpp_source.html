257 bool QETXML::writeXmlFile(const QDomDocument &xml_document,
258  const QString &file_path,
259  QString *error_message)
260 {
261  QFile file(file_path);
262 
263  // Note: we do not set QIODevice::Text to avoid generating CRLF end of lines
264  bool file_opening = file.open(QIODevice::WriteOnly);
265  if (!file_opening)
266  {
267  if (error_message)
268  {
269  *error_message = QString(
270  QObject::tr(
271  "Impossible d'ouvrir le fichier %1 en écriture, erreur %2 rencontrée.",
272  "error message when attempting to write an XML file")
273  ).arg(file_path).arg(file.error());
274  }
275  return(false);
276  }
277 
278  QTextStream out(&file);
279  out.setCodec("UTF-8");
280  out.setGenerateByteOrderMark(false);
281  out << xml_document.toString(4);
282  file.close();
283 
284  return(true);
285 }

341 QVector<QDomElement> QETXML::subChild(const QDomElement &element,
342  const QString parent_tag_name,
343  const QString &children_tag_name)
344 {
345  QVector<QDomElement> return_list;
346 
347  for (
348  QDomNode child = element.firstChild() ;
349  !child.isNull() ;
350  child = child.nextSibling())
351  {
352  QDomElement parents = child.toElement();
353  if (parents.isNull() || parents.tagName() != parent_tag_name)
354  continue;
355 
356  for (
357  QDomNode node_children = parents.firstChild() ;
358  !node_children.isNull() ;
359  node_children = node_children.nextSibling())
360  {
361  QDomElement n_children = node_children.toElement();
362  if (!n_children.isNull() && n_children.tagName() == children_tag_name)
363  return_list.append(n_children);
364  }
365  }
366 
367  return(return_list);

429 QDomElement QETXML::modelHeaderDataToXml(
430  QDomDocument &parent_document,
431  const QAbstractItemModel *model,
432  QHash<int,
433  QList<int>> horizontal_section_role,
434  QHash<int,
435  QList<int>> vertical_section_role)
436 {
437  auto dom_element = parent_document.createElement("header_data");
438 
439  auto orientation_ = Qt::Horizontal;
440  auto data_hash = horizontal_section_role;
441  auto meta_enum_ori = QMetaEnum::fromType<Qt::Orientation>();
442  auto meta_enum_role = QMetaEnum::fromType<Qt::ItemDataRole>();
443 
444  //Iterate twice, first for horizontal header and second to vertical header
445  while (true)
446  {
447  for (auto section : data_hash.keys())
448  {
449  for (auto role : data_hash.value(section))
450  {
451  auto variant = model->headerData(section, orientation_, role);
452  if (variant.isValid())
453  {
454  auto dom_data = parent_document.createElement("data");
455  dom_data.setAttribute("section", QString::number(section));
456  dom_data.setAttribute("orientation", meta_enum_ori.valueToKey(orientation_));
457  dom_data.setAttribute("role", meta_enum_role.valueToKey(role));
458 
459  auto text_node = parent_document.createTextNode("");
460  if (role == Qt::DisplayRole
461  || role == Qt::EditRole
462  || role == Qt::ToolTipRole
463  || role == Qt::StatusTipRole
464  || role == Qt::WhatsThisRole)
465  {
466  text_node.setData(variant.toString());
467  }
468  else if (role == Qt::FontRole)
469  {
470  auto font = variant.value<QFont>();
471  text_node.setData(font.toString());
472  }
473  else if (role == Qt::TextAlignmentRole)
474  {
475  auto me = QMetaEnum::fromType<Qt::Alignment>();
476  text_node.setData(me.valueToKey(variant.toInt()));
477  }
478  dom_data.appendChild(text_node);
479  dom_element.appendChild(dom_data);
480  }
481  }
482  }
483 
484  if(orientation_ == Qt::Vertical) {
485  break;
486  } else {
487  data_hash = vertical_section_role;
488  orientation_ = Qt::Vertical;
489  }
490  }
491 
492  return dom_element;
493 }

501 void QETXML::modelHeaderDataFromXml(const QDomElement &element,
502  QAbstractItemModel *model)
503 {
504  if (element.tagName() != "header_data")
505  return;
506 
507  auto meta_enum_orientation = QMetaEnum::fromType<Qt::Orientations>();
508  auto meta_enum_role = QMetaEnum::fromType<Qt::ItemDataRole>();
509 
510  for (auto child : QETXML::directChild(element, "data"))
511  {
512  auto section_ = child.attribute("section", "-1").toInt();
513  auto orientation_ = Qt::Orientation(
514  meta_enum_orientation.keyToValue(
515  child.attribute(
516  "orientation",
517  "Horizontal"
518  ).toStdString().data()));
519  auto role_ = meta_enum_role.keyToValue(
520  child.attribute(
521  "role",
522  "DisplayRole"
523  ).toStdString().data());
524  auto text_ = child.text();
525  QVariant data_;
526 
527  if (role_ == Qt::DisplayRole
528  || role_ == Qt::EditRole
529  || role_ == Qt::ToolTipRole
530  || role_ == Qt::StatusTipRole
531  || role_ == Qt::WhatsThisRole) {
532  data_ = text_;
533  }
534  else if (role_ == Qt::FontRole)
535  {
536  QFont font;
537  font.fromString(text_);
538  data_ = font;
539  }
540  else if (role_ == Qt::TextAlignmentRole)
541  {
542  auto me = QMetaEnum::fromType<Qt::Alignment>();
543  data_ = me.keyToValue(text_.toStdString().data());
544  }
545 
546  model->setHeaderData(section_, orientation_, data_, role_);
547  }
548 }
