42 ElementsLocation ECHSFileToFile::copy()
43 {
44  //Check if the destination already have an item with the same name of the item to copy
45  ElementsLocation location(m_destination.fileSystemPath() + "/" + m_source.fileName());
46  QString rename;
47  if (location.exist())
48  {
49  RenameDialog rd(location.fileSystemPath());
50  if (rd.exec() == QDialog::Accepted)
51  {
52  if (rd.selectedAction() == QET::Erase)
53  {
54  if (location.isDirectory())
55  {
56  QDir dir(location.fileSystemPath());
57  dir.removeRecursively();
58  }
59  else
60  {
61  QFile file(location.fileSystemPath());
62  file.remove();
63  }
64  }
65  else if (rd.selectedAction() == QET::Rename)
66  {
67  rename = rd.newName();
68  }
69  }
70  else
71  return ElementsLocation();
72  }
73 
74  if (m_source.isElement())
75  return copyElement(m_source, m_destination, rename);
76  else
77  return copyDirectory(m_source, m_destination, rename);

80 ElementsLocation ECHSFileToFile::copyDirectory(ElementsLocation &source, ElementsLocation &destination, const QString& rename)
81 {
82  QDir source_dir(source.fileSystemPath());
83  QDir destination_dir(destination.fileSystemPath());
84 
85  if (!source_dir.exists() || !destination_dir.exists()) return ElementsLocation();
86 
87  QString new_dir_name = rename.isEmpty() ? source_dir.dirName() : rename;
88 
89  //Create a new dir
90  if (destination_dir.mkdir(new_dir_name))
91  {
92  //The new created directory
93  QDir created_dir(destination_dir.canonicalPath() + "/" + new_dir_name);
94 
95  //Copy the qet_directory file
96  QFile::copy(source_dir.canonicalPath() + "/qet_directory", created_dir.canonicalPath() + "/qet_directory");
97 
98  //Copy all dirs found in source_dir to destination_dir
99  ElementsLocation created_location(created_dir.canonicalPath());
100  //Used this bool when user drop a folder into itself to avoid infinite recursive creation of the dropped dir
101  bool copy_itself = false;
102  if (source_dir == destination_dir)
103  copy_itself = true;
104 
105  foreach(QString str, source_dir.entryList(QDir::Dirs | QDir::NoDotAndDotDot, QDir::Name))
106  {
107  if (copy_itself)
108  {
109  if (source_dir.dirName() == str)
110  {
111  copy_itself = false;
112  continue;
113  }
114  }
115 
116  ElementsLocation sub_source(source.fileSystemPath() + "/" + str);
117  copyDirectory(sub_source, created_location);
118  }
119 
120  //Copy all elements found in source_dir to destination_dir
121  source_dir.setNameFilters(QStringList() << "*.elmt");
122  foreach(QString str, source_dir.entryList(QDir::Files | QDir::NoDotAndDotDot, QDir::Name))
123  {
124  ElementsLocation sub_source(source.fileSystemPath() + "/" + str);
125  copyElement(sub_source, created_location);
126  }
127 
128  return created_location;
129  }
130 
131  return ElementsLocation();
132 }

134 ElementsLocation ECHSFileToFile::copyElement(ElementsLocation &source, ElementsLocation &destination, const QString& rename)
135 {
136  QString new_elmt_name = rename.isEmpty() ? source.fileName() : rename;
137  bool rb = QFile::copy(source.fileSystemPath(), destination.fileSystemPath() + "/" + new_elmt_name);
138  if (rb)
139  {
140 #ifdef Q_OS_WIN
141  //On windows when user drag and drop an element from the common elements collection
142  //to the custom elements collection, the element file stay in read only mode, and so
143  //user can't save the element
144  extern Q_CORE_EXPORT int qt_ntfs_permission_lookup;
145  qt_ntfs_permission_lookup++;
146  QFile file(destination.fileSystemPath() + "/" + new_elmt_name);
147  if (!file.isWritable()) {
148  if (!file.setPermissions(file.permissions() | QFileDevice::WriteUser)) {
149  qDebug() << "Failed to change file permission of : " << QFileInfo(file).canonicalFilePath() \
150  << " in ECHSFileToFile::copyElement";
151  }
152  }
153  qt_ntfs_permission_lookup--;
154 #endif
155  return ElementsLocation (destination.fileSystemPath() + "/" + new_elmt_name);
156  }
157  else
158  return ElementsLocation();
159 }

205 ElementsLocation ECHSXmlToFile::copyDirectory(ElementsLocation &source, ElementsLocation &destination, const QString& rename)
206 {
207  QDir destination_dir(destination.fileSystemPath());
208 
209  if (!(destination_dir.exists() && source.exist())) return ElementsLocation();
210 
211  QString new_dir_name = rename.isEmpty() ? source.fileName() : rename;
212 
213  //Create new dir
214  if (destination_dir.mkdir(new_dir_name))
215  {
216  QDir created_dir(destination_dir.canonicalPath() + "/" + new_dir_name);
217  ElementsLocation created_location(created_dir.canonicalPath());
218 
219  //Create the qet-directory file
220  QDomDocument document;
221  QDomElement root = document.createElement("qet-directory");
222  document.appendChild(root);
223  root.appendChild(source.nameList().toXml(document));
224 
225  QString filepath = created_dir.canonicalPath() + "/qet_directory";
226  QET::writeXmlFile(document, filepath);
227 
228  //Create all directory found in source to created_dir
229  XmlElementCollection *project_collection = source.projectCollection();
230 
231  QStringList directories_names = project_collection->directoriesNames( project_collection->directory(source.collectionPath(false)) );
232  foreach(QString name, directories_names)
233  {
234  ElementsLocation sub_source_dir(source.projectCollectionPath() + "/" + name);
235  copyDirectory(sub_source_dir, created_location);
236  }
237 
238  //Create all elements found in source to destination
239  QStringList elements_names = project_collection->elementsNames( project_collection->directory(source.collectionPath(false))) ;
240  foreach (QString name, elements_names)
241  {
242  ElementsLocation source_element(source.projectCollectionPath() + "/" + name);
243  copyElement(source_element, created_location);
244  }
245 
246  return created_location;
247  }
248 
249  return ElementsLocation();

344 ElementsLocation ElementCollectionHandler::createDir(ElementsLocation &parent, const QString &name, const NamesList &name_list)
345 {
346  //Parent must be a directorie and writable
347  if (!(parent.isDirectory() && parent.isWritable() && parent.exist())) {
348  qDebug() << "ElementCollectionHandler::createDir : the prerequisites are not valid. " << parent;
349  return ElementsLocation();
350  }
351 
352  //Directorie to create must not already exist
353  ElementsLocation created_dir = parent;
354  created_dir.addToPath(name);
355  if (created_dir.exist()) {
356  return ElementsLocation();
357  }
358 
359  //Location is a file system
360  if (parent.isFileSystem()) {
361 
362  QDir parent_dir(parent.fileSystemPath());
363 
364  if (parent_dir.mkdir(name)) {
365  //Create the qet-directory file
366  QDomDocument document;
367  QDomElement root = document.createElement("qet-directory");
368  document.appendChild(root);
369  root.appendChild(name_list.toXml(document));
370 
371  QString filepath = created_dir.fileSystemPath() + "/qet_directory";
372  if (!QET::writeXmlFile(document, filepath)) {
373  qDebug() << "ElementCollectionHandler::createDir : write qet-directory file failed";
374  }
375  return created_dir;
376  }
377  else {
378  qDebug() << "ElementCollectionHandler::createDir : error was occurred at creation of new directories in file system. ";
379  return ElementsLocation();
380  }
381  }
382  else if (parent.isProject()) {
383  XmlElementCollection *xmlec = parent.projectCollection();
384  if (xmlec->createDir(parent.collectionPath(false), name, name_list)) {
385  return created_dir;
386  }
387  else {
388  qDebug() << "ElementCollectionHandler::createDir : error was occurred at creation of new directories in embbeded collection.";
389  return ElementsLocation();
390  }
391  }
392 
393  return ElementsLocation();

405 bool ElementCollectionHandler::importFromProject(QETProject *project,
406  ElementsLocation &location)
407 {
408  if (!(location.isElement() && location.exist() && location.isProject()))
409  return false;
410 
411  ElementsLocation destination(location.collectionPath(false), project);
412  if (destination.exist()) return true;
413 
414  QList <QString> names;
415 
416  //Get the parent of location and find if exist in embedded collection of project
417  ElementsLocation source = location.parent();
418  names.append(location.fileName());
419 
420  destination = ElementsLocation(source.collectionPath(), project);
421 
422  //Go back until to find an existing directory in destination
423  while (!destination.exist()) {
424  names.append(source.fileName());
425  source = source.parent();
426  destination = ElementsLocation(source.collectionPath(),
427  project);
428  }
429 
430  XmlElementCollection *collection = project->embeddedElementCollection();
431 
432  while (!names.isEmpty()) {
433  source.addToPath(names.takeLast());
434  destination = collection->copy(source,
435  destination,
436  QString(),
437  false);
438 
439  if (!destination.exist())
440  return false;
441  }
442 
443  return true;
