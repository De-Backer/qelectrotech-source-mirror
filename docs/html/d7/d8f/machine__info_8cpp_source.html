119 void Machine_info::init_get_Screen_info()
120 {
121  const auto screens = qApp->screens();
122  pc.screen.count=screens.count();
123  for (int ii = 0; ii < pc.screen.count; ++ii) {
124  pc.screen.width[ii]=
125  screens[ii]->geometry().width()
126  * screens[ii]->devicePixelRatio();
127  if(pc.screen.Max_width<pc.screen.width[ii])
128  pc.screen.Max_width=pc.screen.width[ii];
129 
130  pc.screen.height[ii]=
131  screens[ii]->geometry().height()
132  * screens[ii]->devicePixelRatio();
133  if(pc.screen.Max_height<pc.screen.height[ii])
134  pc.screen.Max_height=pc.screen.height[ii];
135  }
136 }

158 void Machine_info::init_get_cpu_info_linux()
159 {
160  QProcess linuxcpuinfo;
161  linuxcpuinfo.start("bash", QStringList()
162  << "-c"
163  << "cat /proc/cpuinfo |grep 'model name' | uniq");
164  linuxcpuinfo.waitForFinished();
165  QString linuxOutput = linuxcpuinfo.readAllStandardOutput();
166  pc.cpu.info=QString(linuxOutput.toLocal8Bit().constData());
167 
168  QProcess p;
169  p.start("awk", QStringList()
170  << "/MemTotal/ { print $2 }"
171  << "/proc/meminfo");
172  p.waitForFinished();
173  QString memory = p.readAllStandardOutput();
174  //compilation_info += "<br>"
175  pc.ram.Total=QString("RAM Total : %1 MB").arg(
176  memory.toLong() / 1024);
177  p.close();
178 
179  QProcess qp;
180  qp.start("awk",
181  QStringList()
182  << "/MemAvailable/ {print $2}"
183  << "/proc/meminfo");
184  qp.waitForFinished();
185  QString AvailableMemory = qp.readAllStandardOutput();
186  pc.ram.Available=QString("RAM Available : %1 MB").arg(
187  AvailableMemory.toLong() / 1024);
188  qp.close();
189 
190  QProcess linuxgpuinfo;
191  linuxgpuinfo.start("bash",
192  QStringList()
193  << "-c"
194  << "lspci | grep VGA | cut -d : -f 3");
195  linuxgpuinfo.waitForFinished();
196  QString linuxGPUOutput = linuxgpuinfo.readAllStandardOutput();
197  pc.gpu.info=QString(linuxGPUOutput.toLocal8Bit().constData());
198  pc.gpu.RAM="@ToDo";
199 }

204 void Machine_info::init_get_cpu_info_winnt()

211  pc.cpu.info=QString(windows_output.toLocal8Bit().constData());

218  << "get"
219  << "VideoProcessor ");
220  wingpuinfo.waitForFinished();
221  QString WinGPUOutput = wingpuinfo.readAllStandardOutput();
222  pc.gpu.info=QString(WinGPUOutput.toLocal8Bit().constData());
223 
224  QProcess wingpuraminfo;

227  << "PATH"

231  wingpuraminfo.waitForFinished();
232  QString WinGPURAMOutput = wingpuraminfo.readAllStandardOutput();
233  pc.gpu.RAM=QString("RAM Total : %1 B").arg(

236 
237 #ifdef Q_OS_WIN
238  MEMORYSTATUSEX memory_status;
239  ZeroMemory(&memory_status, sizeof(MEMORYSTATUSEX));
240  memory_status.dwLength = sizeof(MEMORYSTATUSEX);
241  if (GlobalMemoryStatusEx(&memory_status)) {
242  pc.ram.Total .append(
243  QString("RAM Total : %1 MB")
244  .arg(memory_status.ullTotalPhys / (1024 * 1024)));
245  pc.ram.Available .append(
246  QString("RAM Available : %1 MB")
247  .arg(memory_status.ullAvailPhys / (1024 * 1024)));

250  pc.ram.Available.append("Unknown RAM");
