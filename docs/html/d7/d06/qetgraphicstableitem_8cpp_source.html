42 void QetGraphicsTableItem::adjustTableToFolio(QetGraphicsTableItem *table,
43  QMargins margins)
44 {
45  if (!table->diagram()) {
46  return;
47  }
48 
49  auto drawable_rect = table->diagram()->border_and_titleblock.insideBorderRect();
50  table->setPos(drawable_rect.topLeft().x() + margins.left(), drawable_rect.topLeft().y() + margins.top() + table->headerItem()->rect().height());
51 
52  auto size_ = table->size();
53  size_.setWidth(int(drawable_rect.width() - (margins.left() + margins.right())));
54  //Size must be a multiple of 10, because the table adjust itself by step of 10.
55  while (size_.width()%10) {
56  --size_.rwidth(); }
57  table->setSize(size_);
58 
59  //Calcul the maximum row to display to fit the nomenclature into diagram
60  auto available_height = drawable_rect.height() - table->pos().y();
61  auto min_row_height = table->minimumRowHeigth();
62  table->setDisplayNRow(int(floor(available_height/min_row_height))); //Convert a double to int, but max_row_to_display is already rounded an integer so we assume everything is ok
63 }

72 void QetGraphicsTableItem::checkInsufficientRowsCount(
73  QetGraphicsTableItem *table)
74 {
75  if (!table->diagram() || !table->model()) {
76  return;
77  }
78 
79  auto first_table = table;
80  while (first_table->previousTable())
81  first_table = first_table->previousTable();
82 
83  if (first_table->displayNRow() <= 0) //displayed rows is unlimited
84  return;
85 
86  int count_ = first_table->displayNRow();
87  bool several_table = false;
88  while (first_table->nextTable())
89  {
90  several_table = true;
91  first_table = first_table->nextTable();
92  if (first_table->displayNRow() <= 0) { //displayed rows is unlimited
93  return;
94  } else {
95  count_ += first_table->displayNRow();
96  first_table->displayNRowOffset();
97  }
98  }
99 
100  if (count_ < first_table->model()->rowCount())
101  {
102  QWidget *parent = nullptr;
103  if (first_table->diagram() && first_table->diagram()->views().size())
104  parent = first_table->diagram()->views().first();
105 
106  QString text;
107  if (several_table) {
108  text = tr("Les information à afficher sont supérieurs à la quantité maximal pouvant être affiché par les tableaux.\n"
109  "Veuillez ajouter un nouveau tableau ou regler les tableaux existant afin d'afficher l'integralité des informations.");
110  } else {
111  text = tr("Les information à afficher sont supérieurs à la quantité maximal pouvant être affiché par le tableau.\n"
112  "Veuillez ajouter un nouveau tableau ou regler le tableau existant afin d'afficher l'integralité des informations.");
113  }
114  QMessageBox::information(parent, tr("Limitation de tableau"), text);
115  }
116 
117 }

207 void QetGraphicsTableItem::paint(QPainter *painter, const QStyleOptionGraphicsItem *option, QWidget *widget)
208 {
209  Q_UNUSED(option)
210  Q_UNUSED(widget)
211 
212  painter->save();
213 
214  QPen pen;
215  pen.setWidthF(0.7);
216  pen.setColor(Qt::black);
217  painter->setPen(pen);
218  painter->drawRect(0,0, m_header_item->rect().width(), m_current_size.height());
219 
220  if (isSelected())
221  {
222  painter->save();
223  QColor color(Qt::darkBlue);
224  color.setAlpha(20);
225  painter->setBrush(QBrush (color));
226  painter->setPen(Qt::NoPen);
227  painter->drawRect(0,0, m_header_item->rect().width(), m_current_size.height());
228  painter->restore();
229  }
230 
231  if(!m_model)
232  {
233  painter->restore();
234  return;
235  }
236  painter->setFont(m_model->data(m_model->index(0,0), Qt::FontRole).value<QFont>());
237 
238  //Draw vertical lines
239  auto offset= 0;
240  for(auto i=0 ; i<m_model->columnCount() ; ++i)
241  {
242  QPointF p1(offset+m_header_item->sectionSize(i), 0);
243  QPointF p2(offset+m_header_item->sectionSize(i), m_current_size.height());
244  painter->drawLine(p1, p2);
245  offset += m_header_item->sectionSize(i);
246  }
247 
248  //Calcule the number of rows to display.
249  auto row_count = m_model->rowCount();
250 
251  if (m_previous_table) //Remove the number of row already displayed by previous tables
252  row_count -= m_previous_table->displayNRowOffset();
253 
254  if (m_number_of_displayed_row > 0) //User override the number of row to display
255  row_count = std::min(row_count, m_number_of_displayed_row);
256 
257  //Draw horizontal lines
258  auto cell_height = static_cast<double>(m_current_size.height())/static_cast<double>(row_count);
259  for(auto i= 1 ; i-1<row_count ; ++i)
260  {
261  QPointF p1(m_header_item->rect().left(), cell_height*i);
262  QPointF p2(m_header_item->rect().right(), cell_height*i);
263  painter->drawLine(p1, p2);
264  }
265 
266  //Write text of each cell
267  for (auto i=0 ; i<row_count ; ++i)
268  {
269  auto margin_ = QETUtils::marginsFromString(m_model->index(0,0).data(Qt::UserRole+1).toString());
270  QPointF top_left(margin_.left(), i==0? margin_.top() : cell_height*i + margin_.top());
271 
272  for(auto j= 0 ; j<m_model->columnCount() ; ++j)
273  {
274  //In first iteration the top left X is margin left, in all other iteration the top left X is stored in m_column_size
275  if (j>0) {
276  top_left.setX(top_left.x() + m_header_item->sectionSize(j-1));
277  }
278  QSize size(m_header_item->sectionSize(j) - margin_.left() - margin_.right(),
279  static_cast<int>(cell_height) - margin_.top() - margin_.bottom());
280  auto index_row = m_previous_table ? i + m_previous_table->displayNRowOffset() : i;
281  painter->drawText(QRectF(top_left, size),
282  m_model->data(m_model->index(0,0), Qt::TextAlignmentRole).toInt() | Qt::AlignVCenter,
283  m_model->index(index_row, j).data().toString());
284  }
285  }
286 
287  painter->restore();
288 }

772 void QetGraphicsTableItem::adjustColumnsWidth()
773 {
774  if (!m_model)
775  {
776  auto h_ = m_header_item->minimumSectionWidth();
777  for (auto i=0 ; i<h_.size() ; ++i) {
778  m_header_item->resizeSection(i, h_.at(i));
779  }
780  return;
781  }
782 
783  auto a = m_current_size.width() - minimumSize().width();
784  auto b = a/std::max(1,m_model->columnCount()); //avoid divide by 0
785 
786  disconnect(m_header_item, &QetGraphicsHeaderItem::sectionResized, this, &QetGraphicsTableItem::headerSectionResized); //Avoid to resize
787 
788  int sum_=0;
789  for(auto i= 0 ; i<m_model->columnCount() ; ++i)
790  {
791  auto at_a = std::min(m_minimum_column_width.size()-1, i); //In case of the I is higher than m_minimum_column_width or
792  auto at_b = std::min(m_header_item->minimumSectionWidth().size()-1, i); //m_header_item->minimumSectionWidth().size()
793  m_header_item->resizeSection(i, std::max(m_minimum_column_width.at(at_a),
794  m_header_item->minimumSectionWidth().at(at_b))+b);
795  sum_+= m_header_item->sectionSize(i);
796  }
797 
798 
799  //The sum of the header sections width can be less than width of @m_current_size we adjust it in order to have the same width
800  if (m_model->columnCount() > 0 &&
801  sum_ < m_current_size.width())
802  {
803  //add the quotient of the division to each columns
804  auto result = (m_current_size.width()-sum_)/m_model->columnCount();
805  for(auto i= 0 ; i<m_model->columnCount() ; ++i) {
806  m_header_item->resizeSection(i, m_header_item->sectionSize(i) + result);
807  }
808 
809  //add the rest of the division to the last column
810  auto last_section = m_model->columnCount()-1;
811  m_header_item->resizeSection(last_section, m_header_item->sectionSize(last_section) + ((m_current_size.width()-sum_)%m_model->columnCount()));
812  }
813  connect(m_header_item, &QetGraphicsHeaderItem::sectionResized, this, &QetGraphicsTableItem::headerSectionResized);
814  update();
815 }
