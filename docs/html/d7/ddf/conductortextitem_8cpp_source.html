130 void ConductorTextItem::setPos(const QPointF &pos)
131 {
132  /*
133  * In some condition the conductor text item is outside the border of folio in the left.
134  * They cause a margin on the left of folio and in most case this margin is unwanted and annoying the user.
135  * If the text is empty and the scene position is outside the border (left and top),
136  * we can say that this position, is unwanted by user.
137  * So we move this text item to the top left of the bounding rect of parent conductors, because we sure this position is wanted by user.
138  */
139  DiagramTextItem::setPos(pos);
140  if(toPlainText().isEmpty() && (scenePos().x() < 0 || scenePos().y() < 0))
141  {
142  Conductor *cond = parentConductor();
143  if(cond)
144  DiagramTextItem::setPos(cond->boundingRect().topLeft());
145  else
146  DiagramTextItem::setPos(0,0);
147  }
148 }

174 void ConductorTextItem::mouseMoveEvent(QGraphicsSceneMouseEvent *event) {
175  if (textInteractionFlags() & Qt::TextEditable) QGraphicsTextItem::mouseMoveEvent(event);
176 
177  else if ((flags() & QGraphicsItem::ItemIsMovable) && (event -> buttons() & Qt::LeftButton)) {
178 
179  QPointF intended_pos = event ->scenePos() + m_mouse_to_origin_movement;
180 
181  if (parent_conductor_) {
182  if (parent_conductor_->nearShape().contains(intended_pos)) {
183  event->modifiers() == Qt::ControlModifier ? setPos(intended_pos) : setPos(Diagram::snapToGrid(intended_pos));
184  parent_conductor_ -> setHighlighted(Conductor::Normal);
185  } else {
186  parent_conductor_ -> setHighlighted(Conductor::Alert);
187  }
188  }
189 
190  }
191 
192  else event -> ignore();
193 }

199 void ConductorTextItem::mouseReleaseEvent(QGraphicsSceneMouseEvent *e) {
200  if (flags() & QGraphicsItem::ItemIsMovable) {
201 
202  if (Diagram *diagram_ptr = diagram()) {
203  QPointF applied_movement = pos() - before_mov_pos_;
204 
205  if (!applied_movement.isNull()) {
206  //Create an undo object
207  MoveConductorsTextsCommand *undo_object = new MoveConductorsTextsCommand(diagram_ptr);
208  undo_object -> addTextMovement(this, before_mov_pos_, pos(), moved_by_user_);
209 
210  moved_by_user_ = true;
211 
212  diagram_ptr -> undoStack().push(undo_object);
213  }
214 
215  if (parent_conductor_) {
216  parent_conductor_ -> setHighlighted(Conductor::None);
217  }
218  }
219  }
220  if (!(e -> modifiers() & Qt::ControlModifier)) {
221  QGraphicsTextItem::mouseReleaseEvent(e);
222  }
223 }
