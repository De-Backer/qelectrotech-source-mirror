60 void ProjectDBModelPropertiesWidget::on_m_edit_query_pb_clicked()
61 {

64  d.setLayout(l);
65 
66  ElementQueryWidget *nom_w = nullptr;
67  SummaryQueryWidget *sum_w = nullptr;
68  if (m_model->identifier() == "nomenclature")
69  {
70  nom_w = new ElementQueryWidget(&d);
71  nom_w->setQuery(m_model->queryString());
72  l->addWidget(nom_w);
73  }
74  else if (m_model->identifier() == "summary")
75  {
76  sum_w = new SummaryQueryWidget(&d);
77  sum_w->setQuery(m_model->queryString());
78  l->addWidget(sum_w);
79  }
80 
81  auto button_box = new QDialogButtonBox(QDialogButtonBox::Ok | QDialogButtonBox::Cancel);
82  l->addWidget(button_box);
83  connect(button_box, &QDialogButtonBox::accepted, &d, &QDialog::accept);
84  connect(button_box, &QDialogButtonBox::rejected, &d, &QDialog::reject);
85 
86  if (d.exec())
87  {
88  if (nom_w) {
89  m_model->setQuery(nom_w->queryStr());

92  }
93  }
