158 void Diagram::drawBackground(QPainter *p, const QRectF &r) {
159  p -> save();
160 
161  // disable all antialiasing, except for text
162  // desactive tout antialiasing, sauf pour le texte
163  p -> setRenderHint(QPainter::Antialiasing, false);
164  p -> setRenderHint(QPainter::TextAntialiasing, true);
165  p -> setRenderHint(QPainter::SmoothPixmapTransform, false);
166 
167  // draw a white background
168  // dessine un fond blanc
169  p -> setPen(Qt::NoPen);
170  //set brush color to present background color.
171  p -> setBrush(Diagram::background_color);
172  p -> drawRect(r);
173 
174  if (draw_grid_) {
175  /* Draw the point of the grid
176  * if background color is black,
177  * then grid spots shall be white,
178  * else they shall be black in color.
179  */
180  QPen pen;
181  Diagram::background_color == Qt::black? pen.setColor(Qt::white)
182  : pen.setColor(Qt::black);
183  pen.setCosmetic(true);
184  p->setPen(pen);
185 
186  p -> setBrush(Qt::NoBrush);
187 
188  // If user allow zoom out beyond of folio,
189  // we draw grid outside of border.
190  QSettings settings;
191  int xGrid = settings.value("diagrameditor/Xgrid",
192  Diagram::xGrid).toInt();
193  int yGrid = settings.value("diagrameditor/Ygrid",
194  Diagram::yGrid).toInt();
195  QRectF rect = settings.value(
196  "diagrameditor/zoom-out-beyond-of-folio",
197  false).toBool() ? r
198  : border_and_titleblock
199  .insideBorderRect()
200  .intersected(r);
201 
202  qreal limite_x = rect.x() + rect.width();
203  qreal limite_y = rect.y() + rect.height();
204 
205  int g_x = (int)ceil(rect.x());
206  while (g_x % xGrid) ++ g_x;
207  int g_y = (int)ceil(rect.y());
208  while (g_y % yGrid) ++ g_y;
209 
210  QPolygon points;
211  for (int gx = g_x ; gx < limite_x ; gx += xGrid) {
212  for (int gy = g_y ; gy < limite_y ; gy += yGrid) {
213  points << QPoint(gx, gy);
214  }
215  }
216  p -> drawPoints(points);
217  }
218 
219  if (use_border_) border_and_titleblock.draw(p);
220  p -> restore();
221 }

324 void Diagram::keyPressEvent(QKeyEvent *event)
325 {
326  QSettings settings;
327  int xKeyGrid = settings.value("diagrameditor/key_Xgrid",
328  Diagram::xKeyGrid).toInt();
329  int yKeyGrid = settings.value("diagrameditor/key_Ygrid",
330  Diagram::yKeyGrid).toInt();

334  Diagram::yKeyGridFine).toInt();

336 
337  if (m_event_interface) {
338  m_event_interface->keyPressEvent(event);
339  if (event->isAccepted()) {
340  return;
341  }
342  }
343 
344  if (isReadOnly()) return;
345 
346  QPointF movement;
347  qreal top_position = 0;
348  qreal left_position = 0;
349  DiagramContent dc(this);
350  if (dc.items(DiagramContent::All).isEmpty())
351  {
352  event->ignore();
353  QGraphicsScene::keyPressEvent(event);
354  return;
355  }
356 
357 #pragma message("@TODO move code to new function")
358  //Move item with the keyboard arrow
359  if(event->modifiers() == Qt::NoModifier)
360  {
361  switch(event->key())
362  {
363  case Qt::Key_Left:
364  for (Element *item : dc.m_elements)
365  {
366  left_position = item->sceneBoundingRect().x();
367  if(left_position <= 5)
368  return;
369  }
370  movement = QPointF(-xKeyGrid, 0.0);
371  break;
372  case Qt::Key_Right:
373  movement = QPointF(+xKeyGrid, 0.0);
374  break;
375  case Qt::Key_Up:
376  for(Element *item : dc.m_elements)
377  {
378  top_position = item->sceneBoundingRect().y();
379  if(top_position <= 5)
380  return;
381  }
382  movement = QPointF(0.0, -yKeyGrid);
383  break;
384  case Qt::Key_Down:
385  movement = QPointF(0.0, +yKeyGrid);
386  break;
387  }
388 
389  if (!movement.isNull() && !focusItem())
390  {
391  m_elements_mover.beginMovement(this);
392  m_elements_mover.continueMovement(movement);
393  event->accept();
394  return;
395  }
396  }
397  else if(event->modifiers() == Qt::AltModifier)
398  {
399  switch(event->key())
400  {
401  case Qt::Key_Left:
402  for (Element *item : dc.m_elements)
403  {
404  left_position = item->sceneBoundingRect().x();
405  if(left_position <= 5)
406  return;
407  }
408  movement = QPointF(-xKeyGridFine, 0.0);
409  break;
410  case Qt::Key_Right:
411  movement = QPointF(+xKeyGridFine, 0.0);
412  break;
413  case Qt::Key_Up:
414  for(Element *item : dc.m_elements)
415  {
416  top_position = item->sceneBoundingRect().y();
417  if(top_position <= 5)
418  return;
419  }
420  movement = QPointF(0.0, -yKeyGridFine);
421  break;
422  case Qt::Key_Down:
423  movement = QPointF(0.0, +yKeyGridFine);
424  break;
425  }
426 
427  if (!movement.isNull() && !focusItem())
428  {
429  m_elements_mover.beginMovement(this);
430  m_elements_mover.continueMovement(movement);
431  event->accept();
432  return;
433  }
434  }
435  else if(event->modifiers() == Qt::ControlModifier)
436  {
437  //Adjust the alignment of a texts group
438  if(selectedItems().size() == 1
439  && selectedItems().first()->type()
440  == QGraphicsItemGroup::Type)
441  {
442  if(ElementTextItemGroup *etig =
443  dynamic_cast<ElementTextItemGroup *>
444  (selectedItems().first()))
445  {
446  if(event->key() == Qt::Key_Left
447  && etig->alignment() != Qt::AlignLeft)
448  undoStack().push(
449  new AlignmentTextsGroupCommand(

451  Qt::AlignLeft));
452 
453  else if (event->key() == Qt::Key_Up
454  && etig->alignment() != Qt::AlignVCenter)
455  undoStack().push(
456  new AlignmentTextsGroupCommand(
457  etig,
458  Qt::AlignVCenter));
459 
460  else if (event->key() == Qt::Key_Right
461  && etig->alignment() != Qt::AlignRight)
462  undoStack().push(
463  new AlignmentTextsGroupCommand(
464  etig,
465  Qt::AlignRight));
466  }
467  }
468  }
469  event->ignore();
470  QGraphicsScene::keyPressEvent(event);
471 }

591  source_area.setHeight(source_area.height() + 2.0 * margin);
592  } else {
593  source_area = QRectF(
594  0.0,
595  0.0,

605  * si les dimensions ne sont pas precisees,
606  * l'image est exportee a l'echelle 1:1
607  */
608  QSize image_size = (width == -1 && height == -1)
609  ? source_area.size().toSize() : QSize(width, height);
610 
611  // prepare the rendering
612  // prepare le rendu
613  QPainter p;
614  if (!p.begin(&pix)) return(false);
615 
616  // anti-aliasis rendering
617  // rendu antialiase
618  p.setRenderHint(QPainter::Antialiasing, true);
619  p.setRenderHint(QPainter::TextAntialiasing, true);
620  p.setRenderHint(QPainter::SmoothPixmapTransform, true);
621 
622  // deselect all elements
623  // deselectionne tous les elements
624  QList<QGraphicsItem *> selected_elmts = selectedItems();
625  foreach (QGraphicsItem *qgi, selected_elmts) qgi -> setSelected(false);
626 
627  // renders itself
628  // effectue le rendu lui-meme
629  render(&p,

632  aspectRatioMode);

634 
635  // restore selected items
636  // restaure les elements selectionnes
637  foreach (QGraphicsItem *qgi, selected_elmts) qgi -> setSelected(true);
638 
639  return(true);
640 }
