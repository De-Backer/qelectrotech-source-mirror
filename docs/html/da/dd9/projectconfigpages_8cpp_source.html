139 void ProjectMainConfigPage::applyProjectConf() {
140  bool modified_project = false;
141 
142  QString new_title = title_value_ -> text();
143  if (m_project -> title() != new_title) {
144  m_project -> setTitle(new_title);
145  modified_project = true;
146  }
147 
148  DiagramContext new_properties = project_variables_ -> context();
149  if (m_project -> projectProperties() != new_properties) {
150  m_project -> setProjectProperties(new_properties);
151  modified_project = true;
152  }
153  if (modified_project) {
154  m_project -> setModified(true);
155  }
156 }

389 void ProjectAutoNumConfigPage::saveContextElement()
390 {
391  // If the text is the default text "Name of new numerotation" save the edited context
392  // With the the name "No name"
393  if (m_saw_element->contextComboBox()->currentText() == tr("Nom de la nouvelle numérotation"))
394  {
395  QString title(tr("Sans nom"));
396 
397  m_project->addElementAutoNum (title, m_saw_element -> toNumContext());
398  m_project->setCurrrentElementAutonum(title);
399  m_saw_element->contextComboBox()->addItem(tr("Sans nom"));
400  }
401  // If the text isn't yet to the autonum of the project, add this new item to the combo box.
402  else if ( !m_project -> elementAutoNum().keys().contains( m_saw_element->contextComboBox()->currentText()))
403  {
404  m_project->addElementAutoNum(m_saw_element->contextComboBox()->currentText(), m_saw_element->toNumContext());
405  m_project->setCurrrentElementAutonum(m_saw_element->contextComboBox()->currentText());
406  m_saw_element->contextComboBox()->addItem(m_saw_element->contextComboBox()->currentText());
407  }
408  // Else, the text already exist in the autonum of the project, just update the context
409  else
410  {
411  m_project->addElementAutoNum (m_saw_element->contextComboBox() -> currentText(), m_saw_element -> toNumContext());
412  m_project->setCurrrentElementAutonum(m_saw_element->contextComboBox()->currentText());
413  }

486 void ProjectAutoNumConfigPage::applyAutoNum() {
