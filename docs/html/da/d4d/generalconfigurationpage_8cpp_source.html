137 void GeneralConfigurationPage::applyConf()
138 {
139  QSettings settings;
140 
141  //GLOBAL
142  bool was_using_system_colors = settings.value("usesystemcolors", "true").toBool();
143  bool must_use_system_colors = ui->m_use_system_color_cb->isChecked();
144  settings.setValue("usesystemcolors", must_use_system_colors);
145  if (was_using_system_colors != must_use_system_colors) {
146  QETApp::instance()->useSystemPalette(must_use_system_colors);
147  }
148  settings.setValue("border-columns_0",ui->m_border_0->isChecked());
149  settings.setValue("lang", ui->m_lang_cb->itemData(ui->m_lang_cb->currentIndex()).toString());
150 
151  //ELEMENT EDITOR
152  settings.setValue("elementeditor/default-informations", ui->m_default_elements_info->toPlainText());
153 
154  //DIAGRAM VIEW
155  settings.setValue("diagramview/gestures", ui->m_use_gesture_trackpad->isChecked());
156 
157  //DIAGRAM COMMAND
158  settings.setValue("diagramcommands/erase-label-on-copy", ui->m_save_label_paste->isChecked());
159 
160  //GENERIC PANEL
161  settings.setValue("genericpanel/folio",ui->m_use_folio_label->isChecked());
162 
163  //NOMENCLATURE
164  settings.setValue("nomenclature/terminal-exportlist",ui->m_export_terminal->isChecked());
165 
166 
167  //DIAGRAM EDITOR
168  QString view_mode = ui->m_use_tab_mode_rb->isChecked() ? "tabbed" : "windowed";
169  settings.setValue("diagrameditor/viewmode", view_mode) ;
170  settings.setValue("diagrameditor/highlight-integrated-elements", ui->m_highlight_integrated_elements->isChecked());
171  settings.setValue("diagrameditor/zoom-out-beyond-of-folio", ui->m_zoom_out_beyond_folio->isChecked());
172  settings.setValue("diagrameditor/autosave-interval", ui->m_autosave_sb->value());
173  //Grid step and key navigation
174  settings.setValue("diagrameditor/Xgrid", ui->DiagramEditor_xGrid_sb->value());
175  settings.setValue("diagrameditor/Ygrid", ui->DiagramEditor_yGrid_sb->value());
176  settings.setValue("diagrameditor/key_Xgrid", ui->DiagramEditor_xKeyGrid_sb->value());
177  settings.setValue("diagrameditor/key_Ygrid", ui->DiagramEditor_yKeyGrid_sb->value());
178  settings.setValue("diagrameditor/key_fine_Xgrid", ui->DiagramEditor_xKeyGridFine_sb->value());
179  settings.setValue("diagrameditor/key_fine_Ygrid", ui->DiagramEditor_yKeyGridFine_sb->value());
180  //Dynamic text item
181  settings.setValue("diagrameditor/dynamic_text_rotation", ui->m_dyn_text_rotation_sb->value());
182  settings.setValue("diagrameditor/dynamic_text_width", ui->m_dyn_text_width_sb->value());
183  //Independent text item
184  settings.setValue("diagrameditor/independent_text_rotation", ui->m_indi_text_rotation_sb->value());
185 
186  //ELEMENTS COLLECTION
187  QString path = settings.value("elements-collections/common-collection-path").toString();
188  if (ui->m_common_elmt_path_cb->currentIndex() == 1)
189  {
190  QString path = ui->m_common_elmt_path_cb->currentText();
191  QDir dir(path);
192  settings.setValue("elements-collections/common-collection-path",
193  dir.exists() ? path : "default");
194  }
195  else {
196  settings.setValue("elements-collections/common-collection-path", "default");
197  }
198  if (path != settings.value("elements-collections/common-collection-path").toString()) {
199  QETApp::resetUserElementsDir();
200  }
201 
202  path = settings.value("elements-collections/custom-collection-path").toString();
203  if (ui->m_custom_elmt_path_cb->currentIndex() == 1)
204  {
205  QString path = ui->m_custom_elmt_path_cb->currentText();
206  QDir dir(path);
207  settings.setValue("elements-collections/custom-collection-path",
208  dir.exists() ? path : "default");
209  }
210  else {
211  settings.setValue("elements-collections/custom-collection-path", "default");
212  }
213  if (path != settings.value("elements-collections/custom-collection-path").toString()) {
214  QETApp::resetUserElementsDir();
215  }
216 
217  path = settings.value("elements-collections/custom-tbt-path").toString();
218  if (ui->m_custom_tbt_path_cb->currentIndex() == 1)
219  {
220  QString path = ui->m_custom_tbt_path_cb->currentText();
221  QDir dir(path);
222  settings.setValue("elements-collections/custom-tbt-path",
223  dir.exists() ? path : "default");
224  }
225  else {
226  settings.setValue("elements-collections/custom-tbt-path", "default");
227  }
228  if (path != settings.value("elements-collections/custom-tbt-path").toString()) {
229  QETApp::resetUserElementsDir();
230  }
