33 DiagramPropertiesDialog::DiagramPropertiesDialog(Diagram *diagram, QWidget *parent) :
34  QDialog (parent),

36 {
37  bool diagram_is_read_only = diagram -> isReadOnly();
38 
39  // Get some properties of edited diagram
40  TitleBlockProperties titleblock = diagram -> border_and_titleblock.exportTitleBlock();
41  BorderProperties border = diagram -> border_and_titleblock.exportBorder();

43 
44  setWindowModality(Qt::WindowModal);
45 #ifdef Q_OS_MACOS

47 #endif
48 
49  setWindowTitle(tr("Propriétés du folio", "window title"));
50 
51  //Border widget
52  BorderPropertiesWidget *border_infos = new BorderPropertiesWidget(border, this);
53  border_infos -> setReadOnly(diagram_is_read_only);
54 
55  //Title block widget
56  TitleBlockPropertiesWidget *titleblock_infos;
57 
58  if (QETProject *parent_project = diagram -> project())
59  titleblock_infos = new TitleBlockPropertiesWidget(parent_project -> embeddedTitleBlockTemplatesCollection(), titleblock, false, diagram->project(), this);
60  else
61  titleblock_infos = new TitleBlockPropertiesWidget(titleblock, false, diagram->project(), this);
62 
63  titleblock_infos -> setReadOnly(diagram_is_read_only);
64  connect(titleblock_infos,SIGNAL(openAutoNumFolioEditor(QString)),this,SLOT(editAutoFolioNum()));
65  //titleblock_infos->setMinimumSize(590,480); //Minimum Size needed for correct display
66 
67  //Conductor widget
68  m_cpw = new ConductorPropertiesWidget(conductors, this);
69  m_cpw -> setReadOnly(diagram_is_read_only);
70 
71  QComboBox *autonum_combobox = m_cpw->autonumComboBox();
72  autonum_combobox->addItems(diagram->project()->conductorAutoNum().keys());
73  autonum_combobox->setCurrentIndex(autonum_combobox->findText(diagram->conductorsAutonumName()));
74 
75  connect(m_cpw->editAutonumPushButton(), &QPushButton::clicked, this, &DiagramPropertiesDialog::editAutonum);
76 
77  // Buttons
78  QDialogButtonBox boutons(diagram_is_read_only ? QDialogButtonBox::Ok : QDialogButtonBox::Ok | QDialogButtonBox::Cancel);
79  connect(&boutons, SIGNAL(accepted()), this, SLOT(accept()));
80  connect(&boutons, SIGNAL(rejected()), this, SLOT(reject()));
81 
82  QGridLayout *glayout = new QGridLayout;
83  glayout->addWidget(border_infos,0,0);
84  glayout->addWidget(titleblock_infos, 1, 0);
85  glayout->addWidget(m_cpw, 0, 1, 0, 1);
86 
87  QVBoxLayout vlayout(this);
88  vlayout.addLayout(glayout);
89  vlayout.addWidget(&boutons);
90 
91  // if dialog is accepted
92  if (this -> exec() == QDialog::Accepted && !diagram_is_read_only)
93  {
94  TitleBlockProperties new_titleblock = titleblock_infos -> properties();
95  BorderProperties new_border = border_infos -> properties();
96  ConductorProperties new_conductors = m_cpw -> properties();
97 
98  // Title block have change
99  if (new_titleblock != titleblock) {
100  diagram -> undoStack().push(new ChangeTitleBlockCommand(diagram, titleblock, new_titleblock));
101  }
102 
103  // Border have change
104  if (new_border != border) {
105  diagram -> undoStack().push(new ChangeBorderCommand(diagram, border, new_border));
106  }
107 
108  // Conducteur have change
109  if (new_conductors != conductors) {
111  diagram -> defaultConductorProperties = new_conductors;
112  }
113 
114  // Conductor autonum name
115  if (autonum_combobox->currentText() != diagram->conductorsAutonumName())
116  {
117  diagram->setConductorsAutonumName (autonum_combobox->currentText());
118  diagram->project()->conductorAutoNumChanged();
119  }
120  }
121 }
