37 ConductorCreator::ConductorCreator(Diagram *d, QList<Terminal *> terminals_list) :
38  m_terminals_list(terminals_list)
39 {
40  if (m_terminals_list.size() <= 1) {
41  return;
42  }
43  m_properties = m_terminals_list.first()->diagram()->defaultConductorProperties;
44 
45  setUpPropertieToUse();
46  Terminal *hub_terminal = hubTerminal();
47 
48  d->undoStack().beginMacro(QObject::tr("Cr√©ation de conducteurs"));
49 
50  QList<Conductor *> c_list;
51  for (Terminal *t : m_terminals_list)
52  {
53  if (t == hub_terminal) {
54  continue;
55  }
56 
57  Conductor *cond = new Conductor(hub_terminal, t);
58  cond->setProperties(m_properties);
59  cond->setSequenceNum(m_sequential_number);
60  d->undoStack().push(new AddItemCommand<Conductor *>(cond, d));
61 
62  c_list.append(cond);
63  }
64  d->undoStack().endMacro();
65 
66  for(Conductor *c : c_list) {
67  c->refreshText();
68  }
69 }

99 void ConductorCreator::setUpPropertieToUse()
100 {
101  QList<Conductor *> potentials = existingPotential();
102 
103  //There is an existing potential
104  //we get one of them
105  if (!potentials.isEmpty())
106  {
107  if (potentials.size() >= 2)
108  {
109  QList <ConductorProperties> cp_list;
110  for(Conductor *c : potentials) {
111  cp_list.append(c->properties());
112  }
113 
114  m_properties = PotentialSelectorDialog::chosenProperties(cp_list);
115  for (Conductor *c : potentials) {
116  if (c->properties() == m_properties) {
117  m_sequential_number = c->sequenceNum();
118  }
119  }
120  }
121  else if (potentials.size() == 1)
122  {
123  m_properties = potentials.first()->properties();
124  m_sequential_number = potentials.first()->sequenceNum();
125  }
126  return;
127  }
128 
129  //get a new properties
130  ConductorAutoNumerotation::newProperties(m_terminals_list.first()->diagram(), m_properties, m_sequential_number);
131 }

139 QList<Conductor *> ConductorCreator::existingPotential()
140 {
141  QList<Conductor *> c_list;
142  QList<Terminal *> t_exclude;
143 
144  for (Terminal *t : m_terminals_list)
145  {
146  if (t_exclude.contains(t)) {
147  continue;
148  }
149 
150  if (!t->conductors().isEmpty())
151  {
152  c_list.append(t->conductors().first());
153 
154  //We must to check m_terminals_list contain a terminal
155  //in the same potential of c, and if true, exclude this terminal from the search.
156  for (Conductor *c : t->conductors().first()->relatedPotentialConductors(false))
157  {
158  if (m_terminals_list.contains(c->terminal1)) {
159  t_exclude.append(c->terminal1);
160  } else if (m_terminals_list.contains(c->terminal2)) {
161  t_exclude.append(c->terminal2);
162  }
163  }
164  }
165  else if (t->parentElement()->linkType() & Element::AllReport && !t->parentElement()->isFree())
166  {
167  Element *linked_report = t->parentElement()->linkedElements().first();
168  if (!linked_report->conductors().isEmpty()) {
169  c_list.append(linked_report->conductors().first());
170  }
171  }
172  }
173 
174  return c_list;
175 }

181 Terminal *ConductorCreator::hubTerminal()

184 
185  for (Terminal *tt : m_terminals_list)
186  {
187  if (tt->scenePos().x() < hub_terminal->scenePos().x()) {
188  hub_terminal = tt;
189  } else if (tt->scenePos().x() == hub_terminal->scenePos().x()) {
190  if (tt->scenePos().y() < hub_terminal->scenePos().y()) {
191  hub_terminal = tt;
192  }
193  }
194  }
195 
196  return hub_terminal;
