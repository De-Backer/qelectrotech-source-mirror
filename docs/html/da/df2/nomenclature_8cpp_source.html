50 bool nomenclature::saveToCSVFile()
51 {
52  // SAVE IN FILE
53  QString name = QObject::tr("nomenclature_") + QString(m_project -> title() + ".csv");
54  if (!name.endsWith(".csv")) {
55  name += ".csv";
56  }
57  QString filename = QFileDialog::getSaveFileName(this->m_parent, QObject::tr("Enregister sous... "), name, QObject::tr("Fichiers csv (*.csv)"));
58  QFile file(filename);
59  if( !filename.isEmpty() ) {
60  if(QFile::exists ( filename )){
61  // if file already exist -> delete it
62  if(!QFile::remove ( filename ) ){
63  QMessageBox::critical(this->m_parent, QObject::tr("Erreur"),
64  QObject::tr("Impossible de remplacer le fichier!\n\n")+
65  "Destination : "+filename+"\n");
66  return false;
67  }
68  }
69  if (file.open(QIODevice::WriteOnly | QIODevice::Text)){
70  QTextStream stream(&file);
71 #if QT_VERSION < QT_VERSION_CHECK(5, 15, 0) // ### Qt 6: remove
72  stream << getNomenclature() << endl;
73 #else
74 #pragma message("@TODO remove code for QT 5.15 or later")
75  stream << getNomenclature() << &Qt::endl(stream);
76 #endif
77  }
78  else return false;
79  }
80  else return false;
81 
82  return true;
83 }
