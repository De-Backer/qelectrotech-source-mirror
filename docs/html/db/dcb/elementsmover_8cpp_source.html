65 int ElementsMover::beginMovement(Diagram *diagram, QGraphicsItem *driver_item)
66 {
67  // They must be no movement in progress
68  if (movement_running_) return(-1);
69 
70  // Be sure we have diagram to work
71  if (!diagram) return(-1);
72  diagram_ = diagram;
73 
74  // Take count of driver item
75  m_movement_driver = driver_item;
76 
77  // At the beginning of movement, move is NULL
78  current_movement_ = QPointF(0.0, 0.0);
79 
80  m_moved_content = DiagramContent(diagram);
81  m_moved_content.removeNonMovableItems();
82 
83  //Remove element text, if the parent element is selected.
84  QList<DynamicElementTextItem *> deti_list = m_moved_content.m_element_texts.values();
85  for(DynamicElementTextItem *deti : deti_list) {
86  if(m_moved_content.m_elements.contains(deti->parentElement())) {
87  m_moved_content.m_element_texts.remove(deti);
88  }
89  }
90 
91  QList<ElementTextItemGroup *> etig_list = m_moved_content.m_texts_groups.values();
92  for(ElementTextItemGroup *etig : etig_list) {
93  if (m_moved_content.m_elements.contains(etig->parentElement())) {
94  m_moved_content.m_texts_groups.remove(etig);
95  }
96  }
97 
98  if (!m_moved_content.count()) return(-1);
99 
100  /* At this point, we've got all info to manage movement.
101  * There is now a move in progress */
102  movement_running_ = true;
103 
104  return(m_moved_content.count());
105 }

112 void ElementsMover::continueMovement(const QPointF &movement)
113 {
114  if (!movement_running_ || movement.isNull()) return;
115 
116  current_movement_ += movement;
117 
118  //Move every movable item, except conductor
119  typedef DiagramContent dc;
120  for (QGraphicsItem *qgi : m_moved_content.items(dc::Elements | dc::TextFields | dc::Images | dc::Shapes | dc::ElementTextFields | dc::TextGroup | dc::ConductorsToMove))
121  {
122  if (qgi == m_movement_driver)
123  continue;
124  qgi -> setPos(qgi->pos() + movement);
125  }
126 
127  // Move some conductors

133  //For the user this situation appear when the top right of the folio is not at the top right of the graphicsview,

150 void ElementsMover::endMovement()
151 {
152  // A movement must be inited
153  if (!movement_running_) return;
154 
155  //empty command to be used has parent of commands below
156  QUndoCommand *undo_object = new QUndoCommand();
157 
158  //Create undo move if there is a movement
159  if (!current_movement_.isNull()) {
160  QUndoCommand *quc = new MoveElementsCommand(diagram_, m_moved_content, current_movement_, undo_object);
161  undo_object->setText(quc->text());
162  }
163 
164  //There is only one element moved, and project authorize auto conductor,
165  //we try auto connection of conductor;
166  typedef DiagramContent dc;
167  if (m_moved_content.items(dc::TextFields | dc::Images | dc::Shapes).size() == 0 &&
168  m_moved_content.items(dc::Elements).size() == 1 &&
169  diagram_ -> project() -> autoConductor())
170  {
171  Element *elmt = m_moved_content.m_elements.first();
172 
173  int acc = elmt->AlignedFreeTerminals().size();
174 
175  while (!elmt -> AlignedFreeTerminals().isEmpty())
176  {
177  QPair <Terminal *, Terminal *> pair = elmt -> AlignedFreeTerminals().takeFirst();
178 
179  Conductor *conductor = new Conductor(pair.first, pair.second);
180 
181  //Create an undo object for each new auto conductor, with undo_object for parent
182  new AddItemCommand<Conductor *>(conductor, diagram_, QPointF(), undo_object);
183  if (undo_object->text().isEmpty())
184  undo_object->setText(QObject::tr("Ajouter %n conducteur(s)", "add a numbers of conductor one or more", acc));
185 
186  //Get all conductors at the same potential of conductor
187  QSet <Conductor *> conductors_list = conductor->relatedPotentialConductors();
188 
189  //Compare the properties of every conductors stored in conductors_list,
190  //if every conductors properties is equal, we use this properties for conductor.
191  ConductorProperties others_properties;
192  bool use_properties = false;
193  if (!conductors_list.isEmpty())
194  {
195  use_properties = true;
196  others_properties = (*conductors_list.begin())->properties();
197  foreach (Conductor *cond, conductors_list)
198  if (cond->properties() != others_properties)
199  use_properties = false;
200  }
201 
202  if (use_properties)
203  conductor->setProperties(others_properties);
204  else
205  {
206  conductor -> setProperties(diagram_ -> defaultConductorProperties);
207  //Autonum the new conductor, the undo command associated for this, have for parent undo_object
208  ConductorAutoNumerotation can (conductor, diagram_, undo_object);
209  can.numerate();
210  }
211  }
212  }
213 
214  //Add undo_object if have child
215  if (undo_object->childCount() >= 1)
216  diagram_ -> undoStack().push(undo_object);
217  else
218  delete undo_object;
219 
220  // There is no movement in progress now
221  movement_running_ = false;
222  m_moved_content.clear();
223 }
