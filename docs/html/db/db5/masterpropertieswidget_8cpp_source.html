126 void MasterPropertiesWidget::setElement(Element *element)
127 {
128  if (m_element == element)
129  return;
130 
131  if (m_showed_element)
132  {
133  m_showed_element->setHighlighted(false);
134  m_showed_element = nullptr;
135  }
136  if (m_element)
137  m_element->setHighlighted(false);
138 
139  if (m_project)
140  disconnect(m_project, SIGNAL(diagramRemoved(QETProject*,Diagram*)),
141  this, SLOT(diagramWasdeletedFromProject()));
142 
143  if(Q_LIKELY(element->diagram() && element->diagram()->project()))
144  {
145  m_project = element->diagram()->project();
146  connect(m_project, SIGNAL(diagramRemoved(QETProject*,Diagram*)),
147  this, SLOT(diagramWasdeletedFromProject()));
148  }
149  else
150  m_project = nullptr;
151 
152  //Keep up to date this widget when the linked elements of m_element change
153  if (m_element)
154  disconnect(m_element.data(), &Element::linkedElementChanged,
155  this, &MasterPropertiesWidget::updateUi);
156 
157  m_element = element;
158  connect(m_element.data(), &Element::linkedElementChanged,
159  this, &MasterPropertiesWidget::updateUi);
160 
161  updateUi();
162 }

197 QUndoCommand* MasterPropertiesWidget::associatedUndo() const
198 {
199  QList <Element *> to_link;
200  QList <Element *> linked_ = m_element->linkedElements();
201 
202  for (int i=0; i<ui->m_link_tree_widget->topLevelItemCount(); i++)
203  to_link << m_qtwi_hash[ui->m_link_tree_widget->topLevelItem(i)];
204 
205  //The two list contain the same element, there is no change
206  if (to_link.size() == linked_.size())
207  {
208  bool equal = true;
209 
210  foreach(Element *elmt, to_link)
211  if (!linked_.contains(elmt))
212  equal = false;
213 
214  if(equal)
215  return nullptr;
216  }
217 
218  LinkElementCommand *undo = new LinkElementCommand(m_element);
219 
220  if (to_link.isEmpty())
221  undo->unlinkAll();
222  else
223  undo->setLink(to_link);
224 
225  return undo;
226 }
