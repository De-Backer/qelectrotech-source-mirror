53 QString SummaryQueryWidget::queryStr() const
54 {
55  //User define is own query
56  if (ui->m_edit_sql_query_cb->isChecked()) {
57  return ui->m_user_query_le->text();
58  }
59 
60  //Made a string list with the colomns (keys) choosen by the user
61  QStringList keys = selectedKeys();
62 
63  QString select ="SELECT ";

65  QString column;
66  bool first = true;
67  for (auto key: keys) {

69  first = false;
70  } else {
71  column += ", ";
72  }
73  column += key;
74  }
75 
76  QString from = " FROM project_summary_view";
77 
78  QString q(select + column + from);

88 void SummaryQueryWidget::setQuery(const QString &query)
89 {
90  if (query.startsWith("SELECT"))
91  {
92  reset();
93  ui->m_user_query_le->setText(query);
94 
95  QString select = query;
96  select.remove(0,7); //remove SELECT
97  select.truncate(select.indexOf("FROM")); //Truncate at FROM
98  select.replace(" ",""); //Remove white
99 
100  //Get the select -> the item in the right list
101  QStringList split = select.split(",");
102  for (auto str : split)
103  {
104  for (auto item : m_items_list)
105  {
106  if (item->data(Qt::UserRole).toString() == str) {
107  ui->m_available_list->takeItem(ui->m_available_list->row(item));
108  ui->m_choosen_list->addItem(item);
109  continue;
110  }
111  }
112  }
113  }
114 }

291 void SummaryQueryWidget::saveConfig()
292 {
293  QFile file_(QETApp::configDir() + "/summary.json");
294 
295  if (file_.open(QFile::ReadWrite))
296  {
297  QJsonDocument doc_(QJsonDocument::fromJson(file_.readAll()));
298  QJsonObject root_object;
299 
300  if (!doc_.isEmpty())
301  {
302  root_object = doc_.object();
303  if (root_object.contains(ui->m_config_gb->text())) {
304  root_object.remove(ui->m_config_gb->text());
305  }
306  }
307 
308  QJsonObject object_;
309  object_.insert("query", queryStr());
310  root_object[ui->m_config_gb->text()] = object_;
311 
312  doc_.setObject(root_object);
313  file_.resize(0);
314  file_.write(doc_.toJson());
315  }
316 }
