41 void SearchAndReplaceWorker::replaceDiagram(QList<Diagram *> diagram_list)
42 {
43  if (diagram_list.isEmpty()) {
44  return;
45  }
46 
47  QETProject *project = diagram_list.first()->project();
48  for (Diagram *d : diagram_list) {
49  if (d->project() != project) {
50  return;
51  }
52  }
53 
54  QUndoStack *us = project->undoStack();
55  us->beginMacro(QObject::tr("Chercher/remplacer les propriétés de folio"));
56  for (Diagram *d : diagram_list)
57  {
58  TitleBlockProperties old_propertie = d->border_and_titleblock.exportTitleBlock();
59  TitleBlockProperties new_properties = old_propertie;
60 
61  new_properties.title = applyChange(new_properties.title, m_titleblock_properties.title);
62  new_properties.author = applyChange(new_properties.author, m_titleblock_properties.author);
63  new_properties.filename = applyChange(new_properties.filename, m_titleblock_properties.filename);
64  new_properties.plant = applyChange(new_properties.plant, m_titleblock_properties.plant);
65  new_properties.locmach = applyChange(new_properties.locmach, m_titleblock_properties.locmach);
66  new_properties.indexrev = applyChange(new_properties.indexrev, m_titleblock_properties.indexrev);
67  new_properties.folio = applyChange(new_properties.folio, m_titleblock_properties.folio);
68 
69  if (m_titleblock_properties.date.isValid())
70  {
71  if (m_titleblock_properties.date == eraseDate()) {
72  new_properties.date = QDate();
73  } else {
74  new_properties.date = m_titleblock_properties.date;
75  }
76  }
77 
78  new_properties.context.add(m_titleblock_properties.context);
79 
80  if (old_propertie != new_properties) {
81  project->undoStack()->push(new ChangeTitleBlockCommand(d, old_propertie, new_properties));
82  }
83  }
84  us->endMacro();
85 }

103 void SearchAndReplaceWorker::replaceElement(QList<Element *> list)
104 {
105  if (list.isEmpty() || !list.first()->diagram()) {
106  return;
107  }
108 
109  QETProject *project_ = list.first()->diagram()->project();
110  for (Element *elmt : list)
111  {
112  if (elmt->diagram()) {
113  if (elmt->diagram()->project() != project_) {
114  return;
115  }
116  }
117  }
118 
119  project_->undoStack()->beginMacro(QObject::tr("Chercher/remplacer les propriétés d'éléments."));
120  for (Element *elmt : list)
121  {
122  //We apply change only for master, slave, and terminal element.
123  if (elmt->linkType() == Element::Master ||
124  elmt->linkType() == Element::Simple ||
125  elmt->linkType() == Element::Terminale)
126  {
127  DiagramContext old_context;
128  DiagramContext new_context = old_context = elmt->elementInformations();
129  for (QString key : QETApp::elementInfoKeys())
130  {
131  new_context.addValue(key, applyChange(old_context.value(key).toString(),
132  m_element_context.value(key).toString()));
133  }
134 
135  if (old_context != new_context)
136  {
137  ChangeElementInformationCommand *undo = new ChangeElementInformationCommand(elmt, old_context, new_context);
138  project_->undoStack()->push(undo);
139  }
140  }
141  }
142  project_->undoStack()->endMacro();
143 }

197 void SearchAndReplaceWorker::replaceConductor(QList<Conductor *> list)
198 {

200  return;
201  }
202 
203  QETProject *project_ = list.first()->diagram()->project();
204  for (Conductor *c : list) {
205  if (!c->diagram() ||
206  c->diagram()->project() != project_) {
207  return;
208  }
209  }
210 
211  project_->undoStack()->beginMacro(QObject::tr("Chercher/remplacer les propriétés de conducteurs."));
212  for (Conductor *c : list)

214  ConductorProperties cp = applyChange(c->properties(), m_conductor_properties);
215 
216  if (cp != c->properties())
217  {
218  QSet <Conductor *> conductors_list = c->relatedPotentialConductors(true);
219  conductors_list << c;
220  for (Conductor *cc : conductors_list)
221  {
222  QVariant old_value, new_value;
223  old_value.setValue(cc->properties());
224  new_value.setValue(cp);
225  project_->undoStack()->push(new QPropertyUndoCommand(cc, "properties", old_value, new_value));
226  }
227  }
228  }
229  project_->undoStack()->endMacro();

249 void SearchAndReplaceWorker::replaceAdvanced(
250  QList<Diagram *> diagrams,
251  QList<Element *> elements,
252  QList<IndependentTextItem *> texts,
253  QList<Conductor *> conductors)
254 {
255  QETProject *project_ = nullptr;
256 
257  //Some test to check if a least one list have one item
258  //and if all items belong to the same project
259  if (!diagrams.isEmpty()) {
260  project_ = diagrams.first()->project();
261  } else if (!elements.isEmpty() && elements.first()->diagram()) {
262  project_ = elements.first()->diagram()->project();
263  } else if (!texts.isEmpty() && texts.first()->diagram()) {
264  project_ = texts.first()->diagram()->project();
265  } else if (!conductors.isEmpty() && conductors.first()->diagram()) {
266  project_ = conductors.first()->diagram()->project();
267  } else {
268  return;
269  }
270 
271  for (Diagram *dd : diagrams) {
272  if (dd->project() != project_) {
273  return;
274  }
275  }
276  for (Element *elmt : elements) {
277  if (!elmt->diagram() || elmt->diagram()->project() != project_) {
278  return;
279  }
280  }
281  for (IndependentTextItem *text : texts) {
282  if (!text->diagram() || text->diagram()->project() != project_) {
283  return;
284  }
285  }
286  for (Conductor *cc : conductors) {
287  if (!cc->diagram() || cc->diagram()->project() != project_) {
288  return;
289  }
290  }
291  //The end of the test
292 
293  int who = m_advanced_struct.who;
294  if (who == -1) {
295  return;
296  }
297 
298  project_->undoStack()->beginMacro(QObject::tr("Rechercher / remplacer avancé"));
299  if (who == 0)
300  {
301  for (Diagram *diagram : diagrams)
302  {
303  TitleBlockProperties old_properties = diagram->border_and_titleblock.exportTitleBlock();
304  TitleBlockProperties new_properties = replaceAdvanced(diagram);
305  if (old_properties != new_properties) {
306  project_->undoStack()->push(new ChangeTitleBlockCommand(diagram, old_properties, new_properties));
307  }
308  }
309  }
310  else if (who == 1)
311  {
312  for (Element *element : elements)
313  {
314  DiagramContext old_context = element->elementInformations();
315  DiagramContext new_context = replaceAdvanced(element);
316  if (old_context != new_context) {
317  project_->undoStack()->push(new ChangeElementInformationCommand(element, old_context, new_context));
318  }
319  }
320  }
321  else if (who == 2)
322  {
323  for (Conductor *conductor : conductors)
324  {
325  ConductorProperties old_properties = conductor->properties();
326  ConductorProperties new_properties = replaceAdvanced(conductor);
327  if (old_properties != new_properties)
328  {
329  QSet <Conductor *> potential_conductors = conductor->relatedPotentialConductors(true);
330  potential_conductors << conductor;
331 
332  for (Conductor *c : potential_conductors)
333  {
334  QVariant old_value, new_value;
335  old_value.setValue(c->properties());
336  new_value.setValue(new_properties);
337  project_->undoStack()->push(new QPropertyUndoCommand(c, "properties", old_value, new_value));
338  }
339  }
340  }
341  }
342  else if (who == 3)
343  {
344  for (IndependentTextItem *text : texts)
345  {
346  QRegularExpression rx(m_advanced_struct.search);
347  QString replace = m_advanced_struct.replace;
348  QString after = text->toPlainText();
349  after = after.replace(rx, replace);
350 
351  if (after != text->toPlainText()) {
352  project_->undoStack()->push(new ChangeDiagramTextCommand(text, text->toPlainText(), after));
353  }
354  }
355  }
356  project_->undoStack()->endMacro();
357 }

406 ConductorProperties SearchAndReplaceWorker::applyChange(
407  const ConductorProperties &original,
408  const ConductorProperties &change)
409 {
410  ConductorProperties new_properties = original;
411 
412  if (change.text_size > 2) {new_properties.text_size = change.text_size;}
413  new_properties.m_formula = applyChange(new_properties.m_formula, change.m_formula);
414  new_properties.text = applyChange(new_properties.text, change.text);
415  new_properties.m_show_text = change.m_show_text;
416  new_properties.m_function = applyChange(new_properties.m_function, change.m_function);
417  new_properties.m_tension_protocol = applyChange(new_properties.m_tension_protocol, change.m_tension_protocol);
418  new_properties.m_wire_color = applyChange(new_properties.m_wire_color, change.m_wire_color);
419  new_properties.m_wire_section = applyChange(new_properties.m_wire_section, change.m_wire_section);
420  if(change.m_vertical_alignment == Qt::AlignLeft ||
421  change.m_vertical_alignment == Qt::AlignRight) {new_properties.m_vertical_alignment = change.m_vertical_alignment;}
422  if(change.m_horizontal_alignment == Qt::AlignTop ||
423  change.m_horizontal_alignment == Qt::AlignBottom) {new_properties.m_horizontal_alignment = change.m_horizontal_alignment;}
424  if (change.verti_rotate_text >= 0) {new_properties.verti_rotate_text = change.verti_rotate_text;}
425  if (change.horiz_rotate_text >= 0) {new_properties.horiz_rotate_text = change.horiz_rotate_text;}
426  if (change.color.isValid()) {new_properties.color = change.color;}
427  if (change.style != Qt::NoPen) {new_properties.style = change.style;}
428  if (change.cond_size >= 0.4) {new_properties.cond_size = change.cond_size;}
429  new_properties.m_bicolor = change.m_bicolor;
430  if (change.m_color_2.isValid()) {new_properties.m_color_2 = change.m_color_2;}
431  if (change.m_dash_size >= 2) {new_properties.m_dash_size = change.m_dash_size;}
432  new_properties.singleLineProperties = change.singleLineProperties;
433 
434  return new_properties;
435 }
