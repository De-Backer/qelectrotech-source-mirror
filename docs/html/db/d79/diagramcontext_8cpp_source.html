49 QList<QString> DiagramContext::keys(DiagramContext::KeyOrder order) const
50 {
51  if (order == None) {
52  return m_content.keys();
53  }
54  else
55  {
56  QList<QString> keys_list = m_content.keys();
57  if (order == Alphabetical) {
58  std::sort(keys_list.begin(), keys_list.end());
