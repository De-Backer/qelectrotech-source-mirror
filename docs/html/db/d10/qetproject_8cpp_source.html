172 QETProject::ProjectState QETProject::openFile(QFile *file)
173 {
174  bool opened_here = file->isOpen() ? false : true;
175  if (!file->isOpen()
176  && !file->open(QIODevice::ReadOnly
177  | QIODevice::Text)) {
178  return FileOpenFailed;
179  }
180  QFileInfo fi(*file);
181  setFilePath(fi.absoluteFilePath());
182 
183  //Extract the content of the xml
184  QDomDocument xml_project;
185  if (!xml_project.setContent(file))
186  {
187  if(opened_here) {
188  file->close();
189  }
190  return XmlParsingFailed;
191  }
192 
193  //Build the project from the xml
194  readProjectXml(xml_project);
195 
196  if (!fi.isWritable()) {
197  setReadOnly(true);
198  }
199  if(opened_here) {
200  file->close();
201  }
202  return ProjectState::Ok;
203 }

972 ElementsLocation QETProject::importElement(ElementsLocation &location)
973 {
974  //Location isn't an element or doesn't exist
975  if (! (location.isElement() && location.exist()) ) {
976  return ElementsLocation();
977  }
978 
979  //Get the path where the element must be imported
980  QString import_path;
981  if (location.isFileSystem()) {
982  import_path = "import/" + location.collectionPath(false);
983  }
984  else if (location.isProject()) {
985  if (location.project() == this) {
986  return location;
987  }
988 
989  import_path = location.collectionPath(false);
990  }
991 
992  //Element already exist in the embedded collection, we ask what to do to user
993  if (m_elements_collection->exist(import_path)) {
994  ElementsLocation existing_location(import_path, this);
995 
996  //existing_location and location have the same uuid, so it is the same element
997  if (existing_location.uuid() == location.uuid()) {
998  return existing_location;
999  }
1000 
1001  ImportElementDialog ied;
1002  if (ied.exec() == QDialog::Accepted) {
1003  QET::Action action = ied.action();
1004 
1005  //Use the exisitng element
1006  if (action == QET::Ignore) {
1007  return existing_location;
1008  }
1009  //Erase the existing element, and use the newer instead
1010  else if (action == QET::Erase) {
1011  ElementsLocation parent_loc = existing_location.parent();
1012  return m_elements_collection->copy(location, parent_loc);
1013  }
1014  //Add the new element with an other name.
1015  else if (action == QET::Rename) {
