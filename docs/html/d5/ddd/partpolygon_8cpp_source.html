292 QVariant PartPolygon::itemChange(QGraphicsItem::GraphicsItemChange change, const QVariant &value)
293 {
294  if (change == ItemSelectedHasChanged && scene())
295  {
296  if (value.toBool() == true)
297  {
298  //When item is selected, he must to be up to date whene the selection in the scene change, for display or not the handler,
299  //according to the number of selected items.
300  connect(scene(), &QGraphicsScene::selectionChanged, this, &PartPolygon::sceneSelectionChanged);
301 
302  if (scene()->selectedItems().size() == 1)
303  addHandler();
304  }
305  else
306  {
307  disconnect(scene(), &QGraphicsScene::selectionChanged, this, &PartPolygon::sceneSelectionChanged);
308  removeHandler();
309  }
310  }
311  else if (change == ItemPositionHasChanged)
312  {
313  adjusteHandlerPos();
314  }
315  else if (change == ItemSceneChange)
316  {
317  if(scene())
318  disconnect(scene(), &QGraphicsScene::selectionChanged, this, &PartPolygon::sceneSelectionChanged);
319 
320  setSelected(false); //This is item removed from scene, then we deselect this, and so, the handlers is also removed.
321  }
322 
323  return QGraphicsItem::itemChange(change, value);
324 }

332 bool PartPolygon::sceneEventFilter(QGraphicsItem *watched, QEvent *event)
333 {
334  //Watched must be an handler
335  if(watched->type() == QetGraphicsHandlerItem::Type)
336  {
337  QetGraphicsHandlerItem *qghi = qgraphicsitem_cast<QetGraphicsHandlerItem *>(watched);
338 
339  if(m_handler_vector.contains(qghi)) //Handler must be in m_vector_index, then we can start resize
340  {
341  m_vector_index = m_handler_vector.indexOf(qghi);
342  if (m_vector_index != -1)
343  {
344  if(event->type() == QEvent::GraphicsSceneMousePress) //Click
345  {
346  handlerMousePressEvent(qghi, static_cast<QGraphicsSceneMouseEvent *>(event));
347  return true;
348  }
349  else if(event->type() == QEvent::GraphicsSceneMouseMove) //Move
350  {
351  handlerMouseMoveEvent(qghi, static_cast<QGraphicsSceneMouseEvent *>(event));
352  return true;
353  }
354  else if (event->type() == QEvent::GraphicsSceneMouseRelease) //Release
355  {
356  handlerMouseReleaseEvent(qghi, static_cast<QGraphicsSceneMouseEvent *>(event));
357  return true;
358  }
359  }
360  }
361  }
362 
363  return false;
364 }

528 void PartPolygon::removePoint()
529 {
530  if (m_handler_vector.size() == 2)
531  return;
532 
533  QPointF point = mapToScene(m_context_menu_pos);
534  int index = -1;
535  for (int i=0 ; i<m_handler_vector.size() ; i++)
536  {
537  QetGraphicsHandlerItem *qghi = m_handler_vector.at(i);
538  if (qghi->contains(qghi->mapFromScene(point)))
539  {
540  index = i;
541  break;
542  }
543  }
544  if (index > -1 && index<m_handler_vector.count())
545  {
546  QPolygonF polygon = this->polygon();
547  qDebug() << index;
548  polygon.removeAt(index);
549 
550  //Wrap the undo for avoid to merge the undo commands when user add several points.
551  QUndoCommand *undo = new QUndoCommand(tr("Supprimer un point d'un polygone"));
552  new QPropertyUndoCommand(this, "polygon", this->polygon(), polygon, undo);
553  elementScene()->undoStack().push(undo);
554  }
