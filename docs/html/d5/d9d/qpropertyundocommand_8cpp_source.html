129 void QPropertyUndoCommand::redo()
130 {
131  if (m_object->property(m_property_name) != m_new_value)
132  {
133  if (m_animate && m_first_time)
134  {
135  QPropertyAnimation *animation = new QPropertyAnimation(m_object, m_property_name);
136  animation->setStartValue(m_old_value);
137  animation->setEndValue(m_new_value);
138  animation->start(QAbstractAnimation::DeleteWhenStopped);
139  }
140  else
141  {
142  m_object->setProperty(m_property_name, m_new_value);
143  m_first_time = true;
144  }
145  }
146 
147  QUndoCommand::redo();
148 }
