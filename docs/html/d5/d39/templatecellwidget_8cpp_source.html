203 void TitleBlockTemplateCellWidget::edit(TitleBlockCell *cell) {
204  if (!cell) return;
205  edited_cell_ = cell;
206  int type = cell -> type();
207  updateFormType(type);
208 
209  name_input_ -> setText(cell -> value_name);
210  label_checkbox_ -> setChecked(cell -> display_label);
211  label_input_ -> setText(cell -> label.name());
212  value_input_ -> setText(cell -> value.name());
213  font_adjust_input_ -> setChecked(cell -> hadjust);
214  horiz_align_input_ -> setCurrentIndex(horiz_align_indexes_[cell -> horizontalAlign()]);
215  vert_align_input_ -> setCurrentIndex(vert_align_indexes_[cell -> verticalAlign()]);
216 
217  font_size_input_ -> blockSignals(true); // QSpinBox has no signal triggered for each non-programmatic change
218  font_size_input_ -> setValue(TitleBlockTemplate::fontForCell(*cell).pointSize());
219  font_size_input_ -> blockSignals(false);
220 
221  logo_input_ -> setCurrentIndex(logo_input_ -> findData(cell -> logo_reference));
222 }

304 void TitleBlockTemplateCellWidget::updateLogosComboBox(const TitleBlockTemplate *parent_template) {
305  // saves the current value before erasing all entries
306  QVariant current_value = logo_input_ -> itemData(logo_input_ -> currentIndex());
307  logo_input_ -> clear();
308 
309  // default choice (the parent template may have no logo yet)
310  logo_input_ -> addItem(
311  tr("Aucun logo", "text displayed in the combo box when a template has no logo"),
312  QVariant(QString(""))
313  );
314  logo_input_ -> setCurrentIndex(0);
315 
316  if (!parent_template) return;
317  foreach (QString logo, parent_template -> logos()) {
318  logo_input_ -> addItem(logo, QVariant(logo));
319  }
320  int current_value_index = logo_input_ -> findData(current_value);
321  if (current_value_index != -1) {
322  logo_input_ -> setCurrentIndex(current_value_index);
323  }
324 }

407 void TitleBlockTemplateCellWidget::emitModification(const QString &attribute, const QVariant &new_value) const {
408  if (!edited_cell_) return;
409 
410  // avoid creating a QUndoCommand object when no modification was actually done
411  if (edited_cell_ -> attribute(attribute) == new_value) return;
412 
413  ModifyTitleBlockCellCommand *command = new ModifyTitleBlockCellCommand(edited_cell_);
414  command -> addModification(attribute, new_value);
415  command -> setText(
416  tr("Édition d'une cellule : %1", "label of and undo command when editing a cell")
417  .arg(TitleBlockCell::attributeName(attribute))
418  );
419  emit(cellModified(command));
420 }
