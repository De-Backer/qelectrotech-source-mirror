37 DeleteQGraphicsItemCommand::DeleteQGraphicsItemCommand(
38  Diagram *diagram,
39  const DiagramContent &content,
40  QUndoCommand *parent) :
41  QUndoCommand(parent),
42  m_removed_contents(content),
43  m_diagram(diagram)
44 {
45  //If parent element of a dynamic element text item is also in m_removed_content,
46  //we remove it, because when the element will be removed from the scene every child's will also be removed.
47  const QSet<DynamicElementTextItem *> elmt_set = m_removed_contents.m_element_texts;
48  for(DynamicElementTextItem *deti : elmt_set)
49  {
50  if (m_removed_contents.m_elements.contains(deti->parentElement()))
51  m_removed_contents.m_element_texts.remove(deti);
52 
53  }
54 
55  //When remove a deti we must to know his parent item, for re-add deti as child of the parent
56  //when undo this command
57  for(DynamicElementTextItem *deti : m_removed_contents.m_element_texts)
58  {
59  if(deti->parentGroup())
60  m_grp_texts_hash.insert(deti, deti->parentGroup());
61  else
62  m_elmt_text_hash.insert(deti, deti->parentElement());
63  }
64 
65  //If parent element of ElementTextItemGroup is also in m_removed_content,
66  //we remove it, because when the element will be removed from the scene every child's will also be removed.
67  const QSet<ElementTextItemGroup *> group_set = m_removed_contents.m_texts_groups;
68  for(ElementTextItemGroup *group : group_set)
69  {
70  if(m_removed_contents.m_elements.contains(group->parentElement()))
71  m_removed_contents.m_texts_groups.remove(group);
72  }
73 
74  //The deletion of the groups is not managed by this undo, but by a RemoveTextsGroupCommand
75  for(ElementTextItemGroup *group : m_removed_contents.m_texts_groups) {
76  new RemoveTextsGroupCommand(group->parentElement(), group, this);
77  }
78 
79  m_removed_contents.m_texts_groups.clear();
80  setPotentialsOfRemovedElements();
81 
82  //Get all linkeds table of removed table.
83  for (auto table : m_removed_contents.m_tables)
84  {
85  //Table is already managed, jump to next loop
86  if (m_table_scene_hash.keys().contains(table))
87  continue;
88 
89  auto first_table = table; //The first table if the table is linked to another
90  while(first_table->previousTable())
91  first_table = first_table->previousTable();
92  auto current_table = first_table;
93 
94  m_table_scene_hash.insert(first_table, first_table->scene());
95  while (current_table->nextTable())
96  {
97  current_table = current_table->nextTable();
98  m_table_scene_hash.insert(current_table, current_table->scene());
99  }
100  }
101 
102  setText(QString(QObject::tr(
103  "supprimer %1",
104  "undo caption - %1 is a sentence listing the removed content"))
105  .arg(m_removed_contents.sentence(DiagramContent::All)));
106  //Table is now managed by m_table_scene_hash,
107  //we clear the tables of m_removed_content
108  m_removed_contents.m_tables.clear();
109  m_diagram->qgiManager().manage(m_removed_contents.items(DiagramContent::All));
110 }

121 void DeleteQGraphicsItemCommand::setPotentialsOfRemovedElements()
122 {
123  for (Element *elmt : m_removed_contents.m_elements)
124  {
125  //a list of terminals who have at least two conductors docked in.
126  QList<Terminal *> terminals_list;
127  for (Terminal *t : elmt->terminals()) {
128  if (t->conductors().size() >= 2) {
129  terminals_list.append(t);
130  }
131  }
132  if (terminals_list.isEmpty()) {
133  continue;
134  }
135 
136  for (Terminal *t : terminals_list)
137  {
138  //All new created conductors will be docked to hub_terminal
139  Terminal *hub_terminal = nullptr;
140  QList<Terminal *> terminals_to_connect_list;
141 
142  for (Conductor *c : t->conductors())
143  {
144  Terminal *other_terminal = c->terminal1 == t ? c->terminal2 : c->terminal1;
145 
146  if (m_removed_contents.items(DiagramContent::Elements).contains(other_terminal->parentElement()))
147  {
148  other_terminal = terminalInSamePotential(other_terminal, c);
149  if (other_terminal == nullptr) {
150  continue;
151  }
152  }
153 
154  terminals_to_connect_list.append(other_terminal);
155  if (hub_terminal == nullptr) {
156  hub_terminal = other_terminal;
157  }
158  //hub_terminal must be the terminal the more at top left of the diagram.
159  else if (other_terminal->scenePos().x() < hub_terminal->scenePos().x()) {
160  hub_terminal = other_terminal;
161  }
162  else if (other_terminal->scenePos().x() == hub_terminal->scenePos().x()) {
163  if (other_terminal->scenePos().y() < hub_terminal->scenePos().y()) {
164  hub_terminal = other_terminal;
165  }
166  }
167  }
168 
169  terminals_to_connect_list.removeAll(hub_terminal);
170  if (hub_terminal == nullptr || terminals_to_connect_list.isEmpty()) {
171  continue;
172  }
173 
174  ConductorProperties properties = hub_terminal->conductors().first()->properties();
175  for (Terminal *t : terminals_to_connect_list)
176  {
177  //If a conductor was already created between these two terminals
178  //in this undo command, from another removed element, we do nothing
179  bool exist_ = false;
180  for (QPair<Terminal *, Terminal *> pair : m_connected_terminals)
181  {
182  if (pair.first == hub_terminal && pair.second == t) {
183  exist_ = true;
184  continue;
185  } else if (pair.first == t && pair.second == hub_terminal) {
186  exist_ = true;
187  continue;
188  }
189  }
190 
191  if (exist_ == false)
192  {
193  m_connected_terminals.append(qMakePair<Terminal *, Terminal *>(hub_terminal, t));
194  Conductor *new_cond = new Conductor(hub_terminal, t);
195  new_cond->setProperties(properties);
196  new AddItemCommand<Conductor*>(new_cond, t->diagram(), QPointF(), this);
197  }
198  }
199  }
200  }
201 }

214 Terminal *DeleteQGraphicsItemCommand::terminalInSamePotential(
215  Terminal *terminal,
216  Conductor *conductor_to_exclude)
217 {
218  QList<Conductor *> conductor_list = terminal->conductors();
219  conductor_list.removeAll(conductor_to_exclude);
220  for(Conductor *c : conductor_list)
221  {
222  Terminal *other_terminal = c->terminal1 == terminal ? c->terminal2 : c->terminal1;
223  if(!m_removed_contents.items(DiagramContent::Elements).contains(other_terminal->parentElement())) {
224  return other_terminal;
225  }
226  }
227  //No one of direct conductor of terminal are docked to an element which is not removed
228  for(Conductor *c : conductor_list)
229  {
230  Terminal *other_terminal = c->terminal1 == terminal ? c->terminal2 : c->terminal1;
231  Terminal *terminal_to_return = terminalInSamePotential(other_terminal, c);
232  if (terminal_to_return != nullptr) {
233  return terminal_to_return;
234  }
235  }
236 
237  return nullptr;
238 }

244 void DeleteQGraphicsItemCommand::undo()
245 {
246  m_diagram->showMe();
247 
248  for(QGraphicsItem *item : m_removed_contents.items())
249  m_diagram->addItem(item);
250 
251  //We relink element after every element was added to diagram
252  for(Element *e : m_removed_contents.m_elements)
253  for(Element *elmt : m_link_hash[e])
254  e->linkToElement(elmt);
255 
256  for(DynamicElementTextItem *deti : m_removed_contents.m_element_texts)
257  {
258  if(m_elmt_text_hash.keys().contains(deti))
259  m_elmt_text_hash.value(deti)->addDynamicTextItem(deti);
260  else if (m_grp_texts_hash.keys().contains(deti))
261  {
262  Element *elmt = m_grp_texts_hash.value(deti)->parentElement();
263  elmt->addDynamicTextItem(deti);
264  elmt->addTextToGroup(deti, m_grp_texts_hash.value(deti));
265  }
266  }
267 
268  for (auto table : m_table_scene_hash.keys())
269  {
270  if (!m_table_scene_hash.value(table).isNull()) {
271  m_table_scene_hash.value(table)->addItem(table);
272  }
273  }
274 
275  QUndoCommand::undo();
276 }

282 void DeleteQGraphicsItemCommand::redo()
283 {
284  m_diagram -> showMe();
285 
286  for(Conductor *c : m_removed_contents.conductors(DiagramContent::AnyConductor))
287  {
288  //If option one text per folio is enable, and the text item of
289  //current conductor is visible (that mean the conductor have the single displayed text)
290  //We call adjustTextItemPosition to other conductor at the same potential to keep
291  //a visible text on this potential.
292  if (m_diagram -> defaultConductorProperties.m_one_text_per_folio && c -> textItem() -> isVisible())
293  {
294  QList <Conductor *> conductor_list;
295  conductor_list << c -> relatedPotentialConductors(false).values();
296  if (conductor_list.count())
297  conductor_list.first() -> calculateTextItemPosition();
298  }
299  }
300 
301  for(Element *e : m_removed_contents.m_elements)
302  {
303  //Get linked element, for relink it at undo
304  if (!e->linkedElements().isEmpty())
305  m_link_hash.insert(e, e->linkedElements());
306  }
307 
308  for(DynamicElementTextItem *deti : m_removed_contents.m_element_texts)
309  {
310  if(deti->parentGroup() && deti->parentGroup()->parentElement())
311  deti->parentGroup()->parentElement()->removeTextFromGroup(deti, deti->parentGroup());
312 
313  deti->parentElement()->removeDynamicTextItem(deti);
314  deti->setParentItem(nullptr);
315  }
316 
317  for (auto table : m_table_scene_hash.keys())
318  {
319  if (!m_table_scene_hash.value(table).isNull()) {
320  m_table_scene_hash.value(table)->removeItem(table);
321  }
322  }
323 
324  for(QGraphicsItem *item : m_removed_contents.items())
325  m_diagram->removeItem(item);
326 
327  QUndoCommand::redo();
328 }
