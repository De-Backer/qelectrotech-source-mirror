41 void DiagramPropertiesEditorDockWidget::setDiagram(Diagram *diagram)
42 {
43  if (m_diagram == diagram) return;
44 
45  if (m_diagram)
46  {
47  disconnect(m_diagram, SIGNAL(selectionChanged()), this, SLOT(selectionChanged()));
48  disconnect(m_diagram, SIGNAL(destroyed()), this, SLOT(diagramWasDeleted()));

71 void DiagramPropertiesEditorDockWidget::selectionChanged()
72 {

75  }
76 
77  auto editor_ = PropertiesEditorFactory::propertiesEditor(m_diagram->selectedItems(), editors().count()? editors().first() : nullptr, this);
78  if (!editor_) {
79  clear();
80  return;

82  if (editors().count() &&
83  editors().first() != editor_) {
84  clear();
85  }
86
