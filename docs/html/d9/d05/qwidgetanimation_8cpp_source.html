78 void QWidgetAnimation::show()
79 {
80  if (m_state == QWidgetAnimation::Showing)
81  return;
82 
83  stop();
84 
85  m_widget->show();
86 
87  int end_value = 10000;
88  if (m_behavior == QWidgetAnimation::minimumSizeHint)
89  {
90  end_value = m_orientation == Qt::Horizontal
91  ? m_widget->minimumSizeHint().width()
92  : m_widget->minimumSizeHint().height();
93  }
94  else if (m_behavior == QWidgetAnimation::availableSpace
95  && m_widget->parentWidget())
96  {
97  m_widget->parentWidget()->layout();
98  int available_ = m_orientation == Qt::Horizontal
99  ? m_widget->parentWidget()->width()
100  : m_widget->parentWidget()->height();
101  for (auto w : m_widget_to_substract) {
102  available_ -= m_orientation == Qt::Horizontal
103  ? w->minimumSizeHint().width()
104  : w->minimumSizeHint().height();
105  }
106 
107  int mini_ = m_orientation == Qt::Horizontal
108  ? m_widget->minimumSizeHint().width()
109  : m_widget->minimumSizeHint().height();
110 
111  end_value = available_ > mini_ ? available_ : mini_;
112  }
113  else
114  {
115  if (m_last_rect.isValid()) {
116  end_value = m_orientation == Qt::Horizontal
117  ? m_last_rect.width()
118  : m_last_rect.height();
119  } else {
120  end_value = m_orientation == Qt::Horizontal
121  ? m_maximum.width()
122  : m_maximum.height();
123  }
124  }
125 
126  setStartValue(0);
127  setEndValue(end_value);
128  m_state = QWidgetAnimation::Showing;
129  start();
130 }
