152 void PartTerminal::updateSecondPoint() {
153  qreal ts = 4.0; // terminal size
154  switch(d -> m_orientation) {
155  case Qet::North: d -> second_point = QPointF(0.0, ts); break;
156  case Qet::East : d -> second_point = QPointF(-ts, 0.0); break;
157  case Qet::South: d -> second_point = QPointF(0.0, -ts); break;
158  case Qet::West : d -> second_point = QPointF(ts, 0.0); break;
159  }
160 }
