71 bool RectangleEditor::setPart(CustomElementPart *part)
72 {
73  if (!part)
74  {
75  if (m_part)
76  {
77  disconnectChangeConnections();
78  }
79  m_part = nullptr;
80  return(true);
81  }
82 
83  if (PartRectangle *part_rectangle = dynamic_cast<PartRectangle *>(part))
84  {
85  if (m_part == part_rectangle) {
86  return true;
87  }
88  if (m_part)
89  {
90  disconnectChangeConnections();
91  }
92  m_part = part_rectangle;
93  updateForm();
94  setUpChangeConnections();
95  return(true);
96  }
97 
98  return(false);
99 }

134 void RectangleEditor::updateForm()
135 {
136  if (!m_part) {
137  return;
138  }
139  activeConnections(false);
140 
141  QRectF rect = m_part->property("rect").toRectF();
142  QPointF p = m_part->mapToScene(rect.topLeft());
143  ui->m_x_sb->setValue(p.x());
144  ui->m_y_sb->setValue(p.y());
145  ui->m_width_sb->setValue(rect.width());
146  ui->m_height_sb->setValue(rect.height());
147  ui->m_rx_sb->setValue(m_part->XRadius());
148  ui->m_rx_sb->setMaximum(rect.width()/2);
149  ui->m_ry_sb->setValue(m_part->YRadius());
150  ui->m_ry_sb->setMaximum(rect.height()/2);
151 
152  activeConnections(true);
153 }

160 void RectangleEditor::editingFinished()
161 {
162  if (m_locked) {
163  return;
164  }
165  m_locked = true;
166 
167  QUndoCommand *undo = new QUndoCommand();
168  undo->setText(tr("Modifier un rectangle"));
169 
170  QRectF rect(editedTopLeft(), QSizeF(ui->m_width_sb->value(), ui->m_height_sb->value()));
171  if (m_part->rect() != rect)
172  {
173  QPropertyUndoCommand *u = new QPropertyUndoCommand(m_part, "rect", m_part->rect(), rect, undo);
174  u->enableAnimation();
175  }
176  if (m_part->XRadius() != ui->m_rx_sb->value())
177  {
178  QPropertyUndoCommand *u = new QPropertyUndoCommand(m_part, "xRadius", m_part->XRadius(), ui->m_rx_sb->value(), undo);
179  u->setAnimated();
180  }
181  if (m_part->YRadius() != ui->m_ry_sb->value())
182  {
183  QPropertyUndoCommand *u = new QPropertyUndoCommand(m_part, "yRadius", m_part->YRadius(), ui->m_ry_sb->value(), undo);
184  u->setAnimated();
185  }
186  elementScene()->undoStack().push(undo);
187 
188  m_locked = false;
189 }
