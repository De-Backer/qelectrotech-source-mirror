369 void ElementPrimitiveDecorator::applyMovementToRect(int movement_type, const QPointF &movement, QRectF &rect) {
370  qreal new_value;
371  QPointF new_point;
372 
373  switch (movement_type) {
374  case QET::MoveArea:
375  rect.translate(movement.x(), movement.y());
376  break;
377  case QET::ResizeFromTopLeftCorner:
378  new_point = rect.topLeft() + movement;
379  rect.setTopLeft(new_point);
380  break;
381  case QET::ResizeFromTopCenterCorner:
382  new_value = rect.top() + movement.y();
383  rect.setTop(new_value);
384  break;
385  case QET::ResizeFromTopRightCorner:
386  new_point = rect.topRight() + movement;
387  rect.setTopRight(new_point);
388  break;
389  case QET::ResizeFromMiddleLeftCorner:
390  new_value = rect.left() + movement.x();
391  rect.setLeft(new_value);
392  break;
393  case QET::ResizeFromMiddleRightCorner:
394  new_value = rect.right() + movement.x();
395  rect.setRight(new_value);
396  break;
397  case QET::ResizeFromBottomLeftCorner:
398  new_point = rect.bottomLeft() + movement;
399  rect.setBottomLeft(new_point);
400  break;
401  case QET::ResizeFromBottomCenterCorner:
402  new_value = rect.bottom() + movement.y();
403  rect.setBottom(new_value);
404  break;
405  case QET::ResizeFromBottomRightCorner:
406  new_point = rect.bottomRight() + movement;
407  rect.setBottomRight(new_point);
408  break;
409  }

516 void ElementPrimitiveDecorator::handlerMouseMoveEvent(QetGraphicsHandlerItem *qghi, QGraphicsSceneMouseEvent *event)
517 {
518  Q_UNUSED(qghi);
519 
520  QPointF scene_pos = event -> scenePos();
521  QPointF movement = scene_pos - latest_pos_;
522 
523  // For convenience purposes, we may need to adjust mouse movements.
524  QET::ScalingMethod scaling_method = scalingMethod(event);
525  if (scaling_method > QET::FreeScaling)
526  {
527  // real, non-rounded movement from the mouse press event
528  QPointF global_movement = scene_pos - first_pos_;
529 
530  QPointF rounded_global_movement;
531  if (scaling_method == QET::SnapScalingPointToGrid)
532  {
533  // real, rounded movement from the mouse press event
534  rounded_global_movement = snapConstPointToGrid(global_movement);
535  }
536  else
537  {
538  QRectF new_bounding_rect = original_bounding_rect_;
539  applyMovementToRect(current_operation_square_, global_movement, new_bounding_rect);
540 
541  const qreal scale_epsilon = 20.0; // rounds to 0.05
542  QPointF delta = deltaForRoundScaling(original_bounding_rect_, new_bounding_rect, scale_epsilon);
543 
544  // real, rounded movement from the mouse press event
545  rounded_global_movement = global_movement + delta;
546  }
547 
548  // rounded position of the current mouse move event
549  QPointF rounded_scene_pos = first_pos_ + rounded_global_movement;
550 
551  // when scaling the selection, consider the center of the currently dragged resizing rectangle
552  QPointF current_position = mapToScene(getResizingsPoints().at(current_operation_square_));
553  // determine the final, effective movement
554  movement = rounded_scene_pos - current_position;
555  }
556 
557  QRectF bounding_rect = modified_bounding_rect_;
558  applyMovementToRect(current_operation_square_, movement, modified_bounding_rect_);
559  if (modified_bounding_rect_ != bounding_rect) {
560  adjustEffectiveBoundingRect();
561  }
562  latest_pos_ = event -> scenePos();
563  scaleItems(original_bounding_rect_, modified_bounding_rect_);
564 }

640 QPointF ElementPrimitiveDecorator::deltaForRoundScaling(const QRectF &original, const QRectF &current, qreal epsilon) {
641  qreal sx = current.width() / original.width();
642  qreal sy = current.height() / original.height();
643 
644  qreal sx_rounded = QET::round(sx, epsilon);
645  qreal sy_rounded = QET::round(sy, epsilon);
646 
647  QPointF delta(
648  original.width() * (sx_rounded - sx),
649  original.height() * (sy_rounded - sy)
650  );
651  return(delta);

736 bool ElementPrimitiveDecorator::sceneEventFilter(QGraphicsItem *watched, QEvent *event)
737 {
738  //Watched must be an handler
739  if(watched->type() == QetGraphicsHandlerItem::Type)
740  {
741  QetGraphicsHandlerItem *qghi = qgraphicsitem_cast<QetGraphicsHandlerItem *>(watched);
742 
743  if(m_handler_vector.contains(qghi)) //Handler must be in m_vector_index, then we can start resize
744  {
745  m_vector_index = m_handler_vector.indexOf(qghi);
746  if (m_vector_index != -1)
747  {
748  if(event->type() == QEvent::GraphicsSceneMousePress) //Click
749  {
750  handlerMousePressEvent(qghi, static_cast<QGraphicsSceneMouseEvent *>(event));
751  return true;
752  }
753  else if(event->type() == QEvent::GraphicsSceneMouseMove) //Move
754  {
755  handlerMouseMoveEvent(qghi, static_cast<QGraphicsSceneMouseEvent *>(event));
756  return true;
757  }
758  else if (event->type() == QEvent::GraphicsSceneMouseRelease) //Release
759  {
760  handlerMouseReleaseEvent(qghi, static_cast<QGraphicsSceneMouseEvent *>(event));
761  return true;
762  }
763  }
764  }
765  }
766
