105 
106  if (!conductor_list.isEmpty())
107  {
108  if (!m_parent_undo)
109  undo->setText(
110  QObject::tr(
111  "Modifier les propriétés de plusieurs conducteurs",
112  "undo caption"));
113 
114  foreach (Conductor *cond, conductor_list)
115  {
116  ConductorProperties cp2 = cond -> properties();
117  old_value.setValue(cp2);
118  cp2.text = t;
119  new_value.setValue(cp2);
120  new QPropertyUndoCommand(
121  cond,
122  "properties",
123  old_value,

126  }

128 
129  if (!m_parent_undo)

165 void ConductorAutoNumerotation::numeratePotential()
166 {
167  ConductorProperties cp = conductor_list.first()->properties();
168  bool properties_equal = true;
169  foreach (const Conductor *conductor, conductor_list)
170  {
171  if (conductor->properties() != cp)
172  properties_equal = false;
173  }
174  // Every properties of the potential is equal,
175  // so we apply it to m_conductor
176  if (properties_equal)
177  {
178  m_conductor->setProperties(cp);
179  m_conductor->rSequenceNum() = conductor_list.first()->sequenceNum();
180  return;
181  }
182 
183  QStringList text_list;
184  QStringList formula_list;
185  foreach (const Conductor *cc, conductor_list)
186  {
187  ConductorProperties cp = cc->properties();
188  text_list << cp.text;
189  formula_list << cp.m_formula;
190  }
191 
192 
193  //the texts is identicals
194  if (QET::eachStrIsEqual(text_list) && QET::eachStrIsEqual(formula_list))
195  {
196  QList<ConductorProperties> cp_list;
197  foreach(Conductor *c, conductor_list)
198  cp_list<<c->properties();
199 
200  ConductorProperties cp = m_conductor->properties();
201  cp.applyForEqualAttributes(cp_list);
202  m_conductor->rSequenceNum() = conductor_list.first()->sequenceNum();
203  m_conductor->setProperties(cp);
204  }
205  //the texts isn't identicals
206  else
207  {
208  PotentialSelectorDialog psd(
209  m_conductor,
210  m_parent_undo,
211  m_conductor->diagramEditor());
212  psd.exec();
213  }
214 }
