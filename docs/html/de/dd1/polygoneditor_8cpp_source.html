88 bool PolygonEditor::setPart(CustomElementPart *new_part)
89 {
90  if (!new_part)
91  {
92  if (m_part)
93  {
94  disconnectChangeConnections();
95  }
96  m_part = nullptr;
97  return(true);
98  }
99  if (PartPolygon *part_polygon = dynamic_cast<PartPolygon *>(new_part))
100  {
101  if (m_part == part_polygon) return true;
102  if (m_part)
103  {
104  disconnectChangeConnections();
105  }
106  m_part = part_polygon;
107  updateForm();
108  setUpChangeConnections();
109  return(true);
110  }
111  return(false);
112 }

130 void PolygonEditor::updateForm()
131 {
132  if (!m_part) {
133  return;
134  }
135 
136  ui->m_points_list_tree->clear();
137 
138  for(QPointF point : m_part->polygon())
139  {
140  point = m_part->mapToScene(point);
141  QTreeWidgetItem *qtwi = new QTreeWidgetItem();
142  qtwi->setData(0, Qt::EditRole, point.x());
143  qtwi->setData(1, Qt::EditRole, point.y());
144  qtwi -> setFlags(Qt::ItemIsEditable | Qt::ItemIsEnabled);
145  ui->m_points_list_tree->addTopLevelItem(qtwi);
146  }
147  ui->m_close_polygon_cb->setChecked(m_part->isClosed());
148  ui->m_remove_point_action->setEnabled(m_part->polygon().size()
149  > 2 ? true : false);
150 }

157 QVector<QPointF> PolygonEditor::pointsFromTree()
158 {
159  QVector<QPointF> points;
160  if (!m_part) {
161  return points;
162  }
163 
164  for(int i = 0 ; i < ui->m_points_list_tree->topLevelItemCount() ; ++ i)
165  {
166  QTreeWidgetItem *qtwi = ui->m_points_list_tree->topLevelItem(i);
167  bool x_convert_ok, y_convert_ok;
168  qreal x = qtwi->data(0, Qt::EditRole).toReal(&x_convert_ok);
169  qreal y = qtwi->data(1, Qt::EditRole).toReal(&y_convert_ok);
170  if (x_convert_ok && y_convert_ok) {
171  points << m_part->mapFromScene(QPointF(x, y));
172  }
173  }
174  return(points);
175 }

254 void PolygonEditor::on_m_points_list_tree_itemSelectionChanged()
255 {
256  //Prevent when selection change but the widget ins't focused
257  if (!ui->m_points_list_tree->hasFocus()) {
258  return;
259  }
260 
261  QTreeWidgetItem *qtwi = ui->m_points_list_tree->currentItem();
262  if (!qtwi || !m_part) {
263  return;
264  }
265 
266  m_part->resetAllHandlerColor();
267  int index = ui->m_points_list_tree->indexOfTopLevelItem(qtwi);
268 
269  //We need to check if index isn't out of range of polygon
270  //this case can occur when user remove the last point of the polygon
271  //with the context menu of the tree widget
272  if(index >= 0 &&
273  index < m_part->polygon().size())
274  {
275  m_part->setHandlerColor(m_part->polygon().at(index), QColor(0, 255, 128));
276  }

279 void PolygonEditor::on_m_add_point_action_triggered()
280 {
281  QTreeWidgetItem *qtwi = ui->m_points_list_tree->currentItem();
282  if (!qtwi || !m_part) {
283  return;
284  }
285 
286  int index = ui->m_points_list_tree->indexOfTopLevelItem(qtwi);
287  QPolygonF new_polygon = m_part->polygon();
288 
289  //Special case when user add a point after the last point of the polygon
290  if (index == m_part->polygon().size()-1)
291  {
292  QPointF p = m_part->polygon().last();
293  p.rx()+=20;
294  p.ry()+=20;
295  new_polygon.append(p);
296  }
297  else
298  {
299  QPointF p = m_part->polygon().at(index) +
300  m_part->polygon().at(index+1);
301  p/=2;
302  new_polygon.insert(index+1, p);
303  }
304 
305  //Wrap the undo for avoid to merge the undo commands when user add several points.
306  QUndoCommand *undo = new QUndoCommand(tr("Ajouter un point à un polygone"));
307  new QPropertyUndoCommand(m_part, "polygon", m_part->polygon(), new_polygon, undo);
308  elementScene()->undoStack().push(undo);
309 
310  m_part->resetAllHandlerColor();
311  m_part->setHandlerColor(m_part->polygon().at(index+1), QColor(0, 255, 128));
312 }
