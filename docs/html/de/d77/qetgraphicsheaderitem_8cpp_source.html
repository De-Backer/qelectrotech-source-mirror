91 void QetGraphicsHeaderItem::paint(QPainter *painter, const QStyleOptionGraphicsItem *option, QWidget *widget)
92 {
93  Q_UNUSED(option)
94  Q_UNUSED(widget)
95 
96  painter->save();
97 
98  QPen pen;
99  pen.setWidthF(0.7);
100  pen.setColor(Qt::black);
101 
102  QBrush brush = painter->brush();
103  brush.setColor(Qt::lightGray);
104  brush.setStyle(Qt::SolidPattern);
105  painter->setBrush(brush);
106 
107  painter->setPen(pen);
108  painter->drawRect(m_current_rect);
109 
110  if (!m_model)
111  {
112  painter->restore();
113  return;
114  }
115  painter->setFont(m_model->headerData(0, Qt::Horizontal, Qt::FontRole).value<QFont>());
116 
117  //Draw vertical lines
118  auto offset= 0;
119  for(auto size : m_current_sections_width)
120  {
121  QPointF p1(offset+size, m_current_rect.top());
122  QPointF p2(offset+size, m_current_rect.bottom());
123  painter->drawLine(p1, p2);
124  offset += size;
125  }
126 
127  //Write text of each cell
128  auto margins_ = QETUtils::marginsFromString(m_model->headerData(0, Qt::Horizontal, Qt::UserRole+1).toString());
129  QPointF top_left(margins_.left(), margins_.top());
130  for (auto i= 0 ; i<m_model->columnCount() ; ++i)
131  {
132  QSize size(m_current_sections_width.at(i) - margins_.left() - margins_.right(), m_section_height - margins_.top() - margins_.bottom());
133  painter->drawText(QRectF(top_left, size),
134  m_model->headerData(0, Qt::Horizontal, Qt::TextAlignmentRole).toInt(),
135  m_model->headerData(i, Qt::Horizontal, Qt::DisplayRole).toString());
136 
137  top_left.setX(top_left.x() + m_current_sections_width.at(i));
138  }
139 
140  painter->restore();
141 }

156 void QetGraphicsHeaderItem::resizeSection(int logicalIndex, int size)
157 {
158  if (!m_model)
159  {
160  m_current_sections_width.clear();
161  m_current_sections_width.append(no_model_width);
162  m_sections_minimum_width.clear();
163  m_sections_minimum_width.append(no_model_width);
164  m_current_rect.setWidth(no_model_width);
165  setUpBoundingRect();
166  update();
167  emit sectionResized(0, no_model_width);
168  return;
169  }
170 
171  if (logicalIndex >= m_current_sections_width.size() ||
172  m_current_sections_width.at(logicalIndex) == size) {
173  return;
174  }
175 
176  if (m_model &&
177  logicalIndex<m_model->columnCount() &&
178  size >= m_sections_minimum_width.at(logicalIndex))
179  {
180  prepareGeometryChange();
181  m_current_sections_width.replace(logicalIndex, size);
182  m_current_rect.setWidth(std::accumulate(m_current_sections_width.begin(), m_current_sections_width.end(), 0));
183  setUpBoundingRect();
184  update();
185  emit sectionResized(logicalIndex, size);
186  }
187 }

239 void QetGraphicsHeaderItem::setUpMinimumSectionsSize()
240 {
241  if (!m_model)
242  {
243  m_minimum_section_height = no_model_height;
244  m_sections_minimum_width.clear();
245  m_sections_minimum_width.append(no_model_width);
246  m_minimum_width = no_model_width;
247  return;
248  }
249 
250  QFontMetrics metrics(m_model->headerData(0, Qt::Horizontal, Qt::FontRole).value<QFont>());
251  auto margins_ = QETUtils::marginsFromString(m_model->headerData(0, Qt::Horizontal, Qt::UserRole+1).toString());
252  //Set the height of row;
253  m_minimum_section_height = metrics.boundingRect("HEIGHT TEST").height() + margins_.top() + margins_.bottom();
254 
255  m_sections_minimum_width.clear();
256  m_sections_minimum_width.resize(m_model->columnCount());
257 
258  for (auto i= 0 ; i<m_model->columnCount() ; ++i)
259  {
260  auto str = m_model->headerData(i, Qt::Horizontal).toString();
261  m_sections_minimum_width.replace(i, metrics.boundingRect(str).width() + margins_.left() + margins_.right());
262  }
263 
264  m_minimum_width = std::accumulate(m_sections_minimum_width.begin(), m_sections_minimum_width.end(), 0);
265 }

298 void QetGraphicsHeaderItem::adjustSize()
299 {
300  if (m_section_height != m_minimum_section_height)
301  {
302  m_section_height = m_minimum_section_height;
303  m_current_rect.setHeight(m_section_height);
304  emit heightResized();
305  }
306 
307  if(m_current_sections_width.size() == m_sections_minimum_width.size())
308  {
309  auto old_sections_width = m_current_sections_width;
310 
311  for (int i=0 ; i<m_current_sections_width.size() ; ++i)
312  {
313  if (old_sections_width.at(i) < m_sections_minimum_width.at(i)) {
314  resizeSection(i, m_sections_minimum_width.at(i));
315  }
316  }
317  }
318 
319  update();
320 }
