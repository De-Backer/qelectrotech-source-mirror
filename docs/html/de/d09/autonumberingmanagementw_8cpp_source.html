108 void AutoNumberingManagementW::on_m_apply_folios_rb_clicked() {

113  ui->m_from_folios_cb->clear();

119  }
120  }

122  applyEnable(true);
123  else applyEnable(false);

188 void AutoNumberingManagementW::on_buttonBox_clicked(QAbstractButton *button) {
189  //transform button to int
190  int answer = ui -> buttonBox -> buttonRole(button);

195  emit applyPressed();
196  break;
197  case QDialogButtonBox::HelpRole:
198  QMessageBox::information(
199  this,
200  tr("Auto Numbering Management", "title window"),
201  tr("In this Menu you can set whether you want the Auto Numberings to be updated or not."
202  " For Element Auto Numbering you have 4 options of Update Policy:\n"
203  "-Both: both New and Existent Element labels will be updated. This is the default option.\n"
204  "-Update Only New: only new created Elements will be updated. Existent Element labels will be frozen.\n"
205  "-Update Only Existent: only existent Elements will be updated. New Elements will be assigned "
206  "their formula but will not update once created.\n"
207  "-Disable: both New and Existent Element labels will not be updated. This is valid for new folios as well.\n"
208  "Note: These options DO NOT allow or block Auto Numberings, only their Update Policy."
209  ));
210  break;
211  }
212 }
