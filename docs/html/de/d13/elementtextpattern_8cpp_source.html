199 void ImportElementTextPattern::apply(QString name, bool erase) const
200 {
201  if(!name.endsWith(".xml"))
202  name.append(".xml");
203 
204  QFile conf_file(QETApp::configDir() + "/element_texts_pattern/" + name);
205  if(!conf_file.open(QIODevice::ReadOnly | QIODevice::Text))
206  return;
207 
208  //Load the content of the xml file
209  QDomDocument xml_conf;
210  if(!xml_conf.setContent(&conf_file))
211  return;
212 
213  QDomElement root = xml_conf.firstChildElement("Element_texts_pattern");
214 
215  //Get the text
216  QList <QDomElement> texts = QET::findInDomElement(root, "dynamic_texts", "dynamic_elmt_text");
217  if (texts.isEmpty())
218  return;
219 
220  //Replace the original uuid of texts in the xml description, by a new one for every texts
221  QHash<QUuid, QUuid> uuid_hash;
222  for(QDomElement text : texts)
223  {
224  QUuid original_uuid(text.attribute("uuid"));
225  QUuid new_uuid = QUuid::createUuid();
226  text.setAttribute("uuid", new_uuid.toString());
227 
228  uuid_hash.insert(original_uuid, new_uuid);
229  }
230 
231  //In each group of the xml description, replace the original uuids, by the news created upper.
232  QList <QDomElement> groups = QET::findInDomElement(root, "texts_groups", "texts_group");
233  for(const QDomElement& group : groups)
234  {
235  for(QDomElement text : QET::findInDomElement(group, "texts", "text"))
236  {
237  QUuid original_uuid(text.attribute("uuid"));
238  QUuid new_uuid = uuid_hash.value(original_uuid);
239  if(!new_uuid.isNull())
240  text.setAttribute("uuid", new_uuid.toString());
241  }
242  }
243 
244  QUndoStack &undo_stack = m_element->diagram()->undoStack();
245  undo_stack.beginMacro(QObject::tr("Importer la configuration de texte : %1").arg(name.remove(".xml")));
246 
247  //erase existing texts and groups
248  if (erase)
249  {
250  for (ElementTextItemGroup *group : m_element->textGroups()) {
251  undo_stack.push(new RemoveTextsGroupCommand(m_element, group));
252  }
253  for (DynamicElementTextItem *deti : m_element->dynamicTextItems())
254  {
255  DiagramContent dc;
256  dc.m_element_texts << deti;
257  undo_stack.push(new DeleteQGraphicsItemCommand(m_element->diagram(), dc));
258  }
259  }
260 
261  //Add the texts to element
262  for(const QDomElement& text : texts)
263  {
264  DynamicElementTextItem *deti = new DynamicElementTextItem(m_element);
265  undo_stack.push(new AddElementTextCommand(m_element, deti));
266  deti->fromXml(text);
267  }
268  //Add the groups to element
269  for(const QDomElement& xml_group : groups)
270  undo_stack.push(new AddTextsGroupCommand(m_element, xml_group));
271 
272  undo_stack.endMacro();
273 }
