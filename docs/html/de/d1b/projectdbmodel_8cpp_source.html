233 QDomElement ProjectDBModel::toXml(QDomDocument &document) const

236 
237  //Identifier

240  dom_identifier.appendChild(dom_identifier_text);

243  //query
244  auto dom_query = document.createElement("query");
245  auto dom_query_text = document.createTextNode(m_query);
246  dom_query.appendChild(dom_query_text);
247  dom_element.appendChild(dom_query);
248 
249  //Add index 0,0 data
250  auto index_00 = document.createElement("index00");
251  index_00.setAttribute("font", m_index_0_0_data.value(Qt::FontRole).toString());
252  auto me = QMetaEnum::fromType<Qt::Alignment>();
253  index_00.setAttribute("alignment", me.valueToKey(m_index_0_0_data.value(Qt::TextAlignmentRole).toInt()));
254  dom_element.appendChild(index_00);
255  index_00.setAttribute("margins", m_index_0_0_data.value(Qt::UserRole+1).toString());
256 
257  //header data
258  QHash<int, QList<int>> horizontal_;
259  for (auto key : m_header_data.keys())
260  {
261  //We save all data except the display role, because he was generated in the fly
262  auto list = m_header_data.value(key).keys();
263  list.removeAll(Qt::DisplayRole);
264 
265  horizontal_.insert(key, list);
266  }
267 
268  dom_element.appendChild(QETXML::modelHeaderDataToXml(document, this, horizontal_, QHash<int, QList<int>>()));
269 
270  return dom_element;
271 }

311 void ProjectDBModel::dataBaseUpdated()
312 {
313  auto original_record = m_record;
314  fillValue();
315  auto new_record = m_record;
316  m_record = original_record;
317 
318  if (new_record.size() != m_record.size())
319  {
320  emit beginResetModel();
321  m_record = new_record;
322  emit endResetModel();
323  }
324  else
325  {
326  m_record = new_record;
327  auto row = m_record.size();
328  auto col = row ? m_record.first().count() : 1;
329 
330  emit dataChanged(this->index(0,0), this->index(row-1, col-1), QVector<int>(Qt::DisplayRole));
331  }
332 }

334 void ProjectDBModel::setHeaderString()
335 {
336  auto q = m_project->dataBase()->newQuery(m_query);
337  auto record = q.record();
338 
339  for (auto i=0 ; i<record.count() ; ++i)
340  {
341  auto field_name = record.fieldName(i);
342  QString header_name;
343 
344  if (field_name == "position") {
345  header_name = tr("Position");
346  } else if (field_name == "diagram_position") {
347  header_name = tr("Position du folio");
348  } else {
349  header_name = QETApp::elementTranslatedInfoKey(field_name);
350  if (header_name.isEmpty()) {
351  header_name = QETApp::diagramTranslatedInfoKey(field_name);
352  }
353  if (header_name.isEmpty()) {
354  header_name = field_name;
355  }
356  }
357  this->setHeaderData(i, Qt::Horizontal, header_name, Qt::DisplayRole);
358  }
359 }

361 void ProjectDBModel::fillValue()
362 {
363  m_record.clear();
364 
365  auto query_ = m_project->dataBase()->newQuery(m_query);
366  if (!query_.exec()) {
367  qDebug() << "Query error : " << query_.lastError();
368  }
369 
370  while (query_.next())
371  {
372  QStringList record_;
373  auto i=0;
374  while (query_.value(i).isValid())
375  {
376  auto date = query_.value(i).toDate();
377  if (!date.isNull()) {
378  record_ << QLocale::system().toString(query_.value(i).toDate(), QLocale::ShortFormat);
379  } else {
380  record_ << query_.value(i).toString();
381  }
382  ++i;
383  }
384  m_record << record_;
385  }
386 }
