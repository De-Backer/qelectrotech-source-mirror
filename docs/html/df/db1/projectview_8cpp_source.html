211 bool ProjectView::tryClosing() {
212  if (!m_project) return(true);
213 
214  // First step: require external editors closing -- users may either cancel
215  // the whole closing process or save (and therefore add) content into this
216  // project. Of course, they may also discard them.
217  if (!tryClosingElementEditors()) {
218  return(false);
219  }
220 
221  // Check how different the current situation is from a brand new, untouched project
222  if (m_project -> filePath().isEmpty() && !m_project -> projectWasModified()) {
223  return(true);
224  }
225 
226  // Second step: users are presented with a dialog that enables them to
227  // choose whether they want to:
228  // - cancel the closing process,
229  // - discard all modifications,
230  // - or specify what is to be saved, i.e. they choose whether they wants to
231  // save/give up/remove diagrams considered as modified.
232  int user_input = tryClosingDiagrams();
233  if (user_input == QMessageBox::Cancel) {
234  return(false); // the closing process was cancelled
235  } else if (user_input == QMessageBox::Discard) {
236  return(true); // all modifications were discarded
237  }
238 
239  // Check how different the current situation is from a brand new, untouched project (yes , again)
240  if (m_project -> filePath().isEmpty() && !m_project -> projectWasModified()) {
241  return(true);
242  }
243 
244  if (m_project -> filePath().isEmpty()) {
245  QString filepath = askUserForFilePath();
246  if (filepath.isEmpty()) return(false); // users may cancel the closing
247  }
248  QETResult result = m_project -> write();
249  updateWindowTitle();
250  if (!result.isOk()) emit(errorEncountered(result.errorMessage()));
251  return(result.isOk());
252 }

260 bool ProjectView::tryClosingElementEditors() {
261  if (!m_project) return(true);
262  /*
263  La QETApp permet d'acceder rapidement aux editeurs d'element
264  editant un element du projet.
265  */
266  QList<QETElementEditor *> editors = QETApp::elementEditors(m_project);
267  foreach(QETElementEditor *editor, editors) {
268  if (!editor -> close()) return(false);
269  }
270 
271  QList<QETTitleBlockTemplateEditor *> template_editors = QETApp::titleBlockTemplateEditors(m_project);
272  foreach(QETTitleBlockTemplateEditor *template_editor, template_editors) {
273  if (!template_editor -> close()) return(false);
274  }
275  return(true);
276 }

284 int ProjectView::tryClosingDiagrams() {
285  if (!m_project) return(QMessageBox::Discard);
286 
287  if (!project()->projectOptionsWereModified() &&
288  project()->undoStack()->isClean() &&
289  !project()->filePath().isEmpty()) {
290  // nothing was modified, and we have a filepath, i.e. everything was already
291  // saved, i.e we can close the project right now
292  return(QMessageBox::Discard);
293  }
294 
295  QString title = project()->title();
296  if (title.isEmpty()) title = "QElectroTech ";
297 
298  int close_dialog = QMessageBox::question(this, title,
299  tr("Le projet à été modifié.\n"
300  "Voulez-vous enregistrer les modifications ?"),
301  QMessageBox::Save | QMessageBox::Discard
302  | QMessageBox::Cancel,
303  QMessageBox::Save);
304 
305  return(close_dialog);
306 }

352 void ProjectView::removeDiagram(DiagramView *diagram_view)
353 {
354  if (!diagram_view)
355  return;
356  if (m_project -> isReadOnly())
357  return;
358  if (!m_diagram_ids.values().contains(diagram_view))
359  return;
360 
361 
362  //Ask confirmation to user.
363  int answer = QET::QetMessageBox::question(
364  this,
365  tr("Supprimer le folio ?", "message box title"),
366  tr("Êtes-vous sûr de vouloir supprimer ce folio du projet ? Ce changement est irréversible.", "message box content"),
367  QMessageBox::Yes | QMessageBox::No | QMessageBox::Cancel,
368  QMessageBox::No
369  );
370  if (answer != QMessageBox::Yes) {
371  return;
372  }
373 
374  //Remove the diagram view of the tabs widget
375  int index_to_remove = m_diagram_ids.key(diagram_view);
376  m_tab->removeTab(index_to_remove);
377  m_diagram_view_list.removeAll(diagram_view);
378  rebuildDiagramsMap();
379 
380  m_project -> removeDiagram(diagram_view -> diagram());
381  delete diagram_view;
382 
383  emit(diagramRemoved(diagram_view));
384  updateAllTabsTitle();
385  m_project -> setModified(true);
386 }

661 int ProjectView::cleanProject() {
662  if (!m_project) return(0);
663 
664  // s'assure que le schema n'est pas en lecture seule
665  if (m_project -> isReadOnly()) {
666  QET::QetMessageBox::critical(
667  this,
668  tr("Projet en lecture seule", "message box title"),
669  tr("Ce projet est en lecture seule. Il n'est donc pas possible de le nettoyer.", "message box content")
670  );
671  return(0);
672  }
673 
674  // construit un petit dialogue pour parametrer le nettoyage
675  QCheckBox *clean_tbt = new QCheckBox(tr("Supprimer les modèles de cartouche inutilisés dans le projet"));
676  QCheckBox *clean_elements = new QCheckBox(tr("Supprimer les éléments inutilisés dans le projet"));
677  QCheckBox *clean_categories = new QCheckBox(tr("Supprimer les catégories vides"));
678  QDialogButtonBox *buttons = new QDialogButtonBox(QDialogButtonBox::Ok | QDialogButtonBox::Cancel);
679 
680  clean_tbt -> setChecked(true);
681  clean_elements -> setChecked(true);
682  clean_categories -> setChecked(true);
683 
684  QDialog clean_dialog(parentWidget());
685 #ifdef Q_OS_MACOS
686  clean_dialog.setWindowFlags(Qt::Sheet);
687 #endif
688 
689  clean_dialog.setWindowTitle(tr("Nettoyer le projet", "window title"));
690  QVBoxLayout *clean_dialog_layout = new QVBoxLayout();
691  clean_dialog_layout -> addWidget(clean_tbt);
692  clean_dialog_layout -> addWidget(clean_elements);
693  clean_dialog_layout -> addWidget(clean_categories);
694  clean_dialog_layout -> addWidget(buttons);
695  clean_dialog.setLayout(clean_dialog_layout);
696 
697  connect(buttons, SIGNAL(accepted()), &clean_dialog, SLOT(accept()));
698  connect(buttons, SIGNAL(rejected()), &clean_dialog, SLOT(reject()));
699 
700  int clean_count = 0;
701  if (clean_dialog.exec() == QDialog::Accepted)
702  {
703  if (clean_tbt -> isChecked()) {
704  m_project->embeddedTitleBlockTemplatesCollection()->deleteUnusedTitleBlocKTemplates();
705  }
706  if (clean_elements->isChecked()) {
707  m_project->embeddedElementCollection()->cleanUnusedElement();
708  }
709  if (clean_categories->isChecked()) {
710  m_project->embeddedElementCollection()->cleanUnusedDirectory();
711  }
712  }
713 
714  m_project -> setModified(true);
715  return(clean_count);
716 }
