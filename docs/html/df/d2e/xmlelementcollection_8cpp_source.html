333 QString XmlElementCollection::addElement(ElementsLocation &location)
334 {
335  //location must be an element and exist
336  if (!(location.exist() && location.isElement()))
337  return QString();
338 
339  //Add an element from this collection to this collection have no sense
340  if (location.isProject() && location.projectCollection() == this)
341  return QString();
342 
343  //First we check if this location exist in this collection if so, we do nothing
344  if ( exist("import/" + location.collectionPath(false)) )
345  return QString();
346 
347  //Get the import dir of this collection
348  QDomElement parent_element = importCategory();
349  if (parent_element.isNull())
350  return QString();
351 
352  QString integrated_path = parent_element.attribute("name");
353 
354  //Split the path
355  QStringList splitted_path = location.collectionPath(false).split("/");
356  if (splitted_path.isEmpty())
357  return QString();
358 
359  if (location.isFileSystem()) {
360  //Get the root dir of the filesystem collection
361  QDir dir(location.fileSystemPath().remove(
362  location.collectionPath(false)));
363  if (!dir.exists())
364  return QString();
365 
366  foreach(QString str, splitted_path) {
367  QDomElement child_element = child(parent_element, str);
368 
369  //Child doesn't exist, we create it
370  if (child_element.isNull()) {
371  QDomElement created_child;
372 
373  //str is the path of an element, we integrate an element
374  if (str.endsWith(".elmt")) {
375  QFile element_file(dir.filePath(str));
376  if (!element_file.exists())
377  return QString();
378 
379  created_child = QETXML::
380  fileSystemElementToXmlCollectionElement(
381  m_dom_document,
382  element_file);
383  }
384 
385  //str is the path of a directory, we integrate a directory.
386  else {
387  //Dir doesn't exist.
388  if (!dir.cd(str))
389  return QString();
390 
391  created_child = QETXML::
392  fileSystemDirToXmlCollectionDir(
393  m_dom_document,
394  dir);
395  }
396 
397  if(created_child.isNull())
398  return QString();
399 
400  parent_element.appendChild(created_child);
401  parent_element = created_child;
402  }
403  //Child exist
404  else {
405  if (!dir.cd(str))
406  return QString();
407 
408  parent_element = child_element;
409  }
410 
411  integrated_path.append("/"+str);
412  }
413  }
414  else if (location.isProject()) {
415  QString path;
416  foreach(QString str, splitted_path) {
417  if (path.isEmpty())
418  path = str;
419  else
420  path = path + "/" + str;
421 
422  QDomElement child_element = child(parent_element, str);
423 
424  //Child doesn't exist, we create it
425  if (child_element.isNull()) {
426  QDomElement created_child;
427 
428  //str is the path of an element, we integrate an element
429  if (str.endsWith(".elmt")) {
430  created_child = m_dom_document.createElement("element");
431  created_child.setAttribute("name", str);
432 
433  created_child.appendChild(location.xml().cloneNode(true));
434  }
435 
436  //str is the path of a directory, we integrate a directory.
437  else {
438  created_child = m_dom_document.createElement("category");
439  created_child.setAttribute("name", str);
440 
441  ElementsLocation sub_dir_location(path, location.project());
442  QDomElement names_element = sub_dir_location.nameList().toXml(m_dom_document);
443  created_child.appendChild(names_element);
444  }
445 
446  if(created_child.isNull())
447  return QString();
448 
449  parent_element.appendChild(created_child);
450  parent_element = created_child;
451  }
452  //Child exist
453  else
454  parent_element = child_element;
455 
456  integrated_path.append("/"+str);
457  }
458  }
459 
460  emit elementAdded(integrated_path);
461  return integrated_path;
462 }

751  ElementsLocation &destination,
752  const QString& rename,

754 {

756

758  QDomElement parent_dir_dom = directory(destination.collectionPath(false));
759  if (parent_dir_dom.isNull()) return ElementsLocation();

762  QDomElement element = child(destination.collectionPath(false)
763  + "/" + new_dir_name);
764  if (!element.isNull()) {
765  element.parentNode().removeChild(element);
766  emit directoryRemoved(destination.collectionPath(false)
767  + "/" + new_dir_name);
768  }
769 
770  ElementsLocation created_location;
771 
772  //Copy with a file system collection source
773  if (source.isFileSystem())
774  {
775  QDir source_dir(source.fileSystemPath());

777 
778 
779  QDir dir(source.fileSystemPath());

784  parent_dir_dom.appendChild(elmt_dom);
785 
786  created_location.setPath(destination.projectCollectionPath()
787  + "/" + new_dir_name);
788

790  {

792  foreach(QString str, source_dir.entryList(

807  ElementsLocation sub_source(source.fileSystemPath()

809  copyElement(sub_source, created_location);
810  }

812  }

815  else
816  {
817  if (!source.projectCollection()) return ElementsLocation();

826  other_collection_node.appendChild(names.cloneNode(true));

832  other_collection_dom_dir.setAttribute("name", new_dir_name);
833  parent_dir_dom.appendChild(other_collection_dom_dir);
834 
835  created_location.setPath(destination.projectCollectionPath() + "/" + new_dir_name);
836  }
837 
838  emit directorieAdded(created_location.collectionPath(false));
839  return created_location;
840 }
