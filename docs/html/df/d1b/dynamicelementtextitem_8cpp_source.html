324 void DynamicElementTextItem::setTextFrom(DynamicElementTextItem::TextFrom text_from)
325 {
326  if(m_text_from == text_from)
327  return;
328 
329  setNoEditable(text_from == UserText? false : true);

332  TextFrom old_text_from = m_text_from;

334 
335  if(m_text_from == UserText)
336  {
337  setPlainText(m_text);
338  disconnect(m_parent_element.data(), &Element::elementInfoChange, this, &DynamicElementTextItem::elementInfoChanged);
339  }
340  else if (m_text_from == ElementInfo && elementUseForInfo())
341  {
342  if(m_info_name == "label")
343  {
344  setupFormulaConnection();
345  updateLabel();
346  }
347  else
348  setPlainText(elementUseForInfo()->elementInformations().value(m_info_name).toString());
349 
350  if(old_text_from == UserText)
351  connect(elementUseForInfo(), &Element::elementInfoChange, this, &DynamicElementTextItem::elementInfoChanged);
352  }
353  else if (m_text_from == CompositeText && elementUseForInfo())
354  {
355  if(m_composite_text.contains("%{label}"))
356  {
357  setupFormulaConnection();
358  updateLabel();
359  }
360  else
361  setPlainText(autonum::AssignVariables::replaceVariable(m_composite_text, elementUseForInfo()->elementInformations()));
362 
363  if(old_text_from == UserText)
364  connect(elementUseForInfo(), &Element::elementInfoChange, this, &DynamicElementTextItem::elementInfoChanged);
365  }
366

454 void DynamicElementTextItem::setCompositeText(const QString &text)
455 {
456  QString old_composite_text = m_composite_text;
457  m_composite_text = text;
458 
459  if(old_composite_text.contains("%{label}"))
460  {
461  clearFormulaConnection();

464 
465  if (m_parent_element && (m_parent_element.data()->linkType() & Element::AllReport)) //special treatment for report

468  * May be in some case the old and new composite text have both the var %{label},
469  * and so we don't have to remove connection and after set conection,
470  * but for that we must to do several check and because I'm lazy,
471  * in every case I remove connection and set it after ;)
472  */

474  removeConnectionForReportFormula(m_report_formula);
475  if(m_composite_text.contains("%{label}"))
476  setConnectionForReportFormula(m_report_formula);

478  updateReportText();

481  {
482  setupFormulaConnection();
483  updateLabel();
484  updateXref();
485  }

487  {
488  DiagramContext dc;
489  if(elementUseForInfo())
490  dc = elementUseForInfo()->elementInformations();
491  setPlainText(autonum::AssignVariables::replaceVariable(m_composite_text, dc));

682 QVariant DynamicElementTextItem::itemChange(QGraphicsItem::GraphicsItemChange change, const QVariant &value)
683 {
684  //The first time this text is added to a scene, we make several cheking and connection
685  //according to the link type of the parent element
686  if(change == QGraphicsItem::ItemSceneHasChanged && m_first_scene_change)
687  {
688  if(m_parent_element.isNull())
689  return QGraphicsObject::itemChange(change, value);
690 
691  //If the parent is slave, we keep aware about the changement of master.
692  if(m_parent_element.data()->linkType() == Element::Slave)
693  {
694  connect(m_parent_element.data(), &Element::linkedElementChanged, this, &DynamicElementTextItem::masterChanged);
695  //The parent is already linked, wa call master changed for init the connection
696  if(!m_parent_element.data()->linkedElements().isEmpty())
697  masterChanged();
698  }
699  else if(m_parent_element.data()->linkType() & Element::AllReport)
700  {
701  //Get the report formula, and add connection to keep up to date the formula.
702  if (m_parent_element.data()->diagram() && m_parent_element.data()->diagram()->project())
703  {
704  m_report_formula = m_parent_element.data()->diagram()->project()->defaultReportProperties();
705  m_report_formula_con = connect(m_parent_element.data()->diagram()->project(), &QETProject::reportPropertiesChanged, this, &DynamicElementTextItem::reportFormulaChanged);
706  }
707 
708  //Add connection to keep up to date the status of the element linked to the parent folio report of this text.
709  connect(m_parent_element.data(), &Element::linkedElementChanged, this, &DynamicElementTextItem::reportChanged);
710  //The parent is already linked, we call reportChanged for init the connection
711  if(!m_parent_element.data()->linkedElements().isEmpty())
712  reportChanged();
713 
714  if(m_parent_element.data()->terminals().size())
715  {
716  //Add connection to keep up date the conductors added or removed to the parent folio report element
717  connect(m_parent_element.data()->terminals().first(), &Terminal::conductorWasAdded, this, &DynamicElementTextItem::conductorWasAdded);
718  connect(m_parent_element.data()->terminals().first(), &Terminal::conductorWasRemoved, this, &DynamicElementTextItem::conductorWasRemoved);
719  }
720  //Get a conductor in the potential
721  setPotentialConductor();
722  }
723  else if(m_parent_element.data()->linkType() == Element::Master)
724  {
725  connect(m_parent_element.data(), &Element::linkedElementChanged, this, &DynamicElementTextItem::updateXref);
726  if(m_parent_element.data()->diagram())
727  connect(m_parent_element.data()->diagram()->project(), &QETProject::XRefPropertiesChanged, this, &DynamicElementTextItem::updateXref);
728  if(!m_parent_element.data()->linkedElements().isEmpty())
729  updateXref();
730  }
731 
732  m_first_scene_change = false;
733  return QGraphicsObject::itemChange(change, value);
734  }
735  else if (change == QGraphicsItem::ItemParentHasChanged)
736  {
737  updateXref();
738  updateXref();
739  }
740 
741  return QGraphicsObject::itemChange(change, value);
742 }
