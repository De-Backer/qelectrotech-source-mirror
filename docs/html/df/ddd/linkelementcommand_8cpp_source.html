66 bool LinkElementCommand::isLinkable(Element *element_a, Element *element_b, bool already_linked)
67 {
68  switch(element_a->linkType())
69  {
70  case Element::Simple: return false;
71 
72  case Element::NextReport:
73  {
74  //Type isn't good
75  if (element_b->linkType() != Element::PreviousReport) return false;
76  //two report is free
77  if (element_a->isFree() && element_b->isFree()) return true;
78  //Reports aren't free but are already linked between them and and already_linked is true
79  if (element_a->linkedElements().contains(element_b) && already_linked) return true;
80 
81  return false;
82  }
83 
84  case Element::PreviousReport:
85  {
86  //Type isn't good
87  if (element_b->linkType() != Element::NextReport) return false;
88  //two report is free
89  if (element_a->isFree() && element_b->isFree()) return true;
90  //Reports aren't free but are already linked between them and and already_linked is true
91  if (element_a->linkedElements().contains(element_b) && already_linked) return true;
92 
93  return false;
94  }
95 
96  case Element::Master:
97  {
98  //Type isn't good
99  if (element_b->linkType() != Element::Slave) return false;
100  //element_b is free
101  if (element_b->isFree()) return true;
102  //element_b isn't free but already linked to element_a and already_linked is true
103  if (element_a->linkedElements().contains(element_b) && already_linked) return true;
104 
105  return false;
106  }
107 
108  case Element::Slave:
109  {
110  //Type isn't good
111  if (element_b->linkType() != Element::Master) return false;
112  //Element_a is free
113  if (element_a->isFree()) return true;
114  //element_a isn't free but already linked to element_b and already_linked is true;
115  if (element_b->linkedElements().contains(element_a) && already_linked) return true;
116 
117  return false;
118  }
119 
120  case Element::Terminale: return false;
121 
122  default: return false;
123  }
124 }

185 void LinkElementCommand::redo()
186 {
187  if(m_element->diagram()) m_element->diagram()->showMe();
188  makeLink(m_linked_after);
189 
190  //If the action is to link two reports together, we check if the conductors
191  //of the new potential have the same text, function, and protocol.
192  //if not, a dialog ask what do to.
193  if (m_first_redo && (m_element->linkType() & Element::AllReport) \
194  && m_element->conductors().size() \
195  && m_linked_after.size() && m_linked_after.first()->conductors().size())
196  {
197  //fill list of potential
198  QSet <Conductor *> c_list = m_element->conductors().first()->relatedPotentialConductors();
199  c_list << m_element->conductors().first();
200  //fill list of text
201  QStringList str_txt;
202  QStringList str_funct;
203  QStringList str_tens;
204  for (const Conductor *c : c_list)
205  {
206  str_txt << c->properties().text;
207  str_funct << c->properties().m_function;
208  str_tens << c->properties().m_tension_protocol;
209  str_tens << c->properties().m_wire_color;
210  str_tens << c->properties().m_wire_section;
211  }
212 
213  //check text list, isn't same in potential, ask user what to do
214  if (!QET::eachStrIsEqual(str_txt) || !QET::eachStrIsEqual(str_funct) || !QET::eachStrIsEqual(str_tens))
215  {
216  PotentialSelectorDialog psd(m_element, this);
217  psd.exec();
218  }
219  m_first_redo = false;
220  }
221  QUndoCommand::redo();
222 }

233 void LinkElementCommand::setUpNewLink(const QList<Element *> &element_list,
234  bool already_link)
235 {
236  //m_element is a master we can connect several element to it
237  //if m_element isn't master (may be a report or slave) we can connect only one element
238  if (m_element->linkType() == Element::Master || element_list.size() == 1)
239  {
240  foreach(Element *elmt, element_list)
241  if (isLinkable(m_element, elmt, already_link))
242  m_linked_after << elmt;
243  }
244  else
245  {
246  qDebug() << "LinkElementCommand::setUpNewLink : try to link several elements to a report element or slave element,"
247  " only the first element of the list will be taken to be linked";
248  foreach(Element *elmt, element_list)
249  if (isLinkable(m_element, elmt, already_link))
250  {
251  m_linked_after << elmt;
252  return;
253  }
254  }
255 }

263 void LinkElementCommand::makeLink(const QList<Element *> &element_list)
264 {
265  //List is empty, that mean m_element must be free, so we unlink all elements
266  if (element_list.isEmpty())
267  {
268  m_element->unlinkAllElements();
269  return;
270  }
271 
272  //We link all element from element_list
273  foreach(Element *elmt, element_list)
274  m_element->linkToElement(elmt);
275 
276  /* At this point may be there are unwanted linked elements to m_element.
277  * We must to unlink it.
278  * Elements from element_list are wanted so we compare element_list
279  * to current linked element of m_element
280  */
281  QList<Element *> to_unlink = m_element->linkedElements();
282  foreach(Element *elmt, element_list)
283  to_unlink.removeAll(elmt);
284 
285  //All elements stored in to_unlink is unwanted we unlink it from m_element
286  if (!to_unlink.isEmpty())
287  foreach(Element *elmt, to_unlink)
288  m_element->unlinkElement(elmt);
289 }
