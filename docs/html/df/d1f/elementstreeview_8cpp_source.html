70 void ElementsTreeView::startElementDrag(const ElementsLocation &location)
71 {
72  if (!location.exist())
73  return;
74 
75  QDrag *drag = new QDrag(this);
76 
77  QString location_str = location.toString();
78  QMimeData *mime_data = new QMimeData();
79  mime_data->setText(location_str);
80 
81  if (location.isDirectory())
82  {
83  mime_data->setData("application/x-qet-category-uri",
84  location_str.toLatin1());
85  drag->setPixmap(QET::Icons::Folder.pixmap(22, 22));
86  }
87  else if (location.isElement())
88  {
89  mime_data->setData("application/x-qet-element-uri",
90  location_str.toLatin1());
91 
92  //Build the element for set the pixmap of the QDrag
93  int elmt_creation_state;
94  Element *temp_elmt = ElementFactory::Instance()->createElement(
95  location, nullptr,
96  &elmt_creation_state);
97  if (elmt_creation_state)
98  {
99  delete temp_elmt;
100  return;
101  }
102 
103  QPixmap elmt_pixmap(temp_elmt->pixmap());
104  QPoint elmt_hotspot(temp_elmt->hotspot());
105 
106  //Adjust the size of the pixmap if he is too big
107  QPoint elmt_pixmap_size(elmt_pixmap.width(),
108  elmt_pixmap.height());
109  if (elmt_pixmap.width()
110  > MAX_DND_PIXMAP_WIDTH
111  || elmt_pixmap.height()
112  > MAX_DND_PIXMAP_HEIGHT)
113  {
114  elmt_pixmap = elmt_pixmap.scaled(
115  MAX_DND_PIXMAP_WIDTH,
116  MAX_DND_PIXMAP_HEIGHT,
117  Qt::KeepAspectRatio);
118  elmt_hotspot = QPoint(
119  elmt_hotspot.x() * elmt_pixmap.width() / elmt_pixmap_size.x(),
120  elmt_hotspot.y() * elmt_pixmap.height() / elmt_pixmap_size.y()
121  );
122  }
123 
124  drag->setPixmap(elmt_pixmap);
125  drag->setHotSpot(elmt_hotspot);
126 
127 
128  delete temp_elmt;
129  }
130 
131  drag->setMimeData(mime_data);
132  drag->exec(Qt::CopyAction);
133 }
