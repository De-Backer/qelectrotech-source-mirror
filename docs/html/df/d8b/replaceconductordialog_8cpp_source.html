55 void ReplaceConductorDialog::updatePreview(bool b)
56 {
57  const QRect pixmap_rect(0, 0, 96, 96);
58  QPixmap pixmap(pixmap_rect.width(), pixmap_rect.height());
59  QPainter painter;
60  painter.begin(&pixmap);
61  painter.eraseRect(pixmap_rect);
62  painter.drawRect(pixmap_rect.adjusted(0,0,-1,-1));
63  painter.drawLine(QLineF(0, pixmap_rect.height() / 2, pixmap_rect.width(), pixmap_rect.height() / 2));
64 
65  if (b) {
66  properties().singleLineProperties.draw(&painter, QET::Horizontal, pixmap_rect);
67  } else {
68  m_properties.singleLineProperties.draw(&painter, QET::Horizontal, pixmap_rect);
69  }
70 
71  painter.end();
72  ui -> m_preview_label -> setPixmap(pixmap);
73 }

75 void ReplaceConductorDialog::setProperties(const ConductorProperties &properties)
76 {
77  m_properties = properties;
78 
79  if (m_properties.type == ConductorProperties::Single) {

85  ui->m_show_text->setChecked(m_properties.m_show_text);
86  sarw::setupLineEdit(ui->m_formula_le, ui->m_erase_formula_cb, m_properties.m_formula);
87  sarw::setupLineEdit(ui->m_text_le, ui->m_erase_text_cb, m_properties.text);
88  sarw::setupLineEdit(ui->m_function_le, ui->m_erase_function_cb, m_properties.m_function);

94  case Qt::AlignRight: ui->m_vertical_align_cb->setCurrentIndex(2);break;

96  }
97  switch (m_properties.m_horizontal_alignment) {
98  case Qt::AlignTop: ui->m_horizontal_align_cb->setCurrentIndex(1); break;
99  case Qt::AlignBottom: ui->m_horizontal_align_cb->setCurrentIndex(2); break;
100  default: ui->m_horizontal_align_cb->setCurrentIndex(0); break;
101  }
102  ui->m_vertical_angle_sb->setValue(m_properties.verti_rotate_text);
103  ui->m_horizontal_angle_sb->setValue(m_properties.horiz_rotate_text);
104  if (m_properties.color.isValid())
105  {
106  setColorButton(m_properties.color);
107  ui->m_mod_color_cb->setChecked(true);
108  }
109  int index = ui->m_line_style_cb->findData(QPen(m_properties.style));
110  if (index != -1) {
111  ui->m_line_style_cb->setCurrentIndex(index);
112  }
113  ui->m_second_color_gb->setChecked(m_properties.m_bicolor);
114  if (m_properties.m_color_2.isValid())
115  {
116  setColorButton2(m_properties.m_color_2);
117  ui->m_mod_color_2_cb->setChecked(true);
118  }
119  ui->m_color_2_dash_size_sb->setValue(m_properties.m_dash_size);
120 
121  ui->m_earth_cb ->setChecked (m_properties.singleLineProperties.hasGround);
122  ui->m_neutral_cb ->setChecked (m_properties.singleLineProperties.hasNeutral);
123  ui->m_pen_cb ->setChecked (m_properties.singleLineProperties.isPen());
124  ui->m_phase_cb ->setChecked (m_properties.singleLineProperties.phasesCount());
125  ui->m_phase_slider->setValue (m_properties.singleLineProperties.phasesCount());
126 
127  updatePreview(false);
128 }

134 ConductorProperties ReplaceConductorDialog::properties() const
135 {
136  ConductorProperties properties_;
137  if (ui->m_multi_gb->isChecked()) {
138  properties_.type = ConductorProperties::Multi;
139  } else {
140  properties_.type = ConductorProperties::Single;
141  }
142 
143  properties_.text_size = ui->m_text_size_sb->value();
144  properties_.m_formula = ui->m_formula_le->text();
145  properties_.text = ui->m_text_le->text();
146  properties_.m_show_text = ui->m_show_text->isChecked();
147  properties_.m_function = ui->m_function_le->text();
148  properties_.m_tension_protocol = ui->m_tension_protocol_le->text();
149  properties_.m_wire_color =ui->m_wire_color_le->text();
150  properties_.m_wire_section =ui->m_wire_section_le->text();
151  switch (ui->m_vertical_align_cb->currentIndex()) {
152  case 0: properties_.m_vertical_alignment = Qt::AlignAbsolute; break;
153  case 1: properties_.m_vertical_alignment = Qt::AlignLeft; break;
154  case 2: properties_.m_vertical_alignment = Qt::AlignRight; break;
155  default:break;
156  }
157  switch (ui->m_horizontal_align_cb->currentIndex()) {
158  case 0: properties_.m_horizontal_alignment = Qt::AlignAbsolute; break;
159  case 1: properties_.m_horizontal_alignment = Qt::AlignTop; break;
160  case 2: properties_.m_horizontal_alignment = Qt::AlignBottom; break;
161  default: break;
162  }
163  properties_.verti_rotate_text = ui->m_vertical_angle_sb->value();
164  properties_.horiz_rotate_text = ui->m_horizontal_angle_sb->value();
165  properties_.color = ui->m_mod_color_cb->isChecked()
166  ? ui->m_color_pb->palette().color(QPalette::Button)
167  : QColor();
168  properties_.style = ui->m_line_style_cb->itemData(
169  ui->m_line_style_cb->currentIndex())
170  .value<QPen>()
171  .style();
172  properties_.cond_size = ui->m_cond_size_sb->value();
173  properties_.m_bicolor = ui->m_second_color_gb->isChecked();
174  properties_.m_color_2 = ui->m_mod_color_2_cb->isChecked()
175  ? ui->m_color_2_pb->palette().color(QPalette::Button)
176  : QColor();
177  properties_.m_dash_size = ui->m_color_2_dash_size_sb->value();
178 
179  properties_.singleLineProperties.hasGround =
180  ui->m_earth_cb->isChecked();
181  properties_.singleLineProperties.hasNeutral =
182  ui->m_neutral_cb->isChecked();
183  properties_.singleLineProperties.is_pen =
184  ui->m_pen_cb->isChecked();
185  properties_.singleLineProperties.setPhasesCount(
186  ui->m_phase_cb->isChecked()
187  ? ui->m_phase_sb->value()
188  : 0);
189 
190  return properties_;
191 }
