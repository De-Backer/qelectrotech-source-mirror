32 void myMessageOutput(QtMsgType type,
33  const QMessageLogContext &context,
34  const QString &msg)
35 {
36 
37  QString txt=QTime::currentTime().toString("hh:mm:ss.zzz");
38  QByteArray dbs =txt.toLocal8Bit();
39  QByteArray localMsg = msg.toLocal8Bit();
40  const char *file = context.file ? context.file : "";
41  const char *function = context.function ? context.function : "";
42 
43  switch (type) {
44  case QtDebugMsg:
45  fprintf(stderr,
46  "%s Debug: %s (%s:%u, %s)\n",
47  dbs.constData(),
48  localMsg.constData(),
49  file,
50  context.line,
51  function);
52  txt+=" Debug: ";
53  break;
54  case QtInfoMsg:
55  fprintf(stderr,
56  "%s Info: %s \n",
57  dbs.constData(),
58  localMsg.constData());
59  txt+=" Info: ";
60  break;
61  case QtWarningMsg:
62  fprintf(stderr,
63  "%s Warning: %s (%s:%u, %s)\n",
64  dbs.constData(),
65  localMsg.constData(),
66  file, context.line,
67  function);
68  txt+=" Warning: ";
69  break;
70  case QtCriticalMsg:
71  fprintf(stderr,
72  "%s Critical: %s (%s:%u, %s)\n",
73  dbs.constData(),
74  localMsg.constData(),
75  file,
76  context.line,
77  function);
78  txt+=" Critical: ";
79  break;
80  case QtFatalMsg:
81  fprintf(stderr,
82  "%s Fatal: %s (%s:%u, %s)\n",
83  dbs.constData(),
84  localMsg.constData(),
85  file,
86  context.line,
87  function);
88  txt+=" Fatal: ";
89  break;
90  default:
91  fprintf(stderr,
92  "%s Unknown: %s (%s:%u, %s)\n",
93  dbs.constData(),
94  localMsg.constData(),
95  file,
96  context.line,
97  function);
98  txt+=" Unknown: ";
99  }
100  txt+= msg;
101  if(type==QtInfoMsg){
102  txt+=" \n";
103  } else {
104  txt+= " (";
105  txt+= context.file ? context.file : "";
106  txt+= ":";
107  txt+=QString::number(context.line ? context.line :0);
108  txt+= ", ";
109  txt+= context.function ? context.function : "";
110  txt+=")\n";
111  }
112  QFile outFile(QETApp::configDir()
113  +"/"
114  +QDate::currentDate().toString("yyyyMMdd")
115  +".log");
116  if(outFile.open(QIODevice::WriteOnly | QIODevice::Append))
117  {
118  QTextStream ts(&outFile);
119  ts << txt;
120  }
121  outFile.close();
122 }

129 void delete_old_log_files(int days)
130 {
131  const QDate today = QDate::currentDate();
132  const QString path = QETApp::configDir() + "/";
133 
134  QString filter("%1%1%1%1%1%1%1%1.log"); // pattern
135  filter = filter.arg("[0123456789]"); // valid characters
136 
137  Q_FOREACH (auto fileInfo,
138  QDir(path).entryInfoList(
139  QStringList(filter),
140  QDir::Files))
141  {
142  if (fileInfo.lastRead().date().daysTo(today) > days)
143  {
144  QString filepath = fileInfo.absoluteFilePath();
145  QDir deletefile;
146  deletefile.setPath(filepath);
147  deletefile.remove(filepath);
148  qDebug() << "File " + filepath + " is deleted!";
149  }
150  }
151 }
