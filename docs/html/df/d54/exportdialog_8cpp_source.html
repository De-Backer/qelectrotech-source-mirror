237 void ExportDialog::slot_correctWidth(int diagram_id) {
238  // recupere l'ExportDiagramLine concernee
239  ExportDialog::ExportDiagramLine *current_diagram = diagram_lines_[diagram_id];
240  if (!current_diagram) return;
241 
242  // ne fait rien si l'option "Conserver les proportions" n'est pas activee
243  if (!(current_diagram -> keep_ratio -> isChecked())) return;
244 
245  // recupere les proportions du schema
246  qreal diagram_ratio = diagramRatio(current_diagram -> diagram);
247 
248  // ajuste la largeur

250  current_diagram -> width -> setValue(qRound(current_diagram -> height -> value() * diagram_ratio));

402 void ExportDialog::generateDxf(Diagram *diagram,
403  int width,
404  int height,
405  bool keep_aspect_ratio,
406  QString &file_path) {
407  saveReloadDiagramParameters(diagram, true);
408 
409  width -= 2*Diagram::margin;
410  height -= 2*Diagram::margin;
411 
412  Createdxf::xScale = Createdxf::sheetWidth / double(width);
413  Createdxf::yScale = Createdxf::sheetHeight / double(height);
414 
415  Createdxf::dxfBegin(file_path);
416 
417  //Add project elements (lines, rectangles, circles, texts) to dxf file
418  if (epw -> exportProperties().draw_border) {
419  double bx0 = Diagram::margin * Createdxf::xScale;
420  double by0 = Diagram::margin * Createdxf::yScale;
421  Createdxf::drawRectangle(
422  file_path,
423  bx0,
424  -by0,
425  double(width)*Createdxf::xScale,
426  double(height)*Createdxf::yScale,
427  0);
428  }
429  diagram -> border_and_titleblock.drawDxf(width,
430  height,
431  keep_aspect_ratio,
432  file_path,
433  0);
434 
435  // Build the lists of elements.
436  QList<Element *> list_elements;
437  QList<Conductor *> list_conductors;
438  QList<DiagramTextItem *> list_texts;
439  QList<DiagramImageItem *> list_images;
440  QList<QLineF *> list_lines;
441  QList<QRectF *> list_rectangles;
442  //QList<QRectF *> list_ellipses;
443  QList <QetShapeItem *> list_shapes;
444 
445  // Determine les elements a "XMLiser"
446  foreach(QGraphicsItem *qgi, diagram -> items()) {
447  if (Element *elmt = qgraphicsitem_cast<Element *>(qgi)) {
448  list_elements << elmt;
449  } else if (Conductor *f = qgraphicsitem_cast<Conductor *>(qgi)) {
450  list_conductors << f;
451  } else if (IndependentTextItem *iti = qgraphicsitem_cast<IndependentTextItem *>(qgi)) {
452  list_texts << iti;
453  } else if (DiagramImageItem *dii = qgraphicsitem_cast<DiagramImageItem *>(qgi)) {
454  list_images << dii;
455  } else if (QetShapeItem *dii = qgraphicsitem_cast<QetShapeItem *>(qgi)) {
456  list_shapes << dii;
457  } else if (DynamicElementTextItem *deti = qgraphicsitem_cast<DynamicElementTextItem *>(qgi)) {
458  list_texts << deti;
459  }
460  }
461 
462  foreach (QetShapeItem *qsi, list_shapes) qsi->toDXF(file_path, qsi->pen());
463 
464  //Draw elements
465  foreach(Element *elmt, list_elements)
466  {
467  double rotation_angle = elmt -> orientation() * 90;
468 
469  qreal elem_pos_x = elmt -> pos().x();
470  qreal elem_pos_y = elmt -> pos().y();// - (diagram -> margin / 2);
471 
472  qreal hotspot_x = (elem_pos_x) * Createdxf::xScale;
473  qreal hotspot_y = Createdxf::sheetHeight - (elem_pos_y) * Createdxf::yScale;
474 
475  ElementPictureFactory::primitives primitives = ElementPictureFactory::instance()->getPrimitives(elmt->location());
476 
477  for(QGraphicsSimpleTextItem *text : primitives.m_texts)
478  {
479  qreal fontSize = text->font().pointSizeF();
480  if (fontSize < 0)
481  fontSize = text->font().pixelSize();
482 
483  fontSize *= Createdxf::yScale;
484  qreal x = elem_pos_x + text->pos().x();
485  qreal y = elem_pos_y + text->pos().y();
486  x *= Createdxf::xScale;
487  y = Createdxf::sheetHeight - (y * Createdxf::yScale);// - fontSize;
488  QPointF transformed_point = rotation_transformed(x, y, hotspot_x, hotspot_y, rotation_angle);
489  x = transformed_point.x();
490  y = transformed_point.y();
491  QStringList lines = text->text().split('\n');
492  y += (fontSize/2) * (lines.count()-1);
493  for (QString line : lines)
494  {
495  qreal angle = 360 - (text->rotation() + rotation_angle);
496  if (line.size() > 0 && line != "_" ) {
497  Createdxf::drawText(file_path, line, x, y, fontSize, angle, 0);
498  }
499  angle += 1080;
500  // coordinates for next line
501  if (int(angle) % 360 == 0) // no rotation
502  y -= fontSize*1.06;
503  else if (int(angle - 180) % 360 == 0) // 180 degrees rotation
504  y += fontSize*1.06;
505  else if (int(angle - 270) % 360 == 0) // 270 degrees rotation
506  x -= fontSize*1.06;
507  else // ((angle - 90) % 360 == 0) 90 degrees rotation
508  x += fontSize*1.06;
509  }
510  }
511 
512  for (QLineF line : primitives.m_lines)
513  {
514  qreal x1 = (elem_pos_x + line.p1().x()) * Createdxf::xScale;
515  qreal y1 = Createdxf::sheetHeight - (elem_pos_y + line.p1().y()) * Createdxf::yScale;
516  QPointF transformed_point = rotation_transformed(x1, y1, hotspot_x, hotspot_y, rotation_angle);
517  x1 = transformed_point.x();
518  y1 = transformed_point.y();
519  qreal x2 = (elem_pos_x + line.p2().x()) * Createdxf::xScale;
520  qreal y2 = Createdxf::sheetHeight - (elem_pos_y + line.p2().y()) * Createdxf::yScale;
521  transformed_point = rotation_transformed(x2, y2, hotspot_x, hotspot_y, rotation_angle);
522  x2 = transformed_point.x();
523  y2 = transformed_point.y();
524  Createdxf::drawLine(file_path, x1, y1, x2, y2, 0);
525  }
526 
527  for (QRectF rect : primitives.m_rectangles)
528  {
529  qreal x1 = (elem_pos_x + rect.bottomLeft().x()) * Createdxf::xScale;
530  qreal y1 = Createdxf::sheetHeight - (elem_pos_y + rect.bottomLeft().y()) * Createdxf::yScale;
531  qreal w = rect.width() * Createdxf::xScale;
532  qreal h = rect.height() * Createdxf::yScale;
533  // opposite corner
534  qreal x2 = x1 + w;
535  qreal y2 = y1 + h;
536  QPointF transformed_point = rotation_transformed(x1, y1, hotspot_x, hotspot_y, rotation_angle);
537  x1 = transformed_point.x();
538  y1 = transformed_point.y();
539  transformed_point = rotation_transformed(x2, y2, hotspot_x, hotspot_y, rotation_angle);
540  x2 = transformed_point.x();
541  y2 = transformed_point.y();
542  qreal bottom_left_x = (x1 < x2) ? x1 : x2;
543  qreal bottom_left_y = (y1 < y2) ? y1 : y2;
544  w = (x1 < x2) ? x2-x1 : x1-x2;
545  h = (y1 < y2) ? y2-y1 : y1-y2;
546  Createdxf::drawRectangle(file_path, bottom_left_x, bottom_left_y, w, h, 0);
547  }
548 
549  for (QRectF circle_rect : primitives.m_circles)
550  {
551  qreal x1 = (elem_pos_x + circle_rect.center().x()) * Createdxf::xScale;
552  qreal y1 = Createdxf::sheetHeight - (elem_pos_y + circle_rect.center().y()) * Createdxf::yScale;
553  qreal r = circle_rect.width() * Createdxf::xScale / 2;
554  QPointF transformed_point = rotation_transformed(x1, y1, hotspot_x, hotspot_y, rotation_angle);
555  x1 = transformed_point.x();
556  y1 = transformed_point.y();
557  Createdxf::drawCircle(file_path, r, x1, y1, 0);
558  }
559 
560  for (QVector<QPointF> polygon : primitives.m_polygons)
561  {
562  if (polygon.size() == 0)
563  continue;
564  qreal x1 = (elem_pos_x + polygon.at(0).x()) * Createdxf::xScale;
565  qreal y1 = Createdxf::sheetHeight - (elem_pos_y + polygon.at(0).y()) * Createdxf::yScale;
566  QPointF transformed_point = rotation_transformed(x1, y1, hotspot_x, hotspot_y, rotation_angle);
567  x1 = transformed_point.x();
568  y1 = transformed_point.y();
569  for (int i = 1; i < polygon.size(); ++i ) {
570  qreal x2 = (elem_pos_x + polygon.at(i).x()) * Createdxf::xScale;
571  qreal y2 = Createdxf::sheetHeight - (elem_pos_y + polygon.at(i).y()) * Createdxf::yScale;
572  QPointF transformed_point = rotation_transformed(x2, y2, hotspot_x, hotspot_y, rotation_angle);
573  x2 = transformed_point.x();
574  y2 = transformed_point.y();
575  Createdxf::drawLine(file_path, x1, y1, x2, y2, 0);
576  x1 = x2;
577  y1 = y2;
578  }
579  }
580 
581  // Draw arcs and ellipses
582  for (QVector<qreal> arc : primitives.m_arcs)
583  {
584  if (arc.size() == 0)
585  continue;
586  qreal x = (elem_pos_x + arc.at(0)) * Createdxf::xScale;
587  qreal y = Createdxf::sheetHeight - (elem_pos_y + arc.at(1)) * Createdxf::yScale;
588  qreal w = arc.at(2) * Createdxf::xScale;
589  qreal h = arc.at(3) * Createdxf::yScale;
590  qreal startAngle = arc.at(4);
591  qreal spanAngle = arc .at(5);
592  Createdxf::drawArcEllipse(file_path, x, y, w, h, startAngle, spanAngle, hotspot_x, hotspot_y, rotation_angle, 0);
593  }
594  }
595 
596  //Draw conductors
597  foreach(Conductor *cond, list_conductors) {
598  qreal cx = cond->pos().x();
599  qreal cy = cond->pos().y();
600  foreach(ConductorSegment *segment, cond -> segmentsList()) {
601  qreal x1 = (cx + segment -> firstPoint().x()) * Createdxf::xScale;
602  qreal y1 = Createdxf::sheetHeight - ((cy + segment -> firstPoint().y()) * Createdxf::yScale);
603  qreal x2 = (cx + segment -> secondPoint().x()) * Createdxf::xScale;
604  qreal y2 = Createdxf::sheetHeight - ((cy + segment -> secondPoint().y()) * Createdxf::yScale);
605  Createdxf::drawLine(file_path, x1, y1, x2, y2, 0);
606  }
607  //Draw conductor text item
608  ConductorTextItem *textItem = cond -> textItem();
609  if (textItem) {
610  qreal fontSize = textItem -> font().pointSizeF();
611  if (fontSize < 0)
612  fontSize = textItem -> font().pixelSize();
613  fontSize *= Createdxf::yScale;

668 QPointF ExportDialog::rotation_transformed(qreal px,
669  qreal py,
670  qreal origin_x,
671  qreal origin_y,
672  qreal angle) {
673 
674  angle *= -3.14159265 / 180;
675 
676  float s = sin(angle);
677  float c = cos(angle);
678 
679  // Vector to rotate:
680  qreal Vx = px - origin_x;
681  qreal Vy = py - origin_y;
682 
683  // rotate vector

685  float ynew = Vx * s + Vy * c;
686
