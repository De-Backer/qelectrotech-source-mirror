32 ElementsCollectionCache::ElementsCollectionCache(const QString &database_path, QObject *parent) :
33  QObject(parent),
34  locale_("en"),
35  pixmap_storage_format_("PNG")
36 {
37  // initialize the cache SQLite database
38  static int cache_instances = 0;
39  QString connection_name = QString("ElementsCollectionCache-%1").arg(cache_instances++);
40  cache_db_ = QSqlDatabase::addDatabase("QSQLITE", connection_name);
41  cache_db_.setDatabaseName(database_path);
42 
43  if (!cache_db_.open())
44  qDebug() << "Unable to open the SQLite database " << database_path << " as " << connection_name << ": " << cache_db_.lastError();
45  else
46  {
47  cache_db_.exec("PRAGMA temp_store = MEMORY");
48  cache_db_.exec("PRAGMA journal_mode = MEMORY");
49  cache_db_.exec("PRAGMA page_size = 4096");
50  cache_db_.exec("PRAGMA cache_size = 16384");
51  cache_db_.exec("PRAGMA locking_mode = EXCLUSIVE");
52  cache_db_.exec("PRAGMA synchronous = OFF");
53 
54  //TODO This code remove old table with mtime for create table with uuid, created at version 0,5
55  //see to remove this code at version 0,6 or 0,7 when all users will table with uuid.
56 #pragma message("@TODO remove this code for qet 0.6 or later")
57  QSqlQuery table_name(cache_db_);
58  if (table_name.exec("PRAGMA table_info(names)"))
59  {
60  if (table_name.seek(2))
61  {
62  QString str = table_name.value(1).toString();
63  table_name.finish();
64  if (str == "mtime")
65  {
66  QSqlQuery error;
67  error = cache_db_.exec("DROP TABLE names");
68  error = cache_db_.exec("DROP TABLE pixmaps");
69  }
70  }
71  else
72  table_name.finish();
73  }
74 
75  //@TODO the tables could already exist, handle that case.
76  cache_db_.exec("CREATE TABLE names"
77  "("
78  "path VARCHAR(512) NOT NULL,"
79  "locale VARCHAR(2) NOT NULL,"
80  "uuid VARCHAR(512) NOT NULL,"
81  "name VARCHAR(128),"
82  "PRIMARY KEY(path, locale)"
83  ");");
84 
85  cache_db_.exec("CREATE TABLE pixmaps"
86  "("
87  "path VARCHAR(512) NOT NULL UNIQUE,"
88  "uuid VARCHAR(512) NOT NULL,"
89  "pixmap BLOB, PRIMARY KEY(path),"
90  "FOREIGN KEY(path) REFERENCES names (path) ON DELETE CASCADE);");
91 
92  // prepare queries
93  select_name_ = new QSqlQuery(cache_db_);
94  select_pixmap_ = new QSqlQuery(cache_db_);
95  insert_name_ = new QSqlQuery(cache_db_);
96  insert_pixmap_ = new QSqlQuery(cache_db_);
97  select_name_ -> prepare("SELECT name FROM names WHERE path = :path AND locale = :locale AND uuid = :uuid");
98  select_pixmap_ -> prepare("SELECT pixmap FROM pixmaps WHERE path = :path AND uuid = :uuid");
99  insert_name_ -> prepare("REPLACE INTO names (path, locale, uuid, name) VALUES (:path, :locale, :uuid, :name)");
100  insert_pixmap_ -> prepare("REPLACE INTO pixmaps (path, uuid, pixmap) VALUES (:path, :uuid, :pixmap)");
101  }
102 }

161 bool ElementsCollectionCache::fetchElement(ElementsLocation &location)
162 {
163  // can we use the cache with this element?
164  bool use_cache = cache_db_.isOpen() && !location.isProject();
165 
166  // attempt to fetch the element name from the cache database
167  if (!use_cache) {
168  return(fetchData(location));
169  }
170  else
171  {
172  auto uuid = location.uuid();
173  QString element_path = location.toString();
174  bool got_name = fetchNameFromCache(element_path, uuid);
175  bool got_pixmap = fetchPixmapFromCache(element_path, uuid);
176 
177  if (got_name && got_pixmap) {
178  return(true);
179  }
180 
181  if (fetchData(location))
182  {
183  cacheName(element_path, uuid);
184  cachePixmap(element_path, uuid);
185  }
186  return(true);
187  }
188 }

235 bool ElementsCollectionCache::fetchNameFromCache(const QString &path,
236  const QUuid &uuid)
237 {
238  select_name_ -> bindValue(":path", path);
239  select_name_ -> bindValue(":locale", locale_);
240  select_name_ -> bindValue(":uuid", uuid.toString());
241  if (select_name_ -> exec())
242  {
243  if (select_name_ -> first())
244  {
245  current_name_ = select_name_ -> value(0).toString();
246  select_name_ -> finish();
247  return(true);
248  }
249  }
250  else
251  qDebug() << "select_name_->exec() failed";
252 
253  return(false);
254 }

264 bool ElementsCollectionCache::fetchPixmapFromCache(const QString &path,
265  const QUuid &uuid)
266 {
267  select_pixmap_ -> bindValue(":path", path);
268  select_pixmap_ -> bindValue(":uuid", uuid.toString());
269  if (select_pixmap_ -> exec())
270  {
271  if (select_pixmap_ -> first())
272  {
273  QByteArray ba = select_pixmap_ -> value(0).toByteArray();
274  // avoid returning always the same pixmap (i.e. same cacheKey())
275  current_pixmap_.detach();
276  current_pixmap_.loadFromData(ba, qPrintable(pixmap_storage_format_));
277  select_pixmap_ -> finish();
278  }
279  return(true);
280  }
281  else
282  qDebug() << "select_pixmap_->exec() failed";
283 
284  return(false);
285 }

318 bool ElementsCollectionCache::cachePixmap(const QString &path,
319  const QUuid &uuid)
320 {

324  current_pixmap_.save(&buffer, qPrintable(pixmap_storage_format_));
325  insert_pixmap_ -> bindValue(":path", path);
326  insert_pixmap_ -> bindValue(":uuid", uuid.toString());
327  insert_pixmap_ -> bindValue(":pixmap", QVariant(ba));
328  if (!insert_pixmap_->exec())
329  {
330  qDebug() << cache_db_.lastError();
331  return(false);
332  }
333  return(true);
