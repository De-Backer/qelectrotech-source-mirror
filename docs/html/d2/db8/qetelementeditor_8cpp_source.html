467

484  action -> setEnabled(selected_items);

755 bool QETElementEditor::checkElement() {
756  //List of warning and error
757  typedef QPair<QString, QString> QETWarning;
758  QList<QETWarning> warnings;
759  QList<QETWarning> errors;
760 
763  if (!m_elmt_scene -> containsTerminals() && !m_elmt_scene -> elementType().contains("report")) {
764  warnings << qMakePair(
765  tr("Absence de borne", "warning title"),
766  tr(

769  "warning description"

771  );
772  }

791  }
792  }
793 
794  if (!errors.count() && !warnings.count()) {
795  return(true);
796  }
797 
798  // Display warnings
799  QString dialog_message = tr("La vérification de cet élément a généré", "message box content");

806  if (errors.size()) {

810  }
811  dialog_message += " :";
812

814  QList<QETWarning> total = warnings << errors;
815  foreach(QETWarning warning, total) {
816  dialog_message += "<li>";
817  dialog_message += QString(
818  tr("<b>%1</b> : %2", "warning title: warning description")
819  ).arg(warning.first).arg(warning.second);
820  dialog_message += "</li>";
821  }
822  dialog_message += "</ol>";
823 
824  if (errors.size()) {
825  QMessageBox::critical(this, tr("Erreurs"), dialog_message);
826  }
827  else {
828  QMessageBox::warning(this, tr("Avertissements"), dialog_message);
829  }
830 
831  //if error == 0 that means they are only warning, we return true.
832  if (errors.count() == 0) {
833  return(true);
834  }
835  return false;
