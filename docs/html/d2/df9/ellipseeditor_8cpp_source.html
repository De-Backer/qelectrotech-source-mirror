94 bool EllipseEditor::setPart(CustomElementPart *new_part)
95 {
96  if (!new_part)
97  {
98  if (part)
99  disconnectChangeConnections();
100  part = nullptr;
101  return(true);
102  }
103  if (PartEllipse *part_ellipse = dynamic_cast<PartEllipse *>(new_part))
104  {
105  if (part == part_ellipse) return true;
106  if (part)
107  disconnectChangeConnections();
108  part = part_ellipse;
109  updateForm();
110  setUpChangeConnections();
111  return(true);
112  }
113  return(false);
114 }

137 void EllipseEditor::editingFinishedX()
138 {
139  if (m_locked) return;
140  m_locked = true;
141 
142  for (auto part: style_->currentParts()) {
143 
144  PartEllipse* ell = static_cast<PartEllipse*>(part);
145  QRectF rect = ell->property("rect").toRectF();
146  QPointF point = ell->mapFromScene(x->value() - h->value()/2, y->value() - v->value()/2); // does not matter which value y is, because only the x value is used
147  rect.setX(point.x()); // change only the x value
148 
149  if (rect != part->property("rect"))
150  {
151  QPropertyUndoCommand *undo= new QPropertyUndoCommand(ell, "rect", ell->property("rect"), rect);
152  undo->setText("Modifier un arc");
153  undo->enableAnimation();
154  elementScene()->undoStack().push(undo);
155  }
156  }
157 
158  m_locked = false;
159 }

236 void EllipseEditor::updateForm()
237 {
238  if (!part) return;
239  activeConnections(false);
240  QRectF rect = part->property("rect").toRectF();
241  x->setValue(part->mapToScene(rect.topLeft()).x() + (rect.width()/2));
242  y->setValue(part->mapToScene(rect.topLeft()).y() + (rect.height()/2));
243  h->setValue(rect.width());
244  v->setValue(rect.height());
245  activeConnections(true);
