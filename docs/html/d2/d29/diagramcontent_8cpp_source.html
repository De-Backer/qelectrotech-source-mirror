44 DiagramContent::DiagramContent(Diagram *diagram, bool selected) :
45  m_selected_items(diagram->selectedItems())
46 {
47  QList <QGraphicsItem *> item_list;
48  if (selected) {
49  item_list = m_selected_items;
50  } else {
51  item_list = diagram->items();
52  }
53 
54  for (auto item : item_list)
55  {
56  switch (item->type())
57  {
58  case Element::Type: { m_elements << qgraphicsitem_cast<Element *>(item); break;}
59  case IndependentTextItem::Type: { m_text_fields << qgraphicsitem_cast<IndependentTextItem *>(item); break;}
60  case Conductor::Type:
61  {
62  auto c = qgraphicsitem_cast<Conductor *>(item);
63  //Get the isolated selected conductor (= not movable, but deletable)
64  if (!c->terminal1->parentItem()->isSelected() &&\
65  !c->terminal2->parentItem()->isSelected()) {
66  m_other_conductors << c;
67  }
68 
69  if (m_potential_conductors.isEmpty()) {
70  m_potential_conductors << c;
71  } else {
72  if (!potentialIsManaged(c->relatedPotentialConductors(true).values())) {
73  m_potential_conductors << c;
74  }
75  }
76  break;
77  }
78  case DiagramImageItem::Type: { m_images << qgraphicsitem_cast<DiagramImageItem *>(item); break;}
79  case QetShapeItem::Type: { m_shapes << qgraphicsitem_cast<QetShapeItem *>(item); break;}
80  case DynamicElementTextItem::Type: { m_element_texts << qgraphicsitem_cast<DynamicElementTextItem *>(item); break;}
81  case QGraphicsItemGroup::Type: {
82  if (auto *group = dynamic_cast<ElementTextItemGroup *>(item)) {
83  m_texts_groups << group;
84  }
85  break;
86  }
87  case QetGraphicsTableItem::Type: { m_tables << qgraphicsitem_cast<QetGraphicsTableItem *>(item); break;}
88  }
89  }
90 
91 
92  //For each selected element, we determine if conductors must be moved or updated.
93  for(Element *elmt : m_elements)
94  {
95  if (elmt->isSelected())
96  {
97  for(Terminal *terminal : elmt->terminals())
98  {
99  for(Conductor *conductor : terminal->conductors())
100  {
101  Terminal *other_terminal;
102  if (conductor->terminal1 == terminal)
103  other_terminal = conductor->terminal2;
104  else
105  other_terminal = conductor->terminal1;
106 
107  //If the two elements of conductor are movable
108  if (m_elements.contains(other_terminal -> parentElement())) {
109  if (!m_conductors_to_move.contains(conductor))
110  m_conductors_to_move << conductor;
111  }
112  else if (!m_conductors_to_update.contains(conductor))
113  m_conductors_to_update << conductor;
114  }
115  }
116  }
117  }
118 }

247 int DiagramContent::removeNonMovableItems()
248 {
249  int count_ = 0;
250 
251  const QList<Element *> elements_set = m_elements;
252  for(Element *elmt : elements_set) {
253  if (!elmt->isMovable()) {
254  m_elements.removeAll(elmt);
255  ++count_;
256  }
257  }
258 
259  const QSet<DiagramImageItem *> images_set = m_images;
260  for(DiagramImageItem *img : images_set) {
261  if (!img->isMovable()) {
262  m_images.remove(img);
263  ++count_;
264  }
265  }
266 
267  const QSet<QetShapeItem *> shapes_set = m_shapes;
268  for(QetShapeItem *shape : shapes_set) {
269  if (!shape->isMovable()) {
270  m_shapes.remove(shape);
271  ++count_;
272  }
273  }
274 
275  return count_;
276 }

407  if (filter & Images) for(auto dii : m_images) { if (dii -> isSelected()) ++ count; }
408  if (filter & Shapes) for(auto dsi : m_shapes) { if (dsi -> isSelected()) ++ count; }
409  if (filter & ConductorsToMove) for(auto conductor : m_conductors_to_move) { if (conductor -> isSelected()) ++ count; }
410  if (filter & ConductorsToUpdate) for(auto conductor : m_conductors_to_update) { if (conductor -> isSelected()) ++ count; }
411  if (filter & OtherConductors) for(auto conductor : m_other_conductors) { if (conductor -> isSelected()) ++ count; }
412  if (filter & ElementTextFields) for(auto deti : m_element_texts) { if (deti -> isSelected()) ++ count; }
413  if (filter & TextGroup) for(auto etig : m_texts_groups) { if (etig -> isSelected()) ++ count; }
414  if (filter & Tables) for(auto table : m_tables) { if (table -> isSelected()) ++ count; }
415  }
416  else {
417  if (filter & Elements) count += m_elements.count();
418  if (filter & TextFields) count += m_text_fields.count();
419  if (filter & Images) count += m_images.count();
420  if (filter & Shapes) count += m_shapes.count();
421  if (filter & ConductorsToMove) count += m_conductors_to_move.count();
422  if (filter & ConductorsToUpdate) count += m_conductors_to_update.count();
