112 bool TerminalEditor::setPart(CustomElementPart* new_part) {
113  m_terminals.clear();
114  if (!new_part) {
115  if (m_part) {
116  disconnect(m_part, &PartTerminal::orientationChanged, this, &TerminalEditor::updateForm);
117  }
118  m_part = nullptr;
119  return(true);
120  }
121  if (PartTerminal *part_terminal = static_cast<PartTerminal *>(new_part)) {
122  if(m_part == part_terminal) return true;
123  if (m_part) {
124  disconnect(m_part, &PartTerminal::orientationChanged, this, &TerminalEditor::updateForm);
125  }
126  m_part = part_terminal;
127  updateForm();
128  connect(m_part, &PartTerminal::orientationChanged, this, &TerminalEditor::updateForm);
129  return(true);
130  }
131  return(false);
132 }

134 bool TerminalEditor::setParts(QList<CustomElementPart *> parts) {
135  if (parts.isEmpty()) {
136  m_terminals.clear();
137  if (m_part) {
138  disconnect(m_part, &PartTerminal::orientationChanged, this, &TerminalEditor::updateForm);
139  }
140  m_part = nullptr;
141  return(true);
142  }
143 
144  if (PartTerminal *part_terminal = static_cast<PartTerminal *>(parts.first())) {
145  if (m_part) {
146  disconnect(m_part, &PartTerminal::orientationChanged, this, &TerminalEditor::updateForm);
147  }
148  m_part = part_terminal;
149  m_terminals.clear();
150  m_terminals.append(part_terminal);
151  for (int i=1; i < parts.length(); i++) {
152  m_terminals.append(static_cast<PartTerminal*>(parts[i]));
153  }
154  updateForm();
155  connect(m_part, &PartTerminal::orientationChanged, this, &TerminalEditor::updateForm);
156  return(true);
157  }
158  return(false);
159 }

177 void TerminalEditor::updateTerminalO() {
178  if (m_locked) return;
179  m_locked = true;
180  QVariant var(orientation -> itemData(orientation -> currentIndex()));
181 
182  for (int i=0; i < m_terminals.length(); i++) {
183  PartTerminal* term = m_terminals[i];
184  if (var != term->property("orientation"))
185  {
186  QPropertyUndoCommand *undo = new QPropertyUndoCommand(term, "orientation", term->property("orientation"), var);
187  undo->setText(tr("Modifier l'orientation d'une borne"));
188  undoStack().push(undo);
189  }
190  }
191  m_locked = false;
192 }
