59 int BOMExportDialog::exec()

72  // if file already exist -> delete it

75  QMessageBox::critical(this, tr("Erreur"),
76  tr("Impossible de remplacer le fichier!\n\n")+

79  }

95 QString BOMExportDialog::getBom()
96 {
97  m_project->dataBase()->updateDB();
98  auto query_ = m_project->dataBase()->newQuery(m_query_widget->queryStr());
99  QString return_string;
100 
101  if (!query_.exec()) {
102  qDebug() << "BOMExportDialog::getBom : query errir : " << query_.lastError();
103  }
104  else
105  {
106  //HEADERS
107  if (ui->m_include_headers)
108  {
109  auto record_ = query_.record();
110  QStringList header_name;
111  for (auto i=0 ; i<record_.count() ; ++i)
112  {
113  auto field_name = record_.fieldName(i);
114 
115  qDebug() << "field name = " << field_name;
116  if (field_name == "position") {
117  header_name << tr("Position");
118  } else if (field_name == "diagram_position") {
119  header_name << tr("Position du folio");
120  } else if (field_name == "designation_qty") {
121  header_name << tr("Quantité numéro d'article", "Special field with name : designation quantity");
122  } else {
123  header_name << QETApp::elementTranslatedInfoKey(field_name);
124  if (header_name.isEmpty()) {
125  header_name << QETApp::diagramTranslatedInfoKey(field_name);
126  }
127  if (header_name.isEmpty()) {
128  header_name << field_name;
129  }
130  }
131 
132  }
133  return_string = header_name.join(";") + "\n";
134  }
135 
136  //ROWS
137  while (query_.next())
138  {
139  auto i=0;
140  QStringList values;
141  while (query_.value(i).isValid())
142  {
143  auto date = query_.value(i).toDate();
144  if (!date.isNull()) {
145  values << QLocale::system().toString(query_.value(i).toDate(), QLocale::ShortFormat);
146  } else {
147  values << query_.value(i).toString();
148  }
149  ++i;
150  }
151 
152  return_string += values.join(";") + "\n";
153  values.clear();
154  }
155  }
156 
157  qDebug() << return_string;
158  return return_string;
159 }
