175 bool DiagramEventAddElement::buildElement()
176 {
177  ElementsLocation import_loc = m_diagram->project()->importElement(m_location);
178  if (import_loc.exist()) {
179  m_integrate_path = import_loc.projectCollectionPath();
180  }
181  else {
182  qDebug() << "DiagramView::addDroppedElement : Impossible d'ajouter l'element.";
183  return false;
184  }
185 
186  int state;
187  ElementsLocation loc(m_integrate_path);
188  m_element = ElementFactory::Instance() -> createElement(loc, nullptr, &state);
189  //The creation of element failed, we delete it
190  if (state) {
191  delete m_element;
192  return(false);
193  }
194  //Everything is good
195  return true;
196 }

203 void DiagramEventAddElement::addElement()
204 {
205  int state;
206  Element *element;
207  if (m_integrate_path.isEmpty())
208  element = ElementFactory::Instance() -> createElement(m_location, nullptr, &state);
209  else
210  element = ElementFactory::Instance() -> createElement(ElementsLocation(m_integrate_path), nullptr, &state);
211 
212  //Build failed
213  if (state)
214  {
215  delete element;
216  return;
217  }
218 
219  //We must add item to scene (even if addItemCommand do this)
220  //for create the autoconnection below
221  element -> setPos(m_element->pos());
222  element -> setRotation(m_element -> rotation());
223  m_diagram -> addItem(element);
224 
225  //The element is dropped by the user, the dynamic text field stored in m_converted_text_from_xml_description
226  //can be moved to m_dynamic_text_list, because we are sure fromXml will be not called.
227  element->m_dynamic_text_list.append(element->m_converted_text_from_xml_description.keys());
228  element->m_converted_text_from_xml_description.clear();
229 
230  QUndoCommand *undo_object = new QUndoCommand(tr("Ajouter %1").arg(element->name()));
231  new AddItemCommand<Element *>(element, m_diagram, m_element -> pos(), undo_object);
232 
233  while (!element -> AlignedFreeTerminals().isEmpty() && m_diagram -> project() -> autoConductor())
234  {
235  QPair <Terminal *, Terminal *> pair = element -> AlignedFreeTerminals().takeFirst();
236 
237  Conductor *conductor = new Conductor(pair.first, pair.second);
238  new AddItemCommand<Conductor *>(conductor, m_diagram, QPointF(), undo_object);
239 
240  //Autonum the new conductor, the undo command associated for this, have for parent undo_object
241  ConductorAutoNumerotation can (conductor, m_diagram, undo_object);
242  can.numerate();
243  if (m_diagram->freezeNewConductors() || m_diagram->project()->isFreezeNewConductors()) {
244  conductor->setFreezeLabel(true);
245  }
246  }
247 
248  m_diagram -> undoStack().push(undo_object);
249  element->setUpFormula();
250  element->freezeNewAddedElement();
251 }
