155 void PartDynamicTextField::fromXml(const QDomElement &dom_elmt) {

167

225 void PartDynamicTextField::fromTextFieldXml(const QDomElement &dom_element)
226 {
227  if(dom_element.tagName() != "input")
228  return;
229 
230  setFont(QETApp::diagramTextsFont(dom_element.attribute("size", QString::number(9)).toInt()));
231 
232  if(dom_element.attribute("tagg", "none") == "none") {
233  setTextFrom(DynamicElementTextItem::UserText);
234  setText(dom_element.attribute("text", "_"));
235  }
236  else {
237  setTextFrom(DynamicElementTextItem::ElementInfo);
238  setInfoName(dom_element.attribute("tagg", "label"));
239  }
240 
241  QGraphicsTextItem::setRotation(dom_element.attribute("rotation", "0").toDouble());
242 
243  //the origin transformation point of PartDynamicTextField is the top left corner, no matter the font size
244  //The origin transformation point of PartTextField is the middle of left edge, and so by definition, change with the size of the font
245  //We need to use a QTransform to find the pos of this text from the saved pos of text item
246  QTransform transform;
247  //First make the rotation
248  transform.rotate(dom_element.attribute("rotation", "0").toDouble());
249  QPointF pos = transform.map(QPointF(0, -boundingRect().height()/2));
250  transform.reset();
251  //Second translate to the pos
252  transform.translate(
253  dom_element.attribute("x", QString::number(0)).toDouble(),
254  dom_element.attribute("y", QString::number(0)).toDouble()
255  );
256  QGraphicsTextItem::setPos(transform.map(pos));
257 }

486 void PartDynamicTextField::paint(QPainter *painter, const QStyleOptionGraphicsItem *option, QWidget *widget) {
487  QGraphicsTextItem::paint(painter, option, widget);
488 
489  if (m_frame) {
490  painter -> save();
491  painter -> setFont(this -> font());
492 
493  //Adjust the thickness according to the font size,
494  qreal w=0.3;
495  if(this -> font().pointSize() >= 5) {
496  w = this -> font().pointSizeF()*0.1;
497  if(w > 2.5)
498  w = 2.5;
499  }
500 
501  QPen pen;
502  pen.setColor(color());
503  pen.setWidthF(w);
504  painter -> setPen(pen);
505  painter -> setRenderHint(QPainter::Antialiasing);
506 
507  //Get the bounding rectangle of the text
508  QSizeF size = document() -> size();
509  size.setWidth(document() -> idealWidth());
510  //Remove the margin. Size is exactly the bounding rect of the text
511  size.rheight() -= document() -> documentMargin()*2;
512  size.rwidth() -= document() -> documentMargin()*2;
513  //Add a little margin only for a better visual;
514  size.rheight() += 2;
515  size.rwidth() += 2;
516 
517  //The pos of the rect
518  QPointF pos = boundingRect().center();
519  pos.rx() -= size.width()/2;
520  pos.ry() -= size.height()/2;
521 
522  //Adjust the rounding of the rectangle according to the size of the font
523  qreal ro = this -> font().pointSizeF()/3;
524  painter -> drawRoundedRect(QRectF(pos, size), ro, ro);
525 
526  painter -> restore();
527  }
528 }

550 void PartDynamicTextField::finishAlignment() {
551  if(m_block_alignment)
552  return;
553 
554  QTransform transform;
555  transform.rotate(this -> rotation());
556  qreal x,xa, y,ya;
557  x=xa=0;
558  y=ya=0;
559 
560  if(m_alignment &Qt::AlignRight) {
561  x = m_alignment_rect.right();
562  xa = boundingRect().right();
563  }
564  else if(m_alignment &Qt::AlignHCenter) {
565  x = m_alignment_rect.center().x();
566  xa = boundingRect().center().x();
567  }
568 
569  if(m_alignment &Qt::AlignBottom) {
570  y = m_alignment_rect.bottom();
571  ya = boundingRect().bottom();
572  }
573  else if(m_alignment &Qt::AlignVCenter) {
574  y = m_alignment_rect.center().y();
575  ya = boundingRect().center().y();
576  }
577 
578  QPointF p = transform.map(QPointF(x,y));
579  QPointF pa = transform.map(QPointF(xa,ya));
580  QPointF diff = pa-p;
581 
582  setPos(this -> pos() - diff);
583 }
