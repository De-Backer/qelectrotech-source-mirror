112 bool ElementsCollectionModel::canDropMimeData(const QMimeData *data,
113  Qt::DropAction action,
114  int row,
115  int column,
116  const QModelIndex &parent) const
117 {
118  if (!(QStandardItemModel::canDropMimeData(data,
119  action,
120  row,
121  column,
122  parent) && parent.isValid()))
123  return false;
124 
125  QStandardItem *qsi = itemFromIndex(parent.QModelIndex::model()->index(
126  row, column));
127  if (!qsi)
128  qsi = itemFromIndex(parent);
129 
130  //Drop in the common collection is forbiden
131  if (qsi->type() == FileElementCollectionItem::Type)
132  if (static_cast<FileElementCollectionItem *>(qsi)->isCommonCollection())
133  return false;
134 
135  ElementCollectionItem *eci = static_cast<ElementCollectionItem *>(qsi);
136 
137  if (data->hasFormat("application/x-qet-element-uri")
138  || data->hasFormat("application/x-qet-category-uri"))
139  {
140  //Return false if user try to drop a item from a folder to the same folder
141  ElementsLocation drop_location(data->text());
142  for (int i=0 ; i<eci->rowCount() ; i++)
143  if (static_cast<ElementCollectionItem *>(eci->child(i))->collectionPath()
144  == drop_location.collectionPath())
145  return false;
146 
147  return true;
148  }
149  else
150  return false;
151 }

163 bool ElementsCollectionModel::dropMimeData(const QMimeData *data,
164  Qt::DropAction action,
165  int row,
166  int column,
167  const QModelIndex &parent)
168 {
169  Q_UNUSED(action)
170  QStandardItem *qsi = itemFromIndex(
171  parent.QModelIndex::model()->index(row, column));
172  if (!qsi)
173  qsi = itemFromIndex(parent);
174 
175  if (qsi->type() == FileElementCollectionItem::Type)
176  {
177  FileElementCollectionItem *feci = static_cast<FileElementCollectionItem *>(qsi);
178 
179  if (feci->isCommonCollection())
180  return false;
181 
182  if (feci->isElement() && feci->parent() && feci->parent()->type()
183  == FileElementCollectionItem::Type)
184  feci = static_cast<FileElementCollectionItem *>(feci->parent());
185 
186  ElementCollectionHandler ech;
187 
188  ElementsLocation source(data->text());
189  ElementsLocation destination(feci->fileSystemPath());
190  ElementsLocation location = ech.copy(source, destination);
191 
192  if (location.exist())
193  {
194  //If feci have a child with the same path of location,
195  //we remove the existing child befor insert new child
196  for (int i=0 ; i<feci->rowCount() ; i++) {
197  if (static_cast<FileElementCollectionItem *>(
198  feci->child(i))->collectionPath()
199  == location.collectionPath())
200  feci->removeRow(i);
201  }
202  feci->addChildAtPath(location.fileName());
203  return true;
204  }
205 
206  return false;
207  }
208  else if (qsi->type() == XmlProjectElementCollectionItem::Type) {
209  XmlProjectElementCollectionItem *xpeci = static_cast<XmlProjectElementCollectionItem*>(qsi);
210 
211  if (xpeci->isElement() && xpeci->parent() && xpeci->parent()->type()
212  == XmlProjectElementCollectionItem::Type)
213  xpeci = static_cast<XmlProjectElementCollectionItem *>(xpeci->parent());
214 
215  /* before do the copy, we get all collection path of xpeci child,
216  * for remove it
217  * if the copied item have the same path of an existing child.
218  * We can't do this after the copy,
219  * because at the copy
220  * if the xml collection have a DomElement with the same path,
221  * he was removed before the new xml DomElement is inserted
222  * So the existing child of this will return a null QString
223  * when call collectionPath(), because the item
224  * doesn't exist anymore in the xml collection.
225  */
226  QList <QString> child_path_list;
227  for (int i=0 ; i<xpeci->rowCount() ; i++)
228  child_path_list.append(
229  static_cast<XmlProjectElementCollectionItem *>(
230  xpeci->child(i, 0))->collectionPath());
231 
232  ElementCollectionHandler ech;
233 
234  ElementsLocation source(data->text());
235  ElementsLocation destination(xpeci->collectionPath());
236  ElementsLocation location = ech.copy(source, destination);
237 
238  return location.exist();
239  }
240 
241  return false;
242 }

336 void ElementsCollectionModel::addLocation(const ElementsLocation& location)
337 {
338  QModelIndex index = indexFromLocation(location);
339  if (index.isValid())
340  return;
341 
342  ElementCollectionItem *last_item = nullptr;
343  QString collection_name;
344 
345  if (location.isProject()) {
346  QETProject *project = location.project();
347 
348  if (project) {
349  XmlProjectElementCollectionItem *xpeci =
350  m_project_hash.value(project);
351 
352  last_item = xpeci->lastItemForPath(
353  location.collectionPath(false),
354  collection_name);
355  }
356  }
357  else if (location.isCustomCollection()) {
358  QList <ElementCollectionItem *> child_list;
359 
360  for (int i=0 ; i<rowCount() ; i++)
361  child_list.append(static_cast<ElementCollectionItem *>(item(i)));
362 
363  foreach(ElementCollectionItem *eci, child_list) {
364 
365  if (eci->type() == FileElementCollectionItem::Type) {
366  FileElementCollectionItem *feci =
367  static_cast<FileElementCollectionItem *>(eci);
368 
369  if (feci->isCustomCollection()) {
370  last_item = feci->lastItemForPath(
371  location.collectionPath(false),
372  collection_name);
373  if(last_item)
374  break;
375  }
376  }
377  }
378  }
379 
380  if (last_item)
381  last_item->addChildAtPath(collection_name);

543 QModelIndex ElementsCollectionModel::indexFromLocation(
544  const ElementsLocation &location)
545 {
546  QList <ElementCollectionItem *> child_list;
547 
548  for (int i=0 ; i<rowCount() ; i++)
549  child_list.append(static_cast<ElementCollectionItem *>(item(i)));
550 
551  foreach(ElementCollectionItem *eci, child_list) {
552 
553  ElementCollectionItem *match_eci = nullptr;
554 
555  if (eci->type() == FileElementCollectionItem::Type) {
556  if (FileElementCollectionItem *feci = static_cast<FileElementCollectionItem *>(eci)) {
557  if ( (location.isCommonCollection() && feci->isCommonCollection()) ||
558  (location.isCustomCollection() && !feci->isCommonCollection()) ) {
559  match_eci = feci->itemAtPath(location.collectionPath(false));
560  }
561  }
562  }
563  else if (eci->type() == XmlProjectElementCollectionItem::Type) {
564  if (XmlProjectElementCollectionItem *xpeci = static_cast<XmlProjectElementCollectionItem *>(eci)) {
565  match_eci = xpeci->itemAtPath(location.collectionPath(false));
566  }
567  }
568 
569  if (match_eci)
570  return indexFromItem(match_eci);
571  }
572 
573  return QModelIndex();
574 }
