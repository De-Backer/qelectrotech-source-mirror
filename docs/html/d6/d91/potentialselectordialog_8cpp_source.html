36 class NewConductorPotentialSelector : public AbstractPotentialSelector
37 {
38  public:
39  NewConductorPotentialSelector(Conductor *conductor) :
40  m_is_valid(false)
41  {
42  Terminal *terminal_1 = conductor->terminal1;
43  Terminal *terminal_2 = conductor->terminal2;
44  //We temporarily remove the conductor of his two terminals,
45  //to get the two existing potential
46  terminal_1->removeConductor(conductor);
47  terminal_2->removeConductor(conductor);
48 
49  getPotential(terminal_1,
50  m_seq_num_1,
51  m_conductor_number_1,
52  m_properties_list_1,
53  m_conductors_list_1);
54  getPotential(terminal_2,
55  m_seq_num_2,
56  m_conductor_number_2,
57  m_properties_list_2,
58  m_conductors_list_2);
59 
60  //There isn't a potential at terminal 1 or 2.
61  if (m_conductor_number_1 == 0
62  && m_conductor_number_2 == 0) return;
63 
64  //Re-add conductor to his terminals.
65  terminal_1->addConductor(conductor);
66  terminal_2->addConductor(conductor);
67  m_is_valid = true;
68  }

82  void getPotential(
83  Terminal *terminal,
84  autonum::sequentialNumbers &seq_num ,
85  int &number, QList<ConductorProperties> &properties_list,
86  QList<Conductor*> &c_list)
87  {
88  Conductor *conductor_in_potential = nullptr;
89 
90  //Terminal have conductor linked to, we get it.
91  if (!terminal->conductors().isEmpty())
92  conductor_in_potential = terminal->conductors().first();
93 
94  //Terminal haven't got a conductor, but if parent element is a folio report or a terminal element, we search a potential
95  //through the report or in another terminal of the terminal element.
96  else if (terminal->parentElement()->linkType() & (Element::Terminale | Element::AllReport))
97  {
98  Element *elmt_ = terminal->parentElement();
99 
100  if ((elmt_->linkType() & Element::Terminale) && !elmt_->terminals().isEmpty())
101  {
102  foreach(Terminal *t, elmt_->terminals())
103  {
104  if (t->conductors().isEmpty()) continue;
105  conductor_in_potential = t->conductors().first();
106  break;
107  }
108  }
109  else if ((elmt_->linkType() & Element::AllReport) && !elmt_->isFree())
110  {
111  Element *other_report = elmt_->linkedElements().first();
112  if (other_report->terminals().isEmpty()) return;
113  Terminal *t = other_report->terminals().first();
114  if (t->conductors().isEmpty()) return;
115  conductor_in_potential= t->conductors().first();
116  }
117  }
118 
119  if (!conductor_in_potential)
120  return;
121  seq_num = conductor_in_potential->sequenceNum();
122  //We add +1 because conductor_in_potential isn't count by relatedPotentialConductors
123  number = conductor_in_potential->relatedPotentialConductors().size()+1;
124 
125  c_list = conductor_in_potential->relatedPotentialConductors().values();
126  c_list.append(conductor_in_potential);
127  foreach(Conductor *c, c_list)
128  properties_list.append(c->properties());
129  }
130 
131  ~NewConductorPotentialSelector() override {}
132 
133  private :
134  bool m_is_valid;
135 };

143  public:
144  LinkReportPotentialSelector(Element *report) :
145  m_is_valid(false)
146  {
147  if ((report->linkType() & Element::AllReport) && !report->isFree())
148  {
149  //We temporarily unlink report to get the two existing potential
150  Element *other_report = report->linkedElements().first();
151  report->unlinkAllElements();
152 
153  if (report->conductors().isEmpty() || other_report->conductors().isEmpty())
154  return;
155 
156  m_conductor_number_1 = report->conductors().first()->relatedPotentialConductors().size() + 1;
157  m_seq_num_1 = report->conductors().first()->sequenceNum();
158  m_conductors_list_1.append(report->conductors().first()->relatedPotentialConductors().values());
159  m_conductors_list_1.append(report->conductors().first());
160  foreach(Conductor *c, m_conductors_list_1)
161  m_properties_list_1 << c->properties();
162 
163  m_conductor_number_2 = other_report->conductors().first()->relatedPotentialConductors().size() + 1;
164  m_seq_num_2 = other_report->conductors().first()->sequenceNum();
165  m_conductors_list_2.append(other_report->conductors().first()->relatedPotentialConductors().values());
166  m_conductors_list_2.append(other_report->conductors().first());
167  foreach(Conductor *c, m_conductors_list_2)
168  m_properties_list_2 << c->properties();
169 
170  //We relink the report
171  report->linkToElement(other_report);
172  m_is_valid = true;
173  }
174  }

386 void PotentialSelectorDialog::on_buttonBox_accepted()
387 {
388  if (!m_potential_selector->isValid())
389  return;
390 
391  QUndoCommand *undo = nullptr;
392  if (m_parent_undo)
393  undo = m_parent_undo;
394  else
395  undo = new QUndoCommand(
396  tr("Modifier les propriétés de plusieurs conducteurs",
397  "undo caption"));
398 
399  Diagram * diagram = nullptr;
400 
401  if (m_report)
402  {
403  if ((m_report->linkType() & Element::AllReport)
404  && !m_report->isFree())
405  {
406  if (m_report->diagram())
407  diagram = m_report->diagram();
408 
409  QVariant old_value, new_value;
410  QVariant old_seq, new_seq;
411  new_seq.setValue(m_sequential_num);
412 
413  //Set the new properties for each conductors of the new potential
414  foreach(Conductor *cond, m_conductors_to_change)
415  {
416  ConductorProperties new_properties = cond->properties();
417  new_properties.applyForEqualAttributes(m_properties_list);
418  old_value.setValue(cond->properties());
419  new_value.setValue(new_properties);
420  old_seq.setValue(cond->sequenceNum());
421  new QPropertyUndoCommand(cond,
422  "sequenceNum",
423  old_seq,
424  new_seq,
425  undo);
426  new QPropertyUndoCommand(cond,
427  "properties",
428  old_value,
429  new_value,
430  undo);
431  }
432 
433  //Check if formula of the new potential
434  // have incompatible variable with folio report
435  QRegularExpression rx ("%sequf_|%seqtf_|%seqhf_|%id|%F|%M|%LM");
436  foreach(ConductorProperties cp, m_properties_list)
437  {
438  if (cp.m_formula.contains(rx))
439  {
440  QStringList forbidden_str;
441  forbidden_str << "%sequf_"
442  << "%seqtf_"
443  << "%seqhf_"
444  << "%id"
445  << "%F"
446  << "%M"
447  << "%LM";
448 
449  QString text(tr("La formule du nouveau potentiel contient des variables incompatibles avec les reports de folio.\n"
450  "Veuillez saisir une formule compatible pour ce potentiel.\n"
451  "Les variables suivantes sont incompatibles :\n"
452  "%sequf_ %seqtf_ %seqhf_ %id %F %M %LM"));
453  FormulaAssistantDialog fag(this);
454  fag.setForbiddenVariables(forbidden_str);
455  fag.setText(text);
456  fag.setFormula(cp.m_formula);
457  fag.exec();
458 
459  QString new_formula = fag.formula();
460  QSet <Conductor *> c_list = m_report->conductors().first()->relatedPotentialConductors();
461  c_list.insert(m_report->conductors().first());
462  foreach(Conductor *cond, c_list)
463  {
464  old_value.setValue(cond->properties());
465  ConductorProperties new_properties = cond->properties();
466  new_properties.m_formula = new_formula;
467  new_value.setValue(new_properties);
468  new QPropertyUndoCommand(cond,
469  "properties",
470  old_value,
471  new_value,
472  undo);
473  }
474 
475  break;
476  }
477  }
478  }
479  }
480 
481  else if (m_conductor)
482  {
483  if (m_conductor->diagram())
484  diagram = m_conductor->diagram();
485 
486  ConductorProperties new_properties = m_conductor->properties();
487  new_properties.applyForEqualAttributes(m_properties_list);
488 
489  QVariant old_value, new_value;
490  old_value.setValue(m_conductor->properties());
491  new_value.setValue(new_properties);
492 
493  QVariant old_seq, new_seq;
494  old_seq.setValue(m_conductor->sequenceNum());
495  new_seq.setValue(m_sequential_num);
496 
497  new QPropertyUndoCommand(m_conductor,
498  "sequenceNum",
499  old_seq,
500  new_seq,
501  undo);
502  new QPropertyUndoCommand(m_conductor,
503  "properties",
504  old_value,
505  new_value,
506  undo);
507 
508  //Set the new properties for each conductors of the new potential
509  foreach(Conductor *cond, m_conductor->relatedPotentialConductors())
510  {
511  new_properties = cond->properties();
512  new_properties.applyForEqualAttributes(m_properties_list);
513  old_value.setValue(cond->properties());
514  new_value.setValue(new_properties);
515  old_seq.setValue(cond->sequenceNum());
516  new QPropertyUndoCommand(cond,
517  "sequenceNum",
518  old_seq,

531  if (m_parent_undo)
532  return;
533  //There isn't a parent, we push the undo command to diagram undo stack.
534  if (diagram)
535  diagram->undoStack().push(undo);
