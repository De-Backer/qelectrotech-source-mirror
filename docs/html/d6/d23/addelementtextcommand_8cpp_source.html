165 void AddTextsGroupCommand::redo()
166 {
167  if(!m_element)
168  return;
169  if(m_first_undo)
170  {
171  m_group = m_element.data()->addTextGroup(m_name);
172  if(!m_dom_element.isNull())
173  {
174  m_group.data()->fromXml(m_dom_element);
175  /* We get the list of texts (if any)
176  * because when undo is called,
177  * all child text will be removed
178  * from the group, and reparented to m_elemeny.
179  * Then the next time redo is called,
180  * the texts will be added to the group
181  */
182  m_deti_list = m_group.data()->texts();
183  m_group.data()->updateAlignment();
184  }
185  else
186  {
187  for(DynamicElementTextItem *deti : m_deti_list)
188  m_element.data()->addTextToGroup(
189  deti,
190  m_group.data());
191  }
192  m_first_undo = false;
193  }
194  else if(m_group)
195  {
196  m_element.data()->addTextGroup(m_group.data());
197  for(DynamicElementTextItem *deti : m_deti_list)
198  m_element.data()->addTextToGroup(deti, m_group.data());
199  }
200 }
