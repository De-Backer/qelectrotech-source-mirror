209  // we have to create new QSvgRenderer objects from the data
210  // (no copy constructor)
211  foreach (QString logo_key, vector_logos_.keys()) {
212  copy -> vector_logos_[logo_key] =
213  new QSvgRenderer(data_logos_[logo_key]);

216  copy -> data_logos_ = data_logos_;
217  copy -> storage_logos_ = storage_logos_;
218  copy -> type_logos_ = type_logos_;
219  copy -> rows_heights_ = rows_heights_;
220  copy -> columns_width_ = columns_width_;
221 
222  // copy cells basically
223  copy -> cells_ = cells_;
224  for (int j = 0 ; j < rows_heights_.count() ; ++ j) {
225  for (int i = 0 ; i < columns_width_.count() ; ++ i) {
226  copy -> cells_[i][j] = copy -> createCell(cells_[i][j]);
227  }
228  }
229 
230  // ensure the copy has no spanner_cell attribute pointing to a cell
231  // from the original object
232  for (int j = 0 ; j < rows_heights_.count() ; ++ j) {
233  for (int i = 0 ; i < columns_width_.count() ; ++ i) {
234  TitleBlockCell *current_cell = copy -> cells_[i][j];
235  if (TitleBlockCell *original_cell
236  = current_cell -> spanner_cell) {
237  int original_cell_row = original_cell -> num_row;
238  int original_cell_col = original_cell -> num_col;
239  TitleBlockCell *copy_cell =
240  copy -> cells_[original_cell_col]
241  [original_cell_row];
242  current_cell -> spanner_cell = copy_cell;
243  }
244  }
245  }
246

648 bool TitleBlockTemplate::checkCell(const QDomElement &xml_element,
649  TitleBlockCell **titleblock_cell_ptr) {
650  int col_count = columns_width_.count(),
651  row_count = rows_heights_.count();
652 
653 #ifdef TITLEBLOCK_TEMPLATE_DEBUG
654  qDebug() << Q_FUNC_INFO << "begin" << row_count << col_count;
655 #endif
656 
657  int row_num, col_num, row_span, col_span;
658  row_num = col_num = -1;
659  row_span = col_span = 0;
660 
661  // parse the row and col attributes
662  if (!QET::attributeIsAnInteger(xml_element, "row", &row_num)
663  || row_num < 0
664  || row_num >= row_count) {
665  return(false);
666  }
667  if (!QET::attributeIsAnInteger(xml_element, "col", &col_num)
668  || col_num < 0
669  || col_num >= col_count) {
670  return(false);
671  }
672 
673  // check whether the target cell can be used or not
674 #ifdef TITLEBLOCK_TEMPLATE_DEBUG
675  qDebug() << Q_FUNC_INFO << "cell access" << col_num << row_num;
676 #endif
677  TitleBlockCell *cell_ptr = cells_[col_num][row_num];
678  if (cell_ptr -> cell_type != TitleBlockCell::EmptyCell
679  || cell_ptr -> spanner_cell) {
680  return(false);
681  }
682  // ensure the num_row and num_col attributes are alright
683  cell_ptr -> num_row = row_num;
684  cell_ptr -> num_col = col_num;
685 
686  // parse the rowspan and colspan attributes
687  if (QET::attributeIsAnInteger(xml_element, "rowspan", &row_span)
688  && row_span > 0) {
689  cell_ptr -> row_span = row_span;
690  }
691 
692  if (QET::attributeIsAnInteger(xml_element, "colspan", &col_span)
693  && col_span > 0) {
694  cell_ptr -> col_span = col_span;
695  }
696  // these attributes are stored "as is" -- whether they can be applied
697  // directly or must be restricted will be checked later
698 
699  if (titleblock_cell_ptr) *titleblock_cell_ptr = cell_ptr;
700  return(true);
701 }

823 QList<int> TitleBlockTemplate::columnsWidth(int total_width) const {
824  if (total_width < 0) return(QList<int>());
825 
826  // we first iter to determine the absolute and total-width-related widths
827  QVector<int> final_widths(columns_width_.count());
828  int abs_widths_sum = 0, rel_widths_sum = 0;
829  QList<int> relative_columns;
830 
831  for (int i = 0 ; i < columns_width_.count() ; ++ i) {
832  TitleBlockDimension icd = columns_width_.at(i);
833  if (icd.type == QET::Absolute) {
834  abs_widths_sum += icd.value;
835  final_widths[i] = icd.value;
836  } else if (icd.type == QET::RelativeToTotalLength) {
837  int abs_value = qRound(total_width * icd.value / 100.0);
838  relative_columns << i;
839  abs_widths_sum += abs_value;
840  final_widths[i] = abs_value;
841  }
842  }
843 
844  // we can now deduce the remaining width
845  int remaining_width = total_width - abs_widths_sum;
846 
847  // we do a second iteration to build the final widths list
848  for (int i = 0 ; i < columns_width_.count() ; ++ i) {
849  TitleBlockDimension icd = columns_width_.at(i);
850  if (icd.type == QET::RelativeToRemainingLength) {
851  final_widths[i] =
852  qRound(remaining_width * icd.value / 100.0);
853  relative_columns << i;
854  rel_widths_sum += final_widths[i];
855  }
856  }
857 
858  // Have we computed widths from percentage for relative columns?
859  if (relative_columns.count()) {
860  // Due to the rounding process,
861  // we may get a slight difference between the
862  // sum of the columns widths and the total width.
863  int difference = total_width - abs_widths_sum - rel_widths_sum;
864 
865  if (difference) {
866  // We consider we should not attempt to compensate
867  // this difference if it is under
868  // relative_columns_count * 0.5
869  // (which means that each percent-based
870  // columns can "bring" up to 0.5px of difference).
871  qreal max_acceptable_difference =
