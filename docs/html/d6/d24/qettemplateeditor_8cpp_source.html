107 bool QETTitleBlockTemplateEditor::canClose() {
108  if (undo_stack_ -> isClean()) return(true);
109  // ask the user whether he wants to save the current template
110  QMessageBox::StandardButton answer = QET::QetMessageBox::question(
111  this,
112  tr("Enregistrer le modèle en cours ?", "dialog title"),
113  QString(
114  tr(
115  "Voulez-vous enregistrer le modèle %1 ?",
116  "dialog content - %1 is a title block template name"
117  )
118  ).arg(location_.name()),
119  QMessageBox::Yes | QMessageBox::No | QMessageBox::Cancel,
120  QMessageBox::Cancel
121  );
122  bool result;
123  switch(answer) {
124  case QMessageBox::Cancel: result = false; break; // the user hits Cancel or closes the dialog: abort the closing
125  case QMessageBox::Yes: result = save(); break; // the user hits Yes: la reussite depend de l'enregistrement
126  default: result = true; // the user hits no: the editor can be closed
127  }
128  return(result);
129 }

187 bool QETTitleBlockTemplateEditor::edit(const TitleBlockTemplateLocation &location) {
188  // the template name may be empty to create a new one
189  const TitleBlockTemplate *tb_template_orig;
190  if (location.name().isEmpty()) {
191  // loads the default title block template provided by the application
192  // it will be used as a start point to design the title block
193  tb_template_orig = QETApp::defaultTitleBlockTemplate();
194  } else {
195  tb_template_orig = location.getTemplate();
196  }
197  if (!tb_template_orig) {
199  return(false);
200  }
201 
202  opened_from_file_ = false;
203  location_ = location;
204  setReadOnly(location.isReadOnly());
205  editCopyOf(tb_template_orig);
206  return(true);
207 }

215 bool QETTitleBlockTemplateEditor::edit(QETProject *project, const QString &template_name)
216 {
217  // we require a project we will rattach templates to
218  if (!project) return(false);
219 
220  // the template name may be empty to create a new one
221  const TitleBlockTemplate *tb_template_orig;
222  if (template_name.isEmpty())
223  {
224  // loads the default title block template provided by the application
225  // it will be used as a start point to design the title block
226  tb_template_orig = QETApp::defaultTitleBlockTemplate();
227  }
228  else
229  {
230  tb_template_orig = project->embeddedTitleBlockTemplatesCollection()->getTemplate(template_name);
231  }
232 
233  if (!tb_template_orig) {
235  return(false);
236  }
237 
238  opened_from_file_ = false;
239  location_.setParentCollection(project -> embeddedTitleBlockTemplatesCollection());
240  location_.setName(template_name);
241  setReadOnly(project -> isReadOnly());
242  return(editCopyOf(tb_template_orig));
243 }
