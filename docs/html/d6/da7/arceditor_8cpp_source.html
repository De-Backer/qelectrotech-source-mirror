104 bool ArcEditor::setPart(CustomElementPart *new_part)
105 {
106  if (!new_part)
107  {
108  if (part)
109  disconnectChangeConnections();
110 
111  part = nullptr;
112  return(true);
113  }
114 
115  if (PartArc *part_arc = dynamic_cast<PartArc *>(new_part))
116  {
117  if (part == part_arc) return true;
118  if (part)
119  disconnectChangeConnections();
120  part = part_arc;
121  updateForm();
122  setUpChangeConnections();
123  return(true);
124  }
125 
126  return(false);
127 }

155 void ArcEditor::updateArcS()
156 {
157  if (m_locked) return;
158  m_locked = true;
159  double value = start_angle->value() * 16;
160 
161  for (auto part: style_->currentParts()) {
162 
163  PartArc* arc = static_cast<PartArc*>(part);
164 
165  if (value != arc->property("startAngle"))
166  {
167  QPropertyUndoCommand *undo= new QPropertyUndoCommand(arc, "startAngle", arc->property("startAngle"), value);
168  undo->setText("Modifier l'angle de depart d'un arc");
169  undo->enableAnimation();
170  elementScene()->undoStack().push(undo);
171  }
172  }
173 
174  m_locked = false;
175 }

306 void ArcEditor::updateForm()
307 {
308  if (!part) return;
309  activeConnections(false);
310  QRectF rect = part->property("rect").toRectF();
311  x->setValue(part->mapToScene(rect.topLeft()).x() + (rect.width()/2));
312  y->setValue(part->mapToScene(rect.topLeft()).y() + (rect.height()/2));
313  h->setValue(rect.width());
314  v->setValue(rect.height());
315  start_angle->setValue(part->property("startAngle").toInt()/16);
316  angle->setValue(part->property("spanAngle").toInt()/16);
317  activeConnections(true);
318 }
