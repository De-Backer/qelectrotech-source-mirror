97 void CrossRefItem::setUpConnection()
98 {
99  for(const QMetaObject::Connection& c : m_update_connection)
100  disconnect(c);
101 
102  m_update_connection.clear();
103  QETProject *project = m_element->diagram()->project();
104  bool set=false;
105 
106  if(m_properties.snapTo() == XRefProperties::Label && (m_text || m_group)) //Snap to label and parent is a text or a group
107  set=true;
108  else if(m_properties.snapTo() == XRefProperties::Bottom && !m_text && !m_group) //Snap to bottom of element and parent is the element itself
109  {
110  m_update_connection << connect(m_element, SIGNAL(yChanged()), this, SLOT(autoPos()));
111  m_update_connection << connect(m_element, SIGNAL(rotationChanged()), this, SLOT(autoPos()));
112  set=true;
113  }
114 
115  if(set)
116  {
117  m_update_connection << connect(project, &QETProject::projectDiagramsOrderChanged, this, &CrossRefItem::updateLabel);
118  m_update_connection << connect(project, &QETProject::diagramRemoved, this, &CrossRefItem::updateLabel);
119  m_update_connection << connect(m_element, &Element::linkedElementChanged, this, &CrossRefItem::linkedChanged);
120  linkedChanged();
121  updateLabel();
122  }
123 }

149 QString CrossRefItem::elementPositionText(const Element *elmt,
150  const bool &add_prefix) const
151 {
152  XRefProperties xrp =
153  m_element->diagram()->project()->defaultXRefProperties(
154  m_element->kindInformations()["type"].toString());
155  QString formula = xrp.masterLabel();
156  autonum::sequentialNumbers seq;
157  QString txt = autonum::AssignVariables::formulaToLabel(
158  formula,
159  seq, elmt->diagram(),
160  elmt);
161 
162  if (add_prefix)
163  {
164  if (elmt->kindInformations()["type"].toString() == "power")
165  txt.prepend(m_properties.prefix("power"));
166  else if (elmt->kindInformations()["type"].toString().contains("delay"))
167  txt.prepend(m_properties.prefix("delay"));
168  else if (elmt->kindInformations()["state"].toString() == "SW")
169  txt.prepend(m_properties.prefix("switch"));
170  }
171  return txt;
172 }

213  qp.setFont(QETApp::diagramTextsFont(5));
214 
215  //Draw cross or contact, only if master element is linked.
216  if (! m_element->linkedElements().isEmpty())
217  {
218  XRefProperties::DisplayHas dh = m_properties.displayHas();

220  if (dh == XRefProperties::Cross)
221  drawAsCross(qp);

223  drawAsContacts(qp);
224  }
225  qp.end();
226

252 bool CrossRefItem::sceneEvent(QEvent *event)
253 {
254  //By default when a QGraphicsItem is a child of a QGraphicsItemGroup
255  //all events are forwarded to group.
256  //We override it, when this Xref is in a group
257  if(m_group)
258  {
259  switch (event->type())
260  {
261  case QEvent::GraphicsSceneHoverEnter:
262  hoverEnterEvent(static_cast<QGraphicsSceneHoverEvent *>(event));
263  break;
264  case QEvent::GraphicsSceneHoverMove:
265  hoverMoveEvent(static_cast<QGraphicsSceneHoverEvent *>(event));
266  break;
267  case QEvent::GraphicsSceneHoverLeave:
268  hoverLeaveEvent(static_cast<QGraphicsSceneHoverEvent *>(event));
269  break;
270  case QEvent::GraphicsSceneMouseDoubleClick:
271  mouseDoubleClickEvent(static_cast<QGraphicsSceneMouseEvent *>(event));
272  break;
273  default:break;
274  }
275  return true;
276  }
277 
278  return QGraphicsObject::sceneEvent(event);
279 }

337 void CrossRefItem::hoverMoveEvent(QGraphicsSceneHoverEvent *event)
338 {
339  QPointF pos = event->pos();
340 
341  if (m_hovered_contact)
342  {
343  foreach(QRectF rect, m_hovered_contacts_map.values(m_hovered_contact))
344  {
345  //Mouse hover the same rect than previous hover event
346  if (rect.contains(pos))
347  {
348  QGraphicsObject::hoverMoveEvent(event);
349  return;
350  }
351  }
352 
353  //At this point, mouse don't hover previous rect
354  m_hovered_contact = nullptr;
355 
356  foreach (Element *elmt, m_hovered_contacts_map.keys())
357  {
358  foreach(QRectF rect, m_hovered_contacts_map.values(elmt))
359  {
360  //Mouse hover a contact
361  if (rect.contains(pos))
362  {
363  m_hovered_contact = elmt;
364  }
365  }
366  }
367 
368  updateLabel();
369  QGraphicsObject::hoverMoveEvent(event);
370  return;
371  }
372  else
373  {
374  foreach (Element *elmt, m_hovered_contacts_map.keys())
375  {
376  foreach(QRectF rect, m_hovered_contacts_map.values(elmt))
377  {
378  //Mouse hover a contact
379  if (rect.contains(pos))
380  {
381  m_hovered_contact = elmt;
382  updateLabel();
383  QGraphicsObject::hoverMoveEvent(event);
384  return;
385  }
386  }
387  }
388  }

484 void CrossRefItem::setUpCrossBoundingRect(QPainter &painter)
485 {
486  //No need to calcul if nothing is linked
487  if (m_element->isFree()) return;
488 
489  QStringList no_str, nc_str;
490 
491  for (auto elmt : NOElements()) {
492  no_str.append(elementPositionText(elmt, true));
493  }
494  for (auto elmt : NCElements()) {
495  nc_str.append(elementPositionText(elmt, true));
496  }
497 
498 
499  //There is no string to display, we return now
500  if (no_str.isEmpty() && nc_str.isEmpty()) return;
501 
502  //this is the default size of cross ref item
503  QRectF default_bounding(0, 0, 40, header + cross_min_heigth);
504 
505  //Bounding rect of the NO text
506  QRectF no_bounding;
507  for (auto str : no_str)
508  {
509  QRectF bounding = painter.boundingRect(QRectF (), Qt::AlignCenter, str);
510  no_bounding = no_bounding.united(bounding);
511  no_bounding.setHeight(no_bounding.height() + bounding.height());
512  }
513  //Adjust according to the NO
514  if (no_bounding.height() > default_bounding.height() - header)
515  default_bounding.setHeight(no_bounding.height() + header); //adjust the height
516  if (no_bounding.width() > default_bounding.width()/2)
517  default_bounding.setWidth(no_bounding.width()*2); //adjust the width
518 
519  //Bounding rect of the NC text
520  QRectF nc_bounding;
521  for (auto str : nc_str)
522  {
523  QRectF bounding = painter.boundingRect(QRectF (), Qt::AlignCenter, str);
524  nc_bounding = nc_bounding.united(bounding);
525  nc_bounding.setHeight(nc_bounding.height() + bounding.height());
526  }
527  //Adjust according to the NC
528  if (nc_bounding.height() > default_bounding.height() - header)
529  default_bounding.setHeight(nc_bounding.height() + header); //adjust the heigth
530  if (nc_bounding.width() > default_bounding.width()/2)
531  default_bounding.setWidth(nc_bounding.width()*2); //adjust the width
532 
533  //Minor adjustement for better visual
534  default_bounding.adjust(0, 0, 4, 0);
535  m_shape_path.addRect(default_bounding);
536  prepareGeometryChange();
537  m_bounding_rect = default_bounding;
538 }

545 void CrossRefItem::drawAsCross(QPainter &painter)
546 {
547  //calcul the size of the cross
548  setUpCrossBoundingRect(painter);
549  m_hovered_contacts_map.clear();
550 
551  //Bounding rect is empty that mean there's no contact to draw
552  if (boundingRect().isEmpty()) return;
553 
554  //draw the cross
555  QRectF br = boundingRect();
556  painter.drawLine(br.width()/2, 0, br.width()/2, br.height()); //vertical line
557  painter.drawLine(0, header, br.width(), header); //horizontal line
558 
559  //Add the symbolic contacts
560  buildHeaderContact();
561  QPointF p((m_bounding_rect.width()/4) - (m_hdr_no_ctc.width()/2), 0);
562  painter.drawPicture (p, m_hdr_no_ctc);
563  p.setX((m_bounding_rect.width() * 3/4) - (m_hdr_nc_ctc.width()/2));
564  painter.drawPicture (p, m_hdr_nc_ctc);
565 
566  //and fill it
567  fillCrossRef(painter);
568 }

575 void CrossRefItem::drawAsContacts(QPainter &painter)
576 {
577  if (m_element -> isFree())
578  return;
579 
580  m_drawed_contacts = 0;
581  m_hovered_contacts_map.clear();
582  QRectF bounding_rect;
583 
584  //Draw each linked contact
585  foreach (Element *elmt, m_element->linkedElements())
586  {
587  DiagramContext info = elmt->kindInformations();
588 
589  for (int i=0; i<info["number"].toInt(); i++)
590  {
591  int option = 0;
592 
593  QString state = info["state"].toString();
594  if (state == "NO") option = NO;
595  else if (state == "NC") option = NC;
596  else if (state == "SW") option = SW;
597 
598  QString type = info["type"].toString();
599  if (type == "power") option += Power;
600  else if (type == "delayOn") option += DelayOn;
601  else if (type == "delayOff") option += DelayOff;
602  else if (type == "delayOnOff") option += DelayOnOff;
603 
604  QRectF br = drawContact(painter, option, elmt);
605  bounding_rect = bounding_rect.united(br);
606  }
607  }
608 
609  bounding_rect.adjust(-4, -4, 4, 4);
610  prepareGeometryChange();
611  m_bounding_rect = bounding_rect;
612  m_shape_path.addRect(bounding_rect);
613 }

623 QRectF CrossRefItem::drawContact(QPainter &painter, int flags, Element *elmt)
624 {
625  QString str = elementPositionText(elmt);
626  int offset = m_drawed_contacts*10;
627  QRectF bounding_rect;
628 
629  QPen pen = painter.pen();
630  m_hovered_contact == elmt ? pen.setColor(Qt::blue) :pen.setColor(Qt::black);
631  painter.setPen(pen);
632 
633  //Draw NO or NC contact
634  if (flags &NOC)
635  {
636  bounding_rect = QRectF(0, offset, 24, 10);
637 
638  //draw the basic line
639  painter.drawLine(0, offset+6, 8, offset+6);
640  painter.drawLine(16, offset+6, 24, offset+6);
641 
643  /*QFont font = QETApp::diagramTextsFont(4);
644  font.setBold(true);
645  painter.setFont(font);
646  QRectF bt(0, offset, 24, 10);
647  int txt = 10 + m_drawed_contacts;
648  painter.drawText(bt, Qt::AlignLeft|Qt::AlignTop, QString::number(txt));
649  painter.drawText(bt, Qt::AlignRight|Qt::AlignTop, QString::number(txt));
650  painter.setFont(QETApp::diagramTextsFont(5));*/
651 
652  //draw open contact
653  if (flags &NO) {
654  painter.drawLine(8, offset+9, 16, offset+6);
655  }
656  //draw close contact
657  if (flags &NC) {
658  QPointF p1[3] = {
659  QPointF(8, offset+6),
660  QPointF(9, offset+6),
661  QPointF(9, offset+2.5)
662  };
663  painter.drawPolyline(p1,3);
664  painter.drawLine(8, offset+3, 16, offset+6);
665  }
666 
667  //draw half circle for power contact
668  if (flags &Power) {
669  QRectF arc(4, offset+4, 4, 4);
670  if (flags &NO)
671  painter.drawArc(arc, 180*16, 180*16);
672  else
673  painter.drawArc(arc, 0, 180*16);
674  }
675 
676  // draw half circle for delay contact
677  if(flags &Delay) {
678  // for delay on contact
679  if (flags &DelayOn) {
680  if (flags &NO) {
681  painter.drawLine(12, offset+8, 12, offset+11);
682  QRectF r(9.5, offset+9, 5, 3);
683  painter.drawArc(r, 180*16, 180*16);
684  }
685  if (flags &NC) {
686  painter.drawLine(QPointF(12.5, offset+5), QPointF(12.5, offset+8));
687  QRectF r(10, offset+6, 5, 3);
688  painter.drawArc(r, 180*16, 180*16);
689  }
690  }
691  // for delay off contact
692  else if ( flags &DelayOff){
693  if (flags &NO) {
694  painter.drawLine(12, offset+8, 12, offset+9.5);
695  QRectF r(9.5, offset+9.5, 5, 3);
696  painter.drawArc(r, 0, 180*16);
697  }
698  if (flags &NC) {
699  painter.drawLine(QPointF(12.5, offset+5), QPointF(12.5, offset+7));
700  QRectF r(10, offset+7.5, 5, 3);
701  painter.drawArc(r, 0, 180*16);
702  }
703 
704  }
705  else {
706  // for delay on contact
707  if (flags &NO) {
708  painter.drawLine(12, offset+8, 12, offset+11);
709  QRectF r(9.5, offset+11.7, 5, 3);
710  painter.drawArc(r, 0, 180*16);
711  QRectF rl(9.5, offset+9, 5, 3);
712  painter.drawArc(rl, 180*16, 180*16);
713  }
714  if (flags &NC) {
715  painter.drawLine(QPointF(12.5, offset+5), QPointF(12.5, offset+8));
716  QRectF r(9.5, offset+10.7, 5, 3);
717  painter.drawArc(r, 0, 180*16);
718  QRectF rl(9.5, offset+8, 5, 3);
719  painter.drawArc(rl, 180*16, 180*16);
720  }
721  }
722  }
723 
724  QRectF text_rect = painter.boundingRect(QRectF(30, offset, 5, 10), Qt::AlignLeft | Qt::AlignVCenter, str);
725  painter.drawText(text_rect, Qt::AlignLeft | Qt::AlignVCenter, str);
726  bounding_rect = bounding_rect.united(text_rect);
727 
728  if (m_hovered_contacts_map.contains(elmt))
729  {
730  m_hovered_contacts_map.insert(elmt, bounding_rect);
731  }
732  else
733  {
734  m_hovered_contacts_map.insert(elmt, bounding_rect);
735  }
736 
737  ++m_drawed_contacts;
738  }
739 
740  //Draw a switch contact
741  else if (flags &SW)
742  {
743  bounding_rect = QRectF(0, offset, 24, 20);
744 
745  //draw the NO side
746  painter.drawLine(0, offset+6, 8, offset+6);
747  //Draw the NC side
748  QPointF p1[3] = {
749  QPointF(0, offset+16),
750  QPointF(8, offset+16),
751  QPointF(8, offset+12)

753  painter.drawPolyline(p1, 3);
754 
755  //Draw the common side
756  QPointF p2[3] = {
757  QPointF(7, offset+14),
758  QPointF(16, offset+11),
759  QPointF(24, offset+11),
760  };
761  painter.drawPolyline(p2, 3);
762 
763  //Draw the half ellipse off delay
764  if (flags &Delay)
765  {
766  painter.drawLine(12, offset+13, 12, offset+16);
767  if (flags &DelayOn) {
768  QRectF r(9.5, offset+14, 5, 3);
769  painter.drawArc(r, 180*16, 180*16);
770  }
771  else if (flags &DelayOff) {
772  QRectF r(9.5, offset+16.5, 5, 3);
