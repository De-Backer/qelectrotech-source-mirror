88 void SelectAutonumW::setContext(const NumerotationContext &context) {
89  m_context = context;
90 
91  qDeleteAll(num_part_list_);
92  num_part_list_.clear();
93 
94  if (m_context.size() == 0) { //@context contain nothing, build a default numPartEditor
95  on_add_button_clicked();
96  }
97  else {
98  for (int i=0; i<m_context.size(); ++i) { //build with the content of @context
99  NumPartEditorW *part= new NumPartEditorW(m_context, i, m_edited_type, this);
100  connect (part, SIGNAL(changed()), this, SLOT(applyEnable()));
101  num_part_list_ << part;
102  ui -> editor_layout -> addWidget(part);
103  }
104  }
105 
106  num_part_list_.size() == 1 ?
107  ui -> remove_button -> setDisabled(true):
108  ui -> remove_button -> setEnabled (true);
109 
110  applyEnable(false);
111 }
