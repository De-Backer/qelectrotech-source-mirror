51 void MasterElement::linkToElement(Element *elmt)
52 {
53  // check if element is slave and if isn't already linked
54  if (elmt->linkType() == Slave && !connected_elements.contains(elmt))
55  {
56  connected_elements << elmt;
57  elmt->linkToElement(this);
58 
59  XRefProperties xrp = diagram()->project()->defaultXRefProperties(kindInformations()["type"].toString());
60  if (!m_Xref_item && xrp.snapTo() == XRefProperties::Bottom)
61  m_Xref_item = new CrossRefItem(this); //create cross ref item if not yet
62 
63  emit linkedElementChanged();
64  aboutDeleteXref();
65  }
66 }
