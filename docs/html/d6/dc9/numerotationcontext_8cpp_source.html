52 bool NumerotationContext::addValue(const QString &type,
53  const QVariant &value,
54  const int increase,
55  const int initialvalue) {
56  if (!keyIsAcceptable(type) && !value.canConvert(QVariant::String))
57  return false;
58  if (keyIsNumber(type) && !value.canConvert(QVariant::Int))
59  return false;
60 
61  QString valuestr = value.toString();
62  valuestr.remove("|");
63  content_ << type
64  + "|"
65  + valuestr
66  + "|"
67  + QString::number(increase)
68  + "|"
69  + QString::number(initialvalue);
70  return true;
71 }

150 QDomElement NumerotationContext::toXml(QDomDocument &d, const QString& str) {
151  QDomElement num_auto = d.createElement(str);
152  for (int i=0; i<content_.size(); ++i) {
153  QStringList strl = itemAt(i);
154  QDomElement part = d.createElement("part");
155  part.setAttribute("type", strl.at(0));
156  part.setAttribute("value", strl.at(1));
157  part.setAttribute("increase", strl.at(2));
158  if (strl.at(0) == ("unitfolio") ||
159  strl.at(0) == ("tenfolio") ||
160  strl.at(0) == ("hundredfolio")) {
161  part.setAttribute("initialvalue", strl.at(3));
162  }
163  num_auto.appendChild(part);
164  }
165  return num_auto;
166 }
