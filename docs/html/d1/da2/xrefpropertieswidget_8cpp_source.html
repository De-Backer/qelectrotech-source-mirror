118 void XRefPropertiesWidget::saveProperties(int index) {
119  QString type = ui->m_type_cb->itemData(index).toString();
120  XRefProperties xrp = m_properties[type];
121 
122  if (ui->m_display_has_cross_rb->isChecked())
123  xrp.setDisplayHas(XRefProperties::Cross);
124  else if (ui->m_display_has_contacts_rb->isChecked())
125  xrp.setDisplayHas(XRefProperties::Contacts);
126  if (ui->m_snap_to_cb->itemData(
127  ui->m_snap_to_cb->currentIndex()).toString()
128  == "bottom")
129  xrp.setSnapTo(XRefProperties::Bottom);
130  else
131  xrp.setSnapTo(XRefProperties::Label);
132 
133  if(ui->m_xrefpos_cb->itemData(ui->m_xrefpos_cb->currentIndex()).toString() == "bottom") xrp.setXrefPos(Qt::AlignBottom);
134  else if(ui->m_xrefpos_cb->itemData(ui->m_xrefpos_cb->currentIndex()).toString() == "top") xrp.setXrefPos(Qt::AlignTop);
135  else if(ui->m_xrefpos_cb->itemData(ui->m_xrefpos_cb->currentIndex()).toString() == "left") xrp.setXrefPos(Qt::AlignLeft);
136  else if(ui->m_xrefpos_cb->itemData(ui->m_xrefpos_cb->currentIndex()).toString() == "right") xrp.setXrefPos(Qt::AlignRight);
137  else if(ui->m_xrefpos_cb->itemData(ui->m_xrefpos_cb->currentIndex()).toString() == "alignment") xrp.setXrefPos(Qt::AlignBaseline);
138  xrp.setShowPowerContac(ui->m_show_power_cb->isChecked());
139  xrp.setPrefix("power", ui->m_power_prefix_le->text());
140  xrp.setPrefix("delay", ui->m_delay_prefix_le->text());
141  xrp.setPrefix("switch", ui->m_switch_prefix_le->text());
142  xrp.setMasterLabel(ui->m_master_le->text());
143  xrp.setSlaveLabel(ui->m_slave_le->text());
144  xrp.setOffset(ui->m_offset_sb->value());
145 
146  m_properties.insert(type, xrp);
147 }

153 void XRefPropertiesWidget::updateDisplay() {
154  QString type = ui->m_type_cb->itemData(ui->m_type_cb->currentIndex()).toString();
155  XRefProperties xrp = m_properties[type];
156 
157  XRefProperties::DisplayHas dh = xrp.displayHas();
158  if (dh == XRefProperties::Cross) {
159  ui->m_display_has_cross_rb->setChecked(true);
160  }
161  else if (dh == XRefProperties::Contacts) {
162  ui->m_display_has_contacts_rb->setChecked(true);
163  }
164 
165  QString master = xrp.masterLabel();
166  ui->m_master_le->setText(master);
167 
168  QString slave = xrp.slaveLabel();
169  ui->m_slave_le->setText(slave);
170 
171  int offset = xrp.offset();
172  ui->m_offset_sb->setValue(offset);
173 
174  if (xrp.snapTo() == XRefProperties::Bottom){
175  ui->m_snap_to_cb->setCurrentIndex(ui->m_snap_to_cb->findData("bottom"));
176  ui->m_offset_sb->setEnabled(true);
177  }
178  else {
179  ui->m_snap_to_cb->setCurrentIndex(ui->m_snap_to_cb->findData("label"));
180  ui->m_offset_sb->setEnabled(false);
181  }
182 
183  if(xrp.getXrefPos() == Qt::AlignTop) ui->m_xrefpos_cb->setCurrentIndex(ui->m_xrefpos_cb->findData("top"));
184  else if(xrp.getXrefPos() == Qt::AlignLeft) ui->m_xrefpos_cb->setCurrentIndex(ui->m_xrefpos_cb->findData("left"));
185  else if(xrp.getXrefPos() == Qt::AlignRight) ui->m_xrefpos_cb->setCurrentIndex(ui->m_xrefpos_cb->findData("right"));
186  else if(xrp.getXrefPos() == Qt::AlignBaseline) ui->m_xrefpos_cb->setCurrentIndex(ui->m_xrefpos_cb->findData("alignment"));
187  else if(xrp.getXrefPos() == Qt::AlignBottom) ui->m_xrefpos_cb->setCurrentIndex(ui->m_xrefpos_cb->findData("bottom"));
188  ui->m_show_power_cb->setChecked(xrp.showPowerContact());
189  ui->m_power_prefix_le-> setText(xrp.prefix("power"));
190  ui->m_delay_prefix_le-> setText(xrp.prefix("delay"));
191  ui->m_switch_prefix_le->setText(xrp.prefix("switch"));
192  ui->m_cross_properties_gb->setDisabled(!ui->m_display_has_cross_rb->isChecked());
193 }
