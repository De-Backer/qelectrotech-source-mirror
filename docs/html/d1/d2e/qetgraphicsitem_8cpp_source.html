108 void QetGraphicsItem::mouseMoveEvent(QGraphicsSceneMouseEvent *event)
109 {
110  if (isSelected() && event->buttons() & Qt::LeftButton)
111  {
112  //Item is moving
113  if(diagram() && m_first_move) {
114  //It's the first movement, we signal it to parent diagram
115  diagram()->elementsMover().beginMovement(diagram(), this);
116  }
117 
118  //we apply the mouse movement
119  QPointF old_pos = pos();
120  if (m_first_move) {
121  m_mouse_to_origin_movement = old_pos - event -> buttonDownScenePos(Qt::LeftButton);
122  }
123  QPointF expected_pos = event->scenePos() + m_mouse_to_origin_movement;
124  setPos(expected_pos); // setPos() will snap the expected position to the grid
125 
126  //we calcul the real movement apply by setPos()
127  QPointF effective_movement = pos() - old_pos;
128  if (diagram()) {
129  //we signal the real movement apply to diagram,
130  //who he apply to other selected item
131  diagram()->elementsMover().continueMovement(effective_movement);
132  }
133  event->accept();
134  }
135  else {
136  event->ignore();
137  }
138 
139  if (m_first_move) {
140  m_first_move = false;
141  }
142 }
