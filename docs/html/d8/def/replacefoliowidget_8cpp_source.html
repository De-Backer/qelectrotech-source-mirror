45 TitleBlockProperties ReplaceFolioWidget::titleBlockProperties() const

47  TitleBlockProperties prop;

51  prop.plant = ui->m_plant ->text();
52  prop.locmach = ui->m_loc ->text();
53  prop.indexrev = ui->m_indice ->text();
54  prop.folio = ui->m_folio_le ->text();

57  prop.date = QDate();
58  prop.useDate = TitleBlockProperties::UseDateValue;

61  prop.date = SearchAndReplaceWorker::eraseDate();

63  }
64  else if (ui->m_fixed_date_rb->isChecked()) {
65  prop.date = ui->m_date_edit->date();
66  prop.useDate = TitleBlockProperties::UseDateValue;
67  }
68 
69  prop.context = m_diagram_context_widget->context();

109  ui->m_date_edit->setDate(properties.date);

111  }
