191 TitleBlockTemplate *TitleBlockTemplatesProjectCollection::getTemplate(const QString &template_name){
192  // Do we have already loaded this template?
193  if (titleblock_templates_.contains(template_name)) {
194  return(titleblock_templates_[template_name]);
195  }
196 
197  // No? Do we even know of it?
198  if (!titleblock_templates_xml_.contains(template_name)) {
199  return(nullptr);
200  }
201 
202  // Ok, we have its XML description, we have to generate a TitleBlockTemplate object
203  TitleBlockTemplate *titleblock_template = new TitleBlockTemplate(this);
204  if (titleblock_template -> loadFromXmlElement(titleblock_templates_xml_[template_name])) {
205  titleblock_templates_.insert(template_name, titleblock_template);
206  return(titleblock_template);
207  } else {
208  return(nullptr);
209  }
210 }

235 bool TitleBlockTemplatesProjectCollection::setTemplateXmlDescription(const QString &template_name, const QDomElement &xml_elmt) {
236  // check basic stuff
237  if (xml_elmt.tagName() != "titleblocktemplate") {
238  return(false);
239  }
240 
241  // we *require* a project (at least for the moment...)
242  if (!project_) return(false);
243 
244  // we import the provided XML element in the project document
245  QDomElement import = xml_document_.importNode(xml_elmt, true).toElement();
246 
247  // ensure the name stored in the XML description remains consistent with the provided template name
248  import.setAttribute("name", template_name);
249 
250  // we either replace the previous description
251  if (titleblock_templates_xml_.contains(template_name)) {
252  QDomElement old_description = titleblock_templates_xml_[template_name];
253  if (!old_description.parentNode().isNull()) {
254  old_description.parentNode().replaceChild(import, old_description);
255  }
256  }
257  titleblock_templates_xml_.insert(template_name, import);
258 
259  if (titleblock_templates_.contains(template_name)) {
260  titleblock_templates_[template_name] -> loadFromXmlElement(titleblock_templates_xml_[template_name]);
261  }
262  emit(changed(this, template_name));
263 
264  return(true);
265 }

323 void TitleBlockTemplatesProjectCollection::fromXml(const QDomElement &xml_element) {
324  foreach (QDomElement e, QET::findInDomElement(xml_element, "titleblocktemplates", "titleblocktemplate")) {
325  // each titleblock template must have a name
326  if (!e.hasAttribute("name")) continue;
327  QString titleblock_template_name = e.attribute("name");
328 
329  // if several templates have the same name, we keep the first one encountered
330  if (titleblock_templates_xml_.contains(titleblock_template_name)) continue;
331 
332  // we simply store the XML element describing the titleblock template,
333  // without any further analysis for the moment
334  titleblock_templates_xml_.insert(titleblock_template_name, e);
335  }
336 }

422 QDomElement TitleBlockTemplatesFilesCollection::getTemplateXmlDescription(const QString &template_name) {
423  QString xml_file_path = path(template_name);
424 
425  QFileInfo xml_file_info(xml_file_path);
426  if (!xml_file_info.exists() || !xml_file_info.isReadable()) {
427  return(QDomElement());
428  }
429 
430  QFile xml_file(xml_file_path);
431  if (!xml_file.open(QIODevice::ReadOnly)) {
432  return(QDomElement());
433  }
434 
435  QDomDocument *xml_document = new QDomDocument();
436  bool xml_parsing = xml_document -> setContent(&xml_file);
437  if (!xml_parsing) {
438  delete xml_document;
439  return(QDomElement());
440  }
441  return(xml_document -> documentElement());
442 }

449 bool TitleBlockTemplatesFilesCollection::setTemplateXmlDescription(const QString &template_name, const QDomElement &xml_element) {
450  if (template_name.isEmpty()) return(false);
451 
452  // prevent the watcher from emitting signals while we open and write to file
453  blockSignals(true);
454 
455  QDomDocument doc;
456  doc.appendChild(doc.importNode(xml_element, true));
457 
458  bool writing = QET::writeXmlFile(doc, path(template_name));
459  if (!writing) return(false);
460 
461  // emit a single signal for the change
462  blockSignals(false);
463  emit(changed(this, template_name));
