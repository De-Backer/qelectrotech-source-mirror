712 void ElementsCollectionWidget::search()
713 {
714  QString text = m_search_field->text();
715  //Reset the search
716  if (text.isEmpty())
717  {
718  QModelIndex current_index = m_tree_view->currentIndex();
719  m_tree_view->reset();
720 
721  if (m_showed_index.isValid())
722  {
723  hideCollection(true);
724  showAndExpandItem(m_showed_index, true, true);
725  }
726  else
727  expandFirstItems();
728 
729  //Expand the tree and scroll to the last selected index
730  if (current_index.isValid())
731  {
732  showAndExpandItem(current_index);
733  m_tree_view->setCurrentIndex(current_index);
734  m_tree_view->scrollTo(current_index);
735  }
736  return;
737  }
738 
739  hideCollection(true);
740 #if QT_VERSION < QT_VERSION_CHECK(5, 14, 0) // ### Qt 6: remove
741  QStringList text_list = text.split("+", QString::SkipEmptyParts);
742 #else
743 #pragma message("@TODO remove code for QT 5.14 or later")
744  QStringList text_list = text.split("+", Qt::SkipEmptyParts);
745 #endif
746  QModelIndexList match_index;
747  foreach (QString txt, text_list) {
748  match_index << m_model->match(m_showed_index.isValid()
749  ? m_model->index(0,0,m_showed_index)
750  : m_model->index(0,0),
751  Qt::UserRole+1,
752  QVariant(txt),
753  -1,
754  Qt::MatchContains
755  | Qt::MatchRecursive);
756  }
757 
758  foreach(QModelIndex index, match_index)
759  showAndExpandItem(index);
760 }
