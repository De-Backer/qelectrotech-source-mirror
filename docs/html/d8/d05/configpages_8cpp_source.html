116 void NewDiagramPage::applyConf() {
117  if (m_project) { //If project we save to the project
118  if (m_project -> isReadOnly()) return;
119  bool modified_project = false;
120 
121  BorderProperties new_border_prop = bpw -> properties();
122  if (m_project -> defaultBorderProperties() != new_border_prop) {
123  m_project -> setDefaultBorderProperties(bpw -> properties());
124  modified_project = true;
125  }
126 
127  TitleBlockProperties new_tbt_prop = ipw -> properties();
128  if (m_project -> defaultTitleBlockProperties() != new_tbt_prop) {
129  m_project -> setDefaultTitleBlockProperties(ipw -> properties());
130  modified_project = true;
131  }
132 
133  ConductorProperties new_conductor_prop = m_cpw -> properties();
134  if (m_project -> defaultConductorProperties() != new_conductor_prop) {
135  m_project -> setDefaultConductorProperties(m_cpw -> properties());
136  modified_project = true;
137  }
138 
139  QString new_report_prop = rpw -> ReportProperties();
140  if (m_project -> defaultReportProperties() != new_report_prop) {
141  m_project -> setDefaultReportProperties(new_report_prop);
142  modified_project = true;
143  }
144 
145  QHash<QString, XRefProperties> new_xref_properties = xrefpw -> properties();
146  if (m_project -> defaultXRefProperties() != new_xref_properties) {
147  m_project -> setDefaultXRefProperties(new_xref_properties);
148  modified_project = true;
149  }
150 
151  if (modified_project) {
152  m_project -> setModified(modified_project);
153  }
154 
155  } else { //Else we save to the default value
156  QSettings settings;
157 
158  // dimensions des nouveaux schemas
159  bpw -> properties().toSettings(settings, "diagrameditor/default");
160 
161  // proprietes du cartouche
162  ipw-> properties().toSettings(settings, "diagrameditor/default");
163 
164  // proprietes par defaut des conducteurs
165  m_cpw -> properties().toSettings(settings, "diagrameditor/defaultconductor");
166 
167  // default report propertie
168  rpw->toSettings(settings, "diagrameditor/defaultreport");
169 
170  // default xref properties
171  QHash <QString, XRefProperties> hash_xrp = xrefpw -> properties();
172  foreach (QString key, hash_xrp.keys()) {
173  XRefProperties xrp = hash_xrp[key];
174  QString str("diagrameditor/defaultxref");
175  xrp.toSettings(settings, str += key);
176  }
177  }
178 
179 }
