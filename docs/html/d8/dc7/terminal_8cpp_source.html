215 bool Terminal::addConductor(Conductor *conductor)
216 {
217  if (!conductor) return(false);
218 
219  Q_ASSERT_X(((conductor -> terminal1 == this) ^ (conductor -> terminal2 == this)), "Terminal::addConductor", "The conductor must be linked exactly once to this terminal");
220 
221  //Get the other terminal where the conductor must be linked
222  Terminal *other_terminal = (conductor -> terminal1 == this) ? conductor->terminal2 : conductor->terminal1;
223 
224  //Check if this terminal isn't already linked with other_terminal
225  foreach (Conductor* cond, conductors_)
226  if (cond -> terminal1 == other_terminal || cond -> terminal2 == other_terminal)
227  return false; //They already a conductor linked to this and other_terminal
228 
229  conductors_.append(conductor);
230  emit conductorWasAdded(conductor);
231  return(true);
232 }

254  const QStyleOptionGraphicsItem *options,

294  //Draw the help line with same orientation of terminal

299  m_help_line = new QGraphicsLineItem(this);
300  QPen pen;

309  {
310  line.setP2(t -> dockConductor());

312  }
313  }
314 
315  //Map the line (in scene coordinate) to m_help_line coordinate
316  line.setP1(m_help_line -> mapFromScene(line.p1()));
317  line.setP2(m_help_line -> mapFromScene(line.p2()));
318  m_help_line -> setPen(pen);
319  m_help_line -> setLine(line);
320  }
321 
322  //Draw the help line perpendicular to the terminal
323  if (!m_help_line_a)

333

418 QRectF Terminal::boundingRect() const {
419  if (br_ -> isNull())
420  {
421  qreal dcx = d->m_pos.x();
422  qreal dcy = d->m_pos.y();
423  qreal dex = dock_elmt_.x();
424  qreal dey = dock_elmt_.y();
425  QPointF origin = (dcx <= dex && dcy <= dey ? d->m_pos : dock_elmt_);
426  origin += QPointF(-3.0, -3.0);
427  qreal w = qAbs((int)(dcx - dex)) + 7;
428  qreal h = qAbs((int)(dcy - dey)) + 7;
429  *br_ = QRectF(origin, QSizeF(w, h));
430  }
431  return(*br_);
432 }

441 Terminal* Terminal::alignedWithTerminal() const
442 {
443  QLineF line(HelpLine());
444 
445  QPainterPath path;
446  path.moveTo(line.p1());
447  path.lineTo(line.p2());
448 
449  //Get all QGraphicsItem in the alignement of this terminal
450  QList <QGraphicsItem *> qgi_list = diagram() -> items(path);
451 
452  //Remove all terminals of the parent element
453  foreach (Terminal *t, parent_element_ -> terminals())
454  qgi_list.removeAll(t);
455 
456  if (qgi_list.isEmpty()) return nullptr;
457 
458  //Get terminals only if orientation is opposed with this terminal
459  QList <Terminal *> available_terminals;
460  foreach (QGraphicsItem *qgi, qgi_list)
461  {
462  if (Terminal *tt = qgraphicsitem_cast <Terminal *> (qgi))
463  {
464  //Call QET::lineContainsPoint to be sure the line intersect
465  //the dock point and not an other part of terminal
466  if (Qet::isOpposed(orientation(), tt -> orientation()) &&
467  QET::lineContainsPoint(line, tt -> dockConductor()))
468  {
469  available_terminals << tt;
470  }
471  }
472  }
473 
474  if (available_terminals.isEmpty()) return nullptr;
475  if (available_terminals.size() == 1) return (available_terminals.first());
476 
477  //Available_terminals have several terminals, we get the nearest terminal
478  line.setP2(available_terminals.first() -> dockConductor());
479  qreal current_lenght = line.length();
480  Terminal *nearest_terminal = available_terminals.takeFirst();
481 
482  //Search the nearest terminal to this one
483  foreach (Terminal *terminal, available_terminals)
484  {
485  line.setP2(terminal -> dockConductor());
486  if (line.length() < current_lenght)
487  {
488  current_lenght = line.length();
489  nearest_terminal = terminal;
490  }
491  }
492 
493  return nearest_terminal;
494 }

540 void Terminal::mouseMoveEvent(QGraphicsSceneMouseEvent *e) {
541  // pendant la pose d'un conducteur, on adopte un autre curseur
542  //setCursor(Qt::CrossCursor);
543 
544  // d'un mouvement a l'autre, il faut retirer l'effet hover de la borne precedente
545  if (previous_terminal_) {
546  if (previous_terminal_ == this) hovered_ = true;
547  else previous_terminal_ -> hovered_ = false;
548  previous_terminal_ -> hovered_color_ = previous_terminal_ -> neutralColor;
549  previous_terminal_ -> update();
550  }
551 
552  Diagram *diag = diagram();
553  if (!diag) return;
554  // si la scene est un Diagram, on actualise le poseur de conducteur
555  diag -> setConductorStop(e -> scenePos());
556 
557  // on recupere la liste des qgi sous le pointeur
558  QList<QGraphicsItem *> qgis = diag -> items(e -> scenePos());
559 
560  /* le qgi le plus haut
561  = le poseur de conductor
562  = le premier element de la liste
563  = la liste ne peut etre vide
564  = on prend le deuxieme element de la liste
565  */
566  Q_ASSERT_X(!(qgis.isEmpty()), "Terminal::mouseMoveEvent", "La liste d'items ne devrait pas etre vide");
567 
568  // s'il n'y rien d'autre que le poseur de conducteur dans la liste, on arrete la
569  if (qgis.size() <= 1) return;
570 
571  // sinon on prend le deuxieme element de la liste et on verifie s'il s'agit d'une borne
572  QGraphicsItem *qgi = qgis.at(1);
573  // si le qgi est une borne...
574  Terminal *other_terminal = qgraphicsitem_cast<Terminal *>(qgi);
575  if (!other_terminal) return;
576  previous_terminal_ = other_terminal;
577 
578  // s'il s'agit d'une borne, on lui applique l'effet hover approprie
579  if (!canBeLinkedTo(other_terminal)) {
580  other_terminal -> hovered_color_ = forbiddenColor;
581  } else if (other_terminal -> conductorsCount()) {
582  other_terminal -> hovered_color_ = warningColor;
583  } else {
584  other_terminal -> hovered_color_ = allowedColor;
585  }
586 
587  other_terminal -> hovered_ = true;
588  other_terminal -> update();
589 }

596 void Terminal::mouseReleaseEvent(QGraphicsSceneMouseEvent *e)
597 {
598  previous_terminal_ = nullptr;
599  hovered_color_ = neutralColor;
600 
601  if (!diagram()) return;
602 
603  //Stop conductor preview
604  diagram() -> setConductor(false);
605 
606  //Get item under cursor
607  QGraphicsItem *qgi = diagram() -> itemAt(e -> scenePos(), QTransform());
608  if (!qgi) return;
609 
610  //Element must be a terminal
611  Terminal *other_terminal = qgraphicsitem_cast<Terminal *>(qgi);
612  if (!other_terminal) return;
613 
614  other_terminal -> hovered_color_ = neutralColor;
615  other_terminal -> hovered_ = false;
616 
617  //We stop her if we can't link this terminal with other terminal
618  if (!canBeLinkedTo(other_terminal)) return;
619 
620  //Create conductor
621  Conductor *new_conductor = new Conductor(this, other_terminal);
622 
623  //Get all conductors at the same potential of new conductors
624  QSet <Conductor *> conductors_list = new_conductor->relatedPotentialConductors();
625 
626  //Compare the properties of every conductors stored in conductors_list,
627  //if every conductors properties is equal, we use this properties for the new conductor.
628  ConductorProperties others_properties;
629  bool use_properties = false;
630  if (!conductors_list.isEmpty())
631  {
632  use_properties = true;
633  others_properties = (*conductors_list.begin())->properties();
634  foreach (Conductor *conductor, conductors_list) {
635  if (conductor->properties() != others_properties)
636  use_properties = false;
637  }
638  }
639 
640 
641  QUndoCommand *undo = new QUndoCommand();
642  QUndoCommand *aic = new AddItemCommand<Conductor *>(new_conductor, diagram(), QPointF(), undo);
643  undo->setText(aic->text());
644 
645  if (use_properties)
646  {
647  Conductor *other = conductors_list.values().first();
648  new_conductor->rSequenceNum() = other->sequenceNum();
649  new_conductor->setProperties(others_properties);
650  }
651  else
652  {
653  //Autonum it
654  ConductorAutoNumerotation can (new_conductor, diagram(), undo);
655  can.numerate();
656  }
657  //Add undo command to the parent diagram
658  diagram() -> undoStack().push(undo);
659  if (use_properties)
660  {
661  Conductor *other = conductors_list.values().first();
662  new_conductor->setProperties(other->properties());
663  }
664 }

841 QList<Terminal *> relatedPotentialTerminal (const Terminal *terminal, const bool all_diagram)
842 {

844  if (all_diagram && terminal -> parentElement() -> linkType() & Element::AllReport)
845  {
