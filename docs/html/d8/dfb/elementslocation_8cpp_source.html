217 void ElementsLocation::setPath(const QString &path)
218 {
219  QString tmp_path = path;
220 #ifdef Q_OS_WIN32
221  //On windows, we convert backslash to slash
222  tmp_path = QDir::fromNativeSeparators(path);
223 
224 #endif
225 
226  //There is a project, the path is for an embedded coolection.
227  if (m_project)
228  {
229  m_collection_path = path;
230  //Add the protocol to the collection path
231  if (!path.startsWith("embed://"))
232  m_collection_path.prepend("embed://");
233 
234  }
235 
236  //The path start with project, we get the project and the path from the string
237  else if (tmp_path.startsWith("project"))
238  {
239  QRegExp rx("^project([0-9]+)\\+(embed:\\/\\/.*)$", Qt::CaseInsensitive);
240  if (rx.exactMatch(tmp_path))
241  {
242  bool conv_ok;
243  uint project_id = rx.capturedTexts().at(1).toUInt(&conv_ok);
244  if (conv_ok)
245  {
246  QETProject *project = QETApp::project(project_id);
247  if (project)
248  {
249  m_collection_path = rx.capturedTexts().at(2);
250  m_project = project;
251  }
252  }
253  }
254  }
255 
256  // The path is in file system,
257  // the given path is relative to common or custom collection
258  else if (path.startsWith("common://") || path.startsWith("custom://"))
259  {
260  QString p;
261  if (path.startsWith("common://"))
262  {
263  tmp_path.remove("common://");
264  p = QETApp::commonElementsDirN() + "/" + tmp_path;
265  }
266  else
267  {
268  tmp_path.remove("custom://");
269  p = QETApp::customElementsDirN() + "/" + tmp_path;
270  }
271 
272  m_file_system_path = p;
273  m_collection_path = path;
274  }
275  //In this case, the path is supposed to be relative to the file system.
276  else
277  {
278  QString path_ = path;
279  if(path_.endsWith(".elmt"))
280  {
281  m_file_system_path = path_;
282  if (path_.startsWith(QETApp::commonElementsDirN()))
283  {
284  path_.remove(QETApp::commonElementsDirN()+="/");
285  path_.prepend("common://");
286  m_collection_path = path_;
287  }
288  else if (path_.startsWith(QETApp::customElementsDirN()))
289  {
290  path_.remove(QETApp::customElementsDirN()+="/");
291  path_.prepend("custom://");
292  m_collection_path = path_;
293  }
294  }
295  else
296  {
297  m_file_system_path = path_;
298  if (path_.startsWith(QETApp::commonElementsDirN()))
299  {
300  path_.remove(QETApp::commonElementsDirN()+="/");
301  path_.prepend("common://");
302  m_collection_path = path_;
303  }
304  else if (path_.startsWith(QETApp::customElementsDirN()))
305  {
306  path_.remove(QETApp::customElementsDirN()+="/");
307  path_.prepend("custom://");
308  m_collection_path = path_;
309  }
310  }
311  }
312 }

610 pugi::xml_document ElementsLocation::pugiXml() const
611 {
612  pugi::xml_document docu; //empty xml_document();
613  /* Except for linux OS (because linux keep in cache the file),
614  * we keep in memory the xml
615  * to avoid multiple access to file.
616  * keep in memory the XML,
617  * consumes a little more RAM,
618  * for this reason we don't use it for linux to minimize the RAM footprint.
619  */
620 #ifndef Q_OS_LINUX
621  if (!m_string_stream.str().empty())
622  {
623  docu.load_string(m_string_stream.str().c_str());
624  return docu;
625  }
626 #endif
627  if (!m_project)
628  {
629  if (docu.load_file(m_file_system_path.toStdString().c_str()))
630  {
631 #ifndef Q_OS_LINUX
632  docu.save(m_string_stream);
633 #endif
634  }
635  }
636  else
637  {
638  QString str = m_collection_path;
639  //Get the xml dom from Qt xml and copie to pugi xml
640  QDomElement element = m_project
641  ->embeddedElementCollection()
642  ->element(str.remove("embed://"));
643  QDomDocument qdoc;
644  if (isElement())
645  {
646  qdoc.appendChild(qdoc.importNode(
647  element.firstChildElement(
648  "definition"),
649  true));
650  } else {
651  qdoc.appendChild(qdoc.importNode(element, true));
652  }
653  docu.load_string(qdoc.toString(4).toStdString().c_str());
654  }
655  return docu;

667 bool ElementsLocation::setXml(const QDomDocument &xml_document) const
668 {
669  if (!isWritable())
670  return false;
671 
672  if (xml_document.documentElement().tagName() != "definition")
673  {
674  qDebug() << "ElementsLocation::setXml :"
675  " tag name of document element isn't 'definition'";
676  return false;
677  }
678 
679  if (isFileSystem())
680  {
681  QString error;
682  QETXML::writeXmlFile(xml_document, fileSystemPath(), &error);
683 
684  if (!error.isEmpty()) {
685  qDebug() << "ElementsLocation::setXml error : "
686  << error;
687  return false;
688  }
689  else {
690  return true;
691  }
692  }
693  else if (isProject())
694  {
695  //Element exist, we overwrite the existing element.
696  if (exist())
697  {
698  QDomElement dom_element = xml();
699  QDomNode parent_node = dom_element.parentNode();
700  parent_node.removeChild(dom_element);
701  parent_node.appendChild(xml_document
702  .documentElement()
703  .cloneNode(true));
704  return true;
705  }
706  //Element doesn't exist, we create the element
707  else
708  {
709  QString path_ = collectionPath(false);
710  QRegExp rx ("^(.*)/(.*\\.elmt)$");
711 
712  if (rx.exactMatch(path_)) {
713  return project()
714  ->embeddedElementCollection()
715  ->addElementDefinition(
716  rx.cap(1),
717  rx.cap(2),
718  xml_document
719  .documentElement());
720  }
721  else {
722  qDebug() << "ElementsLocation::setXml :"
723  " rx don't match";
724  }
725 
726  }
727  }
728 
729  return false;
730 }
