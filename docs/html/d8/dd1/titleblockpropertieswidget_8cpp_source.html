389 void TitleBlockPropertiesWidget::updateTemplateList()
390 {
391  ui -> m_tbt_cb ->clear();
392 
393  if (m_tbt_collection_list.isEmpty())
394  {
395  setTitleBlockTemplatesVisible(false);
396  return;
397  }
398  setTitleBlockTemplatesVisible(true);
399 
400  //Add the default title block
401  m_map_index_to_collection_type.clear();
402  m_map_index_to_collection_type.append(QET::QetCollection::Common);
403  ui -> m_tbt_cb -> addItem(QET::Icons::QETLogo, tr("Modle par dfaut"));
404 
405  //Add every title block stored in m_tbt_collection_list
406  foreach (TitleBlockTemplatesCollection *tbt_c, m_tbt_collection_list)
407  {
408  QIcon icon;
409  QET::QetCollection qc = tbt_c -> collection();
410  if (qc == QET::QetCollection::Common)
411  icon = QET::Icons::QETLogo;
412  else if (qc == QET::QetCollection::Custom)
413  icon = QET::Icons::Home;
414  else if (qc == QET::QetCollection::Embedded)
415  icon = QET::Icons::TitleBlock;
416 
417  foreach(QString tbt_name, tbt_c -> templates())
418  {
419  m_map_index_to_collection_type.append(qc);
420  ui -> m_tbt_cb -> addItem(icon, tbt_name, tbt_name);
421  }
422  }
423 }

429 void TitleBlockPropertiesWidget::changeCurrentTitleBlockTemplate(int index)
430 {

433  QET::QetCollection qc = m_map_index_to_collection_type.at(index);
434  TitleBlockTemplatesCollection *collection = nullptr;
435  foreach (TitleBlockTemplatesCollection *c, m_tbt_collection_list)
436  if (c -> collection() == qc)
437  collection = c;

439  if (!collection) return;
440
