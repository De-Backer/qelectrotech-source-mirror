194 bool projectDataBase::createDataBase(const QString &connection_name, const QString &name)
195 {
196 
197  QString connect_name = connection_name;
198  if (connect_name.isEmpty()) {
199  connect_name = "qet_project_db_" + m_project->uuid().toString();
200  }
201  if (m_data_base.connectionNames().contains(connect_name)) {
202  m_data_base = QSqlDatabase::database(connect_name);
203  }
204  else
205  {
206  m_data_base = QSqlDatabase::addDatabase("QSQLITE", connect_name);
207  m_data_base.setDatabaseName(name);
208  if(!m_data_base.open()) {
209  m_data_base.close();
210  return false;
211  }
212 
213  m_data_base.exec("PRAGMA temp_store = MEMORY");
214  m_data_base.exec("PRAGMA journal_mode = MEMORY");
215  m_data_base.exec("PRAGMA synchronous = OFF");
216  m_data_base.exec("PRAGMA foreign_keys = ON");
217 
218  QSqlQuery query_(m_data_base);
219  bool first_ = true;
220 
221  //Create diagram table
222  QString diagram_table("CREATE TABLE diagram ("
223  "uuid VARCHAR(50) PRIMARY KEY NOT NULL,"
224  "pos INTEGER)");
225  if (!query_.exec(diagram_table)) {
226  qDebug() << "diagram_table query : "<< query_.lastError();
227  }
228 
229  //Create the table element
230  QString element_table("CREATE TABLE element"
231  "( "
232  "uuid VARCHAR(50) PRIMARY KEY NOT NULL, "
233  "diagram_uuid VARCHAR(50) NOT NULL,"
234  "pos VARCHAR(6) NOT NULL,"
235  "type VARCHAR(50),"
236  "sub_type VARCHAR(50),"
237  "FOREIGN KEY (diagram_uuid) REFERENCES diagram (uuid) ON DELETE CASCADE"
238  ")");
239  if (!query_.exec(element_table)) {
240  qDebug() <<" element_table query : "<< query_.lastError();
241  }
242 
243  //Create the diagram info table
244  QString diagram_info_table("CREATE TABLE diagram_info (diagram_uuid VARCHAR(50) PRIMARY KEY NOT NULL, ");
245  first_ = true;
246  for (auto string : QETApp::diagramInfoKeys())
247  {
248  if (first_) {
249  first_ = false;
250  } else {
251  diagram_info_table += ", ";
252  }
253  diagram_info_table += string += string=="date" ? " DATE" : " VARCHAR(100)";
254  }
255  diagram_info_table += ", FOREIGN KEY (diagram_uuid) REFERENCES diagram (uuid) ON DELETE CASCADE)";
256  if (!query_.exec(diagram_info_table)) {
257  qDebug() << "diagram_info_table query : " << query_.lastError();
258  }
259 
260  //Create the element info table
261  QString element_info_table("CREATE TABLE element_info(element_uuid VARCHAR(50) PRIMARY KEY NOT NULL,");
262  first_=true;
263  for (auto string : QETApp::elementInfoKeys())
264  {
265  if (first_) {
266  first_ = false;
267  } else {
268  element_info_table += ",";
269  }
270 
271  element_info_table += string += " VARCHAR(100)";
272  }
273  element_info_table += ", FOREIGN KEY (element_uuid) REFERENCES element (uuid) ON DELETE CASCADE);";
274 
275  if (!query_.exec(element_info_table)) {
276  qDebug() << " element_info_table query : " << query_.lastError();
277  }
278 
279  createElementNomenclatureView();
280  createSummaryView();
281  }
282 
283  prepareQuery();
284  updateDB();
285  return true;
286 }

451 void projectDataBase::prepareQuery()
452 {
453  //INSERT DIAGRAM
454  m_insert_diagram_query = QSqlQuery(m_data_base);
455  m_insert_diagram_query.prepare("INSERT INTO diagram (uuid, pos) VALUES (:uuid, :pos)");
456 
457  //REMOVE DIAGRAM
458  m_remove_diagram_query = QSqlQuery(m_data_base);
459  m_remove_diagram_query.prepare("DELETE FROM diagram WHERE uuid=:uuid");
460 
461  //INSERT DIAGRAM INFO
462  m_insert_diagram_info_query = QSqlQuery(m_data_base);
463  QStringList bind_diag_info_values;
464  for (auto key : QETApp::diagramInfoKeys()) {
465  bind_diag_info_values << key.prepend(":");
466  }
467  QString insert_diag_info("INSERT INTO diagram_info (diagram_uuid, " +
468  QETApp::diagramInfoKeys().join(", ") +
469  ") VALUES (:uuid, " +
470  bind_diag_info_values.join(", ") +
471  ")");
472  m_insert_diagram_info_query.prepare(insert_diag_info);
473 
474  //INSERT ELEMENT
475  QString insert_element_query("INSERT INTO element (uuid, diagram_uuid, pos, type, sub_type) VALUES (:uuid, :diagram_uuid, :pos, :type, :sub_type)");
476  m_insert_elements_query = QSqlQuery(m_data_base);
477  m_insert_elements_query.prepare(insert_element_query);
478 
479 
480  //INSERT ELEMENT INFO
481  QStringList bind_values;
482  for (auto key : QETApp::elementInfoKeys()) {
483  bind_values << key.prepend(":");
484  }
485  QString insert_element_info("INSERT INTO element_info (element_uuid," +
486  QETApp::elementInfoKeys().join(", ") +
487  ") VALUES (:uuid," +
488  bind_values.join(", ") +
489  ")");
490  m_insert_element_info_query = QSqlQuery(m_data_base);
491  m_insert_element_info_query.prepare(insert_element_info);
492 
493  //REMOVE ELEMENT
494  QString remove_element("DELETE FROM element WHERE uuid=:uuid");
495  m_remove_element_query = QSqlQuery(m_data_base);
496  m_remove_element_query.prepare(remove_element);
497 
498  //UPDATE ELEMENT INFO
499  QString update_str("UPDATE element_info SET ");
500  for (auto string : QETApp::elementInfoKeys()) {
501  update_str.append(string + " = :" + string + ", ");
502  }
503  update_str.remove(update_str.length()-2, 2); //Remove the last ", "
504  update_str.append(" WHERE element_uuid = :uuid");
505  m_update_element_query = QSqlQuery(m_data_base);
506  m_update_element_query.prepare(update_str);
507 }
