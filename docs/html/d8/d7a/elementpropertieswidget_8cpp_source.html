105 void ElementPropertiesWidget::setElement(Element *element)

110

184 void ElementPropertiesWidget::apply()
185 {
186  QList <QUndoCommand *> undo_list;
187 
188  foreach (PropertiesEditorWidget *pew, m_list_editor)
189  if (QUndoCommand *undo = pew->associatedUndo())
190  undo_list << undo;
191 
192  if (undo_list.isEmpty()) return;
193 
194  QString str;
195  foreach(QUndoCommand *uc, undo_list) str += uc->text() += " ";
196 
197  QUndoStack &stack = m_element -> diagram() -> undoStack();
198  stack.beginMacro(str);
199  foreach(QUndoCommand *uc, undo_list) stack.push(uc);
200  stack.endMacro();
201 
202  foreach(PropertiesEditorWidget *pew, m_list_editor)
203  pew->updateUi();
204 }

268 void ElementPropertiesWidget::updateUi()
269 {
270  //We keep the current title of the tab, to return to the same tab
271  //if possible, at the end of this method
272  QString tab_text;
273  tab_text = m_tab->tabText(m_tab->currentIndex());
274 
275  //Purge the tab widget and delete all widget
276  m_tab->clear();
277  qDeleteAll(m_list_editor);
278  m_list_editor.clear();
279  if(m_general_widget) {
280  delete m_general_widget;
281  m_general_widget = nullptr;
282  }
283 
284  //Create editor according to the type of element
285  switch (m_element -> linkType()) {
286  case Element::Simple:
287  m_list_editor << new ElementInfoWidget(m_element, this);
288  break;
289  case Element::NextReport:
290  m_list_editor << new LinkSingleElementWidget(m_element, this);
291  break;
292  case Element::PreviousReport:
293  m_list_editor << new LinkSingleElementWidget(m_element, this);
294  break;
295  case Element::Master:
296  m_list_editor << new MasterPropertiesWidget(m_element, this);
297  m_list_editor << new ElementInfoWidget(m_element, this);
298  break;
299  case Element::Slave:
300  m_list_editor << new LinkSingleElementWidget(m_element, this);
301  break;
302  case Element::Terminale:
303  m_list_editor << new ElementInfoWidget(m_element, this);
304  break;
305  default:
306  break;
307  }
308  m_list_editor << new DynamicElementTextItemEditor(m_element, this);
309 
310  //Add each editors in tab widget
311  for (AbstractElementPropertiesEditorWidget *aepew : m_list_editor)
312  {
313  aepew->setLiveEdit(m_live_edit);
314  m_tab->addTab(aepew, aepew->title());
315  }
316  addGeneralWidget();
317 
318  //Go to the tab, edited at the beginning of this method
319  if (!tab_text.isEmpty())
320  {
321  for(int i=0 ; i<m_tab->count() ; ++i)
322  {
323  if (tab_text == m_tab->tabBar()->tabText(i))
324  {
325  m_tab->setCurrentIndex(i);
326  break;
327  }
328  }
329  }
330 }

353 QWidget *ElementPropertiesWidget::generalWidget()
354 {
355  QString description_string(tr("lement\n"));
356 
357  // some element characteristic
358  description_string += QString(tr("Nom : %1\n")).arg(m_element -> name());
359  int folio_index = m_diagram -> folioIndex();
360  if (folio_index != -1) {
361  description_string += QString(tr("Folio : %1\n")).arg(folio_index + 1);
362  }
363  description_string += QString(tr("Type : %1\n")).arg(m_element ->linkTypeToString());
364  description_string += QString(tr("Sous-type : %1\n")).arg(m_element ->kindInformations()["type"].toString());
365  description_string += QString(tr("Position : %1\n")).arg(m_diagram -> convertPosition(m_element -> scenePos()).toString());
366  description_string += QString(tr("Rotation : %1\n")).arg(m_element.data()->rotation());
367  description_string += QString(tr("Dimensions : %1*%2\n")).arg(m_element -> size().width()).arg(m_element -> size().height());
368  description_string += QString(tr("Bornes : %1\n")).arg(m_element -> terminals().count());
369  description_string += QString(tr("Emplacement : %1\n")).arg(m_element.data()->location().toString());
370 
371  // widget himself
372  QWidget *general_widget = new QWidget (m_tab);
373  QVBoxLayout *vlayout_ = new QVBoxLayout (general_widget);
374  general_widget -> setLayout(vlayout_);
375 
376  //widget for the text
377  QLabel *label = new QLabel (description_string, general_widget);
378  label->setWordWrap(true);
379  label->setTextInteractionFlags(Qt::TextEditorInteraction);
380  vlayout_->addWidget(label);
381 
382  //widget for the pixmap
383  QLabel *pix = new QLabel(general_widget);
384  vlayout_->addWidget(pix, 0, Qt::AlignHCenter);
385  vlayout_ -> addStretch();
386 
387  //button widget
388  QPushButton *find_in_panel = new QPushButton(QET::Icons::ZoomDraw, tr("Retrouver dans le panel"), general_widget);
389  connect(find_in_panel, SIGNAL(clicked()), this, SLOT(findInPanel()));
390  QPushButton *edit_element = new QPushButton(QET::Icons::ElementEdit, tr("diter l'lment"), general_widget);
391  connect(edit_element, SIGNAL(clicked()), this, SLOT(editElement()));
392  QHBoxLayout *hlayout_ = new QHBoxLayout;
393  hlayout_->addWidget(find_in_panel);
394  hlayout_->addWidget(edit_element);
395  vlayout_->addLayout(hlayout_);
396 
397  //Set the maximum size of the pixmap to the minimum size of the layout
398  QPixmap pixmap = m_element->pixmap();
399  int margin = vlayout_->contentsMargins().left() + vlayout_->contentsMargins().right();
400  int widht_ = vlayout_->minimumSize().width()-margin;
401 
402  if (pixmap.size().width() > widht_ || pixmap.size().height() > widht_) {
403  pix->setPixmap(m_element->pixmap().scaled (widht_, widht_, Qt::KeepAspectRatio, Qt::SmoothTransformation));
404  }
405  else {
406  pix->setPixmap(pixmap);
407  }
408 
409  return general_widget;
410 }
