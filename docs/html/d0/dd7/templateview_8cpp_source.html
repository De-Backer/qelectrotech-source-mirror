189 QList<TitleBlockCell> TitleBlockTemplateView::pastedCells() {
190  QList<TitleBlockCell> pasted_cells;
191 
192  // retrieve the clipboard content and parse it as XML
193  QClipboard *clipboard = QApplication::clipboard();
194  QDomDocument xml_import;
195 
196  if (!xml_import.setContent(clipboard -> text().trimmed())) {
197  return(pasted_cells);
198  }
199 
200  // ensure the XML document describes cells that can be pasted
201  if (xml_import.documentElement().tagName() != "titleblocktemplate-partial") {
202  return(pasted_cells);
203  }
204 
205  // load pasted cells
206  QDomElement paste_root = xml_import.documentElement();
207  for (QDomElement e = paste_root.firstChildElement() ; !e.isNull() ; e = e.nextSiblingElement()) {
208  if (e.tagName() == "empty" || e.tagName() == "field" || e.tagName() == "logo") {
209  TitleBlockCell cell;
210  cell.loadContentFromXml(e);
211  int row_num = -1, col_num = -1, row_span = -1, col_span = -1;
212  if (!QET::attributeIsAnInteger(e, "row", &row_num) || row_num < 0) {
213  continue;
214  }
215  if (!QET::attributeIsAnInteger(e, "col", &col_num) || col_num < 0) {
216  continue;
217  }
218  cell.num_row = row_num;
219  cell.num_col = col_num;
220 
221  // parse the rowspan and colspan attributes
222  if (QET::attributeIsAnInteger(e, "rowspan", &row_span) && row_span > 0) {
223  cell.row_span = row_span;
224  }
225 
226  if (QET::attributeIsAnInteger(e, "colspan", &col_span) && col_span > 0) {
227  cell.col_span = col_span;
228  }
229  pasted_cells << cell;
230  }
231  }
232  return(pasted_cells);
233 }

238 void TitleBlockTemplateView::paste() {
239  if (!tbtemplate_) return;
240  QList<TitleBlockCell> pasted_cells = pastedCells();
241  if (!pasted_cells.count()) return;
242 
243  // the top left cell among the selected ones will be used to position the pasted cells
244  TitleBlockTemplateVisualCell *selected_cell = selectedCellsSet().topLeftCell();
245  if (!selected_cell) return;
246  TitleBlockCell *erased_cell = selected_cell -> cell();
247  if (!erased_cell) return;
248 
249  // change num_row and num_col attributes of pasted cells so they get positionned relatively to selected_cell
250  normalizeCells(pasted_cells, erased_cell -> num_row, erased_cell -> num_col);
251 
252  PasteTemplateCellsCommand *paste_command = new PasteTemplateCellsCommand(tbtemplate_);
253  foreach (TitleBlockCell cell, pasted_cells) {
254  TitleBlockCell *erased_cell = tbtemplate_ -> cell(cell.num_row, cell.num_col);
255  if (!erased_cell) continue;
256  paste_command -> addCell(erased_cell, *erased_cell, cell);
257  }
258 
259  requestGridModification(paste_command);

574 void TitleBlockTemplateView::applyColumnsWidths(bool animate) {
575  // the first column is dedicated to helper cells showing the rows height
576  tbgrid_ -> setColumnFixedWidth(0, DEFAULT_ROWS_HELPER_CELLS_WIDTH);
577  tbgrid_ -> setColumnSpacing(0, 0);
578 
579  // we apply the other columns width based on the title block template data
580  QList<int> widths = tbtemplate_ -> columnsWidth(preview_width_);
581  int total_applied_width = 0;
582  for (int i = 0 ; i < widths.count() ; ++ i) {
583  int applied_width = qMax(0, widths.at(i));
584  tbgrid_ -> setColumnSpacing(COL_OFFSET + i, 0);
585  if (!animate) {
586  // no animation on first call
587  tbgrid_ -> setColumnFixedWidth(COL_OFFSET + i, widths.at(i));
588  } else {
589  GridLayoutAnimation *animation = new GridLayoutAnimation(tbgrid_, form_);
590  animation -> setIndex(COL_OFFSET + i);
591  animation -> setActsOnRows(false);
592  animation -> setStartValue(QVariant(tbgrid_ -> columnMinimumWidth(COL_OFFSET + i)));
593  animation -> setEndValue(QVariant(1.0 * applied_width));
594  animation -> setDuration(500);
595  connect(animation, SIGNAL(finished()), this, SLOT(updateColumnsHelperCells()));
596  animation -> start(QAbstractAnimation::DeleteWhenStopped);
597  }
598  total_applied_width += applied_width;
599  }
600  if (!animate) updateColumnsHelperCells();
601  ++ apply_columns_widths_count_;
602 
603  // we systematically parameter some cells
604  total_width_helper_cell_ -> split_size = 0;
605  tbgrid_ -> addItem(total_width_helper_cell_, 0, COL_OFFSET, 1, widths.count());
606  removeItem(extra_cells_width_helper_cell_);
607 
608  if (total_applied_width < preview_width_) {
609  // preview_width is greater than the sum of cells widths
610  // we add an extra column with a helper cell
611  tbgrid_ -> addItem(extra_cells_width_helper_cell_, ROW_OFFSET - 1, COL_OFFSET + widths.count(), tbtemplate_ -> rowsCount() + 1, 1);
612  tbgrid_ -> addItem(total_width_helper_cell_, 0, COL_OFFSET, 1, widths.count() + 1);
613  tbgrid_ -> setColumnFixedWidth(COL_OFFSET + widths.count(), preview_width_ - total_applied_width);
614  extra_cells_width_helper_cell_ -> setLabel(
615  QString(
616  tr("[%1px]","content of the extra cell added when the total width of cells is less than the preview width")
617  ).arg(preview_width_ - total_applied_width)
618  );
619  } else if (total_applied_width > preview_width_) {
620  // preview width is smaller than the sum of cells widths
621  // we draw an extra header within th "preview width" cell.
622  tbgrid_ -> addItem(total_width_helper_cell_, 0, COL_OFFSET, 1, widths.count());
623  total_width_helper_cell_ -> split_background_color = QColor(Qt::red);
624  total_width_helper_cell_ -> split_foreground_color = QColor(Qt::black);
625  total_width_helper_cell_ -> split_label = QString(
626  tr("[%1px]", "content of the extra helper cell added when the total width of cells is greather than the preview width")
627  ).arg(total_applied_width - preview_width_);
628  total_width_helper_cell_ -> split_size = total_applied_width - preview_width_;
629  }
630 
631  updateDisplayedMinMaxWidth();
632 }

703 void TitleBlockTemplateView::addCells() {

705  int row_count = tbtemplate_ -> rowsCount();
706 
707  // we add a big cell to show the total width
708  total_width_helper_cell_ = new SplittedHelperCell();

725  HelperCell *current_col_cell = new HelperCell();

728  current_col_cell -> setActions(columnsActions());

840 void TitleBlockTemplateView::normalizeCells(
841  QList<TitleBlockCell> &cells,
842  int x,
843  int y) const {
844  if (!cells.count()) return;
845 
846  int min_row = cells.at(0).num_row;
847  int min_col = cells.at(0).num_col;
848  for (int i = 1 ; i < cells.count() ; ++ i) {
849  if (cells.at(i).num_row < min_row) min_row = cells.at(i).num_row;
850  if (cells.at(i).num_col < min_col) min_col = cells.at(i).num_col;
851  }
852  for (int i = 0 ; i < cells.count() ; ++ i) {
853  cells[i].num_row = cells[i].num_row - min_row + x;
854  cells[i].num_col = cells[i].num_col - min_col + y;
855  }
