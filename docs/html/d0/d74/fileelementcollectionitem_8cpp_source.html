115 QString FileElementCollectionItem::localName()
116 {
117  if (!text().isNull())
118  return text();
119 
120  else if (isDir()) {
121  if (isCollectionRoot()) {
122  if (m_path == QETApp::commonElementsDirN())
123  setText(QObject::tr("Collection QET"));
124  else if (m_path == QETApp::customElementsDirN())
125  setText(QObject::tr("Collection utilisateur"));
126  else
127  setText(QObject::tr("Collection inconnue"));
128  }
129  else
130  {
131  QString str(fileSystemPath() + "/qet_directory");
132  pugi::xml_document docu;
133  if(docu.load_file(str.toStdString().c_str()))
134  {
135  if (QString(docu.document_element().name())
136  == "qet-directory")
137  {
138  NamesList nl;
139  nl.fromXml(docu.document_element());
140  setText(nl.name());
141  }
142  }
143  }
144  }
145  else if (isElement()) {
146  ElementsLocation loc(collectionPath());
147  setText(loc.name());
148  }
149 
150  return text();
151 }

264 void FileElementCollectionItem::setUpData()
265 {

268  localName();
269  setFlags(Qt::ItemIsSelectable
270  | Qt::ItemIsDragEnabled
271  | Qt::ItemIsDropEnabled
272  | Qt::ItemIsEnabled);
273  }
274  else
275  {
276  setFlags(Qt::ItemIsSelectable
277  | Qt::ItemIsDragEnabled
278  | Qt::ItemIsEnabled);
279 
280  //Set the local name and all informations of the element
281  //in the data Qt::UserRole+1, these data will be use for search.
282  ElementsLocation loc(collectionPath());
283  DiagramContext context = loc.elementInformations();
284  QStringList search_list;
285  for (QString key : context.keys()) {
286  search_list.append(context.value(key).toString());
287  }
288  search_list.append(localName(loc));
289  setData(search_list.join(" "));
290  }
291 
292  setToolTip(collectionPath());

350 void FileElementCollectionItem::populate(bool set_data, bool hide_element)
351 {
352  QDir dir (fileSystemPath());
353 
354  //Get all directory in this directory.
355  for(auto str : dir.entryList(QDir::Dirs | QDir::NoDotAndDotDot,
356  QDir::Name))
357  {
358  FileElementCollectionItem *feci = new FileElementCollectionItem();
359  appendRow(feci);
360  feci->setPathName(str, set_data, hide_element);
361  if (set_data)
362  feci->setUpData();
363  }
364 
365  if (hide_element)
366  return;
367 
368  //Get all elmt file in this directory
369  dir.setNameFilters(QStringList() << "*.elmt");
370  for(auto str : dir.entryList(QDir::Files | QDir::NoDotAndDotDot,
371  QDir::Name))
372  {
373  FileElementCollectionItem *feci = new FileElementCollectionItem();
374  appendRow(feci);
375  feci->setPathName(str, set_data);
376  if (set_data)
377  feci->setUpData();
378  }
