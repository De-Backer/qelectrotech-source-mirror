405 void DiagramPrintDialog::printDiagram(Diagram *diagram,
406  bool fit_page,
407  const ExportProperties &options,
408  QPainter *qp,
409  QPrinter *printer) {
410  //qDebug() << printer -> paperSize() << printer -> paperRect() << diagram -> title();
411  // l'imprimante utilise-t-elle toute la feuille ?
412  bool full_page = printer -> fullPage();
413 
414  // impression physique (!= fichier PDF)
415  if (printer -> outputFileName().isEmpty()) {
416  // utiliser cette condition pour agir differemment en cas d'impression physique
417  }
418 
419  saveReloadDiagramParameters(diagram, options, true);
420 
421  // deselectionne tous les elements
422  QList<QGraphicsItem *> selected_elmts = diagram -> selectedItems();
423  foreach (QGraphicsItem *qgi, selected_elmts) qgi -> setSelected(false);
424 
425  // enleve le flag focusable de tous les elements concernes pour eviter toute reprise de focus par un champ de texte editable
426  QList<QGraphicsItem *> focusable_items;
427  foreach (QGraphicsItem *qgi, diagram -> items()) {
428  if (qgi -> flags() & QGraphicsItem::ItemIsFocusable) {
429  focusable_items << qgi;
430  qgi -> setFlag(QGraphicsItem::ItemIsFocusable, false);
431  }
432  }
433 
434  // evite toute autre forme d'interaction
435  foreach (QGraphicsView *view, diagram -> views()) {
436  view -> setInteractive(false);
437  }
438 
439  QRect diagram_rect = diagramRect(diagram, options);
440  if (fit_page) {
441  // impression adaptee sur une seule page
442  diagram -> render(qp, QRectF(), diagram_rect, Qt::KeepAspectRatio);
443  } else {
444  // impression sur une ou plusieurs pages
445  QRect printed_area = full_page ? printer -> paperRect() : printer -> pageRect();
446  //qDebug() << "impression sur une ou plusieurs pages";
447  //qDebug() << " schema :" << diagram_rect;
448  //qDebug() << " page :" << printed_area;
449 
450  int used_width = printed_area.width();
451  int used_height = printed_area.height();
452  int h_pages_count = horizontalPagesCount(diagram, options, full_page);
453  int v_pages_count = verticalPagesCount(diagram, options, full_page);
454 
455  QVector< QVector< QRect > > pages_grid;
456  // le schema est imprime sur une matrice de feuilles
457  // parcourt les lignes de la matrice
458  int y_offset = 0;
459  for (int i = 0 ; i < v_pages_count ; ++ i) {
460  pages_grid << QVector< QRect >();
461 
462  // parcourt les feuilles de la ligne
463  int x_offset = 0;
464  for (int j = 0 ; j < h_pages_count ; ++ j) {
465  pages_grid.last() << QRect(
466  QPoint(x_offset, y_offset),
467  QSize(
468  qMin(used_width, diagram_rect.width() - x_offset),
469  qMin(used_height, diagram_rect.height() - y_offset)
470  )
471  );
472  x_offset += used_width;
473  }
474 
475  y_offset += used_height;
476  }
477 
478  // ne retient que les pages a imprimer
479  QVector<QRect> pages_to_print;
480  for (int i = 0 ; i < v_pages_count ; ++ i) {
481  for (int j = 0 ; j < h_pages_count ; ++ j) {
482  pages_to_print << pages_grid.at(i).at(j);
483  }
484  }
485  //qDebug() << " " << pages_to_print.count() << " pages a imprimer :";
486 
487  // parcourt les pages pour impression
488  for (int i = 0 ; i < pages_to_print.count() ; ++ i) {
489  QRect current_rect(pages_to_print.at(i));
490  //qDebug() << " " << current_rect;
491  diagram -> render(
492  qp,
493  QRect(QPoint(0,0), current_rect.size()),
494  current_rect.translated(diagram_rect.topLeft()),
495  Qt::KeepAspectRatio
496  );
497  if (i != pages_to_print.count() - 1) {
498  printer -> newPage();
499  }
500  }
501  }
502 
503  // remet en place les interactions
504  foreach (QGraphicsView *view, diagram -> views()) {
505  view -> setInteractive(true);
506  }
507 
508  // restaure les flags focusable
509  foreach (QGraphicsItem *qgi, focusable_items) {
510  qgi -> setFlag(QGraphicsItem::ItemIsFocusable, true);
511  }
512 
513  // restaure les elements selectionnes
514  foreach (QGraphicsItem *qgi, selected_elmts) qgi -> setSelected(true);
515 
516  saveReloadDiagramParameters(diagram, options, false);
517 }

543 void DiagramPrintDialog::savePageSetupForCurrentPrinter()
544 {
545  QSettings settings;
546  QString printer_section = settingsSectionName(printer_);
547 
548  while (!settings.group().isEmpty()) settings.endGroup();
549  settings.beginGroup("printers");
550  settings.beginGroup(printer_section);
551 
552  settings.setValue("orientation", printer_ -> orientation() == QPrinter::Portrait ? "portrait" : "landscape");
553  settings.setValue("papersize", int(printer_ -> paperSize()));
554  if (printer_ -> paperSize() == QPrinter::Custom) {
555  QSizeF size = printer_ -> paperSize(QPrinter::Millimeter);
556  settings.setValue("customwidthmm", size.width());
557  settings.setValue("customheightmm", size.height());
558  } else {
559  settings.remove("customwidthmm");
560  settings.remove("customheightmm");
561  }
562  qreal left, top, right, bottom;
563  printer_ -> getPageMargins(&left, &top, &right, &bottom, QPrinter::Millimeter);
564  settings.setValue("marginleft", left);
565  settings.setValue("margintop", top);
566  settings.setValue("marginright", right);
567  settings.setValue("marginbottom", bottom);
568  settings.setValue("fullpage", printer_ -> fullPage() ? "true" : "false");
569  settings.endGroup();
570  settings.endGroup();
571  settings.sync();
572 }

579 void DiagramPrintDialog::loadPageSetupForCurrentPrinter()
580 {
581  QSettings settings;
582  QString printer_section = settingsSectionName(printer_);
583 
584  while (!settings.group().isEmpty()) settings.endGroup();
585  settings.beginGroup("printers");
586  if (!settings.childGroups().contains(printer_section)) {
587  settings.endGroup();
588  return;
589  }
590 
591  settings.beginGroup(printer_section);
592  if (settings.contains("orientation")) {
593  QString value = settings.value("orientation", "landscape").toString();
594  printer_ -> setOrientation(value == "landscape" ? QPrinter::Landscape : QPrinter::Portrait);
595  }
596  if (settings.contains("papersize")) {
597  int value = settings.value("papersize", QPrinter::A4).toInt();
598  if (value == QPrinter::Custom) {
599  bool w_ok, h_ok;
600  int w = settings.value("customwidthmm", -1).toInt(&w_ok);
601  int h = settings.value("customheightmm", -1).toInt(&h_ok);
602  if (w_ok && h_ok && w != -1 && h != -1) {
603  printer_ -> setPaperSize(QSizeF(w, h), QPrinter::Millimeter);
604  }
605  } else if (value < QPrinter::Custom) {
606  printer_ -> setPaperSize(static_cast<QPrinter::PaperSize>(value));
607  }
608  }
609 
610  qreal margins[4];
611  printer_ -> getPageMargins(&margins[0], &margins[1], &margins[2], &margins[3], QPrinter::Millimeter);
612  QStringList margins_names(QStringList() << "left" << "top" << "right" << "bottom");
613  for (int i = 0 ; i < 4 ; ++ i) {
614  bool conv_ok;
615  qreal value = settings.value("margin" + margins_names.at(i), -1.0).toReal(&conv_ok);
616  if (conv_ok && value != -1.0) margins[i] = value;
617  }
618  printer_ -> setPageMargins(margins[0], margins[1], margins[2], margins[3], QPrinter::Millimeter);
619  printer_ -> setFullPage(settings.value("fullpage", "false").toString() == "true");
620 
621  settings.endGroup();
622  settings.endGroup();
623 }
