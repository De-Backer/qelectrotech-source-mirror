266 void ConductorPropertiesWidget::updatePreview(bool b) {
267  const QRect pixmap_rect(0, 0, 96, 96);
268  QPixmap pixmap(pixmap_rect.width(), pixmap_rect.height());
269  QPainter painter;
270  painter.begin(&pixmap);
271  painter.eraseRect(pixmap_rect);
272  painter.drawRect(pixmap_rect.adjusted(0,0,-1,-1));
273  painter.drawLine(QLineF(0, pixmap_rect.height() / 2, pixmap_rect.width(), pixmap_rect.height() / 2));
274 
275  if (b) properties().singleLineProperties.draw(&painter, QET::Horizontal, pixmap_rect);
276  else m_properties.singleLineProperties.draw(&painter, QET::Horizontal, pixmap_rect);
277 
278  painter.end();
