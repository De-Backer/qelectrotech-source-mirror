166 bool QET::orthogonalProjection(const QPointF &point,
167  const QLineF &line,
168  QPointF *intersection) {
169  // recupere le vecteur normal de `line'
170  QLineF line_normal_vector(line.normalVector());
171  QPointF normal_vector(line_normal_vector.dx(), line_normal_vector.dy());
172 
173  // cree une droite perpendiculaire a `line' passant par `point'
174  QLineF perpendicular_line(point, point + normal_vector);
175 
176  // determine le point d'intersection des deux droites = le projete orthogonal
177  QPointF intersection_point;
178 #if QT_VERSION < QT_VERSION_CHECK(5, 14, 0)
179  QLineF::IntersectType it = line.intersect(perpendicular_line,
180  &intersection_point); // ### Qt 6: remove
181 #else
182 #pragma message("@TODO remove code for QT 5.14 or later")
183  QLineF::IntersectType it = line.intersects(perpendicular_line,
184  &intersection_point);
185 #endif
186 
187  // ne devrait pas arriver (mais bon...)
188  if (it == QLineF::NoIntersection) return(false);
189 
190  // fournit le point d'intersection a l'appelant si necessaire
191  if (intersection) {
192  *intersection = intersection_point;
193  }
194 
195  // determine si le point d'intersection appartient au segment de droite
196  if (QET::lineContainsPoint(line, intersection_point)) {
197  return(true);
198  }
199  return(false);
200 }

352 QList<QDomElement> QET::findInDomElement(const QDomElement &e,
353  const QString &parent,
354  const QString &children) {
355  QList<QDomElement> return_list;
356 
357  // parcours des elements parents
358  for (QDomNode enfant = e.firstChild() ; !enfant.isNull() ; enfant = enfant.nextSibling()) {
359  // on s'interesse a l'element XML "parent"
360  QDomElement parents = enfant.toElement();
361  if (parents.isNull() || parents.tagName() != parent) continue;
362  // parcours des enfants de l'element XML "parent"
363  for (QDomNode node_children = parents.firstChild() ; !node_children.isNull() ; node_children = node_children.nextSibling()) {
364  // on s'interesse a l'element XML "children"
365  QDomElement n_children = node_children.toElement();
366  if (!n_children.isNull() && n_children.tagName() == children) return_list.append(n_children);
367  }
368  }
369  return(return_list);
370 }
