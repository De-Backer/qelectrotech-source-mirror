57 void TitleBlockTemplateLocationSaver::setLocation(const TitleBlockTemplateLocation &location) {
58  // hack: if no suitable index was found, set it to 1, which is supposed to be the user collection
59  int index = indexForCollection(location.parentCollection());
60  if (index == -1 && collections_ -> count() > 1) index = 1;
61  collections_ -> setCurrentIndex(index);
62 
63  if (!location.name().isEmpty()) {
64  int template_index = templates_ -> findText(location.name());
65  if (template_index != -1) {
66  templates_ -> setCurrentIndex(template_index);
67  return;
68  }
69  }
70  templates_ -> setCurrentIndex(0);
71 }
