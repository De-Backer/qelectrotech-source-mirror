113 void NamesList::fromXml(const QDomElement &xml_element, const QHash<QString, QString> &xml_options) {
114  QHash<QString, QString> xml_opt = getXmlOptions(xml_options);
115  // parcourt les enfants "names" de l'element XML
116  for (QDomNode node = xml_element.firstChild() ; !node.isNull() ; node = node.nextSibling()) {
117  QDomElement names = node.toElement();
118  if (names.isNull() || names.tagName() != xml_opt["ParentTagName"]) continue;
119  // parcourt les petits-enfants "name"
120  for (QDomNode n = names.firstChild() ; !n.isNull() ; n = n.nextSibling()) {
121  QDomElement name = n.toElement();
122  if (name.isNull() || name.tagName() != xml_opt["TagName"]) continue;
123  addName(name.attribute(xml_opt["LanguageAttribute"]), name.text());
124  }
125  }
126 }

138 void NamesList::fromXml(const pugi::xml_node &xml_element, const QHash<QString, QString> &xml_options)
139 {
140  QHash<QString, QString> xml_opt = getXmlOptions(xml_options);
141 
142  //Walk the childs "names" of the xml element
143  for (auto names = xml_element.first_child() ; names ; names = names.next_sibling())
144  {
145  if (names.type() != pugi::node_element ||
146  QString(names.name()) != xml_opt["ParentTagName"]) {
147  continue;
148  }
149  for (auto name = names.first_child(); name; name = name.next_sibling()) {
150  if (name.type() != pugi::node_element ||
151  QString(name.name()) != xml_opt["TagName"]) {
152  continue;
153  }
154  QString lang_str(name.attribute(xml_opt["LanguageAttribute"].toStdString().c_str()).as_string());
155  QString name_str(name.text().get());
156  addName(lang_str, name_str);
157  }
158  }

169 QDomElement NamesList::toXml(QDomDocument &xml_document, const QHash<QString, QString> &xml_options) const {
170  QHash<QString, QString> xml_opt = getXmlOptions(xml_options);
171  QDomElement names_elmt = xml_document.createElement(xml_opt["ParentTagName"]);
172  QHashIterator<QString, QString> names_iterator(hash_names);
173  while (names_iterator.hasNext()) {
174  names_iterator.next();
175  QDomElement name_elmt = xml_document.createElement(xml_opt["TagName"]);
176  name_elmt.setAttribute(xml_opt["LanguageAttribute"], names_iterator.key());
177  name_elmt.appendChild(xml_document.createTextNode(names_iterator.value()));
178  names_elmt.appendChild(name_elmt);
179  }
180  return(names_elmt);
181 }
