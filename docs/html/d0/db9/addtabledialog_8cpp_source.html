193 void AddTableDialog::saveConfig()
194 {
195  QFile file_(QETApp::configDir() + "/graphics_table.json");
196 
197  if (file_.open(QFile::ReadWrite))
198  {
199  QJsonDocument doc_(QJsonDocument::fromJson(file_.readAll()));
200  QJsonObject root_object;
201 
202  if (!doc_.isEmpty())
203  {
204  root_object = doc_.object();
205  if (root_object.contains(ui->m_config_gb->text())) {
206  root_object.remove(ui->m_config_gb->text());
207  }
208  }
209 
210  QJsonObject header_object;
211  header_object.insert("margins", QETUtils::marginsToString(this->headerMargins()));
212  auto me = QMetaEnum::fromType<Qt::Alignment>();
213  header_object.insert("alignment", me.valueToKey(int(this->headerAlignment())));
214  header_object.insert("font", this->headerFont().toString());
215 
216  QJsonObject table_object;
217  table_object.insert("margins", QETUtils::marginsToString(this->tableMargins()));
218  table_object.insert("alignment", me.valueToKey(int(this->tableAlignment())));
219  table_object.insert("font", this->tableFont().toString());
220 
221  QJsonObject config_object;
222  config_object.insert("header", header_object);
223  config_object.insert("table", table_object);
224 
225  root_object.insert(ui->m_config_gb->text(), config_object);
226 
227  doc_.setObject(root_object);
228  file_.resize(0);
229  file_.write(doc_.toJson());
230  }
231 }

233 void AddTableDialog::loadConfig()
234 {
235  auto name = ui->m_config_gb->selectedText();
236  if (name.isEmpty()) {
237  return;
238  }
239 
240  QFile file_(QETApp::configDir() + "/graphics_table.json");
241  if (!file_.open(QFile::ReadOnly)) {
242  return;
243  }
244 
245  QJsonDocument doc_(QJsonDocument::fromJson(file_.readAll()));
246  QJsonObject root_object(doc_.object());
247  auto config_object = root_object.value(name).toObject();
248  if (config_object.isEmpty()) {
249  return;
250  }
251 
252  auto me = QMetaEnum::fromType<Qt::Alignment>();
253 
254  //Header
255  auto header_object = config_object.value("header").toObject();
256  m_header_margins = QETUtils::marginsFromString(header_object.value("margins").toString());
257  switch (me.keyToValue(header_object.value("alignment").toString().toStdString().data())) {
258  case Qt::AlignLeft :
259  ui->m_header_alignment_cb->setCurrentIndex(0);
260  break;
261  case Qt::AlignCenter :
262  ui->m_header_alignment_cb->setCurrentIndex(1);
263  break;
264  default:
265  ui->m_header_alignment_cb->setCurrentIndex(2);
266  }
267  m_header_font.fromString(header_object.value("font").toString());
268  ui->m_header_font_pb->setText(m_header_font.family());
269 
270  //Table
271  auto table_object = config_object.value("table").toObject();
272  m_table_margins = QETUtils::marginsFromString(table_object.value("margins").toString());
273  switch (me.keyToValue(table_object.value("alignment").toString().toStdString().data())) {
274  case Qt::AlignLeft :
275  ui->m_table_alignment_cb->setCurrentIndex(0);
276  break;
277  case Qt::AlignCenter :
278  ui->m_table_alignment_cb->setCurrentIndex(1);
279  break;
280  default:
281  ui->m_table_alignment_cb->setCurrentIndex(2);
282  }
283  m_table_font.fromString(table_object.value("font").toString());
284  ui->m_table_font_pb->setText(m_table_font.family());
285 
286 }
