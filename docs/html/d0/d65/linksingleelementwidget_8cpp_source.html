110 void LinkSingleElementWidget::setElement(Element *element)
111 {
112  if (m_element == element)
113  return;
114 
115  //Remove connection of previous edited element
116  if (m_element)
117  {
118  disconnect(m_element->diagram()->project(),
119  &QETProject::diagramRemoved,
120  this,
121  &LinkSingleElementWidget::diagramWasRemovedFromProject);
122  disconnect(m_element.data(),
123  &Element::linkedElementChanged,
124  this,
125  &LinkSingleElementWidget::updateUi);
126  m_element->setHighlighted(false);
127  }
128 
129  if(m_showed_element)
130  m_showed_element->setHighlighted(false);
131 
132  m_unlink = false;
133  m_showed_element = nullptr;
134  m_element_to_link = nullptr;
135  m_pending_qtwi = nullptr;
136 
137  //Setup the new element, connection and ui
138  m_element = element;
139 
140  if (m_element->linkType() & Element::Slave)
141  m_filter = Element::Master;
142  else if (m_element->linkType() & Element::AllReport)
143  m_filter = m_element->linkType() == Element::NextReport
144  ? Element::PreviousReport
145  : Element::NextReport;
146  else
147  m_filter = Element::Simple;
148 
149  connect(m_element->diagram()->project(), &QETProject::diagramRemoved,
150  this, &LinkSingleElementWidget::diagramWasRemovedFromProject);
151  connect(m_element.data(), &Element::linkedElementChanged,
152  this, &LinkSingleElementWidget::updateUi, Qt::QueuedConnection);
153 
154  updateUi();
155 }

368 QList <Element *> LinkSingleElementWidget::availableElements()
369 {
370  QList <Element *> elmt_list;
371  //if element isn't free and unlink isn't pressed, return an empty list
372  if (!m_element->isFree() && !m_unlink)
373  return elmt_list;
374 
375  if (!m_element->diagram() || !m_element->diagram()->project()) return elmt_list;
376 
377  ElementProvider ep(m_element->diagram()->project());
378  if (m_filter & Element::AllReport)
379  elmt_list = ep.freeElement(m_filter);
380  else
381  elmt_list = ep.find(m_filter);
382 
383  //If element is linked, remove is parent from the list
384  if(!m_element->isFree()) elmt_list.removeAll(m_element->linkedElements().first());
385 
386  return elmt_list;
387 }
