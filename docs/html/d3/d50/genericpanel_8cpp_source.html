227 QTreeWidgetItem *GenericPanel::fillProjectItem(QTreeWidgetItem *project_qtwi,
228  QETProject *project,
229  PanelOptions options,
230  bool freshly_created) {
231  if (!project_qtwi || !project) return(nullptr);
232 
233 
234  if (options & AddChildDiagrams) {
235  if (freshly_created) {
236  connect(
237  project,&QETProject::diagramAdded,
238  this,&GenericPanel::diagramAdded);
239  connect(
240  project,&QETProject::diagramRemoved,
241  this,&GenericPanel::diagramRemoved);
242  connect(
243  project,&QETProject::projectDiagramsOrderChanged,
244  this,&GenericPanel::projectDiagramsOrderChanged);
245  } else {
246  // remove diagrams unknown to the project (presumably removed)
247  removeObsoleteItems(project -> diagrams(),
248  project_qtwi, QET::Diagram, false);
249  }
250  int index = 0;
251  foreach (Diagram *diagram, project -> diagrams()) {
252  QTreeWidgetItem *diagram_qtwi = addDiagram(diagram,
253  nullptr,
254  options);
255  project_qtwi -> insertChild(index, diagram_qtwi);
256  ++ index;
257  }
258  }
259 
260  if (options & AddChildTemplatesCollection) {
261  if (freshly_created) {
262  connect(
263  project,&QETProject::diagramUsedTemplate,
264  this,&GenericPanel::diagramUsedTemplate);
265  }
266  addTemplatesCollection(
267  project -> embeddedTitleBlockTemplatesCollection(),
268  project_qtwi,
269  options);
270  }
271 
272  return(fillItem(project_qtwi, options, freshly_created));
273 }

747 void GenericPanel::projectDiagramsOrderChanged(QETProject *project,
748  int from,
749  int to)
750 {
751  // get the item representing the provided project
752  QTreeWidgetItem *qtwi_project = itemForProject(project);
753  if (!qtwi_project) return;
754 
755  // get the item representing the moved diagram
756  QTreeWidgetItem *moved_qtwi_diagram = qtwi_project -> child(from);
757  if (!moved_qtwi_diagram) return;
758 
759  // remove the QTWI then insert it back at the adequate location
760  bool was_selected = moved_qtwi_diagram -> isSelected();
761  qtwi_project -> removeChild (moved_qtwi_diagram);
762  qtwi_project -> insertChild (to, moved_qtwi_diagram);
763 
764  // update the QTWI labels because they may display the folio index
765  for (int i = qMin(from, to); i < qMax(from, to) + 1; i++)
766  {
767  QTreeWidgetItem *qtwi_diagram = qtwi_project -> child(i);
768  if (!qtwi_diagram) continue;
769 
770  Diagram *diagram = valueForItem<Diagram *>(qtwi_diagram);
771  if (diagram)
772  updateDiagramItem(qtwi_diagram, diagram);
773  }
774 
775  if (was_selected)
776  setCurrentItem(moved_qtwi_diagram);
777 
778  emit(panelContentChanged());
779 }

932 QList<QTreeWidgetItem *> GenericPanel::childItems(
933  QTreeWidgetItem *item,
934  QET::ItemType type,
935  bool recursive) const {
936  QList<QTreeWidgetItem *> items;
937  if (!item) return(items);
938  for (int i = 0 ; i < item -> childCount() ; ++ i) {
939  QTreeWidgetItem *current_item = item -> child(i);
940  if (!current_item) continue;
941  if (current_item -> type() == type) {
942  items << current_item;
943  }
944  if (recursive) {
945  items << childItems(current_item, type, true);
946  }
947  }
948  return(items);
949 }
