55 ExportProperties ExportPropertiesWidget::exportProperties() const {
56  ExportProperties export_properties;
57 
58  export_properties.destination_directory = QDir(dirpath -> text());
59  export_properties.format = format -> itemData(format -> currentIndex()).toString();
60  export_properties.draw_grid = draw_grid -> isChecked();
61  export_properties.draw_border = draw_border -> isChecked();
62  export_properties.draw_titleblock = draw_titleblock -> isChecked();
63  export_properties.draw_terminals = draw_terminals -> isChecked();
64  export_properties.draw_colored_conductors = draw_colored_conductors -> isChecked();
65  export_properties.exported_area = export_border -> isChecked() ? QET::BorderArea : QET::ElementsArea;
66 
67  return(export_properties);

74 void ExportPropertiesWidget::setExportProperties(const ExportProperties &export_properties) {
75  dirpath -> setText(QDir::toNativeSeparators(export_properties.destination_directory.absolutePath()));
76 
77  int index = format -> findData(export_properties.format);
78  if (index == -1) index = 0;
79  format -> setCurrentIndex(index);
80 
81  draw_grid -> setChecked(export_properties.draw_grid);
82  draw_border -> setChecked(export_properties.draw_border);
83  draw_titleblock -> setChecked(export_properties.draw_titleblock);
84  draw_terminals -> setChecked(export_properties.draw_terminals);
85  draw_colored_conductors -> setChecked(export_properties.draw_colored_conductors);
86 
87  if (export_properties.exported_area == QET::BorderArea) {
88  export_border -> setChecked(true);
89  } else {
90  export_elements -> setChecked(true);
91  }
92 }
