147 bool ElementPictureFactory::build(const ElementsLocation &location,
148  QPicture *picture,
149  QPicture *low_picture)
150 {
151  QDomElement dom = location.xml();
152 
153  //Check if the curent version can read the xml description
154  if (dom.hasAttribute("version"))
155  {
156  bool conv_ok;
157  qreal element_version = dom.attribute("version").toDouble(&conv_ok);
158  if (conv_ok && QET::version.toDouble() < element_version)
159  {
160  std::cerr << qPrintable(
161  QObject::tr("Avertissement : l'élément "
162  " a été enregistré avec une version"
163  " ultérieure de QElectroTech.")
164  ) << std::endl;
165  }
166  }
167 
168  //This attributes must be present and valid
169  int w, h, hot_x, hot_y;
170  if (!QET::attributeIsAnInteger(dom, QString("width"), &w) ||\
171  !QET::attributeIsAnInteger(dom, QString("height"), &h) ||\
172  !QET::attributeIsAnInteger(dom, QString("hotspot_x"), &hot_x) ||\
173  !QET::attributeIsAnInteger(dom, QString("hotspot_y"), &hot_y))
174  {
175  return(false);
176  }
177 
178  QPainter painter;
179  QPicture pic;
180  primitives primitives_;
181  if (picture) {
182  painter.begin(picture);
183  }
184  else {
185  painter.begin(&pic);
186  }
187  painter.setRenderHint(QPainter::Antialiasing, true);
188  painter.setRenderHint(QPainter::TextAntialiasing, true);
189  painter.setRenderHint(QPainter::SmoothPixmapTransform,true);
190 
191 
192  QPainter low_painter;
193  QPicture low_pic;
194  if (low_picture) {
195  low_painter.begin(low_picture);
196  }
197  else {
198  low_painter.begin(&low_pic);
199  }
200  low_painter.setRenderHint(QPainter::Antialiasing, true);
201  low_painter.setRenderHint(QPainter::TextAntialiasing, true);
202  low_painter.setRenderHint(QPainter::SmoothPixmapTransform,true);
203 
204  QPen tmp;
205  tmp.setWidthF(1.0); //Vaudoo line to take into account the setCosmetic - don't remove
206  tmp.setCosmetic(true);
207  low_painter.setPen(tmp);
208 
209  //scroll of the Children of the Definition: Parts of the Drawing
210  for (QDomNode node = dom.firstChild() ; !node.isNull() ; node = node.nextSibling())
211  {
212  QDomElement elmts = node.toElement();
213  if (elmts.isNull()) {
214  continue;
215  }
216 
217  if (elmts.tagName() == "description")
218  {
219  //Manage the graphic description = part of drawing
220  for (QDomNode n = node.firstChild() ; !n.isNull() ; n = n.nextSibling())
221  {
222  QDomElement qde = n.toElement();
223  if (qde.isNull()) {
224  continue;
225  }
226  parseElement(qde, painter, primitives_);
227  primitives fake_prim;
228  parseElement(qde, low_painter, fake_prim);
229  }
230  }
231  }
232 
233  //End of the drawing
234  painter.end();
235  low_painter.end();
236 
237  if (!picture) {
238  m_pictures_H.insert(location.uuid(), pic);
239  m_primitives_H.insert(location.uuid(), primitives_);
240  }
241  if (!low_picture) {
242  m_low_pictures_H.insert(location.uuid(), low_pic);
243  m_primitives_H.insert(location.uuid(), primitives_);
244  }
245  return true;
246 }

259 void ElementPictureFactory::parseLine(const QDomElement &dom, QPainter &painter, primitives &prim) const
260 {
261  //This attributes must be present and valid
262  qreal x1, y1, x2, y2;
263  if (!QET::attributeIsAReal(dom, QString("x1"), &x1)) return;
264  if (!QET::attributeIsAReal(dom, QString("y1"), &y1)) return;
265  if (!QET::attributeIsAReal(dom, QString("x2"), &x2)) return;
266  if (!QET::attributeIsAReal(dom, QString("y2"), &y2)) return;
267 
268  Qet::EndType first_end = Qet::endTypeFromString(dom.attribute("end1"));
269  Qet::EndType second_end = Qet::endTypeFromString(dom.attribute("end2"));
270  qreal length1, length2;
271  if (!QET::attributeIsAReal(dom, QString("length1"), &length1)) length1 = 1.5;
272  if (!QET::attributeIsAReal(dom, QString("length2"), &length2)) length2 = 1.5;
273 
274  painter.save();
275  setPainterStyle(dom, painter);
276  QPen t = painter.pen();
277  t.setJoinStyle(Qt::MiterJoin);
278  painter.setPen(t);
279 
280  QLineF line(x1, y1, x2, y2);
281 
282  prim.m_lines << line;
283 
284  QPointF point1(line.p1());
285  QPointF point2(line.p2());
286 
287  qreal line_length(line.length());
288  qreal pen_width = painter.pen().widthF();
289 
290  //Check if we must to draw extremity
291  bool draw_1st_end, draw_2nd_end;
292  qreal reduced_line_length = line_length - (length1 * PartLine::requiredLengthForEndType(first_end));
293  draw_1st_end = first_end && reduced_line_length >= 0;
294  if (draw_1st_end) {
295  reduced_line_length -= (length2 * PartLine::requiredLengthForEndType(second_end));
296  } else {
297  reduced_line_length = line_length - (length2 * PartLine::requiredLengthForEndType(second_end));
298  }
299  draw_2nd_end = second_end && reduced_line_length >= 0;
300 
301  //Draw first extremity
302  QPointF start_point, stop_point;
303  if (draw_1st_end) {
304  QList<QPointF> four_points1(PartLine::fourEndPoints(point1, point2, length1));
305  if (first_end == Qet::Circle) {
306  painter.drawEllipse(QRectF(four_points1[0] - QPointF(length1, length1), QSizeF(length1 * 2.0, length1 * 2.0)));
307  start_point = four_points1[1];
308  } else if (first_end == Qet::Diamond) {
309  painter.drawPolygon(QPolygonF() << four_points1[1] << four_points1[2] << point1 << four_points1[3]);
310  start_point = four_points1[1];
311  } else if (first_end == Qet::Simple) {
312  painter.drawPolyline(QPolygonF() << four_points1[3] << point1 << four_points1[2]);
313  start_point = point1;
314 
315  } else if (first_end == Qet::Triangle) {
316  painter.drawPolygon(QPolygonF() << four_points1[0] << four_points1[2] << point1 << four_points1[3]);
317  start_point = four_points1[0];
318  }
319 
320  //Adjust the begining according to the width of the pen
321  if (pen_width && (first_end == Qet::Simple || first_end == Qet::Circle)) {
322  start_point = QLineF(start_point, point2).pointAt(pen_width / 2.0 / line_length);
323  }
324  } else {
325  start_point = point1;
326  }
327 
328  //Draw second extremity
329  if (draw_2nd_end) {
330  QList<QPointF> four_points2(PartLine::fourEndPoints(point2, point1, length2));
331  if (second_end == Qet::Circle) {
332  painter.drawEllipse(QRectF(four_points2[0] - QPointF(length2, length2), QSizeF(length2 * 2.0, length2 * 2.0)));
333  stop_point = four_points2[1];
334  } else if (second_end == Qet::Diamond) {
335  painter.drawPolygon(QPolygonF() << four_points2[2] << point2 << four_points2[3] << four_points2[1]);
336  stop_point = four_points2[1];
337  } else if (second_end == Qet::Simple) {
338  painter.drawPolyline(QPolygonF() << four_points2[3] << point2 << four_points2[2]);
339  stop_point = point2;
340  } else if (second_end == Qet::Triangle) {
341  painter.drawPolygon(QPolygonF() << four_points2[0] << four_points2[2] << point2 << four_points2[3] << four_points2[0]);
342  stop_point = four_points2[0];
343  }
344 
345  //Adjust the end according to the width of the pen
346  if (pen_width && (second_end == Qet::Simple || second_end == Qet::Circle)) {
347  stop_point = QLineF(point1, stop_point).pointAt((line_length - (pen_width / 2.0)) / line_length);
348  }
349  } else {
350  stop_point = point2;
351  }
352 
353  painter.drawLine(start_point, stop_point);
354 
355  painter.restore();
356 }

488 {

502  }

505  }

514 
515  painter.setTransform(QTransform(), false);

517  painter.rotate(dom.attribute("rotation", "0").toDouble());
