376 bool Element::buildFromXml(const QDomElement &xml_def_elmt, int *state)
377 {
378  m_state = QET::GIBuildingFromXml;
379 
380  if (xml_def_elmt.tagName() != "definition"
381  || xml_def_elmt.attribute("type") != "element")
382  {
383  if (state) *state = 4;
384  m_state = QET::GIOK;
385  return(false);
386  }
387 
388  //Check if the curent version can read the xml description
389  if (xml_def_elmt.hasAttribute("version"))
390  {
391  bool conv_ok;
392  qreal element_version = xml_def_elmt.attribute(
393  "version").toDouble(&conv_ok);
394  if (conv_ok && QET::version.toDouble() < element_version)
395  {
396  std::cerr << qPrintable(
397  QObject::tr("Avertissement : l'élément "
398  " a été enregistré avec une version"
399  " ultérieure de QElectroTech.")
400  ) << std::endl;
401  }
402  }
403 
404  //This attribute must be present and valid
405  int w, h, hot_x, hot_y;
406  if (
407  !QET::attributeIsAnInteger(xml_def_elmt, QString("width"), &w) ||\
408  !QET::attributeIsAnInteger(xml_def_elmt, QString("height"), &h) ||\
409  !QET::attributeIsAnInteger(xml_def_elmt, QString("hotspot_x"), &hot_x) ||\
410  !QET::attributeIsAnInteger(xml_def_elmt, QString("hotspot_y"), &hot_y)
411  ) {
412  if (state) *state = 5;
413  m_state = QET::GIOK;
414  return(false);
415  }
416 
417  setSize(w, h);
418  setHotspot(QPoint(hot_x, hot_y));
419 
420  //the definition must have childs
421  if (xml_def_elmt.firstChild().isNull())
422  {
423  if (state) *state = 6;
424  m_state = QET::GIOK;
425  return(false);
426  }
427  //Extract the names
428  m_names.fromXml(xml_def_elmt);
429  setToolTip(name());
430 
431  //load kind informations
432  m_kind_informations.fromXml(
433  xml_def_elmt.firstChildElement("kindInformations"),
434  "kindInformation");
435  //load element information
436  m_element_informations.fromXml(
437  xml_def_elmt.firstChildElement("elementInformations"),
438  "elementInformation");
439 
440  //scroll of the Children of the Definition: Parts of the Drawing
441  int parsed_elements_count = 0;
442  for (QDomNode node = xml_def_elmt.firstChild() ;
443  !node.isNull() ;
444  node = node.nextSibling())
445  {
446  QDomElement elmts = node.toElement();
447  if (elmts.isNull())
448  continue;
449 
450  if (elmts.tagName() == "description")
451  {
452  //Minor workaround to find if there is a "input" tagg as label.
453  //If not, we set the tagg "label" to the first "input.
454  QList <QDomElement> input_field;
455  bool have_label = false;
456  for (QDomElement input_node = node.firstChildElement("input") ;
457  !input_node.isNull() ;
458  input_node = input_node.nextSiblingElement("input"))
459  {
460  if (!input_node.isNull())
461  {
462  input_field << input_node;
463  if (input_node.attribute("tagg", "none")
464  == "label")
465  have_label = true;
466  }
467  }
468  if(!have_label && !input_field.isEmpty())
469  input_field.first().setAttribute("tagg",
470  "label");
471 
472  //Parse the definition
473  for (QDomNode n = node.firstChild() ;
474  !n.isNull() ;
475  n = n.nextSibling())
476  {
477  QDomElement qde = n.toElement();
478  if (qde.isNull())
479  continue;
480 
481  if (parseElement(qde)) {
482  ++ parsed_elements_count;
483  }
484  else
485  {
486  if (state)
487  *state = 7;
488  m_state = QET::GIOK;
489  return(false);
490  }
491  }
492  }
493  }
494 
495  ElementPictureFactory *epf = ElementPictureFactory::instance();
496  epf->getPictures(m_location,
497  const_cast<QPicture&>(m_picture),
498  const_cast<QPicture&>(m_low_zoom_picture));
499 
500  if(!m_picture.isNull())
501  ++ parsed_elements_count;
502 
503  //They must be at least one parsed graphics part
504  if (!parsed_elements_count)
505  {
506  if (state)
507  *state = 8;
508  m_state = QET::GIOK;
509  return(false);
510  }
511  else
512  {
513  if (state)
514  *state = 0;
515  m_state = QET::GIOK;
516  return(true);
517  }
518 }

542 bool Element::parseInput(const QDomElement &dom_element)
543 {
544  qreal pos_x, pos_y;
545  int size;
546  if (
547  !QET::attributeIsAReal(dom_element, "x", &pos_x) ||\
548  !QET::attributeIsAReal(dom_element, "y", &pos_y) ||\
549  !QET::attributeIsAnInteger(dom_element, "size", &size)
550  ) return(false);
551  else
552  {
553  DynamicElementTextItem *deti = new DynamicElementTextItem(this);
554  deti->setText(dom_element.attribute("text", "_"));
555  QFont font = deti->font();
556  font.setPointSize(dom_element.attribute("size",
557  QString::number(9)).toInt());
558  deti->setFont(font);
559  deti->setRotation(dom_element.attribute("rotation",
560  QString::number(0)).toDouble());
561 
562  if(dom_element.attribute("tagg", "none") != "none")
563  {
564  deti->setTextFrom(DynamicElementTextItem::ElementInfo);
565  deti->setInfoName(dom_element.attribute("tagg"));
566  }
567 
568  //the origin transformation point of PartDynamicTextField is the top left corner, no matter the font size
569  //The origin transformation point of ElementTextItem is the middle of left edge, and so by definition, change with the size of the font
570  //We need to use a QTransform to find the pos of this text from the saved pos of text item
571  QTransform transform;
572  //First make the rotation
573  transform.rotate(dom_element.attribute("rotation",
574  "0").toDouble());
575  QPointF pos = transform.map(
576  QPointF(0,
577  -deti->boundingRect().height()/2));
578  transform.reset();
579  //Second translate to the pos
580  QPointF p(dom_element.attribute("x",
581  QString::number(0)).toDouble(),
582  dom_element.attribute("y",
583  QString::number(0)).toDouble());
584  transform.translate(p.x(), p.y());
585  deti->setPos(transform.map(pos));
586  m_converted_text_from_xml_description.insert(deti, p);
587  return true;
588  }

691  les bornes vont maintenant etre recensees pour associer leurs id a leur adresse reelle
692  ce recensement servira lors de la mise en place des fils
693  */
694  QList<QDomElement> liste_terminals;
695  foreach(QDomElement qde,
696  QET::findInDomElement(e, "terminals", "terminal")) {
697  if (Terminal::valideXml(qde)) liste_terminals << qde;
698  }
699 
700  QHash<int, Terminal *> priv_id_adr;
701  int terminals_non_trouvees = 0;
702  foreach(QGraphicsItem *qgi, childItems()) {
703  if (Terminal *p = qgraphicsitem_cast<Terminal *>(qgi)) {
704  bool terminal_trouvee = false;
705  foreach(QDomElement qde, liste_terminals) {
706  if (p -> fromXml(qde)) {
707  priv_id_adr.insert(
708  qde.attribute(
709  "id").toInt(),
710  p);

712  // We used to break here, because we did not expect
713  // several terminals to share the same position.
714  // Of course, it finally happened.
715  }
716  }
717  if (!terminal_trouvee) ++ terminals_non_trouvees;
718  }
719  }
720 
721  if (terminals_non_trouvees > 0)
722  {
723  m_state = QET::GIOK;
724  return(false);
725  }
726  else
727  {

729  foreach(int id_trouve, priv_id_adr.keys())
730  {

733  // cet element possede un id qui est deja reference (= conflit)

739  foreach(int id_trouve, priv_id_adr.keys()) {

747  "links_uuids",
748  "link_uuid");
749  foreach (QDomElement qdo, uuid_list)
750  tmp_uuids_link << qdo.attribute("uuid");

752  //uuid of this element
753  m_uuid= QUuid(e.attribute("uuid", QUuid::createUuid().toString()));
754 
755  //load prefix
756  m_prefix = e.attribute("prefix");
757 
758  QString fl = e.attribute("freezeLabel", "false");

760 
761  //Load Sequential Values
762  if (e.hasAttribute("sequ_1")
763  || e.hasAttribute("sequf_1")
764  || e.hasAttribute("seqt_1")
765  || e.hasAttribute("seqtf_1")
766  || e.hasAttribute("seqh_1")
767  || e.hasAttribute("sequf_1"))
768  ElementXmlRetroCompatibility::loadSequential(e, this);
769  else
770  m_autoNum_seq.fromXml(e.firstChildElement("sequentialNumbers"));
771 
772  //Position and selection.
773  //We directly call setPos from QGraphicsObject, because QetGraphicsItem will snap to grid
774  QGraphicsObject::setPos(e.attribute("x").toDouble(),
775  e.attribute("y").toDouble());
776  setZValue(e.attribute("z", QString::number(this->zValue())).toDouble());
777  setFlags(QGraphicsItem::ItemIsMovable
778  | QGraphicsItem::ItemIsSelectable);
779 
780  // orientation
781  bool conv_ok;
782  int read_ori = e.attribute("orientation").toInt(&conv_ok);
783  if (!conv_ok || read_ori < 0 || read_ori > 3) read_ori = 0;
784  if (handle_inputs_rotation) {
785  setRotation(rotation() + (90*read_ori));
786  } else {
787  setRotation(90*read_ori);
788  }
789 
790  //Befor load the dynamic text field,
791  //we remove the dynamic text field created from the description of this element, to avoid doublons.
792  for(DynamicElementTextItem *deti : m_dynamic_text_list)
793  delete deti;
794  m_dynamic_text_list.clear();
795 
796  //************************//
797  //***Dynamic texts item***//
798  //************************//
799  for (const QDomElement& qde : QET::findInDomElement(
800  e,
801  "dynamic_texts",
802  DynamicElementTextItem::xmlTagName()))
803  {

805  addDynamicTextItem(deti);

807  }

809

811  //***Element texts item***//
812  //************************//
813  QList<QDomElement> inputs = QET::findInDomElement(e, "inputs", "input");
814 
815  //First case, we check for the text item converted to dynamic text item
816  const QList <DynamicElementTextItem *> conv_deti_list =
817  m_converted_text_from_xml_description.keys();
818  QList <DynamicElementTextItem *> successfully_converted;
819  const QList <QDomElement> dom_inputs = inputs;
820 
821  for (DynamicElementTextItem *deti : conv_deti_list)
822  {
823  for(const QDomElement& dom_input : dom_inputs)
