359 void SearchAndReplaceWidget::search()
360 {
361  QString str = ui->m_search_le->text();
362  if(str.isEmpty())
363  {
364  for (QTreeWidgetItemIterator it(m_root_qtwi) ; *it ; ++it) {
365  (*it)->setHidden(false);
366  }
367 
368  for (QTreeWidgetItem *item : m_category_qtwi) {
369  item->setExpanded(false);
370  }
371  m_root_qtwi->setExpanded(true);
372 
373  ui->m_tree_widget->setCurrentItem(m_root_qtwi);
374  ui->m_search_le->setPalette(QPalette());
375  }
376  else
377  {
378  for (QTreeWidgetItemIterator it(m_root_qtwi) ; *it ; ++it) {
379  (*it)->setHidden(true);
380  }
381 
382  bool match = false;
383 
384  //Extended search,
385  //on each string stored in column 0 with role : UserRole
386  QList<QTreeWidgetItem *> qtwi_list;
387  qtwi_list.append(m_diagram_hash.keys());
388  qtwi_list.append(m_element_hash.keys());
389  qtwi_list.append(m_text_hash.keys());
390  qtwi_list.append(m_conductor_hash.keys());
391  for (QTreeWidgetItem *qtwi : qtwi_list)
392  {
393  QStringList list = qtwi->data(0, Qt::UserRole)
394  .toStringList();
395 
396  if(ui->m_mode_cb->currentIndex() == 0)
397  {
398  //Contain string
399  list = list.filter(
400  str,
401  ui->m_case_sensitive_cb->isChecked()
402  ? Qt::CaseSensitive
403  : Qt::CaseInsensitive);
404  }
405  else
406  {
407  //entire word
408  QRegularExpression rx("\\b" + str + "\\b");
409  if (!ui->m_case_sensitive_cb->isChecked()) {
410  rx.setPatternOptions(
411  QRegularExpression::CaseInsensitiveOption);
412  }
413  list = list.filter(rx);
414  }
415  if (!list.isEmpty())
416  {
417  match = true;
418  qtwi->setHidden(false);
419  setVisibleAllParents(qtwi);
420  }
421  }
422 
423  QPalette background = ui->m_search_le->palette();
424  background.setColor(QPalette::Base, match
425  ? QColor("#E0FFF0")
426  : QColor("#FFE0EF"));
427  ui->m_search_le->setPalette(background);
428 
429  //Go to the first occurence
430  ui->m_tree_widget->setCurrentItem(m_root_qtwi);
431  on_m_next_pb_clicked();
432  }
433 }

611 void SearchAndReplaceWidget::updateParentCheckState(QTreeWidgetItem *item,
612  bool all_parents)
613 {
614  QTreeWidgetItem *parent = item->parent();
615 
616  if (!parent) {
617  parent = item;
618  }
619 
620  int check=0,
621  partially=0;
622 
623  for (int i=0 ; i<parent->childCount() ; ++i)
624  {
625  switch (parent->child(i)->checkState(0)) {
626  case Qt::Checked:
627  ++check;
628  break;
629  case Qt::PartiallyChecked:
630  ++partially;
631  break;
632  default:
633  break;
634  }
635  }
636 
637  if (check == parent->childCount()) {
638  parent->setCheckState(0, Qt::Checked);
639  }
640  else if (partially || check) {
641  parent->setCheckState(0, Qt::PartiallyChecked);
642  }
643  else {
644  parent->setCheckState(0, Qt::Unchecked);
645  }
646 
647  if (all_parents && item->parent()) {
648  updateParentCheckState(parent, all_parents);
649  }
650 }

789 QStringList SearchAndReplaceWidget::searchTerms(Element *element)
790 {
791  QStringList list;
792  DiagramContext context = element->elementInformations();
793  for (QString key : QETApp::elementInfoKeys())
794  {
795  QString str = context.value(key).toString();
796  if (!str.isEmpty()) {
797  list.append(str);
798  }
799  }
800 
801  for (DynamicElementTextItem *deti : element->dynamicTextItems())
802  {
803  if (deti->textFrom()
804  == DynamicElementTextItem::UserText
805  || deti->textFrom()
806  == DynamicElementTextItem::CompositeText) {
807  list.append(deti->toPlainText());
808  }
