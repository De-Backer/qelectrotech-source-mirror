111 void ElementQueryWidget::setQuery(const QString &query)
112 {
113  if (query.startsWith("SELECT"))
114  {
115  reset();
116 
117  ui->m_sql_query->setText(query);
118 
119  QString select = query;
120  select.remove(0,7); //Remove SELECT
121  select.truncate(select.indexOf("FROM")); //Truncate at FROM
122  select.replace(" ", ""); //Remove white space
123 
124  //Get the select -> the item in the right list
125  QStringList split = select.split(",");
126  for (auto str : split)
127  {
128  for (auto item : m_items_list)
129  {
130  if (item->data(Qt::UserRole).toString() == str) {
131  ui->m_var_list->takeItem(ui->m_var_list->row(item));
132  ui->m_choosen_list->addItem(item);
133  continue;
134  }
135  }
136  }
137 
138  //There is not filter return now.
139  if (!query.contains("WHERE"))
140  {
141  ui->m_all_cb->setCheckState(Qt::Checked);
142  for (auto b : m_button_group.buttons())
143  b->setChecked(true);
144  return;
145  }
146 
147  //Get the filter
148  auto where = query;
149  where.remove(0, where.indexOf("WHERE") + 6);
150  where.truncate(where.indexOf("ORDER BY"));
151 
152  //Element type filter
153  if (where.contains("element_sub_type") || where.contains("element_type"))
154  {
155  int c=0;
156  ui->m_simple_cb->setChecked (where.contains("Simple") ? true : false);
157  if (ui->m_simple_cb->isChecked()) {
158  ++c;
159  where.remove("element_type = 'Simple' ");
160  }
161  ui->m_terminal_cb->setChecked (where.contains("Terminale") ? true : false);
162  if (ui->m_terminal_cb->isChecked()) {
163  ++c;
164  where.remove("element_type = 'Terminale'");
165  }
166  ui->m_coil_cb->setChecked (where.contains("coil") ? true : false);
167  if (ui->m_coil_cb->isChecked()) {
168  ++c;
169  where.remove("element_sub_type = 'coil' ");
170  }
171  ui->m_button_cb->setChecked (where.contains("commutator") ? true : false);
172  if (ui->m_button_cb->isChecked()) {
173  ++c;
174  where.remove("element_sub_type = 'commutator' ");
175  }
176  ui->m_protection_cb->setChecked(where.contains("protection") ? true : false);
177  if (ui->m_protection_cb) {
178  ++c;
179  where.remove("element_sub_type = 'protection'");
180  }
181 
182  if (c == 5) {
183  ui->m_all_cb->setCheckState(Qt::Checked);
184  } else if (c > 0) {
185  ui->m_all_cb->setCheckState(Qt::PartiallyChecked);
186  }
187 
188  where.remove("OR");
189  }
190  else // There is not "element_sub_type" or "element_type" that mean every element are selected
191  {
192  ui->m_all_cb->setCheckState(Qt::Checked);
193  for (auto b : m_button_group.buttons())
194  b->setChecked(true);
195  }
196 
197 
198  //Filter for selected data
199  QStringList strl;
200  for (auto item : m_items_list) {
201  strl.append(item->data(Qt::UserRole).toString());
202  }
203 
204  QString beginning_rx;
205  beginning_rx.append(QString("^(").append(strl.join("|")));
206  beginning_rx.append(")");
207 
208  QRegularExpression rx_is_not_null(beginning_rx + " IS NOT NULL$");
209  QRegularExpression rx_is_null (beginning_rx + " IS NULL$");
210  QRegularExpression rx_like (beginning_rx + QString(" LIKE'%(.+)%'$"));
211  QRegularExpression rx_not_like (beginning_rx + QString(" NOT LIKE'%(.+)%'$"));
212  QRegularExpression rx_equal (beginning_rx + QString("='(.+)'$"));
213  QRegularExpression rx_not_equal (beginning_rx + QString("!='(.+)'$"));
214 
215 
216  QStringList split_where;
217 
218  //Remove the white space at begin and end of each string
219  for (auto str : where.split("AND "))
220  {
221  if (str.startsWith(" "))
222  str.remove(0,1);
223  if (str.endsWith(" "))
224  str.remove(str.size()-1, 1);
225 
226  split_where.append(str);
227  }
228 
229 
230  QRegularExpressionMatch rxm;
231  for (auto str : split_where)
232  {
233  rxm = rx_is_not_null.match(str);
234  if (rxm.hasMatch()) {
235  m_filter.insert(rxm.captured(1), qMakePair(1, QString()));
236  continue;
237  }
238 
239  rxm = rx_is_null.match(str);
240  if (rxm.hasMatch()) {
241  m_filter.insert(rxm.captured(1), qMakePair(2, QString()));
242  continue;
243  }
244 
245  rxm = rx_like.match(str);
246  if (rxm.hasMatch()) {
247  m_filter.insert(rxm.captured(1), qMakePair(3, rxm.captured(2)));
248  continue;
249  }
250 
251  rxm = rx_not_like.match(str);
252  if (rxm.hasMatch()) {
253  m_filter.insert(rxm.captured(1), qMakePair(4, rxm.captured(2)));
254  continue;
255  }
256 
257  rxm = rx_equal.match(str);
258  if (rxm.hasMatch()) {
259  m_filter.insert(rxm.captured(1), qMakePair(5, rxm.captured(2)));
260  continue;
261  }
262 
263  rxm = rx_not_equal.match(str);
264  if (rxm.hasMatch()) {
265  m_filter.insert(rxm.captured(1), qMakePair(6, rxm.captured(2)));
266  continue;
267  }
268 
269  }
270  }
271 }

277 QString ElementQueryWidget::queryStr() const
278 {
279  //User define is own query
280  if (ui->m_edit_sql_query_cb->isChecked()) {
281  return ui->m_sql_query->text();
282  }
283  //Made a string list with the colomns (keys) choosen by the user
284  QStringList keys = selectedKeys();
285 
286  QString select ="SELECT ";
287  QString order_by = " ORDER BY ";
288  QString filter_;
289 
290  QString column;
291  bool first = true;
292  for (auto key: keys) {
293  if (first) {
294  first = false;
295  } else {
296  column += ", ";
297  order_by += ", ";
298  }
299  column += key;
300  order_by += key;
301 
302  auto f = FilterFor(key);
303  switch (f.first)
304  {
305  case 0: //No filter
306  break;
307  case 1: //Not empty
308  filter_ += QString(" AND ") += key += " IS NOT NULL";
309  break;
310  case 2: //empty
311  filter_ += QString(" AND ") += key += " IS NULL";
312  break;
313  case 3: // contain
314  filter_ += QString(" AND ") += key += QString(" LIKE'%") += f.second += "%'";
315  break;
316  case 4: // not contain
317  filter_ += QString(" AND ") += key += QString(" NOT LIKE'%") += f.second += "%'";
318  break;
319  case 5: // is equal
320  filter_ += QString(" AND ") += key += QString("='") += f.second += "'";
321  break;
322  case 6: // is not equal
323  filter_ += QString(" AND ") += key += QString("!='") += f.second += "'";
324  break;
325  }
326  }
327 
328  QString from = " FROM element_nomenclature_view";
329 
330  QString where;
331  where = " WHERE (";
332  bool b = false;
333  if (ui->m_terminal_cb->isChecked()) {
334  if (b) where +=" OR";
335  where += " element_type = 'Terminale'";
336  b = true;
337  }
338  if (ui->m_simple_cb->isChecked()) {
339  if (b) where +=" OR";
340  where += " element_type = 'Simple'";
341  b = true;
342  }
343  if (ui->m_button_cb->isChecked()) {
344  if (b) where +=" OR";
345  where += " element_sub_type = 'commutator'";
346  b = true;
347  }
348  if (ui->m_coil_cb->isChecked()) {
349  if (b) where +=" OR";
350  where += " element_sub_type = 'coil'";
351  b = true;
352  }
353  if (ui->m_protection_cb->isChecked()) {
354  if (b) where +=" OR";
355  where += " element_sub_type = 'protection'";
356  }
357  where.append(")");
358 
359  if (where == " WHERE ()") {
360  where.clear();
361  }
362 
363  if (where.isEmpty() && !filter_.isEmpty()) {
364  filter_.remove(0, 4); //Remove the first " AND" of filter.
365  filter_.prepend( " WHERE");
366  }
367 
368  QString q(select + column + m_count + from + where + filter_ + m_group_by + order_by);
369  return q;
370 }
