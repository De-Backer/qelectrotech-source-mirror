257 QPainterPath QetShapeItem::shape() const
258 {
259  QPainterPath path;
260 
261  switch (m_shapeType)
262  {
263  case Line:
264  path.moveTo(m_P1);
265  path.lineTo(m_P2);
266  break;
267  case Rectangle:
268  path.addRoundedRect(QRectF(m_P1, m_P2),
269  m_xRadius,
270  m_yRadius);
271  break;
272  case Ellipse:
273  path.addEllipse(QRectF(m_P1, m_P2));
274  break;
275  case Polygon:
276  path.addPolygon(m_polygon);
277  if (m_closed) {
278  path.closeSubpath();
279  }
280  break;
281  }
282 
283  QPainterPathStroker pps;
284  pps.setWidth(m_hovered? m_pen.widthF()+10 : m_pen.widthF());
285  pps.setJoinStyle(Qt::RoundJoin);
286  path = pps.createStroke(path);
287 
288  return (path);
289 }

299  const QStyleOptionGraphicsItem *option,
300  QWidget *widget)
301 {
302  Q_UNUSED(option)
303  Q_UNUSED(widget)
304 
305  painter->save();
306  painter->setRenderHint(QPainter::Antialiasing, true);
307  painter->setPen(m_pen);
308  painter->setBrush(m_brush);
309 
310  //Draw hovered shadow
311  if (m_hovered)
312  {
313  painter->save();
314  QColor color(Qt::darkBlue);
315  color.setAlpha(50);
316  painter -> setBrush (QBrush (color));
317  painter -> setPen (Qt::NoPen);
318  painter -> drawPath (shape());
319  painter -> restore ();
320  }
321 
322  switch (m_shapeType)
323  {
324  case Line: painter->drawLine(QLineF(m_P1, m_P2)); break;
325  case Rectangle: painter->drawRoundedRect(QRectF(m_P1, m_P2),
326  m_xRadius,

375 QVariant QetShapeItem::itemChange(QGraphicsItem::GraphicsItemChange change,
376  const QVariant &value)
377 {
378  if (change == ItemSelectedHasChanged)
379  {
380  if (value.toBool() == true) {
381  //If this is selected, wa add handlers.
382  addHandler();
383  }
384  else //Else this is deselected, we remove handlers
385  {
386  if(!m_handler_vector.isEmpty())
387  {
388  qDeleteAll(m_handler_vector);
389  m_handler_vector.clear();
390  }
391  m_resize_mode = 1;
392  }
393  }
394  else if (change == ItemPositionHasChanged) {
395  adjusteHandlerPos();
396  }
397  else if (change == ItemSceneHasChanged)
398  {
399  if (!scene()) //This is removed from scene, then we deselect this, and so, the handlers is also removed.
400  {
401  setSelected(false);
402  }
403  }
404 
405  return QGraphicsItem::itemChange(change, value);

414 bool QetShapeItem::sceneEventFilter(QGraphicsItem *watched, QEvent *event)
415 {
416  //Watched must be an handler
417  if(watched->type() == QetGraphicsHandlerItem::Type)
418  {
419  QetGraphicsHandlerItem *qghi = qgraphicsitem_cast<QetGraphicsHandlerItem *>(watched);
420 
421  if(m_handler_vector.contains(qghi)) //Handler must be in m_vector_index, then we can start resize
422  {
423  m_vector_index = m_handler_vector.indexOf(qghi);
424  if (m_vector_index != -1)
425  {
426  if(event->type() == QEvent::GraphicsSceneMousePress) //Click
427  {
428  handlerMousePressEvent(qghi, static_cast<QGraphicsSceneMouseEvent *>(event));
429  return true;
430  }
431  else if(event->type() == QEvent::GraphicsSceneMouseMove) //Move
432  {
433  handlerMouseMoveEvent(qghi, static_cast<QGraphicsSceneMouseEvent *>(event));
434  return true;
435  }
436  else if (event->type() == QEvent::GraphicsSceneMouseRelease) //Release
437  {
438  handlerMouseReleaseEvent(qghi, static_cast<QGraphicsSceneMouseEvent *>(event));
439  return true;
440  }
441  }
442  }
443  }
444 
445  return false;
446 }

561 void QetShapeItem::addHandler()
562 {
563  if (m_handler_vector.isEmpty())
564  {
565  QVector <QPointF> points_vector;
566  switch (m_shapeType)
567  {
568  case Line:
569  points_vector << m_P1 << m_P2;
570  break;
571  case Rectangle:
572  if (m_resize_mode == 3) {
573  points_vector = QetGraphicsHandlerUtility::pointForRadiusRect(QRectF(m_P1, m_P2), m_xRadius, m_yRadius);
574  }
575  else {
576  points_vector = QetGraphicsHandlerUtility::pointsForRect(QRectF(m_P1, m_P2));
577  }
578  break;
579  case Ellipse:
580  points_vector = QetGraphicsHandlerUtility::pointsForRect(QRectF(m_P1, m_P2));
581  break;
582  case Polygon:
583  points_vector = m_polygon;
584  break;
585  }
586 
587  if(!points_vector.isEmpty() && scene())
588  {
589  m_handler_vector = QetGraphicsHandlerItem::handlerForPoint(mapToScene(points_vector));
590 
591  for(QetGraphicsHandlerItem *handler : m_handler_vector)
592  {
593  handler->setZValue(this->zValue()+1);
594  handler->setColor(Qt::blue);
595  scene()->addItem(handler);
596  handler->installSceneEventFilter(this);
597  }
598  }
599  }

606 void QetShapeItem::adjusteHandlerPos()
607 {
608  if (m_handler_vector.isEmpty()) {
609  return;
610  }
611 
612  QVector <QPointF> points_vector;
613  switch (m_shapeType)
614  {
615  case Line: {
616  points_vector << m_P1 << m_P2;
617  break;
618  }
619  case Rectangle: {
620  if (m_resize_mode != 3) {
621  points_vector = QetGraphicsHandlerUtility::pointsForRect(QRectF(m_P1, m_P2));
622  }
623  else {
624  points_vector = QetGraphicsHandlerUtility::pointForRadiusRect(QRectF(m_P1, m_P2), m_xRadius, m_yRadius);
625  }
626  break;
627  }
628  case Ellipse: {
629  points_vector = QetGraphicsHandlerUtility::pointsForRect(QRectF(m_P1, m_P2));
630  break;
631  }
632  case Polygon: {
633  points_vector = m_polygon;
634  break;
635  }
636  }
637 
638  if (m_handler_vector.size() == points_vector.size())
639  {
640  points_vector = mapToScene(points_vector);
641  for (int i = 0 ; i < points_vector.size() ; ++i)
642  m_handler_vector.at(i)->setPos(points_vector.at(i));
643  }
644  else
645  {
646  qDeleteAll(m_handler_vector);
647  m_handler_vector.clear();
648  addHandler();
649  }
650 }

668 void QetShapeItem::removePoint()
669 {
670  if (m_shapeType != QetShapeItem::Polygon) {
671  return;
672  }
673 
674  if (m_handler_vector.size() == 2) {
675  return;
676  }
677 
678  QPointF point = mapToScene(m_context_menu_pos);
679  int index = -1;
680  for (int i=0 ; i<m_handler_vector.size() ; i++)
681  {

683  if (qghi->contains(qghi->mapFromScene(point)))
684  {
685  index = i;

687  }
688  }
689  if (index > -1 && index<m_handler_vector.count())
690  {
691  QPolygonF polygon = this->polygon();

693 
694  //Wrap the undo for avoid to merge the undo commands when user add several points.
695  QUndoCommand *undo = new QUndoCommand(tr("Supprimer un point d'un polygone"));
696  new QPropertyUndoCommand(this, "polygon", this->polygon(), polygon, undo);
697  diagram()->undoStack().push(undo);
698  }

726 void QetShapeItem::handlerMouseMoveEvent(QetGraphicsHandlerItem *qghi, QGraphicsSceneMouseEvent *event)
727 {
728  Q_UNUSED(qghi)
729 
730  QPointF new_pos = event->scenePos();
731  if (event->modifiers() != Qt::ControlModifier)
732  new_pos = Diagram::snapToGrid(event->scenePos());
733  new_pos = mapFromScene(new_pos);
734 
735  switch (m_shapeType)
736  {
737  case Line:
738  prepareGeometryChange();
739  m_vector_index == 0 ? m_P1 = new_pos : m_P2 = new_pos;
740  adjusteHandlerPos();
741  break;
742 
743  case Rectangle:
744  if (m_resize_mode == 1) {
745  setRect(QetGraphicsHandlerUtility::rectForPosAtIndex(QRectF(m_P1, m_P2), new_pos, m_vector_index));
746  break;
747  }
748  else if (m_resize_mode == 2) {
749  setRect(QetGraphicsHandlerUtility::mirrorRectForPosAtIndex(QRectF(m_P1, m_P2), new_pos, m_vector_index));
750  break;
751  }
752  else {
753  qreal radius = QetGraphicsHandlerUtility::radiusForPosAtIndex(QRectF(m_P1, m_P2), new_pos, m_vector_index);
754  if(m_modifie_radius_equaly) {
755  setXRadius(radius);
756  setYRadius(radius);
757  }
758  else if(m_vector_index == 0) {
759  setXRadius(radius);
760  }
761  else {
762  setYRadius(radius);
763  }
764  adjusteHandlerPos();
765  break;
766  }
767  case Ellipse:
768  if (m_resize_mode == 1) {
769  setRect(QetGraphicsHandlerUtility::rectForPosAtIndex(QRectF(m_P1, m_P2), new_pos, m_vector_index));
770  break;
771  }
772  else {
773  setRect(QetGraphicsHandlerUtility::mirrorRectForPosAtIndex(QRectF(m_P1, m_P2), new_pos, m_vector_index));
774  break;
775  }
776 
777  case Polygon:
778  prepareGeometryChange();
779  m_polygon.replace(m_vector_index, new_pos);
780  adjusteHandlerPos();
781  break;
782  } //End switch
783 }

790 void QetShapeItem::handlerMouseReleaseEvent(QetGraphicsHandlerItem *qghi, QGraphicsSceneMouseEvent *event)
791 {
792  Q_UNUSED(qghi);
793  Q_UNUSED(event);
794 
795  m_modifie_radius_equaly = false;
796 
797  if (diagram())
798  {
799  QPropertyUndoCommand *undo = nullptr;
800  if ((m_shapeType & (Line | Rectangle | Ellipse)) && ((m_P1 != m_old_P1 || m_P2 != m_old_P2) ||
801  (m_old_xRadius != XRadius() || m_old_yRadius != m_yRadius))
802  )
803  {
804  switch(m_shapeType)
805  {
806  case Line: {
807  undo = new QPropertyUndoCommand(this, "line",QLineF(m_old_P1, m_old_P2), QLineF(m_P1, m_P2));
808  break;
809  }
810  case Rectangle: {
811  if (m_resize_mode == 1 || m_resize_mode == 2) {
812  undo = new QPropertyUndoCommand(this, "rect",QRectF(m_old_P1, m_old_P2), QRectF(m_P1, m_P2).normalized());
813  }
814  else if (m_resize_mode == 3)
815  {
816  undo = new QPropertyUndoCommand(this, "xRadius", m_old_xRadius, m_xRadius);
817  QPropertyUndoCommand *undo_ = new QPropertyUndoCommand(this, "yRadius", m_old_yRadius, m_yRadius, undo);
818  undo_->setAnimated();
819  }
820  break;
821  }
822  case Ellipse: {
823  undo = new QPropertyUndoCommand(this, "rect",QRectF(m_old_P1, m_old_P2), QRectF(m_P1, m_P2).normalized());
824  break;
825  }
826  case Polygon: break;
827  }
828  if (undo) {
829  undo->setAnimated(true, false);
830  }
831  }
832  else if (m_shapeType == Polygon && (m_polygon != m_old_polygon))
833  undo = new QPropertyUndoCommand(this, "polygon", m_old_polygon, m_polygon);
834 
835  if(undo)
836  {
837  undo->setText(tr("Modifier %1").arg(name()));
838  diagram()->undoStack().push(undo);
839  }
840  }

853  is_movable_ = (e.attribute("is_movable").toInt());
854  m_closed = e.attribute("closed", "0").toInt();
855  m_pen = QETXML::penFromXml(e.firstChildElement("pen"));
856  m_brush = QETXML::brushFromXml(e.firstChildElement("brush"));
857 
858  QString type = e.attribute("type");
859  //@TODO Compatibility for version older than N4075, shape type was stored with an int
860  if (type.size() == 1)
861  {
862  switch(e.attribute("type","0").toInt())
863  {
864  case 0: m_shapeType = Line; break;
865  case 1: m_shapeType = Rectangle; break;
866  case 2: m_shapeType = Ellipse; break;
867  case 3: m_shapeType = Polygon; break;
868  }
869  }
870  //For version after N4075, shape is stored with a string
871  else
872  {
873  QMetaEnum me = metaObject()->enumerator(metaObject()->indexOfEnumerator("ShapeType"));
874  m_shapeType = QetShapeItem::ShapeType(me.keysToValue(type.toStdString().data()));
875  }
876 
877  if (m_shapeType != Polygon)
878  {
879  m_P1.setX(e.attribute("x1", nullptr).toDouble());
880  m_P1.setY(e.attribute("y1", nullptr).toDouble());
881  m_P2.setX(e.attribute("x2", nullptr).toDouble());
882  m_P2.setY(e.attribute("y2", nullptr).toDouble());
883 
884  if (m_shapeType == Rectangle)
885  {
886  setXRadius(e.attribute("rx", "0").toDouble());
887  setYRadius(e.attribute("ry", "0").toDouble());
888  }
889  }
890  else {
891  for(const QDomElement& de : QET::findInDomElement(e, "points", "point")) {
892  m_polygon << QPointF(de.attribute("x", nullptr).toDouble(), de.attribute("y", nullptr).toDouble());
893  }
894  }
895  setZValue(e.attribute("z", QString::number(this->zValue())).toDouble());
896 
897  return (true);
898 }

906 QDomElement QetShapeItem::toXml(QDomDocument &document) const

909 
910  //write some attribute
911  QMetaEnum me = metaObject()->enumerator(metaObject()->indexOfEnumerator("ShapeType"));
912  result.setAttribute("type", me.valueToKey(m_shapeType));
913  result.appendChild(QETXML::penToXml(document, m_pen));
914  result.appendChild(QETXML::brushToXml(document, m_brush));
915  result.setAttribute("is_movable", bool(is_movable_));
916  result.setAttribute("closed", bool(m_closed));
917 
918  if (m_shapeType != Polygon)
919  {
920  result.setAttribute("x1", QString::number(mapToScene(m_P1).x()));
921  result.setAttribute("y1", QString::number(mapToScene(m_P1).y()));
922  result.setAttribute("x2", QString::number(mapToScene(m_P2).x()));
923  result.setAttribute("y2", QString::number(mapToScene(m_P2).y()));
924

926  {
927  QRectF rect(m_P1, m_P2);
928  rect = rect.normalized();
929  qreal x = m_xRadius;
930  if (x > rect.width()/2) {
931  x = rect.width()/2;
932  }
933  qreal y = m_yRadius;
934  if (y > rect.height()/2) {
935  y = rect.height()/2;
936  }
937 
938  result.setAttribute("rx", QString::number(m_xRadius));
939  result.setAttribute("ry", QString::number(m_yRadius));
940  }
941  }
942  else
943  {
944  QDomElement points = document.createElement("points");
945  for (QPointF p : m_polygon)
946  {
947  QDomElement point = document.createElement("point");

949  point.setAttribute("x", QString::number(pf.x()));
950  point.setAttribute("y", QString::number(pf.y()));
951  points.appendChild(point);
952  }
953  result.appendChild(points);
954  }
955  result.setAttribute("z", QString::number(this->zValue()));
956
