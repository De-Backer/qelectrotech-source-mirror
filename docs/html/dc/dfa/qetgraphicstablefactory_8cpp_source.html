74 void QetGraphicsTableFactory::create(Diagram *diagram, AddTableDialog *dialog)
75 {
76  auto table_ = newTable(diagram, dialog);
77  if (dialog->adjustTableToFolio()) {
78  QetGraphicsTableItem::adjustTableToFolio(table_);
79  }
80 
81  //Add new table if needed and option checked
82  dialog->addNewTableToNewDiagram();
83  table_->model()->rowCount();
84  table_->displayNRow();
85  if (dialog->addNewTableToNewDiagram() && table_->model()->rowCount() > table_->displayNRow())
86  {
87  auto already_displayed_rows = table_->displayNRow();
88  auto project_ = diagram->project();
89  auto actual_diagram = diagram;
90  auto previous_table = table_;
91 
92  table_->setTableName(dialog->tableName() + QString(" 1"));
93  int table_number = 2;
94  while (already_displayed_rows < table_->model()->rowCount())
95  {
96  //Add a new diagram after the current one
97  actual_diagram = project_->addNewDiagram(project_->folioIndex(actual_diagram)+1);
98  table_ = newTable(actual_diagram, dialog, previous_table);
99  table_->setTableName(dialog->tableName() + QString(" %1").arg(table_number));
100  //Adjust table
101  if (dialog->adjustTableToFolio()) {
102  QetGraphicsTableItem::adjustTableToFolio(table_);
103  }
104  //Update some variable for the next loop
105  already_displayed_rows += table_->displayNRow();
106  previous_table = table_;
107  ++table_number;
108  }
109  }
110 }

121 QetGraphicsTableItem *QetGraphicsTableFactory::newTable(Diagram *diagram, AddTableDialog *dialog, QetGraphicsTableItem *previous_table)
122 {
123  auto table = new QetGraphicsTableItem();
124  table->setTableName(dialog->tableName());
125 
126  if (!previous_table)
127  {
128  QString identifier_;
129  QString query_;
130 
131  if (auto query_widget = dynamic_cast<ElementQueryWidget *>(dialog->contentWidget())) {
132  identifier_ = query_widget->modelIdentifier();
133  query_ = query_widget->queryStr();
134  } else if (auto query_widget = dynamic_cast<SummaryQueryWidget *>(dialog->contentWidget())) {
135  identifier_ = query_widget->modelIdentifier();
136  query_ = query_widget->queryStr();
137  }
138 
139  auto model = new ProjectDBModel(diagram->project(), diagram->project());
140  model->setIdentifier(identifier_);
141  model->setQuery(query_);
142  model->setData(model->index(0,0), int(dialog->tableAlignment()), Qt::TextAlignmentRole);
143  model->setData(model->index(0,0), dialog->tableFont(), Qt::FontRole);
144  model->setData(model->index(0,0), QETUtils::marginsToString(dialog->headerMargins()), Qt::UserRole+1);
145  model->setHeaderData(0, Qt::Horizontal, int(dialog->headerAlignment()), Qt::TextAlignmentRole);
146  model->setHeaderData(0, Qt::Horizontal, dialog->headerFont(), Qt::FontRole);
147  model->setHeaderData(0, Qt::Horizontal, QETUtils::marginsToString(dialog->headerMargins()), Qt::UserRole+1);
148  table->setModel(model);
149  }
150  else {
151  table->setPreviousTable(previous_table);
152  }
153 
154  diagram->addItem(table);
155  table->setPos(50,50);
156 
157  return table;
158 }
