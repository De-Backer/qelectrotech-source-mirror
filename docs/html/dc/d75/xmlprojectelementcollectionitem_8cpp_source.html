136 void XmlProjectElementCollectionItem::addChildAtPath(
137  const QString &collection_name)
138 {
139  if (collection_name.isEmpty())
140  return;
141 
142  QString str (collection_name.endsWith(".elmt")? "element" : "category");
143  QDomElement child_element = m_dom_element.firstChildElement(str);
144 
145  while (!child_element.isNull()) {
146  if (child_element.attribute("name") == collection_name) {
147  XmlProjectElementCollectionItem *xpeci = new XmlProjectElementCollectionItem ();
148  insertRow(rowForInsertItem(collection_name), xpeci);
149  xpeci->setXmlElement(child_element, m_project);
150  xpeci->setUpData();
151  return;
152  }
153  else
154  child_element = child_element.nextSiblingElement(str);
155  }
156 }

244 void XmlProjectElementCollectionItem::populate(bool set_data, bool hide_element)
245 {
246  QList <QDomElement> dom_category = m_project->embeddedElementCollection()->directories(m_dom_element);
247  std::sort(dom_category.begin(), dom_category.end(), [](QDomElement a, QDomElement b){return (a.attribute("name") < b.attribute("name"));});
248 
249  foreach (QDomElement element, dom_category)
250  {
251  XmlProjectElementCollectionItem *xpeci = new XmlProjectElementCollectionItem();
252  appendRow(xpeci);
253  xpeci->setXmlElement(element, m_project, set_data, hide_element);
254  if (set_data)
255  xpeci->setUpData();
256  }
257 
258  if (hide_element)
259  return;
260 
261  QList <QDomElement> dom_elements = m_project->embeddedElementCollection()->elements(m_dom_element);
262  std::sort(dom_elements.begin(), dom_elements.end(), [](QDomElement a, QDomElement b){return (a.attribute("name") < b.attribute("name"));});
263 
264  foreach (QDomElement element, dom_elements)
265  {
266  XmlProjectElementCollectionItem *xpeci = new XmlProjectElementCollectionItem();
267  appendRow(xpeci);
268  xpeci->setXmlElement(element, m_project, set_data);
269  if (set_data)
270  xpeci->setUpData();
271  }
272 }
