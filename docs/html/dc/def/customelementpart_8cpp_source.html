69 QList<QPointF> CustomElementPart::mapPoints(
70  const QRectF &initial_selection_rect,
71  const QRectF &new_selection_rect,
72  const QList<QPointF> &points) {
73  QList<QPointF> new_points;
74  if (!points.count()) return(new_points);
75 
76  // compare the new selection rectangle with the stored one to get the scaling ratio
77  qreal sx = new_selection_rect.width() / initial_selection_rect.width();
78  qreal sy = new_selection_rect.height() / initial_selection_rect.height();
79 
80  QPointF initial_top_left = initial_selection_rect.topLeft();
81  qreal new_top_left_x = new_selection_rect.x();
82  qreal new_top_left_y = new_selection_rect.y();
83 
84  foreach (QPointF point, points) {
85  QPointF point_offset = point - initial_top_left;
86  new_points << QPointF(
87  new_top_left_x + (point_offset.rx() * sx),
88  new_top_left_y + (point_offset.y() * sy)
89  );
90  }
91 
92  return(new_points);
93 }
