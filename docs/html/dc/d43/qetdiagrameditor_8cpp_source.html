980 bool QETDiagramEditor::openAndAddProject(const QString &filepath, bool interactive)
981 {
982  if (filepath.isEmpty()) return(false);
983 
984  QFileInfo filepath_info(filepath);
985 
986  //Check if project is not open in another editor
987  if (QETDiagramEditor *diagram_editor = QETApp::diagramEditorForFile(filepath))
988  {
989  if (diagram_editor == this)
990  {
991  if (ProjectView *project_view = viewForFile(filepath))
992  {
993  activateWidget(project_view);
994  show();
995  activateWindow();
996  }
997  return(false);
998  }
999  else
1000  {
1001  //Ask to the other editor to display the file
1002  return(diagram_editor -> openAndAddProject(filepath));
1003  }
1004  }
1005 
1006  // check the file exists
1007  if (!filepath_info.exists())
1008  {
1009  if (interactive)
1010  {
1011  QET::QetMessageBox::critical(
1012  this,
1013  tr("Impossible d'ouvrir le fichier", "message box title"),
1014  QString(
1015  tr("Il semblerait que le fichier %1 que vous essayez d'ouvrir"
1016  " n'existe pas ou plus.")
1017  ).arg(filepath)
1018  );
1019  }
1020  return(false);
1021  }
1022 
1023  //Check if file readable
1024  if (!filepath_info.isReadable())
1025  {
1026  if (interactive) {
1027  QET::QetMessageBox::critical(
1028  this,
1029  tr("Impossible d'ouvrir le fichier", "message box title"),
1030  tr("Il semblerait que le fichier que vous essayez d'ouvrir ne "
1031  "soit pas accessible en lecture. Il est donc impossible de "
1032  "l'ouvrir. Veuillez vérifier les permissions du fichier.")
1033  );
1034  }
1035  return(false);
1036  }
1037 
1038  //Check if file is read only
1039  if (!filepath_info.isWritable())
1040  {
1041  if (interactive) {
1042  QET::QetMessageBox::warning(
1043  this,
1044  tr("Ouverture du projet en lecture seule", "message box title"),
1045  tr("Il semblerait que le projet que vous essayez d'ouvrir ne "
1046  "soit pas accessible en écriture. Il sera donc ouvert en "
1047  "lecture seule.")
1048  );
1049  }
1050  }
1051 
1052  //Create the project
1053  DialogWaiting::instance(this);
1054 
1055  QETProject *project = new QETProject(filepath);
1056  if (project -> state() != QETProject::Ok)
1057  {
1058  if (interactive && project -> state() != QETProject::FileOpenDiscard)
1059  {
1060  QET::QetMessageBox::warning(
1061  this,
1062  tr("Échec de l'ouverture du projet", "message box title"),
1063  QString(
1064  tr(
1065  "Il semblerait que le fichier %1 ne soit pas un fichier"
1066  " projet QElectroTech. Il ne peut donc être ouvert.",
1067  "message box content"
1068  )
1069  ).arg(filepath)
1070  );
1071  }
1072  delete project;
