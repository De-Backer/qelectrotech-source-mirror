119 bool LineEditor::setPart(CustomElementPart *new_part)
120 {
121  if (!new_part)
122  {
123  if (part)
124  {
125  disconnectChangeConnections();
126  }
127  part = nullptr;
128  return(true);
129  }
130  if (PartLine *part_line = dynamic_cast<PartLine *>(new_part))
131  {
132  if (part == part_line) return true;
133  if (part)
134  {
135  disconnectChangeConnections();
136  }
137  part = part_line;
138  updateForm();
139  setUpChangeConnections();
140  return(true);
141  }
142  return(false);
143 }

192

196  undo->setText(tr("Modifier une ligne"));
