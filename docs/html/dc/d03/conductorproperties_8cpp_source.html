65 void SingleLineProperties::draw(QPainter *painter,
66  QET::ConductorSegmentType direction,
67  const QRectF &rect) {
68  // s'il n'y a rien a dessiner, on retourne immediatement
69  if (!hasNeutral && !hasGround && !phases) return;
70 
71  // prepare le QPainter
72  painter -> save();
73  QPen pen(painter -> pen());
74  pen.setCapStyle(Qt::FlatCap);
75  pen.setJoinStyle(Qt::MiterJoin);
76  pen.setStyle(Qt::SolidLine);
77  pen.setWidthF(1);
78  painter -> setPen(pen);
79  painter -> setRenderHint(QPainter::Antialiasing, true);
80 
81  uint symbols_count = (hasNeutral ? 1 : 0) + (hasGround ? 1 : 0) - (isPen() ? 1 : 0) + phases;
82  qreal interleave_base = (direction == QET::Horizontal ? rect.width() : rect.height());
83  qreal interleave = interleave_base / (symbols_count + 1);;
84  qreal symbol_width = interleave_base / 12;
85 
86  for (uint i = 1 ; i <= symbols_count ; ++ i) {
87  // dessine le tronc du symbole
88  QPointF symbol_p1, symbol_p2;
89  if (direction == QET::Horizontal) {
90  symbol_p1 = QPointF(rect.x() + (i * interleave) + symbol_width, rect.y() + rect.height() * 0.75);
91  symbol_p2 = QPointF(rect.x() + (i * interleave) - symbol_width, rect.y() + rect.height() * 0.25);
92  } else {
93  symbol_p2 = QPointF(rect.x() + rect.width() * 0.75, rect.y() + (i * interleave) - symbol_width);
94  symbol_p1 = QPointF(rect.x() + rect.width() * 0.25, rect.y() + (i * interleave) + symbol_width);
95  }
96  painter -> drawLine(QLineF(symbol_p1, symbol_p2));
97 
98  // dessine le reste des symboles terre et neutre
99  if (isPen()) {
100  if (i == 1) {
101  drawPen(painter, direction, symbol_p2, symbol_width);
102  }
103  } else {
104  if (hasGround && i == 1) {
105  drawGround(painter, direction, symbol_p2, symbol_width * 2.0);
106  } else if (hasNeutral && ((i == 1 && !hasGround) || (i == 2 && hasGround))) {
107  drawNeutral(painter, direction, symbol_p2, symbol_width * 1.5);
108  }
109  }
110  }
111  painter -> restore();
112 }

121 void SingleLineProperties::drawGround(QPainter *painter,
122  QET::ConductorSegmentType direction,
123  QPointF center,
124  qreal size) {
125  painter -> save();
126 
127  // prepare le QPainter
128  painter -> setRenderHint(QPainter::Antialiasing, false);
129  QPen pen2(painter -> pen());
130  pen2.setCapStyle(Qt::SquareCap);
131  painter -> setPen(pen2);
132 
133  // dessine le segment representant la terre
134  qreal half_size = size / 2.0;
135  QPointF offset_point(
136  (direction == QET::Horizontal) ? half_size : 0.0,
137  (direction == QET::Horizontal) ? 0.0 : half_size
138  );
139  painter -> drawLine(
140  QLineF(
141  center + offset_point,
142  center - offset_point
143  )
144  );
145 
146  painter -> restore();

321  // read style of conductor

341  m_wire_section = e.attribute("conductor_section");
342  text_size = e.attribute("numsize", QString::number(9)).toInt();

344  m_show_text = e.attribute("displaytext", QString::number(1)).toInt();
345  m_one_text_per_folio = e.attribute("onetextperfolio", QString::number(0)).toInt();
346  verti_rotate_text = e.attribute("vertirotatetext").toDouble();
347  horiz_rotate_text = e.attribute("horizrotatetext").toDouble();
348 
349  QMetaEnum me = QMetaEnum::fromType<Qt::Alignment>();
350  m_horizontal_alignment = Qt::Alignment(
351  me.keyToValue(
352  e.attribute(
353  "horizontal-alignment",
354  "AlignBottom"
355  ).toStdString().data()));
356  m_vertical_alignment = Qt::Alignment(
357  me.keyToValue(
358  e.attribute(
359  "vertical-alignment",
360  "AlignRight"
361  ).toStdString().data()));
